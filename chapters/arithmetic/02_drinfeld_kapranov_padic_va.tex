\chapter{The $p$-adic Vertex Algebra Locality Problem}
\label{v4-arith:drinfeld-kapranov}

\emph{In the voice of Drinfeld + Kapranov. Drinfeld built vertex
algebras on curves via chiral/factorization algebras with Beilinson;
Kapranov (1995, unpublished ``Analogies between the Langlands
correspondence and topological quantum field theory'') proposed
loops-at-a-point for $\mathrm{Spec}\,\mathbb{Z}$ and Kapranov--Vasserot
(Publ.~IHES~100, 2004, arXiv:math/0107143) built the formal-loop
chiral framework for curves over fields. No published construction
over an arithmetic base exists.}

\section{The Algebraic Lane: Formal VA over $\mathbb{Q}_{p}$}
\label{v4-arith:dk:algebraic-lane}

A vertex algebra over a field $k$ of characteristic zero is a
vector space $V_{k}$ with operations
\begin{equation}
Y\colon V_{k}\otimes V_{k}\to V_{k}((z))
\end{equation}
and axioms which are polynomial identities (Kac, \emph{Vertex
Algebras for Beginners}, 2nd ed., 1998, \S1.3). Replacing $\mathbb{C}$
by $\mathbb{Q}_{p}$ as the coefficient field is painless
\emph{at the purely algebraic level}: the Heisenberg Fock space
$\mathcal{H}^{(p)}_{\mathbb{Q}_{p}}=\mathrm{Sym}(\mathbb{Q}_{p}[t]\cdot a)$
with modes $[a_{m},a_{n}]=m\delta_{m+n,0}$ and the OPE
$a(z)a(w)\sim 1/(z-w)^{2}$ are identities of formal power series in
$\mathrm{End}(\mathcal{H}^{(p)})[[z^{\pm 1},w^{\pm 1}]]$.

\begin{proposition}
\label{v4-arith:prop:formal-padic-va}
\ClaimStatusProvedHere (citation-level; Kac Thm~4.3a). For any
field $k$ of characteristic zero, the rank-$2$ Heisenberg VA
$\mathcal{H}^{(p)}_{k}=\mathrm{Sym}_{k}(k[t](a\oplus b))$ with standard
OPE exists as a formal VA over $k$.
\end{proposition}

The bluff is not here. It is in the arithmetic content claimed for
this formal VA.

\section{The Conformal Vector Scaling is Not $p$-adic}
\label{v4-arith:dk:conformal-vector-scaling}

\begin{observation}
\label{v4-arith:obs:log-p-not-padic}
The proposed conformal vector
$\omega_{p}=\tfrac{1}{2}(\log p)(a_{-1}^{2}+b_{-1}^{2})|0\rangle$
has a complex scalar $\log p\in\mathbb{R}$. If
$\mathcal{H}^{(p)}$ is a $\mathbb{Q}_{p}$-VA, then
$\omega_{p}\in\mathcal{H}^{(p)}$ requires $\log p\in\mathbb{Q}_{p}$.
The only candidate is the Iwasawa $p$-adic logarithm
$\log_{p}\colon\mathbb{Z}_{p}^{\times}\to\mathbb{Q}_{p}$, which by
convention sets $\log_{p}(p)=0$ (Iwasawa, \emph{Lectures on p-adic
L-functions}, Princeton 1972, \S1; Koblitz, \emph{p-adic Analysis},
GTM~58, Ch.~IV).
\end{observation}

Under Iwasawa's $\log_{p}$, the $p$-adic interpretation of
$\omega_{p}$ gives the zero conformal vector. Reinterpretation:
the scalar $\log p\in\mathbb{R}$ is the \emph{scale} at which
$\mathcal{H}^{(p)}$ contributes to the global partition function
(Hartnoll--Yang arXiv:2502.02661~\S3), not an internal $\mathbb{Q}_{p}$-parameter of the VA. The central charge of
$\omega_{p}$ is $2$ regardless; the $L_{0}$-grading carries the
$\log p$.

\begin{convention}
\label{v4-arith:conv:log-p-archimedean}
$\log p$ enters as archimedean-real scaling data, not as an
internal $p$-adic coefficient. At finite place $p$ the VA is
$\mathcal{H}^{(p)}_{\mathbb{Q}_{p}}$ with vacuum conformal vector
$\tfrac{1}{2}(a_{-1}^{2}+b_{-1}^{2})|0\rangle$; the rescaling
enters through external grading data at the archimedean
completion.
\end{convention}

\section{The Restricted Tensor Product is Not Over a Field}
\label{v4-arith:dk:restricted-tensor-not-field}

\begin{observation}
\label{v4-arith:obs:no-single-field}
The restricted tensor product $\bigotimes_{p}^{!}\mathcal{H}^{(p)}$
cannot be a vector space over any single field because the
$\mathcal{H}^{(p)}$ are vector spaces over different fields
$\mathbb{Q}_{p}$. The colimit of finite tensor products of
$\mathbb{Q}_{p}$ over distinct primes does not exist in the
category of fields.
\end{observation}

The three resolutions:
\begin{enumerate}
\item \textbf{(Base-extend to $\mathbb{C}$):} Each
$\mathcal{H}^{(p)}$ is base-extended to $\mathbb{C}$ before
tensoring; the whole $\mathbb{Q}_{p}$-structure is lost and we are
back to the standard complex Heisenberg. Tautological.
\item \textbf{(Base-extend to $\mathbb{A}_{\mathbb{Q}}$):} Tensor
over the adele ring, which is a ring, not a field. The result is
an adelic module, not a VA in the classical sense.
\item \textbf{(Form functor):} Define
$\mathcal{V}^{\mathrm{prim}}\colon\mathrm{Fields}/\mathbb{Q}\to\mathrm{VA}$
with $\mathcal{V}^{\mathrm{prim}}(\mathbb{Q}_{p})=\mathcal{H}^{(p)}$
and $\mathcal{V}^{\mathrm{prim}}(\mathbb{C})=$ complex Heisenberg.
This is a motive-like structure.
\end{enumerate}

\begin{warning}
Option~(a) is tautological. Option~(b) is adelic and interesting
but not a VA. Option~(c) requires arithmetic input (Galois action,
Frobenius) absent in the naive construction.
\end{warning}

\section{The $p$-adic Diagonal Has No Thickening}
\label{v4-arith:dk:padic-diagonal-no-thickening}

\begin{proposition}
\label{v4-arith:prop:padic-diagonal-collapse}
\ClaimStatusProvedHere. The formal completion
$\mathbb{Z}_{p}\,\widehat{\otimes}_{\mathbb{Z}}\,\mathbb{Z}_{p}\cong\mathbb{Z}_{p}$,
because $\mathbb{Z}_{p}$ is $p$-adically complete and the diagonal
is already an isomorphism after completion.
\end{proposition}

\begin{proof}
The naive $\mathbb{Z}_{p}\otimes_{\mathbb{Z}}\mathbb{Z}_{p}=
\varprojlim_{n}(\mathbb{Z}/p^{n})\otimes_{\mathbb{Z}}\varprojlim_{m}(\mathbb{Z}/p^{m})$ is not
$\varprojlim_{n,m}(\mathbb{Z}/p^{n}\otimes\mathbb{Z}/p^{m})=
\varprojlim(\mathbb{Z}/p^{\min(n,m)})$; the tensor product of
$p$-adic integers does not commute with inverse limits on the
nose. After $p$-adic completion,
$\mathrm{Spec}(\mathbb{Z}_{p}\,\widehat{\otimes}\,\mathbb{Z}_{p})
=\mathrm{Spec}\,\mathbb{Z}_{p}$, which is one-dimensional, and the
diagonal is trivial.
\end{proof}

\begin{corollary}[No $p$-adic OPE]
\label{v4-arith:cor:no-padic-ope}
\ClaimStatusProvedHere. The Beilinson--Drinfeld chiral bracket
$\mu^{\mathrm{ch}}\colon j_{*}j^{*}(\mathcal{A}\boxtimes\mathcal{A})
\to\Delta_{!}\mathcal{A}$ as a residue at the diagonal does not
exist at a finite place, because the formal completion of the
diagonal is trivial. The archimedean place supports the full
holomorphic OPE; each finite place supports only the
\emph{locally constant} structure on
$\mathrm{Spf}\,\mathbb{Z}_{p}$.
\end{corollary}

\section{The Locally Constant (\texorpdfstring{$E_{0}$}{E0}) Local Model at Each Finite Place}
\label{v4-arith:dk:E0-local-model}

\begin{definition}
\label{v4-arith:def:E0-local-factor}
The local factor of $\mathcal{V}^{\mathrm{prim}}$ at a finite place
$p$ is the locally constant prestack
$\mathcal{V}^{\mathrm{prim}}|_{p}$ on $\mathrm{Spf}\,\mathbb{Z}_{p}$
whose constant stalk is $\mathcal{H}^{(p)}$ equipped with the
$p$-adic filtration and a compatible Hecke--Adams endomorphism
$T_{p}\colon\mathcal{H}^{(p)}\to\mathcal{H}^{(p)}$. This is a
factorization algebra in the degenerate ($E_{0}$) sense: the
``fusion'' $\mathcal{H}^{(p)}\otimes\mathcal{H}^{(p)}\to\mathcal{H}^{(p)}$
is the commutative multiplication, not an OPE.
\end{definition}

\begin{theorem}[global structure at archimedean place]
\label{v4-arith:thm:archimedean-is-holomorphic}
\ClaimStatusProvedHere. At $\infty$, $\mathcal{V}^{\mathrm{prim}}|_{\infty}$
is the ordinary Heisenberg vertex algebra on a germ
$(\mathbb{C},0)$, a genuine Beilinson--Drinfeld/Costello--Gwilliam
factorization algebra with non-trivial diagonal singularity.
\end{theorem}

\section{The True Hidden Structure: Adelic Automorphic Object with Local Langlands at Each Place}
\label{v4-arith:dk:true-hidden-structure}

\begin{theorem}[healed statement]
\label{v4-arith:thm:adelic-automorphic-placement}
\ClaimStatusConjectured. $\mathcal{V}^{\mathrm{prim}}$ is an adelic
automorphic object: a compatible system of local factors
$\{\mathcal{V}^{\mathrm{prim}}|_{v}\}_{v\in|\overline{C}|}$ with
$\mathcal{V}^{\mathrm{prim}}|_{\infty}$ a genuine Heisenberg VA at
the archimedean place and $\mathcal{V}^{\mathrm{prim}}|_{p}$ an
$E_{0}$-algebra (Fock module with Hecke-Adams endomorphism) at
each finite place. Under Tate's restricted tensor product over
spherical vectors $|0\rangle_{p}$ (Tate~1950; Weil~1967,
\emph{Basic Number Theory}, Ch.~VII), the system assembles into an
automorphic representation of $GL_{2}(\mathbb{A}_{\mathbb{Q}})$
with trivial ramification at unramified primes.
\end{theorem}

\begin{remark}
The classical chiral-algebras formalism of Beilinson--Drinfeld
(AMS~Colloq.~51, 2004) does not apply directly to
$\overline{C}=\overline{\mathrm{Spec}\,\mathbb{Z}}$. What replaces
it: a compatible system of factorization-algebra factors (at the
archimedean place) and locally constant factors (at finite
places), tied together by automorphic compatibility (Tate-style
restriction at each place). The global object is a spectral
datum; its OPE lives only at $\infty$.
\end{remark}

\section{What Kapranov and Kapranov--Vasserot Actually Built}
\label{v4-arith:dk:kapranov-actually-built}

\begin{remark}
Kapranov--Vasserot's four-paper series \emph{Vertex algebras and
the formal loop space} (arXiv:math/0107143, Publ.~IHES~100, 2004;
and subsequent math/0612371, etc.) introduces the ind-scheme
$\mathcal{L}X$ of formal loops into a finite-type scheme $X$ and
shows that factorization structure on $\mathcal{L}X$ recovers the
VA structure on chiral de Rham. There is no $\mathrm{Spec}\,\mathbb{Z}$
version in the Kapranov--Vasserot series. The arithmetic extension
is what this branch of Vol~IV attempts to formulate.
\end{remark}

\begin{remark}
Kapranov's 1995 unpublished preprint ``Analogies between the
Langlands correspondence and topological quantum field theory''
\emph{motivates} the search for an arithmetic chiral-algebra
formalism but does not construct one. Kapranov--Smirnov's
\emph{Cohomology determinants and reciprocity laws: number field
case} (also unpublished, ca.~1995) gives an $\mathbb{F}_{1}$-style
analogy via cohomology determinants feeding reciprocity laws; this
is the foundational informal reference for the $\mathbb{F}_{1}$
literature (Manin arXiv:1312.5160; Thas, \emph{F$_{1}$ for
Everyone}, 2018; Lorscheid arXiv:1811.01341 survey).
\end{remark}

\section{Summary of This Chapter}
\label{v4-arith:dk:summary}

\begin{enumerate}
\item A formal VA over $\mathbb{Q}_{p}$ exists algebraically
(\Cref{v4-arith:prop:formal-padic-va}) but carries no arithmetic
content: the $\log p$ scaling is not $p$-adic
(\Cref{v4-arith:obs:log-p-not-padic}).
\item The restricted tensor product $\bigotimes^{!}_{p}\mathcal{H}^{(p)}$
is not over any single field
(\Cref{v4-arith:obs:no-single-field}); it must be an adelic module
or a form-functor $\mathrm{Fields}/\mathbb{Q}\to\mathrm{VA}$.
\item The $p$-adic diagonal has no thickening
(\Cref{v4-arith:prop:padic-diagonal-collapse}), so no $p$-adic OPE
exists (\Cref{v4-arith:cor:no-padic-ope}).
\item Each finite place carries an $E_{0}$-algebra
(\Cref{v4-arith:def:E0-local-factor}); the archimedean place
carries a genuine Heisenberg VA
(\Cref{v4-arith:thm:archimedean-is-holomorphic}).
\item The true object is an adelic automorphic datum
(\Cref{v4-arith:thm:adelic-automorphic-placement}), not a VA on
$\overline{C}$.
\end{enumerate}

This chapter identifies one joint at which the naive speculation
fails and the structural correction that survives. The remaining
chapters address additional joints and converge on the common
hidden structure.
