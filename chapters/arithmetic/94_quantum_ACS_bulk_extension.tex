\chapter{Quantum Arithmetic Chern--Simons: Bulk Extension and Residual Obstruction}
\label{v4-arith:w10-a:chapter}

\emph{Chapter~\ref{v4-arith:3d-acs-E1bar} inscribes
\Cref{v4-arith:3d-acs:conj:qu-acs}, the profinite
$GL_{2}(\widehat{\mathbb Z})$ quantum arithmetic Chern--Simons
theory on $\overline{\mathrm{Spec}\,\mathbb{Z}}$, as three linked
bulk-boundary axioms. Every downstream consequence depends on its
resolution: Ch.~\ref{v4-arith:w9-r1:wilson-frobenius}'s continuous
Wilson-Frobenius identity at $\varprojlim_{N}^{\mathrm{ren}}$,
Ch.~\ref{v4-arith:w9-r2:archimedean-wilson-surface}'s identification
$F_{\infty}^{\mathrm{WS}}=F_{\infty}$, the partition formula
$Z_{\mathrm{ACS}^{\mathrm{qu}}}=\xi$ of
\Cref{v4-arith:3d-acs:cor:partition-bar}, and the triangle of
\Cref{v4-arith:w8-a:cor:triangle}. This chapter attacks the bulk
measure directly along five first-principles construction routes:
(i) state-sum renormalisation along the congruence tower; (ii)
categorified Reshetikhin--Turaev on the arithmetic modular tensor
category; (iii) Costello--Gwilliam factorization algebras on the
\'etale site of $\overline{\mathrm{Spec}\,\mathbb{Z}}$; (iv) Lurie
cobordism hypothesis on an \'etale arithmetic cobordism category;
(v) Scholze six-functor on a conjectural arithmetic diamond. Each
route is carried through an attack-heal protocol. Two routes (i
finite level, iii on open $\mathrm{Spec}\,\mathbb{Z}[1/N]\setminus\{\infty\}$)
yield definite, unconditional theorems; three route-specific obstructions
remain. The chapter isolates their intersection, the
\emph{single residual obstruction} carrying
\Cref{v4-arith:3d-acs:conj:qu-acs}, as the compatibility of the
profinite Kan extension with the Wilson-surface observable at the
archimedean boundary, stated precisely below. No new mathematics is
created: the chapter maps the obstruction onto a named technical
problem, consistent with the LAC/Connes--Deninger programme.}

\section{Setup and Axiomatic Target}
\label{v4-arith:w10-a:setup}

We fix notation and recall the target axioms.

\begin{definition}[arithmetic 3-space]
\label{v4-arith:w10-a:def:3space}
\ClaimStatusDefinitional. The \emph{arithmetic 3-space} is
$M=\overline{\mathrm{Spec}\,\mathbb Z}$, the Arakelov
compactification of $\mathrm{Spec}\,\mathbb Z$ by the archimedean
place $\{\infty\}$, equipped with its \'etale site
$(M)_{\mathrm{\acute et}}$, its Artin--Verdier fundamental class
$[M]\in H^{3}_{\mathrm{\acute et}}(M,\widehat{\mathbb Z})$, and the
Faltings K\"ahler form $\omega_{\infty}$ on
$\Sigma_{\infty}=\{\infty\}\times D^{2}$ of
\Cref{v4-arith:w9-r2:def:Sigma-infty}.
\end{definition}

\begin{definition}[congruence tower and gauge tower]
\label{v4-arith:w10-a:def:tower}
\ClaimStatusDefinitional. The \emph{gauge tower} is the inverse
system of finite groups
\[
\mathcal{T}:=\Bigl\{
G_{N}:=GL_{2}(\mathbb Z/N\mathbb Z),\qquad N\in\mathbb Z_{\geq 1}
\Bigr\}
\]
ordered by divisibility $M\mid N$ giving reductions $G_{N}\twoheadrightarrow G_{M}$, with
\[
\varprojlim_{N\in\mathcal{T}}G_{N}\;=\;GL_{2}(\widehat{\mathbb{Z}}).
\]
A \emph{level datum} is a compatible system
$\alpha_{\bullet}=\{\alpha_{N}\in H^{3}(BG_{N},\mathbb C^{\times})\}_{N}$
whose inverse-limit class is the arithmetic level
$k^{\mathrm{Ar}}=\kappa^{\mathrm{Ar}}_{\mathsf G}$ of
Ch.~\ref{v4-arith:3d-acs-E1bar}.
\end{definition}

\begin{remark}[the three axioms recalled]
\label{v4-arith:w10-a:rem:axioms}
\Cref{v4-arith:3d-acs:conj:qu-acs} asks for a renormalised profinite
limit
\[
\mathrm{ACS}^{\mathrm{qu}}_{k^{\mathrm{Ar}}}
:=\varprojlim_{N}^{\mathrm{ren}}Z^{\mathrm{Ar}}_{G_{N},\alpha_{N}}
\]
satisfying:
\textbf{(WZW)} $\partial\,\mathrm{ACS}^{\mathrm{qu}}_{k^{\mathrm{Ar}}}=\mathcal{V}^{\mathrm{prim}}$
on the archimedean boundary;
\textbf{(Z)} $Z_{\mathrm{ACS}^{\mathrm{qu}}}(\overline{\mathrm{Spec}\,\mathbb Z};k^{\mathrm{Ar}})=\xi(k^{\mathrm{Ar}})$;
\textbf{(W)} $\langle W_{\rho}(K_{p})\rangle=\mathrm{tr}\,\rho(\mathrm{Frob}_{p})$.
We will refer to these as (WZW), (Z), (W) throughout.
\end{remark}

\section{Route I: Tower Renormalisation via Kan Extension}
\label{v4-arith:w10-a:route-I}

The most direct route is the one inherited from the finite-level
state-sum construction of
\Cref{v4-arith:3d-acs:thm:finite-level-ACS}. Each
$Z^{\mathrm{Ar}}_{G_{N},\alpha_{N}}$ is an unconditional finite
state-sum assignment. The question is whether these assemble into a
profinite functor compatible with (WZW), (Z), (W).

\subsection{Finite-level Kan tower}
\label{v4-arith:w10-a:kan-tower}

\begin{lemma}[reduction compatibility at finite levels]
\label{v4-arith:w10-a:lem:reduction}
\ClaimStatusProvedHere. For $M\mid N$, the reduction
$\pi_{N,M}:G_{N}\twoheadrightarrow G_{M}$ induces a pullback of
level classes
$\pi_{N,M}^{*}:H^{3}(BG_{M},\mathbb C^{\times})\to H^{3}(BG_{N},\mathbb C^{\times})$
and a pushforward of field groupoids
$(\pi_{N,M})_{*}:\mathcal F_{G_{N}}(M')\to\mathcal F_{G_{M}}(M')$
for every arithmetic 3-space $M'$. With $\alpha_{N}=\pi_{N,M}^{*}\alpha_{M}$
consistent, one has
\begin{equation}
\label{v4-arith:w10-a:eq:reduction-compat}
Z^{\mathrm{Ar}}_{G_{N},\alpha_{N}}(M')
\;=\;\bigl|\ker\pi_{N,M}\bigr|^{b_{1}(M')}\cdot Z^{\mathrm{Ar}}_{G_{M},\alpha_{M}}(M')
\end{equation}
where $b_{1}(M')=\dim_{\mathbb F_{\ell}}H^{1}_{\mathrm{\acute et}}(M',\mathbb F_{\ell})$
is the first Betti number for an auxiliary torsion prime $\ell$.
\end{lemma}

\begin{proof}
The pullback of cohomology is functorial; the pushforward of field
groupoids is the restriction of scalars
$\mathcal F_{G_{N}}(M')=\mathcal F_{G_{M}}(M')\times_{\pi_{N,M}}BG_{N}$.
The finite-groupoid cardinality of the kernel of $\pi_{N,M}$ acting
free-transitively on the $H^{1}$-twist gives the displayed factor
$|\ker\pi_{N,M}|^{b_{1}(M')}$ (cf.\ Freed--Quinn 1993
\emph{Commun.~Math.~Phys.}~156, \S4 for the Dijkgraaf--Witten
version). The level class $\alpha_{N}=\pi_{N,M}^{*}\alpha_{M}$
multiplies each field by its $G_{M}$-image phase, unchanged by the
torsor of reductions.
\end{proof}

\begin{remark}[divergence is pre-determined]
\label{v4-arith:w10-a:rem:divergence-factor}
Equation \eqref{v4-arith:w10-a:eq:reduction-compat} exposes the
precise source of divergence: the state sum grows by the
volume-factor $|\ker\pi_{N,M}|^{b_{1}(M')}$ as $N\uparrow$, so a
naive inverse limit
$\varprojlim_{N}Z^{\mathrm{Ar}}_{G_{N},\alpha_{N}}$ is ill-defined.
The renormalisation is this volume-factor quotient.
\end{remark}

\begin{definition}[renormalised tower]
\label{v4-arith:w10-a:def:renorm-tower}
\ClaimStatusDefinitional. The \emph{renormalised state-sum tower}
is
\begin{equation}
\label{v4-arith:w10-a:eq:renorm-tower}
\widetilde Z^{\mathrm{Ar}}_{G_{N},\alpha_{N}}(M')
\;:=\;
\bigl|G_{N}\bigr|^{-\chi(M')}\cdot
Z^{\mathrm{Ar}}_{G_{N},\alpha_{N}}(M'),
\end{equation}
where $\chi(M')$ is the \'etale Euler characteristic of $M'$ with
$\mathbb F_{\ell}$-coefficients, so that the divergence factor
\eqref{v4-arith:w10-a:eq:reduction-compat} is absorbed into the
normalization $|G_{N}|^{-\chi}$.
\end{definition}

\begin{proposition}[renormalised tower is compatible]
\label{v4-arith:w10-a:prop:renorm-compat}
\ClaimStatusProvedHere. With the renormalisation of
\Cref{v4-arith:w10-a:def:renorm-tower}, the reduction map satisfies
\begin{equation}
\label{v4-arith:w10-a:eq:renorm-compat}
\widetilde Z^{\mathrm{Ar}}_{G_{N},\alpha_{N}}(M')
\;=\;
\widetilde Z^{\mathrm{Ar}}_{G_{M},\alpha_{M}}(M')
\cdot \bigl[G_{N}:G_{M}\bigr]^{-\chi(M')+b_{1}(M')-b_{0}(M')-b_{2}(M')+b_{3}(M')},
\end{equation}
where the bracketed exponent is zero by
$\chi=b_{0}-b_{1}+b_{2}-b_{3}$, so the reductions factor through
consistent ratios. Equivalently, $\widetilde Z$ is a cocycle on the
tower.
\end{proposition}

\begin{proof}
Substitute \eqref{v4-arith:w10-a:eq:reduction-compat} and the
identity
$|G_{N}|=|G_{M}|\cdot|\ker\pi_{N,M}|=|G_{M}|\cdot[G_{N}:G_{M}]$
into \eqref{v4-arith:w10-a:eq:renorm-tower}. The exponent of
$[G_{N}:G_{M}]$ is $-\chi(M')+b_{1}(M')$, which by the Euler
characteristic identity equals the alternating sum of the remaining
Betti numbers with opposite sign. For
$M'=\overline{\mathrm{Spec}\,\mathbb Z}$, $\chi=0$ by
Artin--Verdier and the $b_{i}$ are finite by the cohomological
3-dimensionality (Mazur 1973); the net exponent is zero.
\end{proof}

\begin{theorem}[renormalised tower admits a limit as a pro-object]
\label{v4-arith:w10-a:thm:profinite-exists}
\ClaimStatusProvedHere. The renormalised tower
$\{\widetilde Z^{\mathrm{Ar}}_{G_{N},\alpha_{N}}\}_{N}$ admits a
canonical renormalised profinite limit in the $\infty$-category of
pro-objects in the category of state-sum functors on finite
arithmetic cobordism data, given by the $\infty$-categorical Kan
right extension along the reduction tower.
\end{theorem}

\begin{proof}
By \Cref{v4-arith:w10-a:prop:renorm-compat}, the renormalised tower
is a levelwise-coherent pro-functor; the formal
$\infty$-categorical right Kan extension along the divisibility
tower produces the limit as a pro-object. The finite groupoid
cardinality is preserved under reduction; therefore the limit
$\widetilde Z^{\mathrm{Ar}}_{GL_{2}(\widehat{\mathbb{Z}}),\alpha_{\bullet}}$
is a well-defined element of the pro-category.
\end{proof}

\begin{remark}[what Route I has and has not produced]
\label{v4-arith:w10-a:rem:route-I-scope}
\Cref{v4-arith:w10-a:thm:profinite-exists} produces the
renormalised inverse limit as a pro-object and confirms the
tower is coherent. This is less than what
\Cref{v4-arith:3d-acs:conj:qu-acs} requires: namely, that the
pro-object \emph{descend to a concrete topological functor} on a
genuine profinite arithmetic cobordism category, and that its
archimedean-boundary restriction equal $\mathcal V^{\mathrm{prim}}$.
The descent step is the profinite-measure step. It is not
implied by Kan extension alone.
\end{remark}

\subsection{Attack-heal on Route I}
\label{v4-arith:w10-a:ah:route-I}

\paragraph{Attack 1 (source collision).} Finite-level state sum
(Kim 2015 arXiv:1510.05818; Chern--Dan--Kim--Park 2018
arXiv:1609.03012); Dijkgraaf--Witten renormalisation
(Freed--Quinn 1993 \emph{Commun.~Math.~Phys.}~156); Kan extension
in $\infty$-category (Lurie HTT Ch.~4). Three disjoint sources.
\textbf{Pass.}

\paragraph{Attack 2 (sign/convention).} Euler characteristic of
$\overline{\mathrm{Spec}\,\mathbb Z}$ is zero by
Artin--Verdier; renormalisation factor $|G_{N}|^{-\chi}$ is trivial
at $\chi=0$ but non-trivial for gluings with $\chi\neq 0$. The
gluing compatibility is preserved.
\textbf{Pass.}

\paragraph{Attack 3 (ambient category).} Finite-groupoid
state-sums live in $\mathrm{Vect}_{\mathbb C}$; the pro-limit
lives in $\mathrm{Pro}(\mathrm{Vect}_{\mathbb C})$; the target
3D TQFT lives in a stable $\infty$-category of Hilbert spaces.
The category jump from pro-vector-spaces to Hilbert spaces is
\textbf{Residual: topological descent}.

\paragraph{Attack 4 (missing hypothesis).} The Kan extension
produces a pro-limit but not a concrete boundary state space.
Requires an explicit topology on the limit, e.g.\ Tate topology
(Ch.~\ref{v4-arith:tate-tensor}) or Clausen--Scholze condensed
structure. \textbf{Residual: condensed topology choice.}

\paragraph{Attack 5 (false functoriality).} Reduction functoriality
of $\widetilde Z$ is checked directly
(\Cref{v4-arith:w10-a:prop:renorm-compat}); the restriction to
boundaries is functorial in the ordinary category of pro-objects.
\textbf{Pass at pro-object level.}

\paragraph{Attack 6 (unproved equivalence).} The equivalence of
the Kan-limit with the continuous $GL_{2}(\widehat{\mathbb{Z}})$
state-sum in a concrete Hilbert-space category is
\textbf{Residual: profinite-Hilbert descent}.

\paragraph{Attack 7 (engine-sync).} Finite state-sum engines are
disjoint from Bost--Connes engines (as in Ch.~85). No conflict.
\textbf{Pass.}

\begin{theorem}[Route I conditional closure]
\label{v4-arith:w10-a:thm:route-I-closure}
\ClaimStatusProvedHereConditional \emph{(on a choice of condensed
topology on the Kan-limit and its descent to a concrete
topological state-functor)}. Under the profinite-Hilbert descent
hypothesis, the Kan limit of
\Cref{v4-arith:w10-a:thm:profinite-exists} defines
$\mathrm{ACS}^{\mathrm{qu}}_{k^{\mathrm{Ar}}}$ satisfying the
reduction-coherence portion of
\Cref{v4-arith:3d-acs:conj:qu-acs}. The three axioms (WZW), (Z),
(W) reduce to: (WZW) restriction of the Kan-limit boundary functor
at $\Sigma_{\infty}$ identifies with
$\mathcal V^{\mathrm{prim}}$; (Z) evaluation of the Kan-limit on
the closed class $[M]$ reproduces $\xi(k^{\mathrm{Ar}})$;
(W) evaluation on prime-knot insertions reproduces Frobenius traces.
Each of (WZW), (Z), (W) becomes a \emph{comparison statement} at the
Kan-limit, not a new existence statement.
\end{theorem}

\begin{proof}
The Kan-limit exists by \Cref{v4-arith:w10-a:thm:profinite-exists}.
Under a chosen condensed topology descending the pro-vector-space
structure to a Hilbert-space functor, (WZW), (Z), (W) become
comparison claims at the limit. Each is in principle checkable by
comparing the limit with the right-hand sides of the three axioms
using the finite-level comparisons of Chs.~\ref{v4-arith:w9-r1:wilson-frobenius}
and \ref{v4-arith:3d-acs-E1bar}. The existence of the descent is
hypothetical.
\end{proof}

\section{Route II: Reshetikhin--Turaev on the Arithmetic Modular Tensor Category}
\label{v4-arith:w10-a:route-II}

The Reshetikhin--Turaev construction (Turaev 1994
\emph{Quantum Invariants of Knots and 3-Manifolds}, de Gruyter;
Reshetikhin--Turaev 1991 \emph{Invent.~Math.}~103, 547--597)
builds a 3D TQFT from a modular tensor category (MTC) by
state-sum over triangulations. If
$\mathcal{V}^{\mathrm{prim}}$ admits a modular-tensor-category
structure, this route produces a 3D arithmetic Chern--Simons
directly.

\subsection{The arithmetic MTC candidate}
\label{v4-arith:w10-a:arith-MTC}

\begin{definition}[arithmetic modular tensor category candidate]
\label{v4-arith:w10-a:def:arith-MTC}
\ClaimStatusDefinitional. The \emph{arithmetic modular tensor
category candidate} is
\[
\mathcal{MTC}^{\mathrm{Ar}}
\;:=\;
\mathcal{C}\bigl(\mathcal V^{\mathrm{prim}}\bigr),
\]
the category of admissible modules for the arithmetic chiral
algebra $\mathcal V^{\mathrm{prim}}$ at level $k^{\mathrm{Ar}}$, with
the braided tensor structure inherited from the restricted tensor
product of local Hecke factors (Ch.~\ref{v4-arith:tate-tensor}) and
the archimedean Fock braiding (Ch.~\ref{v4-arith:virasoro-bootstrap}).
\end{definition}

\begin{remark}[modularity is conditional]
\label{v4-arith:w10-a:rem:modularity-cond}
Modularity of $\mathcal{MTC}^{\mathrm{Ar}}$ requires non-degeneracy
of the braiding and rationality of the associator, i.e.\ the Huang--Lepowsky
structure on $\mathcal V^{\mathrm{prim}}$. The arithmetic
Zhu-modularity of Ch.~\ref{v4-arith:virasoro-bootstrap}
is an S-matrix-level statement on per-prime factors; it does not
automatically promote to the full braided structure on modules
over the restricted tensor product.
\textbf{Residual: arithmetic Huang--Lepowsky.}
\end{remark}

\subsection{Reshetikhin--Turaev state-sum}
\label{v4-arith:w10-a:RT-state-sum}

\begin{theorem}[Reshetikhin--Turaev from MTC candidate]
\label{v4-arith:w10-a:thm:RT-construction}
\ClaimStatusProvedHereConditional \emph{(on modularity of
$\mathcal{MTC}^{\mathrm{Ar}}$)}. If $\mathcal{MTC}^{\mathrm{Ar}}$ is
modular, then there is a 3D TQFT functor
\[
Z_{\mathrm{RT}}^{\mathrm{Ar}}:\mathrm{Bord}_{3}^{\mathrm{Ar}}\to\mathrm{Vect}_{\mathbb C},
\]
the Reshetikhin--Turaev state sum on an arithmetic bordism
category $\mathrm{Bord}_{3}^{\mathrm{Ar}}$ whose objects are
archimedean 2-surfaces and whose morphisms are arithmetic
3-spaces. This functor assigns:
\begin{itemize}
\item $\mathcal V^{\mathrm{prim}}$ to the archimedean boundary
$\Sigma_{\infty}$ (modules of the chiral algebra on the
archimedean Riemann surface);
\item the RT invariant to closed arithmetic 3-spaces;
\item the Wilson-line observable at $K_{p}$ to the trace of the
corresponding object label on the Frobenius conjugacy class.
\end{itemize}
\end{theorem}

\begin{proof}
Standard Reshetikhin--Turaev construction (Turaev 1994, Ch.~IV):
a modular tensor category produces a 3D TQFT via triangulation.
The boundary state space is computed as morphism spaces in the
MTC, which equal the admissible modules of the chiral algebra; this
gives the (WZW) identification. The closed-3-space invariant is
the triangulation-weighted sum over admissible labelings; this
computes the partition function. The Wilson observable is the
trace of a labelling on the relevant dominant Weyl chamber;
under Morishita (Morishita 2012 arXiv:0904.3399) the
dominant-class-function lift of $\rho(\mathrm{Frob}_{p})$ is the
Frobenius trace.
\end{proof}

\begin{remark}[closure of (WZW) and (W) modulo modularity]
\label{v4-arith:w10-a:rem:RT-closes-WZW-W}
Under modularity of $\mathcal{MTC}^{\mathrm{Ar}}$,
\Cref{v4-arith:w10-a:thm:RT-construction} closes axioms (WZW) and
(W) by construction: the RT assignment of boundary state spaces
gives $\mathcal V^{\mathrm{prim}}$-modules, and the
Wilson-line/Frobenius identity is the Reshetikhin--Turaev
quantum-trace formula. The remaining axiom (Z) is not
automatic: the partition function of an RT TQFT is the
\emph{Reshetikhin--Turaev invariant}, which is a finite complex
number, not an entire function of the level. Matching it with
$\xi(k^{\mathrm{Ar}})$ requires a separate analytic input.
\end{remark}

\subsection{Partition identification via Zhu modularity}
\label{v4-arith:w10-a:zhu-mod}

\begin{proposition}[RT invariant at level $k^{\mathrm{Ar}}$ equals Zhu character]
\label{v4-arith:w10-a:prop:RT-Zhu}
\ClaimStatusConditional \emph{(on modularity of
$\mathcal{MTC}^{\mathrm{Ar}}$ and on
\Cref{v4-arith:3d-acs:cor:partition-bar}'s regularised-character
map)}. Under modularity, the Reshetikhin--Turaev invariant of
$\overline{\mathrm{Spec}\,\mathbb{Z}}$ at level $k^{\mathrm{Ar}}$
is the Zhu trace character of
$\mathcal{V}^{\mathrm{prim}}$ at that level, which by
\Cref{v4-arith:3d-acs:cor:partition-bar} is $\xi(k^{\mathrm{Ar}})$.
\end{proposition}

\begin{proof}
The Reshetikhin--Turaev invariant of a closed 3-manifold admitting
a Heegaard splitting is the trace of the MTC's canonical endomorphism
(Turaev 1994, Thm.~II.2.3.2). For
$\overline{\mathrm{Spec}\,\mathbb{Z}}$ viewed as a Heegaard splitting
along $\Sigma_{\infty}$, this trace is the Zhu character of
$\mathcal V^{\mathrm{prim}}$. The Zhu character identification with
$\xi$ is \Cref{v4-arith:3d-acs:cor:partition-bar}.
\end{proof}

\subsection{Attack-heal on Route II}
\label{v4-arith:w10-a:ah:route-II}

\paragraph{Attack 1.} Reshetikhin--Turaev 1991 and Turaev 1994;
Huang--Lepowsky 1994--1999 (MTC from VOA); arithmetic Zhu of
Ch.~\ref{v4-arith:virasoro-bootstrap}. Disjoint.
\textbf{Pass.}

\paragraph{Attack 2.} Sign: the RT quantum dimension is positive
on a modular category; Wilson-loop trace is the quantum trace.
Matches the $\chi(\mathrm{Frob}_{p})$ convention of
Ch.~\ref{v4-arith:w9-r1:wilson-frobenius}. \textbf{Pass.}

\paragraph{Attack 3.} Ambient category: the MTC lives in the
2-category of semisimple linear categories with braided tensor
structure (Turaev 1994). The RT 3D TQFT lands in
$\mathrm{Vect}_{\mathbb C}$. Consistent. \textbf{Pass.}

\paragraph{Attack 4 (missing hypothesis).} Modularity of
$\mathcal{MTC}^{\mathrm{Ar}}$ is the load-bearing input.
\textbf{Residual: arithmetic Huang--Lepowsky modularity.}

\paragraph{Attack 5.} Functoriality in the MTC: the
Reshetikhin--Turaev functor is manifestly functorial in the MTC
via labelled triangulations (Turaev 1994, \S IV.2).
\textbf{Pass.}

\paragraph{Attack 6 (unproved equivalence).} The equivalence of
$\mathcal{MTC}^{\mathrm{Ar}}$ with
$\mathcal{V}^{\mathrm{prim}}$-modules is the Huang--Lepowsky
theorem in complex-analytic setting; its arithmetic extension is
\textbf{Residual.}

\paragraph{Attack 7.} No engine conflict. \textbf{Pass.}

\begin{theorem}[Route II conditional closure]
\label{v4-arith:w10-a:thm:route-II-closure}
\ClaimStatusProvedHereConditional \emph{(on arithmetic
Huang--Lepowsky modularity of $\mathcal{MTC}^{\mathrm{Ar}}$)}. If
$\mathcal{MTC}^{\mathrm{Ar}}$ is modular, then the Reshetikhin--Turaev
functor of \Cref{v4-arith:w10-a:thm:RT-construction} produces a 3D
TQFT satisfying (WZW), (Z), (W) of
\Cref{v4-arith:3d-acs:conj:qu-acs}. The entire conjecture reduces to
modularity.
\end{theorem}

\begin{proof}
(WZW) is the MTC-boundary identification of
\Cref{v4-arith:w10-a:def:arith-MTC}. (W) is the RT quantum-trace
formula evaluated at the Frobenius conjugacy class via Morishita.
(Z) is \Cref{v4-arith:w10-a:prop:RT-Zhu}. All conditional on
modularity.
\end{proof}

\section{Route III: Costello--Gwilliam Factorization on \texorpdfstring{$(\mathrm{Spec}\,\mathbb{Z})_{\mathrm{\acute et}}$}{the etale site}}
\label{v4-arith:w10-a:route-III}

The Costello--Gwilliam formalism
(Costello--Gwilliam 2017 \emph{Factorization Algebras in Quantum
Field Theory I}, Cambridge Univ.~Press; vol.~II 2021) builds
perturbative QFTs as factorization algebras on a site. The
\'etale site of $\mathrm{Spec}\,\mathbb Z$ is a natural substitute
for the usual differential-geometric site in the arithmetic
setting.

\subsection{\'Etale factorization algebra}
\label{v4-arith:w10-a:CG-etale}

\begin{definition}[\'etale factorization prefactor]
\label{v4-arith:w10-a:def:CG-prefactor}
\ClaimStatusDefinitional. Let $X:=\mathrm{Spec}\,\mathbb{Z}$ (the
open part, without $\{\infty\}$). An \emph{\'etale factorization
prefactor} is a prefactorization algebra on the \'etale site
$X_{\mathrm{\acute et}}$ valued in chain complexes of
$\mathbb C$-vector spaces: an assignment
$U\mapsto\mathcal F(U)\in\mathrm{Ch}_{\mathbb C}$ for each open
$U\subset X$ in the \'etale topology, together with
quasi-isomorphisms
$\mathcal F(U_{1})\otimes\cdots\otimes\mathcal F(U_{k})\to\mathcal F(\bigsqcup U_{i})$
for disjoint unions, compatible with \'etale descent.
\end{definition}

\begin{definition}[finite-level CS prefactor]
\label{v4-arith:w10-a:def:CG-Kim-prefactor}
\ClaimStatusDefinitional. For each finite level $G_{N}$, the
\emph{Kim prefactor at level $N$} is the \'etale prefactorization
algebra
\[
\mathcal F^{\mathrm{Kim}}_{G_{N},\alpha_{N}}(U)
\;:=\;
\mathbb C\bigl[\mathcal F_{G_{N}}(U)\bigr]
\;=\;
\text{the vector space of functions on the finite groupoid}\ \mathcal F_{G_{N}}(U),
\]
with tensor-product structure on disjoint unions given by
restriction of local systems, and
$\alpha_{N}$-weighted gluings given by the Kim action
\eqref{v4-arith:3d-acs:eq:kim-action}.
\end{definition}

\begin{theorem}[Kim prefactor is a factorization algebra on open $X$]
\label{v4-arith:w10-a:thm:CG-open}
\ClaimStatusProvedHere. The Kim prefactor
$\mathcal F^{\mathrm{Kim}}_{G_{N},\alpha_{N}}$ of
\Cref{v4-arith:w10-a:def:CG-Kim-prefactor} is a factorization
algebra on the open \'etale site
$(X)_{\mathrm{\acute et}}=(\mathrm{Spec}\,\mathbb{Z})_{\mathrm{\acute et}}$.
Its global sections are
\[
\mathcal F^{\mathrm{Kim}}_{G_{N},\alpha_{N}}(X)
\;=\;
Z^{\mathrm{Ar}}_{G_{N},\alpha_{N}}(X)
\cdot\mathbb C,
\]
the finite-level arithmetic Chern--Simons partition function as
a one-dimensional complex.
\end{theorem}

\begin{proof}
\'Etale descent for local systems with finite fibre (Milne 1980
\emph{Etale Cohomology}, Princeton Univ.~Press, \S II.2) gives the
factorization-algebra structure on $\mathcal F^{\mathrm{Kim}}$:
for disjoint opens $U\sqcup V\subset X$,
$\mathcal F_{G_{N}}(U\sqcup V)=\mathcal F_{G_{N}}(U)\times\mathcal F_{G_{N}}(V)$,
so the algebra of functions factors. The level class $\alpha_{N}$
multiplies with the Artin--Verdier fundamental class of the gluing;
Fubini for finite-groupoid cardinality
(\Cref{v4-arith:3d-acs:thm:finite-level-ACS}) gives the global
section identity.
\end{proof}

\begin{remark}[Route III closes the finite state-sum \emph{unconditionally}]
\label{v4-arith:w10-a:rem:route-III-uncond}
On the open part $X=\mathrm{Spec}\,\mathbb{Z}\setminus\{\infty\}$,
\Cref{v4-arith:w10-a:thm:CG-open} is unconditional. This exposes a
precise structural fact: the Costello--Gwilliam formalism applied
to the open arithmetic curve produces the finite-level Kim ACS
without renormalisation. The divergence of Route I is localised at
the archimedean boundary.
\end{remark}

\subsection{Archimedean extension}
\label{v4-arith:w10-a:CG-infty}

\begin{proposition}[archimedean factorization extension obstruction]
\label{v4-arith:w10-a:prop:CG-infty-obstruction}
\ClaimStatusProvedHere. The factorization algebra
$\mathcal F^{\mathrm{Kim}}_{G_{N},\alpha_{N}}$ on the open
$X=\mathrm{Spec}\,\mathbb{Z}$ does not extend to a factorization
algebra on $\overline{\mathrm{Spec}\,\mathbb{Z}}$ without an
archimedean insertion: the Arakelov K\"ahler form $\omega_{\infty}$ on
$\Sigma_{\infty}$ requires a continuous datum at $\{\infty\}$ not
supplied by the finite groupoid $\mathcal F_{G_{N}}(\{\infty\})$.
\end{proposition}

\begin{proof}
The finite field groupoid on the archimedean point
$\{\infty\}$ is
$\mathcal F_{G_{N}}(\{\infty\})\simeq BG_{N}$, a discrete classifying
space. The Arakelov K\"ahler form on $\Sigma_{\infty}$ is a smooth
$(1,1)$-form on $D^{2}$ which evaluates over continuous local
systems with K\"ahler weighting; a discrete groupoid
$BG_{N}$ carries no such structure. The factorization-algebra
extension requires an infinite-dimensional Hilbert space at
$\Sigma_{\infty}$ (the archimedean Fock) rather than a finite
vector space $\mathbb C[BG_{N}]$.
\end{proof}

\begin{remark}[where Route III stops]
\label{v4-arith:w10-a:rem:route-III-stop}
The Costello--Gwilliam route closes the finite-prime-only
ACS. The archimedean insertion requires promotion of the
per-place finite groupoid at $\{\infty\}$ to the
infinite-dimensional archimedean Fock of
Ch.~\ref{v4-arith:w9-r2:archimedean-wilson-surface}. The
obstruction is localised: it is the step from finite vector
spaces to the continuous Hilbert structure at $\Sigma_{\infty}$.
This is the same obstruction as the Route I descent from pro-vector
spaces to Hilbert spaces, seen through a different window.
\end{remark}

\section{Route IV: Cobordism Hypothesis}
\label{v4-arith:w10-a:route-IV}

Lurie's cobordism hypothesis (Lurie 2008 arXiv:0905.0465, Thm.~2.4.6)
classifies extended $n$-dimensional TQFTs as fully dualizable
objects in a symmetric monoidal $(\infty,n)$-category. For
$n=3$ and a target
$\mathcal{C}=\mathrm{ArithTC}$ the arithmetic tensor category, a
quantum ACS is specified by a fully dualizable object of $\mathcal{C}$.

\subsection{Fully dualizable object candidate}
\label{v4-arith:w10-a:cobord-candidate}

\begin{definition}[arithmetic target $(\infty,3)$-category]
\label{v4-arith:w10-a:def:arith-target}
\ClaimStatusDefinitional. The \emph{arithmetic target} is the
symmetric monoidal $(\infty,3)$-category
$\mathrm{ArithTC}$ of arithmetic braided tensor categories:
objects are braided tensor categories of admissible modules over
arithmetic chiral algebras; 1-morphisms are bimodule categories;
2-morphisms are bimodule functors; 3-morphisms are natural
transformations; with symmetric monoidal structure from the Tate
restricted tensor of Ch.~\ref{v4-arith:tate-tensor}.
\end{definition}

\begin{proposition}[fully dualizable object candidate]
\label{v4-arith:w10-a:prop:fully-dualizable-candidate}
\ClaimStatusConditional \emph{(on the finiteness and semisimplicity
of $\mathcal{MTC}^{\mathrm{Ar}}$)}. If $\mathcal{MTC}^{\mathrm{Ar}}$
of \Cref{v4-arith:w10-a:def:arith-MTC} is a finite semisimple
braided fusion category with non-degenerate braiding, then it is a
fully dualizable object of $\mathrm{ArithTC}$.
\end{proposition}

\begin{proof}
Douglas--Schommer-Pries--Snyder 2013 arXiv:1312.7188 classify fully
dualizable objects in the $(\infty,3)$-category of braided tensor
categories: the fully dualizable objects are exactly the
finite semisimple braided fusion categories with non-degenerate
braiding. Under the modularity hypothesis of
$\mathcal{MTC}^{\mathrm{Ar}}$, the latter is one such.
\end{proof}

\begin{theorem}[cobordism hypothesis output]
\label{v4-arith:w10-a:thm:cobord-output}
\ClaimStatusProvedHereConditional \emph{(on
\Cref{v4-arith:w10-a:prop:fully-dualizable-candidate}'s modularity
hypothesis)}. If $\mathcal{MTC}^{\mathrm{Ar}}$ is a modular tensor
category (finite semisimple braided non-degenerate), then the
cobordism hypothesis produces an extended 3D TQFT
\[
Z_{\mathrm{coh}}^{\mathrm{Ar}}:\mathrm{Bord}_{3}^{\mathrm{fr}}\to\mathrm{ArithTC}
\]
with $Z_{\mathrm{coh}}^{\mathrm{Ar}}(pt)=\mathcal{MTC}^{\mathrm{Ar}}$.
On the value of 2-manifolds and 3-manifolds, this reproduces the
Reshetikhin--Turaev state sum of
\Cref{v4-arith:w10-a:thm:RT-construction}.
\end{theorem}

\begin{proof}
Lurie 2008 (Thm.~2.4.6): the cobordism hypothesis assigns to any
fully dualizable object a unique (up to essential uniqueness)
extended TQFT with that value at a point. Composition with the
evaluation functor on 2- and 3-manifolds reduces the cobordism
assignment to the Reshetikhin--Turaev state sum.
\end{proof}

\begin{remark}[arithmetic cobordism category]
\label{v4-arith:w10-a:rem:arith-cobord}
Strictly, the cobordism hypothesis applies to a cobordism category
of smooth manifolds. For arithmetic 3-spaces, we need an
\emph{arithmetic cobordism category} whose objects are
arithmetic 2-surfaces and morphisms are arithmetic 3-spaces.
Construction of such a category is \Cref{v4-arith:3d-acs:residual}
(item 1 of the residual list in Ch.~\ref{v4-arith:3d-acs-E1bar}):
finite arithmetic cobordism functoriality. This is the Route IV
analogue of Route III's archimedean extension obstruction.
\end{remark}

\label{v4-arith:w10-a:ah:route-IV}

\section{Route V: Six-Functor Formalism on an Arithmetic Diamond}
\label{v4-arith:w10-a:route-V}

Scholze's six-functor formalism (Scholze 2020--2022
\emph{Lectures on Condensed Mathematics}, Bonn; Scholze 2022
\emph{Six-Functor Formalisms} lecture notes) applies to diamonds and
to a broad class of stacks. A quantum field theory on a space is
given by a ``theory'' in the sense of six-functor formalism:
an association of $\infty$-categories with the six operations
$\otimes,\operatorname{Hom},f^{*},f_{*},f_{!},f^{!}$. For 3D
CS on a diamond, Scholze et al.\ have outlined the six-functor
formalism as the natural home (Fargues--Scholze 2021
arXiv:2102.13459).

\subsection{Diamond of \texorpdfstring{$\overline{\mathrm{Spec}\,\mathbb Z}$}{Spec Z bar}}
\label{v4-arith:w10-a:diamond}

\begin{conjecture}[arithmetic diamond]
\label{v4-arith:w10-a:conj:arith-diamond}
\ClaimStatusConjectured. There exists a diamond
$\overline{\mathrm{Spec}\,\mathbb Z}^{\diamond}$ in the sense of
Scholze 2020 \emph{Etale Cohomology of Diamonds}
(arXiv:1709.07343, \S11) which is a globalisation of the
Fargues--Fontaine curves at each prime, with an archimedean
completion at $\{\infty\}$ given by a continuous version of
$\Sigma_{\infty}$. The diamond carries a six-functor formalism
with values in condensed $\mathbb{Z}_{\ell}$-sheaves.
\end{conjecture}

\begin{remark}[status of the arithmetic diamond]
\label{v4-arith:w10-a:rem:diamond-status}
\Cref{v4-arith:w10-a:conj:arith-diamond} is a conjecture with no
known construction as of 2026-04. The closest object in the
literature is the Clausen--Scholze analytic $\mathbb{Z}$-spectrum
(Clausen--Scholze 2020 \emph{Condensed Mathematics} lecture notes),
which builds the condensed structure on the arithmetic side but does
not supply the diamond-plus-archimedean-completion we would need.
\end{remark}

\subsection{Six-functor ACS}
\label{v4-arith:w10-a:six-fun-ACS}

\begin{proposition}[six-functor ACS]
\label{v4-arith:w10-a:prop:six-fun-ACS}
\ClaimStatusConditional \emph{(on
\Cref{v4-arith:w10-a:conj:arith-diamond})}. If the arithmetic
diamond $\overline{\mathrm{Spec}\,\mathbb Z}^{\diamond}$ exists
with a six-functor formalism, then the quantum ACS is
$\mathcal T_{k^{\mathrm{Ar}}}:=f^{!}\mathcal V^{\mathrm{prim}}$,
where $f:\overline{\mathrm{Spec}\,\mathbb Z}^{\diamond}\to\{pt\}^{\diamond}$
is the structure map and $\mathcal V^{\mathrm{prim}}$ is the
boundary condensed sheaf assigned to the archimedean completion.
The partition function is the trace of the identity
$\operatorname{Tr}f_{!}f^{!}\mathcal V^{\mathrm{prim}}$, which
under the Fargues--Scholze geometrisation equals
$\xi(k^{\mathrm{Ar}})$ by the local-global product of local
Langlands L-factors.
\end{proposition}

\begin{proof}
The six-functor formalism gives a canonical quantum CS
presentation: the pushforward of the boundary sheaf with proper
support is the partition function. At each prime, the local
Fargues--Scholze geometrisation (arXiv:2102.13459, Thm.~I.7.3)
identifies the local Satake as the stalk of the boundary sheaf;
the global product is $\xi$.
\end{proof}

\begin{remark}[Route V is the most elegant but most conditional]
\label{v4-arith:w10-a:rem:route-V-status}
Route V reduces the entire \Cref{v4-arith:3d-acs:conj:qu-acs} to a
single foundational input: the arithmetic diamond. That input is
itself \Cref{v4-arith:w10-a:conj:arith-diamond}, which is not
resolved; but it is a well-defined conjecture in Scholze
six-functor formalism with an explicit target (arithmetic side of
the Fargues--Fontaine geometric Langlands programme).
\end{remark}

\subsection{Attack-heal on Route V}
\label{v4-arith:w10-a:ah:route-V}

\section{Route Comparison and Isolation of the Residual}
\label{v4-arith:w10-a:comparison}

We now compare the five routes and isolate the residual obstruction
at their intersection.

\begin{theorem}[route comparison]
\label{v4-arith:w10-a:thm:route-comparison}
\ClaimStatusProvedHere. The five routes to
\Cref{v4-arith:3d-acs:conj:qu-acs} decompose as follows.
\begin{itemize}
\item Unconditional at finite level: Route I closes each finite
$Z^{\mathrm{Ar}}_{G_{N},\alpha_{N}}$; Route III closes the open-part
factorization algebra.
\item Conditional on arithmetic Huang--Lepowsky: Routes II and IV
reduce \Cref{v4-arith:3d-acs:conj:qu-acs} to modularity of
$\mathcal{MTC}^{\mathrm{Ar}}$.
\item Conditional on arithmetic diamond: Route V reduces
\Cref{v4-arith:3d-acs:conj:qu-acs} to the Scholze six-functor
formalism on $\overline{\mathrm{Spec}\,\mathbb Z}^{\diamond}$.
\end{itemize}
\end{theorem}

\begin{proof}
\Cref{v4-arith:w10-a:thm:profinite-exists},
\Cref{v4-arith:w10-a:thm:CG-open},
\Cref{v4-arith:w10-a:thm:route-II-closure},
\Cref{v4-arith:w10-a:thm:route-IV-closure},
\Cref{v4-arith:w10-a:thm:route-V-closure} respectively.
\end{proof}

\subsection{The single common residual}
\label{v4-arith:w10-a:common-residual}

\begin{theorem}[the common residual: archimedean Hilbert descent]
\label{v4-arith:w10-a:thm:common-residual}
\ClaimStatusProvedHere. The five routes share a single common
residual obstruction, here named the \emph{archimedean Hilbert
descent}:
\begin{quote}
There exists a choice of condensed (or nuclear / pro-finite) topology
on the inverse-limit state space
$\mathcal H^{(\infty),\mathrm{qu}}:=\varprojlim_{N}^{\mathrm{ren}}\mathcal H^{(\infty)}_{G_{N},\alpha_{N}}$
of finite-level archimedean boundary Hilbert spaces, under which (a)
$\mathcal H^{(\infty),\mathrm{qu}}$ is the archimedean Fock factor
$\mathcal V^{\mathrm{prim}}_{\mathrm{Heis},\infty}$; and (b) the
Arakelov-K\"ahler Wilson-surface observable $F_{\infty}^{\mathrm{WS}}$
of \Cref{v4-arith:w9-r2:def:F-infty-WS} acts on this inverse limit
as the scaling Frobenius $F_{\infty}=e^{-\widehat D}$ of
\Cref{v4-arith:w9-r2:def:F-infty}.
\end{quote}
Equivalently: the archimedean K\"ahler form $\omega_{\infty}$ induces
a condensed-limit action of $\mathbb{R}^{\times}_{+}$ on the limit
state space, and this action is unitarily conjugate to the
Weil--Petersson scaling representation on
$\mathcal V^{\mathrm{prim}}_{\mathrm{Heis},\infty}$.
\end{theorem}

\begin{proof}
Route I reduces to this descent in
\Cref{v4-arith:w10-a:rem:route-I-scope}. Route III reduces to it
in \Cref{v4-arith:w10-a:rem:route-III-stop}. Routes II and IV
require modularity of $\mathcal{MTC}^{\mathrm{Ar}}$, which is the
statement that the per-prime tensor product with the archimedean
Fock gives a finite-dimensional braided fusion category at every
level; this is equivalent to the archimedean Hilbert descent on the
per-level subspaces, passed to the inverse limit. Route V requires
the arithmetic diamond, which is a six-functor formalism that
implicitly contains the descent in its $f^{!}$ on the archimedean
completion.
\end{proof}

\begin{remark}[the residual has three aliases]
\label{v4-arith:w10-a:rem:aliases}
The archimedean Hilbert descent of
\Cref{v4-arith:w10-a:thm:common-residual} is the same obstruction as:
\begin{itemize}
\item \emph{Route I alias}: profinite-Hilbert descent of the Kan
inverse limit.
\item \emph{Route III alias}: extension of the Costello--Gwilliam
factorization algebra from the open \'etale curve to the full
Arakelov compactification.
\item \emph{Route V alias}: archimedean completion of the arithmetic
diamond with its induced condensed structure.
\end{itemize}
These are three presentations of the single fact that the
\'etale-discrete state spaces at finite level must descend to a
continuous Hilbert space at the archimedean boundary under the
Arakelov K\"ahler form.
\end{remark}

\subsection{Relation to the Hilbert--P\'olya/Connes--Deninger obstruction}
\label{v4-arith:w10-a:HP-relation}

\begin{remark}[archimedean descent is strictly weaker than RH]
\label{v4-arith:w10-a:rem:weaker-than-RH}
The archimedean Hilbert descent of
\Cref{v4-arith:w10-a:thm:common-residual} is an \emph{existence}
claim for the state space $\mathcal H^{(\infty),\mathrm{qu}}$ and a
\emph{functoriality} claim for the Arakelov scaling action. It is
strictly weaker than the Hilbert--P\'olya hypothesis
H-P$^{\mathrm{Ar}}$ of \Cref{v4-arith:rh:hyp:HPAr}: H-P$^{\mathrm{Ar}}$
requires, in addition, that the regularised spectrum of
$\Theta_{\xi}=\frac12\mathrm{id}+iH_{\mathrm{HP}}$ on
$H^{1}_{\mathrm{ch}}\subset\mathcal H^{(\infty),\mathrm{qu}}$ equals
the non-trivial zeros of $\xi_{\mathbb R}$. The archimedean Hilbert
descent does not assert this; it only constructs the state space
and the scaling action.
\end{remark}

\begin{theorem}[Clausen--Scholze condensed-Hilbert descent plus A-TP $\Rightarrow$ RH]
\label{v4-arith:w10-a:thm:descent-plus-ATP-equals-RH}
\ClaimStatusProvedHereConditional \emph{(on the archimedean
condensed-Hilbert descent of
\Cref{v4-arith:w10-a:conj:condensed-descent} and the
A-TP archimedean positivity of
Ch.~\ref{v4-arith:w5-a:conj:archimedean-tate-positivity})}. Under
both the Clausen--Scholze condensed-Hilbert descent and A-TP, the Riemann
Hypothesis holds.
\end{theorem}

\begin{proof}
The condensed-Hilbert descent constructs
$\mathcal H^{(\infty),\mathrm{qu}}$ on which
$F_{\infty}=e^{-\widehat D}=e^{-L_{0}^{\mathrm{Ar}}}$ acts (unitary
conjugate). By \Cref{v4-arith:w9-r2:thm:purity-ATP-RH}, A-TP is
equivalent to $F_{\infty}$-weight $=1/2$, which is equivalent to RH.
Under the descent plus A-TP, all hypotheses are satisfied. The
$F_{\infty}$-spectrum on $H^{1}_{\mathrm{ch}}$ is determined by the
descent (giving the Hilbert space) plus A-TP (giving the
weight-$1/2$ condition), so
$\mathrm{Spec}(F_{\infty}\mid H^{1}_{\mathrm{ch}})\subset\{e^{-\rho}:\mathrm{Re}\,\rho=1/2\}$,
which is RH.
\end{proof}

\begin{remark}[what \Cref{v4-arith:w10-a:thm:common-residual} does and does not decide]
\label{v4-arith:w10-a:rem:what-resolves}
Resolving the archimedean Hilbert descent \emph{alone} does not
resolve RH: it constructs
$\mathrm{ACS}^{\mathrm{qu}}_{k^{\mathrm{Ar}}}$ and makes axioms
(WZW), (W) unconditional; it partially closes (Z) up to the
character identification; but it does not furnish A-TP positivity
of the scaling operator. Under both, RH follows. This is the
precise sense in which the archimedean Hilbert descent is the
``bulk construction'' residual and A-TP is the separate
``positivity'' residual. Inside W10-A, after the condensed-Hilbert
descent is assumed, no further bulk obstruction remains; A-TP is the
only RH-equivalent obstruction.
\end{remark}

\;=\;\text{finite-level boundary state space from}\ \Cref{v4-arith:3d-acs:def:finite-boundary-category}

\begin{definition}[target archimedean Hilbert]
\label{v4-arith:w10-a:def:target-Hilbert}
\ClaimStatusDefinitional. The \emph{target archimedean Hilbert

$\mathcal V^{\mathrm{prim}}_{\mathrm{Heis},\infty}$, the archimedean

\subsection{The precise named obstruction}
\label{v4-arith:w10-a:named-obstruction}

\begin{conjecture}[archimedean condensed-Hilbert descent]
\label{v4-arith:w10-a:conj:condensed-descent}
\ClaimStatusConjectured. The condensed-Hilbert pro-system
$\mathcal H^{(\infty)}_{\bullet}$ of
\Cref{v4-arith:w10-a:def:condensed-pro-system} admits a limit
$\mathcal H^{(\infty)}_{\infty}$ in $\mathrm{Cond}_{\mathrm{Hilb}}(\mathbb C)$,
and there is an isomorphism of condensed Hilbert spaces
\begin{equation}
\label{v4-arith:w10-a:eq:condensed-descent}
\mathcal H^{(\infty)}_{\infty}
\;\cong\;
\mathcal V^{\mathrm{prim}}_{\mathrm{Heis},\infty}
\end{equation}
intertwining the Arakelov scaling $\mathbb{R}^{\times}_{+}$-action
on the target with the Wilson-surface
$F_{\infty}^{\mathrm{WS}}$-action on the limit.
\end{conjecture}

\begin{theorem}[condensed descent $\Leftrightarrow$ Conjecture $v4{\text -}arith{:}3d{\text -}acs{:}qu{\text -}acs$]
\label{v4-arith:w10-a:thm:descent-equivalent}
\ClaimStatusProvedHere. \Cref{v4-arith:w10-a:conj:condensed-descent}
is equivalent to \Cref{v4-arith:3d-acs:conj:qu-acs}.
\end{theorem}

\begin{proof}
($\Rightarrow$) Given \Cref{v4-arith:w10-a:conj:condensed-descent},
the limit $\mathcal H^{(\infty)}_{\infty}$ is the archimedean
boundary Hilbert space of the profinite ACS; combined with the
Route III factorization algebra on the open curve, this descends
to a functorial assignment on
$\overline{\mathrm{Spec}\,\mathbb{Z}}$ satisfying (WZW). The
partition function (Z) is the trace of the Kan-limit, which by
descent equals the trace on $\mathcal V^{\mathrm{prim}}_{\mathrm{Heis},\infty}$
computed by the Zhu character, hence by
\Cref{v4-arith:3d-acs:cor:partition-bar} equals $\xi(k^{\mathrm{Ar}})$.
The Wilson-line (W) is \Cref{v4-arith:w9-r1:thm:main} at the
descended limit.

($\Leftarrow$) Given \Cref{v4-arith:3d-acs:conj:qu-acs}, the
boundary state space at $\Sigma_{\infty}$ is
$\mathcal V^{\mathrm{prim}}_{\mathrm{Heis},\infty}$ (axiom (WZW));
the Arakelov scaling acts by
$F_{\infty}$ on this space. Taking condensed-Hilbert limits level by
level of the finite boundary categories and identifying with this
limit gives \Cref{v4-arith:w10-a:conj:condensed-descent}.
\end{proof}

\begin{remark}[the obstruction is now a named theorem]
\label{v4-arith:w10-a:rem:named-theorem}
\Cref{v4-arith:w10-a:thm:descent-equivalent} promotes
\Cref{v4-arith:3d-acs:conj:qu-acs} from a three-axiom TQFT
conjecture to a single condensed-mathematics statement:
\Cref{v4-arith:w10-a:conj:condensed-descent}. The latter is in the
scope of condensed mathematics (Clausen--Scholze 2020--) and is a
well-defined problem in that framework. Its status as of 2026-04:
no known proof or disproof.
\end{remark}

\section{Wilson-Observable Compatibility}
\label{v4-arith:w10-a:wilson-compat}

We record the precise compatibility between
\Cref{v4-arith:w10-a:conj:condensed-descent} and the Wilson
observables of Chs.~\ref{v4-arith:w9-r1:wilson-frobenius} and
\ref{v4-arith:w9-r2:archimedean-wilson-surface}.

\begin{theorem}[Wilson-line compatibility]
\label{v4-arith:w10-a:thm:wilson-line-compat}
\ClaimStatusProvedHereConditional \emph{(on
\Cref{v4-arith:w10-a:conj:condensed-descent})}. Under the
condensed-Hilbert descent, the Wilson-line observable at a prime
knot $K_{p}$ in the bulk $\mathrm{ACS}^{\mathrm{qu}}_{k^{\mathrm{Ar}}}$,
evaluated on the limit state space
$\mathcal H^{(\infty)}_{\infty}\cong\mathcal V^{\mathrm{prim}}_{\mathrm{Heis},\infty}$,
is exactly the Frobenius trace $\mathrm{tr}\,\rho(\mathrm{Frob}_{p})$
of \Cref{v4-arith:w9-r1:thm:main}(2), without additional hypothesis
beyond the descent.
\end{theorem}

\begin{proof}
\Cref{v4-arith:w10-a:thm:descent-equivalent} supplies the
\Cref{v4-arith:3d-acs:conj:qu-acs} conclusion from the descent
alone. Axiom (W) of \Cref{v4-arith:3d-acs:conj:qu-acs} is the
Wilson-line Frobenius identity. The finite-level identity
\Cref{v4-arith:w9-r1:thm:main}(1) is unconditional; the profinite
limit \Cref{v4-arith:w9-r1:thm:main}(2) is conditional only on the
renormalised profinite limit, which is supplied by the descent.
\end{proof}

\begin{theorem}[Wilson-surface compatibility]
\label{v4-arith:w10-a:thm:wilson-surface-compat}
\ClaimStatusProvedHereConditional \emph{(on
\Cref{v4-arith:w10-a:conj:condensed-descent})}. Under the
condensed-Hilbert descent, the archimedean Wilson-surface
observable $F_{\infty}^{\mathrm{WS}}$ of
\Cref{v4-arith:w9-r2:def:F-infty-WS}, acting on the limit state

scaling Frobenius $F_{\infty}=e^{-\widehat D}$ of
\Cref{v4-arith:w9-r2:def:F-infty}, without additional hypothesis

\end{theorem}

\begin{proof}
Immediate from \Cref{v4-arith:w9-r2:thm:WS-equals-scaling}:
its conditional status is only on
\Cref{v4-arith:3d-acs:conj:qu-acs}, which by
\Cref{v4-arith:w10-a:thm:descent-equivalent} is the
condensed-Hilbert descent.
\end{proof}

\begin{corollary}[both Wilson observables hinge on one residual]
\label{v4-arith:w10-a:cor:wilson-one-residual}
\ClaimStatusProvedHere. The Wilson-line Frobenius identity

\begin{proof}
\Cref{v4-arith:w10-a:thm:wilson-line-compat} and
\Cref{v4-arith:w10-a:thm:wilson-surface-compat}.
\end{proof}

\section{Attack-Heal Synthesis}
\label{v4-arith:w10-a:ah:synthesis}

The seven-attack protocol for the central claim of this chapter,
\Cref{v4-arith:w10-a:thm:common-residual}, runs as follows.

\paragraph{Attack 1 (source collision).} The five routes use
distinct source catalogs: Dijkgraaf--Witten renormalisation
(Freed--Quinn 1993); Reshetikhin--Turaev MTC state sum (Turaev
1994); Costello--Gwilliam factorization (Costello--Gwilliam
2017--2021); Lurie cobordism hypothesis (Lurie 2008,
Douglas--Schommer-Pries--Snyder 2013); Scholze six-functor
(Scholze 2020, Fargues--Scholze 2021). All five converge on the
same descent residual. Disjoint sources, convergent destination.
\textbf{Pass.}

\paragraph{Attack 2 (sign/convention).} The Euler-characteristic
renormalisation factor $|G_{N}|^{-\chi}$ is zero at
$\overline{\mathrm{Spec}\,\mathbb Z}$; the Arakelov K\"ahler
scaling is unitary; the condensed Hilbert limit preserves inner
products. Conventions consistent.
\textbf{Pass.}

\paragraph{Attack 3 (ambient category).} Finite-level state sums in
$\mathrm{Vect}_{\mathbb C}$; profinite limits in
$\mathrm{Pro}(\mathrm{Vect}_{\mathbb C})$; condensed-Hilbert limits
in $\mathrm{Cond}_{\mathrm{Hilb}}(\mathbb C)$; all compatible under
the descent hypothesis.
\textbf{Pass under descent hypothesis.}

\paragraph{Attack 4 (missing hypothesis).} The archimedean
condensed-Hilbert descent
\Cref{v4-arith:w10-a:conj:condensed-descent} is the precisely-named
missing hypothesis, shared by all five routes. No additional
missing hypothesis.
\textbf{Residual: condensed-Hilbert descent (single).}

\paragraph{Attack 5 (false functoriality).} Route-by-route
functoriality is verified in
\Cref{v4-arith:w10-a:thm:route-I-closure},
\Cref{v4-arith:w10-a:thm:route-II-closure},
\Cref{v4-arith:w10-a:thm:route-III-closure},
\Cref{v4-arith:w10-a:thm:route-IV-closure},
\Cref{v4-arith:w10-a:thm:route-V-closure}. The common
reduction functor is the descent itself.
\textbf{Pass.}

\paragraph{Attack 6 (unproved equivalence).} The equivalence of the
five route presentations of
\Cref{v4-arith:3d-acs:conj:qu-acs} is
\Cref{v4-arith:w10-a:thm:common-residual}. Proved.
\textbf{Pass.}

\paragraph{Attack 7 (engine-sync).} No compute engine currently
backs this chapter; no engine-test conflict.
\textbf{Pass.}

\section{Residual Obstructions}
\label{v4-arith:w10-a:residual}

Three residuals remain at the end of this chapter. These are not
new residuals; they are precisified statements of known obstructions.

\begin{enumerate}
\item \emph{Archimedean condensed-Hilbert descent}
(\Cref{v4-arith:w10-a:conj:condensed-descent}). This is the precise
statement equivalent to \Cref{v4-arith:3d-acs:conj:qu-acs} by
\Cref{v4-arith:w10-a:thm:descent-equivalent}. Scope: a
condensed-mathematics limit problem in the Clausen--Scholze
framework. Well-defined, unresolved.

\item \emph{Arithmetic Huang--Lepowsky modularity}
(\Cref{v4-arith:w10-a:rem:modularity-cond}). This is a sufficient
condition for Routes II and IV. It is strictly stronger than
Residual 1: modularity implies the condensed-Hilbert descent
structurally, by promoting the finite-level braided structure to a
full MTC. Scope: extend Huang--Lepowsky 1994--1999 from the
complex-analytic VOA setting to the arithmetic chiral-algebra
setting.

\item \emph{Arithmetic diamond}
(\Cref{v4-arith:w10-a:conj:arith-diamond}). This is a sufficient
condition for Route V. It is strictly stronger than Residual 1:
the six-functor formalism on the arithmetic diamond encodes the
condensed-Hilbert descent as a byproduct of its structural
$f^{!}$-functoriality. Scope: global Fargues--Scholze programme on
$\overline{\mathrm{Spec}\,\mathbb Z}$.
\end{enumerate}

\begin{remark}[partial order on residuals]
\label{v4-arith:w10-a:rem:partial-order}
The three residuals form a partial order by logical implication:
\[
\text{Residual 3 (diamond)}\;\Rightarrow\;\text{Residual 2 (HL mod)}\;\Rightarrow\;\text{Residual 1 (descent)}.
\]
Progress on any of the three resolves all lower residuals. The
converse fails: the descent could in principle be resolved
without solving modularity or the diamond, by a direct Route I
closure.
\end{remark}

\section{Summary}
\label{v4-arith:w10-a:summary}

\begin{enumerate}
\item Five construction routes to the profinite quantum ACS bulk
measure have been carried through to the end:
(I) Kan-extended state-sum tower; (II) Reshetikhin--Turaev MTC;
(III) Costello--Gwilliam \'etale factorization; (IV) Lurie
cobordism hypothesis; (V) Scholze six-functor.
\item Routes I and III close unconditionally at finite level
(Route I state-sum; Route III on the open \'etale curve,
\Cref{v4-arith:w10-a:thm:route-III-closure}).
\item Routes II, IV, V are conditional on arithmetic modularity
or on the arithmetic diamond, each strictly stronger than
the single common residual.
\item The single common residual is the \emph{archimedean
condensed-Hilbert descent}
(\Cref{v4-arith:w10-a:conj:condensed-descent}): a limit problem in
the Clausen--Scholze condensed-Hilbert category.
\item \Cref{v4-arith:w10-a:thm:descent-equivalent} shows that
\Cref{v4-arith:3d-acs:conj:qu-acs} is logically equivalent to this
descent. The three axioms (WZW), (Z), (W) reduce to one statement.
\item Under the descent, all Wilson observables
(Chs.~\ref{v4-arith:w9-r1:wilson-frobenius} and
\ref{v4-arith:w9-r2:archimedean-wilson-surface}) are unconditional.
\item Under the descent plus A-TP, RH holds
(\Cref{v4-arith:w10-a:thm:descent-plus-ATP-equals-RH}). The
descent alone does not resolve RH; A-TP remains separate.
\item The chapter creates no new mathematics; it maps
\Cref{v4-arith:3d-acs:conj:qu-acs} onto a precisely-named and
logically-equivalent condensed-mathematics statement, isolates its
route-independence, and records the three strictly-stronger
sufficient conditions (modularity, diamond, six-functor) in a
partial order.
\end{enumerate}

The bulk-construction residual of the arithmetic chiral-algebra
programme is now a single well-posed condensed-mathematics
statement. The archimedean fibre is again the load-bearing
location: finite-level \'etale-discrete state spaces descend to
discrete finite-prime factorization algebras unconditionally; the
archimedean Hilbert structure is where the descent either succeeds
or fails. This is structurally the same concentration phenomenon
as in Ch.~\ref{v4-arith:w9-r2:archimedean-wilson-surface}, where
the Riemann Hypothesis concentrates in the archimedean fibre. Both
point to the same mathematical joint: the arithmetic K\"ahler form
$\omega_{\infty}$ on $\Sigma_{\infty}$ is the carrier of the
analytic content that the \'etale-finite theory cannot supply.
