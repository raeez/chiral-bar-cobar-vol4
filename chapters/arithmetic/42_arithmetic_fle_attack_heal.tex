\section{Source Theorems}
\label{v4-arith:fle-audit:sources}

\begin{theorem}[geometric critical-level FLE]
\label{v4-arith:fle-audit:thm:critical-FLE-source}
\ClaimStatusProvedElsewhere. The 2024 geometric-Langlands proof
installs the critical-level fundamental local equivalence in the
geometric setting over a smooth projective curve in characteristic
$0$:
\[
\mathrm{KL}(G)_{\mathrm{crit}}\;\simeq\;
\mathrm{IndCoh}^{*}\bigl(\mathrm{Op}^{\mathrm{mf}}_{\check G}\bigr),
\]
with compatibility with Kac--Moody localization. Its source is
Arinkin--Beraldo--Campbell--Chen--Faergeman--Gaitsgory--Lin--Raskin--
Rozenblyum, \emph{Proof of the geometric Langlands conjecture II:
Kac--Moody localization and the FLE}, arXiv:2405.03648.
\end{theorem}

\begin{theorem}[quantum and small FLE sources]
\label{v4-arith:fle-audit:thm:quantum-FLE-source}
\ClaimStatusProvedElsewhere. The quantum/local FLE layer is supplied
by Gaitsgory's quantum Langlands notes, Campbell--Dhillon--Raskin's
\emph{Fundamental local equivalences in quantum geometric Langlands}
(arXiv:1907.03204), and Gaitsgory--Lysenko's small-FLE work
(arXiv:1903.02279). These are local geometric or metaplectic/quantum
statements. They are not number-field adelic equivalences.
\end{theorem}

\begin{remark}[source boundary]
\label{v4-arith:fle-audit:rem:source-boundary}
The FLE sources live in the geometric Langlands, Kac--Moody,
Whittaker, and root-of-unity quantum-group environment. They do not
construct a local categorical $p$-adic Langlands equivalence over
$\mathbb{Q}_{p}$, do not globalize over
$\overline{\mathrm{Spec}\,\mathbb{Z}}$, and do not supply the
Emerton--Gee full Weil/$(\varphi,\Gamma)$ target comparison.
\end{remark}

\section{Arithmetic Non-Identification}
\label{v4-arith:fle-audit:non-identification}

\begin{proposition}[F1/F2/F3 are not an arithmetic FLE]
\label{v4-arith:fle-audit:prop:not-FLE}
\ClaimStatusNeedsVerification. The arithmetic branch's F1/F2/F3
package is not an arithmetic instance of Gaitsgory's FLE. It is a
number-field local-global Langlands and Hochschild-trace programme.
\end{proposition}

\begin{proof}
The source and target types differ. Gaitsgory's FLE compares
critical-level Kac--Moody or Whittaker/factorizable categories with
oper or quantum-group categories over a geometric curve. F1 compares
the sph-finite automorphic category for $GL_{2}/\mathbb{Q}$ with
ind-coherent sheaves on an Emerton--Gee stack. F2 asks for a
mixed-characteristic $E_{2}$ enhancement and restricted tensor
product. F3 asks for a Ben-Zvi--Nadler Hochschild-trace comparison
against a full Weil/$(\varphi,\Gamma)$ arithmetic target. None of
these targets is the oper target of critical FLE, and none of the
finite-prime inputs is a Beilinson--Drinfeld chiral algebra over a
complex curve.
\end{proof}

\begin{remark}[finite-prime replacement]
\label{v4-arith:fle-audit:rem:finite-prime-replacement}
At a finite prime the correct language is Fargues--Fontaine,
diamonds, Hecke correspondences, spectral action, and
$(\varphi,\Gamma)$-modules. It is not a literal chiral OPE or a
critical Kac--Moody category. The archimedean fibre may carry the
classical chiral/Kac--Moody intuition, but it does not make the
finite-prime package a global arithmetic FLE.
\end{remark}
