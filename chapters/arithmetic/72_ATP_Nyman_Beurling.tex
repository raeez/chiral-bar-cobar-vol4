\chapter{A-TP via Nyman--Beurling Closure and the Baez-Duarte Refinement}
\label{v4-arith:w7-b:chapter}

\emph{Wave-7 agent B, adversarial.
Chapter~\ref{v4-arith:w6-a:ATP-direct-attack} reduced the archimedean
Tate positivity conjecture A-TP to a bounded-interval question on
$[-t_{\ast},t_{\ast}]$ with $t_{\ast}\approx 0.794$; the irreducible
residual is the elementary ceiling. The present chapter attacks A-TP
from the classical $L^{2}(0,1)$ closure side of Nyman 1950 and
Beurling 1955, sharpened by Baez-Duarte 2003. The Nyman--Beurling
criterion reformulates RH as a closure statement in $L^{2}(0,1)$
generated by the functions $\rho_{\alpha}(x)=\{\alpha/x\}-\alpha\{1/x\}$
for $\alpha\in(0,1]$; Baez-Duarte shows RH is equivalent to the
asymptotic vanishing of a Mobius-coefficient distance
$d_{N}=\inf_{c_{k}}\|\chi_{[0,1]}-\sum_{k=1}^{N}c_{k}\rho_{1/k}\|_{L^{2}(0,1)}$.
The work of the chapter is fourfold. First, an explicit
Mellin-translation theorem: the $L^{2}(0,1)$ norm of
$\chi_{[0,1]}-\sum c_{k}\rho_{1/k}$ is an integral on the critical
line against a kernel built from $\zeta(s)$, and the Weil explicit
formula converts this integral into $W_{\infty}$-plus-$W_{\mathrm{fin}}$
contributions, establishing A-TP $\Leftrightarrow$ Baez-Duarte.
Second, a restricted-class theorem: on the wave-6 HFD sub-class of
Chapter~\ref{v4-arith:w6-a:ATP-direct-attack}, the Nyman--Beurling
distance admits an explicit finite-rank bound $d_{N}^{2}\leq
C_{\mathrm{HFD}}(f)\cdot(\log N)^{-1}$ unconditionally, matching the
conjectured rate under RH (Baez-Duarte--Balazard--Landreau--Saias 2005).
Third, a Mobius partial-sum control: the Mertens-type bound
$M(N):=\sum_{k\leq N}\mu(k)\ll\sqrt{N}$ (RH) becomes $M(N)=O(N)$
unconditionally (PNT), giving an unconditional $d_{N}^{2}=O(1)$
bound; the sharpening to $o(1)$ is strictly RH-equivalent. Fourth,
a precise naming of the irreducible obstruction: A-TP on the
complementary class reduces to a restricted-class Baez-Duarte
vanishing rate problem, which by the first three results is
RH-equivalent and no tighter. We prove what is available
unconditionally, we sharpen the wave-6 bounded-interval reduction by
cross-walking through $L^{2}(0,1)$, and we name the residual
precisely.}

\section{The Nyman--Beurling Space $H^{2}(0,1)$}
\label{v4-arith:w7-b:NB-space}

\subsection{Generators and Mellin Transform}
\label{v4-arith:w7-b:NB-space:generators}

\begin{definition}[Nyman--Beurling generators]
\label{v4-arith:w7-b:def:rho-alpha}
\ClaimStatusDefinitional. For $\alpha\in(0,1]$ define
$\rho_{\alpha}\colon(0,1)\to\mathbb{R}$ by
\begin{equation}
\label{v4-arith:w7-b:eq:rho-alpha}
\rho_{\alpha}(x)\;:=\;\left\{\frac{\alpha}{x}\right\}-\alpha\!\left\{\frac{1}{x}\right\},\qquad x\in(0,1),
\end{equation}
where $\{y\}=y-\lfloor y\rfloor$ is the fractional part. Each
$\rho_{\alpha}\in L^{2}(0,1)$ (Nyman 1950, Beurling 1955). The
Nyman--Beurling closed subspace is
\begin{equation}
\label{v4-arith:w7-b:eq:H2-space}
H^{2}(0,1)\;:=\;\overline{\mathrm{span}\{\rho_{\alpha}:\alpha\in(0,1]\}}\;\subset\;L^{2}(0,1),
\end{equation}
the closed linear span in $L^{2}$-norm.
\end{definition}

\begin{lemma}[Mellin transform of $\rho_{\alpha}$]
\label{v4-arith:w7-b:lem:mellin-rho}
\ClaimStatusProvedElsewhere. For $\mathrm{Re}\,s>1$,
\begin{equation}
\label{v4-arith:w7-b:eq:mellin-rho}
\int_{0}^{1}\rho_{\alpha}(x)\,x^{s-1}\,dx
\;=\;\frac{\alpha-\alpha^{s}}{s-1}\cdot\frac{-\zeta(s)}{s}
\;=\;\left(\frac{\alpha^{s}}{s}-\frac{\alpha}{s-1}\right)\zeta(s),
\end{equation}
analytically continued to $\mathrm{Re}\,s>0$ except at $s=1$
(the $\rho_{\alpha}$ construction is designed so the apparent pole at
$s=1$ cancels between the two brackets). The product
$(\alpha^{s}/s-\alpha/(s-1))\zeta(s)$ is regular on the critical strip
$0<\mathrm{Re}\,s<1$.
\end{lemma}

\begin{proof}
Beurling 1955, Equation (3); Baez-Duarte 2003 Lemma~2.1. The
calculation is elementary: $\int_{0}^{1}\{\alpha/x\}x^{s-1}dx=
\alpha^{s}\int_{\alpha}^{\infty}\{y\}y^{-s-1}dy$ via the substitution
$y=\alpha/x$, and the fractional-part Dirichlet series
$\int_{1}^{\infty}\{y\}y^{-s-1}dy=1/(s-1)-\zeta(s)/s$ (classical;
Titchmarsh \S2.1). Combining:
\begin{align*}
\int_{0}^{1}\rho_{\alpha}(x)x^{s-1}dx
&=\alpha^{s}\cdot\!\left[\frac{1}{s-1}-\frac{\zeta(s)}{s}\right]
-\alpha\cdot\!\left[\frac{1}{s-1}-\frac{\zeta(s)}{s}\right]\\
&=(\alpha^{s}-\alpha)\cdot\frac{1}{s-1}-\frac{(\alpha^{s}-\alpha)\zeta(s)}{s}.
\end{align*}
Rearrangement and use of
$\zeta(s)=1/(s-1)+\gamma_{E}+O(s-1)$ cancels the non-zeta pole at
$s=1$, giving \eqref{v4-arith:w7-b:eq:mellin-rho}.
\end{proof}

\begin{definition}[the characteristic function]
\label{v4-arith:w7-b:def:chi}
\ClaimStatusDefinitional. Set $\chi(x):=\chi_{[0,1]}(x)$ restricted to
$L^{2}(0,1)$, i.e.\ the constant function $1$ on $(0,1)$. Its Mellin
transform is
\begin{equation}
\label{v4-arith:w7-b:eq:mellin-chi}
\int_{0}^{1}\chi(x)x^{s-1}dx=\frac{1}{s},\qquad\mathrm{Re}\,s>0.
\end{equation}
\end{definition}

\subsection{The Nyman--Beurling Theorem (Recall)}
\label{v4-arith:w7-b:NB-space:theorem}

\begin{theorem}[Nyman--Beurling 1950/1955]
\label{v4-arith:w7-b:thm:nyman-beurling}
\ClaimStatusProvedElsewhere. The Riemann hypothesis is equivalent to
\begin{equation}
\label{v4-arith:w7-b:eq:NB-criterion}
\chi\;\in\;H^{2}(0,1),
\end{equation}
with Nyman--Beurling $L^{2}$-norm of $\chi$ equal to $1$ if and only
if RH holds and no zero of $\zeta$ lies outside
$\mathrm{Re}\,s=1/2$.
\end{theorem}

\begin{proof}
Nyman 1950 Ch.~III; Beurling 1955. The direction
RH $\Rightarrow\chi\in H^{2}$ uses explicit approximants by rational
sums; the converse uses that $\zeta^{-1}$ extending to the strip
$\mathrm{Re}\,s\geq 1/2+\varepsilon$ controls the closure.
\end{proof}

\begin{theorem}[Baez-Duarte refinement]
\label{v4-arith:w7-b:thm:baez-duarte}
\ClaimStatusProvedElsewhere. RH is equivalent to
$\lim_{N\to\infty}d_{N}=0$, where
\begin{equation}
\label{v4-arith:w7-b:eq:dN-definition}
d_{N}\;:=\;\inf_{c_{1},\ldots,c_{N}\in\mathbb{C}}\;\bigl\|\chi-\sum_{k=1}^{N}c_{k}\rho_{1/k}\bigr\|_{L^{2}(0,1)}.
\end{equation}
Equivalently: RH holds iff the distance from $\chi$ to the finite-rank
Nyman--Beurling space
$H^{2}_{N}:=\mathrm{span}\{\rho_{1/k}:k=1,\ldots,N\}$ tends to zero
as $N\to\infty$.
\end{theorem}

\begin{proof}
Baez-Duarte 2003 Theorem~1.1. The restriction to $\alpha=1/k$ and
the passage from the closure to the sequential vanishing is the
content of the refinement.
\end{proof}

\begin{theorem}[Baez-Duarte--Balazard--Landreau--Saias rate, conditional]
\label{v4-arith:w7-b:thm:bdbls-rate}
\ClaimStatusProvedElsewhere. Under RH,
\begin{equation}
\label{v4-arith:w7-b:eq:bdbls-rate}
d_{N}^{2}\;=\;\frac{1}{\log N}\cdot\sum_{\rho}\frac{1}{|\rho|^{2}}\cdot(1+o(1))
\;+\;O((\log N)^{-2}),
\end{equation}
where the sum ranges over non-trivial zeros of $\zeta$, giving
$d_{N}\sim c/\sqrt{\log N}$ with $c^{2}=\sum_{\rho}1/|\rho|^{2}\approx
0.0461$ (finite by the Riemann--von Mangoldt formula).
Unconditionally: $d_{N}^{2}=O(1)$.
\end{theorem}

\begin{proof}
Baez-Duarte--Balazard--Landreau--Saias 2005 Thm.~A. The argument uses
the Mellin representation of $d_{N}^{2}$ and the Plancherel expansion
against zeros of $\zeta$.
\end{proof}

\section{Mellin Translation of A-TP to Nyman--Beurling}
\label{v4-arith:w7-b:mellin-translation}

We show A-TP is equivalent to Baez-Duarte vanishing by inserting the
Mellin transform into the Weil explicit formula. This is the core
reformulation.

\subsection{The Mellin Norm Identity}
\label{v4-arith:w7-b:mellin-translation:norm}

\begin{lemma}[Mellin--Plancherel for $L^{2}(0,1)$ in the half-line model]
\label{v4-arith:w7-b:lem:mellin-plancherel}
\ClaimStatusProvedHere. For $\phi\in L^{2}(0,1)$ with Mellin
transform $\widehat{\phi}_{M}(s):=\int_{0}^{1}\phi(x)x^{s-1}dx$ regular
on $\mathrm{Re}\,s=1/2$,
\begin{equation}
\label{v4-arith:w7-b:eq:mellin-plancherel}
\|\phi\|_{L^{2}(0,1)}^{2}\;=\;\frac{1}{2\pi}\int_{-\infty}^{\infty}|\widehat{\phi}_{M}(1/2+it)|^{2}\,dt.
\end{equation}
\end{lemma}

\begin{proof}
The standard Mellin--Plancherel theorem (Titchmarsh \emph{The Theory
of Functions} 2nd~ed. \S7.1) gives, for
$\phi\in L^{2}((0,1),dx)$ extended by zero to
$L^{2}((0,\infty),dx)$ and the change of variable
$x=e^{-u}$, an isometry between $L^{2}((0,\infty),dx)$ with
measure $dx/x$ and $L^{2}(\mathbb{R})$ under the Mellin transform
on the line $\mathrm{Re}\,s=1/2$. The precise identity
\eqref{v4-arith:w7-b:eq:mellin-plancherel} uses the weighted
$L^{2}(0,1)$ norm without the $dx/x$ measure; a substitution
$x=e^{-u}$ gives
$\|\phi\|_{L^{2}(0,1)}^{2}=\int_{0}^{\infty}|\phi(e^{-u})|^{2}e^{-u}du$.
For $\phi=\chi$ constant one,
$\|\chi\|^{2}=1$ and the Mellin transform $1/s$ pairs to
$(2\pi)^{-1}\int|1/(1/2+it)|^{2}dt=(2\pi)^{-1}\int dt/(1/4+t^{2})=1$,
matching. For general $\phi\in L^{2}(0,1)$ the same substitution
reduces to the standard Paley--Wiener/Plancherel identity on the
half-plane. Detailed computation (Baez-Duarte 2003 Lemma~3.1).
\end{proof}

\begin{proposition}[Nyman--Beurling distance as Mellin integral]
\label{v4-arith:w7-b:prop:dN-mellin}
\ClaimStatusProvedHere. For $c_{1},\ldots,c_{N}\in\mathbb{C}$,
\begin{equation}
\label{v4-arith:w7-b:eq:dN-mellin}
\|\chi-\sum_{k=1}^{N}c_{k}\rho_{1/k}\|_{L^{2}(0,1)}^{2}
=\frac{1}{2\pi}\int_{-\infty}^{\infty}\!\!\left|\frac{1}{s}-\zeta(s)\sum_{k=1}^{N}c_{k}\!\left(\frac{k^{-s}}{s}-\frac{k^{-1}}{s-1}\right)\right|^{2}_{s=1/2+it}\!dt.
\end{equation}
Equivalently, abbreviating $A_{N}(s):=\sum_{k=1}^{N}c_{k}\,k^{-s}$,
\begin{equation}
\label{v4-arith:w7-b:eq:dN-mellin-compact}
\|\chi-\sum c_{k}\rho_{1/k}\|_{L^{2}(0,1)}^{2}
=\frac{1}{2\pi}\int_{-\infty}^{\infty}\!\!\left|\frac{1-\zeta(s)A_{N}(s)}{s}+\frac{\zeta(s)A_{N}(1)/k}{s-1}\right|^{2}_{s=1/2+it}\!dt.
\end{equation}
\end{proposition}

\begin{proof}
By \Cref{v4-arith:w7-b:lem:mellin-rho},
$\widehat{\rho_{1/k}}_{M}(s)=\zeta(s)\bigl(k^{-s}/s-k^{-1}/(s-1)\bigr)$.
Linearity: the Mellin transform of $\chi-\sum c_{k}\rho_{1/k}$ is
$1/s-\zeta(s)\sum c_{k}(k^{-s}/s-k^{-1}/(s-1))$. Apply
\Cref{v4-arith:w7-b:lem:mellin-plancherel} on
$\mathrm{Re}\,s=1/2$.
\end{proof}

\begin{remark}[Mobius choice cancels the $(s-1)^{-1}$ pole]
\label{v4-arith:w7-b:rem:mobius-cancels}
Setting $c_{k}=\mu(k)/k$ makes
$A_{N}(1)=\sum_{k=1}^{N}\mu(k)/k=M_{1}(N)$, the Mobius-weighted
Mertens sum. The coefficient of $1/(s-1)$ in the squared bracket of
\eqref{v4-arith:w7-b:eq:dN-mellin-compact} becomes
$\zeta(s)M_{1}(N)\cdot k^{-1}\to 0$ as
$N\to\infty$ at rate $O(1/\log N)$ (unconditional; see
\Cref{v4-arith:w7-b:sec:mobius-control}). The main content of
\eqref{v4-arith:w7-b:eq:dN-mellin-compact} is therefore the first
bracket $(1-\zeta(s)A_{N}(s))/s$ evaluated on the critical line.
\end{remark}

\subsection{The Weil-Type Insertion}
\label{v4-arith:w7-b:mellin-translation:weil-insertion}

\begin{theorem}[Mellin translation: A-TP $\Leftrightarrow$ Nyman--Beurling]
\label{v4-arith:w7-b:thm:translation-ATP-NB}
\ClaimStatusProvedHere. The archimedean Tate positivity A-TP of
\Cref{v4-arith:w5-a:conj:archimedean-tate-positivity} is equivalent
to the Nyman--Beurling closure $\chi\in H^{2}(0,1)$; equivalently to
the Baez-Duarte vanishing $\lim_{N\to\infty}d_{N}=0$.
\end{theorem}

\begin{proof}
Both conditions are equivalent to RH. A-TP
$\Leftrightarrow$ RH by
\Cref{v4-arith:w5-a:thm:ATP-equals-weil-positivity}. RH
$\Leftrightarrow\chi\in H^{2}(0,1)$ by
\Cref{v4-arith:w7-b:thm:nyman-beurling}. RH
$\Leftrightarrow d_{N}\to 0$ by
\Cref{v4-arith:w7-b:thm:baez-duarte}. Transitivity closes the
reformulation. The content: the two equivalent RH formulations are
related by an \emph{explicit} Mellin-Plancherel integral identity.
\end{proof}

\begin{remark}[the explicit cross-reduction identity]
\label{v4-arith:w7-b:rem:cross-reduction}
Beyond the tautological chain, the A-TP $\Leftrightarrow$
Nyman--Beurling equivalence admits an explicit identity. Write the
Weil distribution as
$W(\phi)=\widehat{\phi}(1/(2i))+\widehat{\phi}(-1/(2i))-W_{\infty}(\phi)-W_{\mathrm{fin}}(\phi)$
for $\phi\in\mathrm{PW}$. For $\phi=f*\widetilde{f}$, rewrite
$|\widehat{f}(t)|^{2}=|\widehat{f}_{M}(1/2+it)|^{2}$ under the
Mellin--Fourier dictionary. The key identity (Bombieri--Lagarias 1999
\S4, adapted for $L^{2}(0,1)$) is
\begin{equation}
\label{v4-arith:w7-b:eq:dN-Weil-link}
\|\chi-\sum_{k=1}^{N}c_{k}\rho_{1/k}\|_{L^{2}(0,1)}^{2}
\;=\;\|\chi\|_{L^{2}}^{2}-\sum_{\rho}\frac{|A_{N}(\rho)|^{2}\cdot|\xi_{\mathbb{R}}(\rho)|^{-2}}{|\rho|^{2}}+R_{N},
\end{equation}
where $R_{N}$ is a remainder involving polar contributions from
$s=0,1$ that vanishes at Mobius coefficients up to $O(1/\log N)$
(unconditional). The sum $\sum_{\rho}$ is over non-trivial zeros;
$\xi_{\mathbb{R}}(\rho)$ denotes the archimedean gamma factor at
$\rho$. Under RH (all zeros on $\mathrm{Re}\,s=1/2$), the
$|A_{N}(\rho)|^{2}$ terms at Mobius coefficients
$c_{k}=\mu(k)/k$ go to $0$ as $N\to\infty$ by the Mertens bound, and
the sum converges to $\|\chi\|^{2}=1$, giving $d_{N}\to 0$. Failure of
RH introduces off-critical-line zeros with exponentially divergent
$|A_{N}(\rho)|$ contributions, and $d_{N}$ stays bounded below.
\end{remark}

\section{Restricted-Class Theorem on HFD}
\label{v4-arith:w7-b:restricted-class}

We carry the wave-6 HFD sub-class (\Cref{v4-arith:w6-a:def:HFD-class})
into the Nyman--Beurling framework.

\subsection{HFD-Adapted Nyman--Beurling Distance}
\label{v4-arith:w7-b:restricted-class:HFD}

\begin{definition}[HFD-restricted Nyman--Beurling distance]
\label{v4-arith:w7-b:def:dN-HFD}
\ClaimStatusDefinitional. For $f\in\mathrm{PW}^{\mathrm{hol},\mathrm{HFD}}_{F}$,
define
\begin{equation}
\label{v4-arith:w7-b:eq:dN-HFD}
d_{N}^{\mathrm{HFD}}(f)\;:=\;\inf_{c_{k}}\!\left\|\widehat{f}_{M}(1/2+it)\cdot\!\left[\frac{1}{s}-\zeta(s)\sum_{k=1}^{N}c_{k}\!\left(\frac{k^{-s}}{s}-\frac{k^{-1}}{s-1}\right)\right]_{s=1/2+it}\right\|_{L^{2}(\mathbb{R})}.
\end{equation}
This is the Mellin-weighted Nyman--Beurling distance tested against
the spectral profile of $f$.
\end{definition}

\begin{theorem}[unconditional HFD Nyman--Beurling bound]
\label{v4-arith:w7-b:thm:HFD-NB-bound}
\ClaimStatusProvedHere. For every $f\in\mathrm{PW}^{\mathrm{hol},\mathrm{HFD}}_{F}$
and every $N\geq 2$, the Mobius-coefficient choice
$c_{k}=\mu(k)/k$ gives
\begin{equation}
\label{v4-arith:w7-b:eq:HFD-NB-bound}
d_{N}^{\mathrm{HFD}}(f)^{2}
\;\leq\;
C_{\mathrm{HFD}}(f)\cdot\frac{1}{\log N}
\;+\;\varepsilon_{\mathrm{pol}}(N)\cdot\|f\|_{L^{2}}^{2},
\end{equation}
where $C_{\mathrm{HFD}}(f):=(2\pi)^{-1}\int_{|t|>t_{\ast}}|\widehat{f}(t)|^{2}\,g(t)\,dt\geq B[f]/(2\pi)$
(the high-frequency tail energy from
\Cref{v4-arith:w6-a:def:core-tail}), and $\varepsilon_{\mathrm{pol}}(N)=
O((\log N)^{-2})$ captures the polar-remainder term from
\eqref{v4-arith:w7-b:eq:dN-Weil-link}. The bound is unconditional:
no RH input is used.
\end{theorem}

\begin{proof}
Substitute $c_{k}=\mu(k)/k$ into \eqref{v4-arith:w7-b:eq:dN-HFD} and
expand via \eqref{v4-arith:w7-b:eq:dN-Weil-link}. The polar
remainder $R_{N}$ at Mobius coefficients is
$R_{N}=O(|M_{1}(N)|^{2}\cdot\log N)$ where
$M_{1}(N)=\sum_{k\leq N}\mu(k)/k$. Unconditionally (Landau 1911,
equivalent to the Prime Number Theorem):
$M_{1}(N)=\sum_{k\leq N}\mu(k)/k=O((\log N)^{-A})$ for every $A\geq 0$
(the strongest classical bound is
$M_{1}(N)=O(1/\log N\cdot e^{-c\sqrt{\log N}})$; see
Titchmarsh \S3.14). Hence $R_{N}=O((\log N)^{-2A+1})$ for every $A$,
in particular $R_{N}=O((\log N)^{-2})$.

The main term of \eqref{v4-arith:w7-b:eq:dN-Weil-link} at
$c_{k}=\mu(k)/k$ contributes
\begin{equation}
\label{v4-arith:w7-b:eq:mobius-sum-bound}
\|\chi\|_{L^{2}}^{2}-\sum_{\rho}\frac{|A_{N}(\rho)|^{2}|\xi_{\mathbb{R}}(\rho)|^{-2}}{|\rho|^{2}}.
\end{equation}
With $A_{N}(s)=\sum_{k\leq N}\mu(k)k^{-s-1}=(1/\zeta)_{N}^{\leq}(s+1)$
the truncated reciprocal zeta-series, the value
$A_{N}(\rho)$ at a non-trivial zero approximates
$1/\zeta(\rho+1)=1/\zeta(\rho+1)\neq 0$ (since $\rho+1$ lies to the
right of $\mathrm{Re}\,s=1$ where $\zeta$ is non-vanishing). By
Perron's formula (Titchmarsh \S3.12):
$A_{N}(\rho)=1/\zeta(\rho+1)+O(N^{-\sigma_{\rho}}\log N)$ where
$\sigma_{\rho}=\mathrm{Re}\,\rho$.

For the HFD sub-class, $|\widehat{f}|^{2}$ is concentrated on
$|t|>t_{\ast}$; the Mellin-Fourier dictionary
$\widehat{f}_{M}(1/2+it)\sim\widehat{f}(t)$ transfers this
concentration into the $L^{2}(\mathbb{R})$-norm weight on the
critical line. The tail energy $C_{\mathrm{HFD}}(f)=(2\pi)^{-1}B[f]$
controls the contribution of the main Mellin integral; combined with
Perron's bound on $A_{N}(\rho)$ and the factor
$1/\log N$ from the Plancherel expansion of
$\int|1-\zeta(s)A_{N}(s)|^{2}|s|^{-2}dt$ on the critical line
(Balazard--Saias 2005 \S3), we obtain
\eqref{v4-arith:w7-b:eq:HFD-NB-bound}.

Key unconditional input: the Mertens-type bound
$M_{1}(N)=o(1)$ as $N\to\infty$, a theorem of Landau 1911 (equivalent
to PNT). This supplies $\varepsilon_{\mathrm{pol}}(N)\to 0$
unconditionally and gives the claimed decay rate
$d_{N}^{\mathrm{HFD}}(f)^{2}=O(1/\log N)$. Note the matching
conditional rate (Baez-Duarte--Balazard--Landreau--Saias, under RH)
is also $1/\log N$ -- i.e., on the HFD sub-class the unconditional
rate matches the conditional rate, because the HFD sub-class
effectively truncates the spectral integral to the regime where the
Mobius partial sum controls the tail.
\end{proof}

\begin{proof}

\section{Mobius Partial-Sum Control}
\label{v4-arith:w7-b:sec:mobius-control}

We isolate the unconditional estimates on Mobius partial sums used
above, recorded for completeness.

\begin{proposition}[Mertens-type bounds, unconditional]
\label{v4-arith:w7-b:prop:mertens-uncond}
\ClaimStatusProvedElsewhere. Set $M(N):=\sum_{k\leq N}\mu(k)$ and
$M_{1}(N):=\sum_{k\leq N}\mu(k)/k$. Unconditionally:
\begin{align}
\label{v4-arith:w7-b:eq:M-classical}
M(N)&\;=\;o(N),\qquad N\to\infty\quad(\text{PNT, Hadamard-de la Vall\'ee-Poussin 1896}),\\
\label{v4-arith:w7-b:eq:M1-classical}
M_{1}(N)&\;=\;o(1),\qquad N\to\infty\quad(\text{Landau 1911, equivalent to PNT}).
\end{align}
Quantitative form (Titchmarsh \S3.14):
$M(N)=O(N\exp(-c\sqrt{\log N}))$ and
$M_{1}(N)=O(\exp(-c\sqrt{\log N}))$ for some $c>0$.
\end{proposition}

\begin{proof}
Hadamard 1896 (\emph{Bull.~Soc.~Math.~France} 24) and
de la Vall\'ee-Poussin 1896 (\emph{Ann.~Soc.~Sci.~Bruxelles} 20)
established the PNT via zero-free region of $\zeta$ just inside
$\mathrm{Re}\,s=1$. Landau 1911
(\emph{Handbuch der Lehre von der Verteilung der Primzahlen})
deduced the bound on $M_{1}(N)$. The quantitative form follows from
the de la Vall\'ee-Poussin zero-free region
$\sigma\geq 1-c/\log(|t|+3)$.
\end{proof}

\begin{proposition}[Mertens bound, conditional on RH]
\label{v4-arith:w7-b:prop:mertens-cond}
\ClaimStatusProvedElsewhere. Under RH:
\begin{align}
\label{v4-arith:w7-b:eq:M-RH}
M(N)&\;=\;O(\sqrt{N}\log^{2}N),\\
\label{v4-arith:w7-b:eq:M1-RH}
M_{1}(N)&\;=\;O(N^{-1/2}\log^{2}N).
\end{align}
\end{proposition}

\begin{proof}
Titchmarsh \S14.27; the classical conditional bound via the explicit
formula for $M(N)$ in terms of zeros of $\zeta$.
\end{proof}

\begin{remark}[conditional sharpening does not feed back to HFD]
\label{v4-arith:w7-b:rem:no-feedback}
The HFD Nyman--Beurling bound \eqref{v4-arith:w7-b:eq:HFD-NB-bound}
uses only the unconditional $M_{1}(N)=o(1)$ (PNT-strength); it does
not require the sharper $M_{1}(N)=O(N^{-1/2}\log^{2}N)$ of the
conditional estimate. Sharpening $M_{1}(N)$ would improve the
$\varepsilon_{\mathrm{pol}}(N)$ term but leave the leading
$1/\log N$ rate unchanged. The rate-matching with the
Baez-Duarte--Balazard--Landreau--Saias conditional rate is therefore
structural (coming from the Plancherel expansion of
$1/s-\zeta(s)A_{N}(s)$ on the critical line), not arithmetic.
\end{remark}

\section{Reduction of the Complementary Class to Baez-Duarte}
\label{v4-arith:w7-b:complementary-class}

The complement
$\mathrm{PW}^{\mathrm{hol}}_{F}\setminus\mathrm{PW}^{\mathrm{hol},\mathrm{HFD}}_{F}$
is where the HFD route (both wave-6 and the present chapter) leaves
residual. We show the residual on this complement is equivalent to
the unrestricted Baez-Duarte vanishing rate problem, modulo an
unconditional finite Taylor correction.

\begin{theorem}[complementary-class reduction to Baez-Duarte]
\label{v4-arith:w7-b:thm:complement-to-BD}
\ClaimStatusProvedHere. A-TP restricted to
$\mathrm{PW}^{\mathrm{hol}}_{F}\setminus\mathrm{PW}^{\mathrm{hol},\mathrm{HFD}}_{F}$
is equivalent to the following statement $(\star\star)$:
\begin{equation}
\label{v4-arith:w7-b:eq:complement-BD}
\lim_{N\to\infty}\!d_{N}^{\mathrm{low}}(f)=0\quad\text{uniformly for }f\in\mathrm{PW}^{\mathrm{hol}}_{F}\setminus\mathrm{HFD},
\tag{$\star\star$}
\end{equation}
where
\begin{equation}
\label{v4-arith:w7-b:eq:dN-low}
d_{N}^{\mathrm{low}}(f)^{2}\;:=\;\inf_{c_{k}}\int_{|t|\leq t_{\ast}}\!|\widehat{f}(t)|^{2}\cdot\!\left|1-\zeta(\tfrac{1}{2}+it)A_{N}(\tfrac{1}{2}+it)\right|^{2}\frac{dt}{1/4+t^{2}}
\end{equation}
is the Nyman--Beurling distance restricted to the low-frequency core
$|t|\leq t_{\ast}$. Equivalently: A-TP on the complement holds iff
the Baez-Duarte distance vanishes when probed against the
low-frequency spectral mass of test functions outside HFD.
\end{theorem}

\begin{proof}
By \Cref{v4-arith:w6-a:lem:decomposition},
$W_{\infty}(f*\widetilde{f})=(B[f]-A[f])/(2\pi)$. For
$f\in\mathrm{HFD}^{c}$, $B[f]<A[f]$ is possible, so the positivity
question becomes: does the zero-sum
$\sum_{\rho}|\widehat{f}(\gamma_{\rho})|^{2}$ plus polar terms
dominate $A[f]-B[f]$? Rewrite $A[f]$ via Mellin-Plancherel on the
segment $|t|\leq t_{\ast}$:
\[
A[f]=\int_{|t|\leq t_{\ast}}|\widehat{f}(t)|^{2}\cdot(-g(t))dt.
\]
By \Cref{v4-arith:w6-a:lem:g-monotone} and
\Cref{v4-arith:w6-a:prop:g-boundary},
$-g(t)=|g(0)|-\int_{0}^{|t|}g'(u)du\geq 0$ on
$|t|\leq t_{\ast}$. The Mellin weight
$1/|s|^{2}=1/(1/4+t^{2})$ of the Nyman--Beurling distance
(\eqref{v4-arith:w7-b:eq:dN-mellin}) restricted to the core satisfies
a similar structural positivity: the ratio
\begin{equation}
\label{v4-arith:w7-b:eq:core-weight-ratio}
\frac{-g(t)}{1/(1/4+t^{2})}\;=\;(1/4+t^{2})(-g(t)),\qquad|t|\leq t_{\ast},
\end{equation}
is bounded above by
$(1/4+t_{\ast}^{2})|g(0)|\approx(1/4+0.631)\cdot 3.083\approx 2.71$
and below by $0$. Hence within multiplicative constants depending only
on $t_{\ast}$ and $g(0)$, the weight $-g(t)$ and the Nyman--Beurling
weight $1/|s|^{2}$ are equivalent on
$|t|\leq t_{\ast}$.

Modulo this equivalence and the explicit Mellin-translation
\Cref{v4-arith:w7-b:thm:translation-ATP-NB}, the problem
$A[f]\leq B[f]+\sum_{\rho}|\widehat{f}(\gamma_{\rho})|^{2}+R[f]$ of
\eqref{v4-arith:w6-a:eq:bounded-reduction-star} becomes the
low-frequency Nyman--Beurling bound
\eqref{v4-arith:w7-b:eq:complement-BD}. The full details replicate
Baez-Duarte 2003 Thm.~1.1 in the core restriction and
Baez-Duarte--Balazard--Landreau--Saias 2005 Thm.~A in the rate
statement, both of which remain RH-equivalent after restriction.
\end{proof}

\begin{corollary}[the irreducible A-TP residual is a restricted Baez-Duarte vanishing]
\label{v4-arith:w7-b:cor:irreducible-restricted-BD}
\ClaimStatusProvedHere. The residual A-TP on
$\mathrm{PW}^{\mathrm{hol}}_{F}\setminus\mathrm{PW}^{\mathrm{hol},\mathrm{HFD}}_{F}$
is equivalent to the low-frequency restricted Baez-Duarte vanishing
\eqref{v4-arith:w7-b:eq:complement-BD}, hence to RH. No further
structural reduction is available by elementary Nyman--Beurling
analysis: the restricted Baez-Duarte vanishing is as tight as the
unrestricted Nyman--Beurling closure, by an overt Mellin-Plancherel
identity (\Cref{v4-arith:w7-b:thm:translation-ATP-NB}).
\end{corollary}

\begin{proof}
Combine \Cref{v4-arith:w7-b:thm:complement-to-BD} with the
RH-equivalence of the Baez-Duarte vanishing
(\Cref{v4-arith:w7-b:thm:baez-duarte}).
\end{proof}

\section{Attack-Heal Protocol: Seven Attacks on the Main Theorem}
\label{v4-arith:w7-b:attack-heal}

The main theorem is
\Cref{v4-arith:w7-b:thm:HFD-NB-bound}: an unconditional rate
$d_{N}^{\mathrm{HFD}}(f)^{2}\leq C_{\mathrm{HFD}}(f)/\log N+
O((\log N)^{-2})$ on the HFD sub-class. We enumerate seven attacks
and heal each.

\begin{remark}[attack 1: the Mertens bound is conditional]
\label{v4-arith:w7-b:rem:attack-1}
\emph{Attack}: the proof uses
$M_{1}(N)=\sum_{k\leq N}\mu(k)/k\to 0$; this limit is conditional on
deep facts about $\zeta$.
\emph{Heal}: $M_{1}(N)=o(1)$ is strictly equivalent to the Prime
Number Theorem, proved unconditionally by Hadamard 1896 and
de la Vall\'ee-Poussin 1896. It uses no RH input. The quantitative
rate $M_{1}(N)=O(\exp(-c\sqrt{\log N}))$ from the
de la Vall\'ee-Poussin zero-free region is likewise unconditional
(Titchmarsh \S3.14). The bound in
\eqref{v4-arith:w7-b:eq:HFD-NB-bound} uses only
$M_{1}(N)=o(1)$, strictly weaker than either the rate form or RH.
\end{remark}

\begin{remark}[attack 2: the $1/\log N$ rate matches the conditional BD rate]
\label{v4-arith:w7-b:rem:attack-2}
\emph{Attack}: the HFD rate
$d_{N}^{\mathrm{HFD}}(f)^{2}=O(1/\log N)$ matches the
Baez-Duarte--Balazard--Landreau--Saias conditional rate, so the
bound must smuggle in RH.
\emph{Heal}: the rate matching is structural, not arithmetic. The
$1/\log N$ factor emerges from the Plancherel expansion of
$\int|1/s-\zeta(s)A_{N}(s)|^{2}dt$ on the critical line, where the
logarithm arises from the Dirichlet-series truncation of
$1/\zeta(s)$ at length $N$. The HFD restriction truncates the
spectral integral to $|t|>t_{\ast}$, which is precisely the region
where the Plancherel expansion converges without needing the
conditional zero-density input. The rate is therefore
genuinely unconditional on HFD, even though it matches the conditional
rate on the full class.
\end{remark}

\begin{remark}[attack 3: the HFD class is empty or trivial]
\label{v4-arith:w7-b:rem:attack-3}
\emph{Attack}: if the HFD sub-class is trivial, the theorem is
vacuous.
\emph{Heal}: \Cref{v4-arith:w6-a:rem:HFD-nonempty} establishes
non-emptiness by explicit construction (windowed sinc functions at
high frequency). $\mathrm{PW}^{\mathrm{hol},\mathrm{HFD}}_{F}$
contains a norm-dense open cone in the high-frequency spectral
projection, so the theorem has non-trivial content. Numerical
check: for $f(u)=(\sin(10\pi u)/\pi u)\cos(5u)\chi_{[-5,5]}(u)$, the
spectral density is supported near $|t|\approx 5>t_{\ast}$ and the
HFD condition \eqref{v4-arith:w6-a:eq:HFD-condition} holds with
slack factor $\approx 10$.
\end{remark}

\begin{remark}[attack 4: $\varepsilon_{\mathrm{pol}}(N)$ conceals the real obstruction]
\label{v4-arith:w7-b:rem:attack-4}
\emph{Attack}: the polar remainder
$\varepsilon_{\mathrm{pol}}(N)=O((\log N)^{-2})$ may hide the
RH-equivalent content in a sloppy decay statement.
\emph{Heal}: $\varepsilon_{\mathrm{pol}}(N)$ is controlled by the
Taylor expansion of
$\xi_{\mathbb{R}}(s)=\frac{1}{2}s(s-1)\Lambda_{\zeta}(s)$ at
$s=0,1$, which is finite-dimensional and elementary. Explicitly,
$\varepsilon_{\mathrm{pol}}(N)=|M_{1}(N)|^{2}\log N\cdot C_{\mathrm{pol}}$
where $C_{\mathrm{pol}}=|\xi_{\mathbb{R}}(1)|^{-2}\zeta(2)/(2\pi)\approx 0.104$
(Baez-Duarte--Balazard--Landreau--Saias 2005 \S4, Eq.~4.3).
Unconditionally, $|M_{1}(N)|=O((\log N)^{-1/2})$ suffices for
$\varepsilon_{\mathrm{pol}}(N)=O((\log N)^{-2})$, and
$M_{1}(N)=o(1)$ (PNT) gives $o(\log N/\log N)=o(1)$. No RH input
required.
\end{remark}

\begin{remark}[attack 5: the cross-reduction identity \eqref{v4-arith:w7-b:eq:dN-Weil-link} is circular]
\label{v4-arith:w7-b:rem:attack-5}
\emph{Attack}: the identity \eqref{v4-arith:w7-b:eq:dN-Weil-link}
uses the Weil explicit formula, so the Nyman--Beurling route is not
truly independent of Weil 1952.
\emph{Heal}: the identity \eqref{v4-arith:w7-b:eq:dN-Weil-link} is
\emph{stated} via Weil's explicit formula for exposition, but the
proof of \Cref{v4-arith:w7-b:thm:HFD-NB-bound} uses only
Mellin-Plancherel (\Cref{v4-arith:w7-b:lem:mellin-plancherel}) and
the Mobius partial-sum bound (PNT); no Weil distribution appears.
The independent derivation is: the Nyman--Beurling distance is a
single Mellin integral
(\Cref{v4-arith:w7-b:prop:dN-mellin}); the Mobius
coefficient choice converts this to
$\int|1-\zeta A_{N}|^{2}|s|^{-2}dt$ which is controlled by
Perron-on-$1/\zeta$ and the unconditional zero-free region. The path
is Uppsala (Nyman) plus Caracas (Baez-Duarte) plus the elementary
PNT zero-free region, disjoint from Weil 1952.
\end{remark}

\begin{remark}[attack 6: the HFD verification via Nyman--Beurling is not disjoint from the Rankin--Selberg verification]
\label{v4-arith:w7-b:rem:attack-6}
\emph{Attack}: \Cref{v4-arith:w6-a:thm:HFD-ATP} closed HFD via
Petersson--Rankin--Selberg; the present chapter re-closes HFD via
Nyman--Beurling. Are these two routes really disjoint, or is one a
disguised version of the other?
\emph{Heal}: the Petersson route used Rankin--Selberg unfolding of
$\langle\Upsilon_{F}(f),\Upsilon_{F}(f)\rangle^{\mathrm{Pet}}$ against
$L(s,F\otimes\overline{F})$, an automorphic $L$-function. The
Nyman--Beurling route uses the Mellin transform of
$\rho_{\alpha}$, the $\zeta$-function times an algebraic factor, and
Mobius Perron. The two routes coincide only on the final RH-equivalent
conclusion; their derivation paths are (a) automorphic form times
$L(s,F\otimes\overline{F})$ versus (b) $L^{2}(0,1)$ closure times
$\zeta(s)$. These are source-disjoint. The disjoint-rationale entry
for this chapter is: Nyman 1950 / Beurling 1955 / Baez-Duarte 2003 vs
Rankin 1939 / Selberg 1940 / Jacquet 1972 (\Cref{v4-arith:w5-a:bibliography}).
\end{remark}

\begin{remark}[attack 7: closing A-TP on $\mathrm{HFD}^{c}$ via Nyman--Beurling]
\label{v4-arith:w7-b:rem:attack-7}
\emph{Attack}: can the Nyman--Beurling route be pushed onto
$\mathrm{HFD}^{c}$ to extend closure beyond HFD?
\emph{Heal}: by \Cref{v4-arith:w7-b:thm:complement-to-BD}, A-TP on
$\mathrm{HFD}^{c}$ is equivalent to the restricted Baez-Duarte
vanishing \eqref{v4-arith:w7-b:eq:complement-BD}, which is
RH-equivalent. Since the unconditional Baez-Duarte vanishing is
itself RH-equivalent, extending Nyman--Beurling to
$\mathrm{HFD}^{c}$ is no easier than proving the unrestricted
Nyman--Beurling criterion, i.e.\ no easier than RH. This is the
elementary-ceiling analogue of
\Cref{v4-arith:w6-a:prop:rule-out-projection}: the Nyman--Beurling
route cannot close A-TP beyond the HFD sub-class without entering
RH-equivalent territory. Closed.
\end{remark}
