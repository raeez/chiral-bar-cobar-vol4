\section{The Synthesis}
\label{v4-arith:syn:synthesis}

\begin{theorem}[six-window placement]
\label{v4-arith:thm:six-window-placement}
\ClaimStatusConjectured. $\mathcal{V}^{\mathrm{prim}}$ admits the
following six equivalent presentations:
\begin{enumerate}
\item \textbf{(Langlands / automorphic)}
$\mathcal{V}^{\mathrm{prim}}$ is the universal Hecke character on
$\mathrm{D}([GL_{2}])^{\mathrm{sph\text{-}fin}}$; its spectral
counterpart is the structure sheaf of the global Emerton--Gee
stack $\mathcal{X}^{\mathrm{EG,glob}}_{GL_{2}}$.
\item \textbf{(TQFT / Chern--Simons)}
$\mathcal{V}^{\mathrm{prim}}=\partial\,\mathrm{ACS}[\overline{\mathrm{Spec}\,\mathbb{Z}},\mathbb{G}_{m},1]$
at level~$1$; Wilson lines at prime $p$ are ramified Hecke
characters.
\item \textbf{(Categorical / Hochschild trace)}
$\mathcal{V}^{\mathrm{prim}}=\mathrm{HH}_{\bullet}(\mathrm{Hk}^{\mathrm{Iw}}_{GL_{2},\mathrm{glob}})$,
the trace class of the identity on the global affine Iwahori
Hecke category.
\item \textbf{(Integrability / Bethe algebra)}
$\mathcal{V}^{\mathrm{prim}}$ is an arithmetic Yangian-module spin
chain with primes as lattice sites, Hecke = transfer matrices,
Hecke eigenforms = Bethe eigenvectors; the arithmetic Yangian is
the commutative Bethe subalgebra $\bigotimes_{p}\mathcal{H}_{p}^{\mathrm{sph}}$.
\item \textbf{(Statistical mechanics)} $\mathcal{V}^{\mathrm{prim}}_{N}\cong(\text{Bost--Connes at rank }\pi(N))\otimes(\text{universal Heisenberg descendant tower of rank }\pi(N))$;
all arithmetic content in the left factor, the VA factor
universal.
\item \textbf{(Factorization / homotopical)}
$\mathcal{V}^{\mathrm{prim}}\in\mathrm{FactAlg}_{\mathbb{C}_{\infty}}(\mathrm{Mod}_{\widehat{\mathbb{Z}},\psi})$
with (F1)--(F5) of \Cref{v4-arith:def:CG-arithmetic-factorization}:
holomorphic factorization at the archimedean place, locally
constant with Adams operations at each finite place.
\end{enumerate}
These are the same object seen through different cohomology
theories. Ben-Zvi--Nadler's identification
$\mathrm{HH}_{\bullet}(\mathrm{D}(G/G))=\mathcal{O}(\mathrm{LocSys}_{G}(S^{1}))^{\mathrm{flat}}$
glues (1), (2), (3) via the CS-Hecke-trace triangle.
\end{theorem}

\section{The Load-Bearing Theorem}
\label{v4-arith:syn:load-bearing}

\begin{theorem}[$\mathsf{G}$-row arithmetic complementarity = Riemann functional equation]
\label{v4-arith:thm:G-row-Riemann-functional-equation}
\ClaimStatusConjectured.
\begin{equation}
\boxed{\;\kappa^{\mathrm{Ar}}_{\mathsf{G}}\,+\,\bigl(\kappa^{!}\bigr)^{\mathrm{Ar}}_{\mathsf{G}}\;=\;0\;\Longleftrightarrow\;\xi(s)\;=\;\xi(1-s).\;}
\end{equation}
The arithmetic Heisenberg sits in archetype $\mathsf{G}$ with
$\kappa=0$. Its Koszul dual is Bost--Connes. The complementarity
$\kappa+\kappa^{!}=0$ translates, through the Koszul-resolution
computation of Hochschild cochains, to the meromorphic
continuation and functional equation of $\zeta$. The archimedean
$\Gamma(s/2)\pi^{-s/2}$-factor is the derived-centre shift.
\end{theorem}

\begin{remark}[significance]
This is the first genuine mathematical content the fifteen-agent
process produced. The arithmetic Heisenberg's Koszul duality
\emph{implies} the Riemann functional equation. The Vol~I chain
``Koszul duality $\Rightarrow$ derived-centre complementarity''
becomes ``Koszul duality $\Rightarrow$ functional equation of the
Dirichlet series generating bar cohomology''.
\end{remark}

\begin{remark}[what the fifteen voices agree on]
\begin{enumerate}
\item $\mathcal{V}^{\mathrm{prim}}$ is not a vertex algebra in the
strict sense: no Virasoro action, no BD factorization on
$\overline{C}$, no Borger $\Lambda$-ring structure on the natural
integral form.
\item It is a spectral datum: Hochschild trace class, boundary of
arithmetic CS, automorphic representation, Bethe algebra module.
\item All arithmetic content lives on the Bost--Connes /
automorphic / Hecke side; the VA structure that appears is the
universal rank-$\pi(N)$ Heisenberg descendant tower, which does
not see the primes.
\item Its partition function is an $L$-function; character is
$\zeta(s)$ for the trivial sector or $L(s,\chi)$ for Hecke
characters $\chi$.
\item The complementarity $\kappa+\kappa^{!}=0$ encodes the
Riemann functional equation, through Koszul duality.
\end{enumerate}
\end{remark}

\begin{remark}
This is the Beilinson principle applied at the programme level.
The attack--heal process is the algorithmic analogue of adversarial
mathematics: propose a speculation; subject it to independent
adversarial attack from multiple perspectives; retain only what
survives. What survived the fifteen-voice assault on arithmetic
chiral algebras is a small, rigorous, structurally forced core
that connects the Vol~I chiral-bar-cobar programme to the full
arithmetic Langlands tower.
\end{remark}
