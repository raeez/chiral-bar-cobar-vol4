\chapter{Central Charge as Residue}
\label{v4-arith:polyakov}

\emph{In the voice of Polyakov. Central charge is a physical
observable, not a label. A regularisation is honest only if the
regularised object controls a computable correlator. We compute the
central charge of $\mathcal{V}^{\mathrm{prim}}$.}

\section{Rescaling Is Not a Conformal Rescaling}
\label{v4-arith:pol:rescaling-not-conformal}

\begin{observation}[rescaling breaks Virasoro]
\label{v4-arith:obs:rescaling-breaks-virasoro}
For a rescaled stress tensor $T'=\lambda T$:
\begin{equation*}
T'(z)T'(w)\;=\;\lambda^{2}\Bigl(\frac{c/2}{(z-w)^{4}}+\frac{2T(w)}{(z-w)^{2}}+\frac{\partial T(w)}{z-w}\Bigr)+\cdots
\end{equation*}
For this to be a conformal OPE
$\tfrac{c'/2}{(z-w)^{4}}+\tfrac{2T'(w)}{(z-w)^{2}}+\cdots$ we need
$\lambda^{2}c=c'$ but $\lambda^{2}\cdot 2T=2T'=2\lambda T$, i.e.
$\lambda^{2}=\lambda$, forcing $\lambda=1$. Rescaling by
$\lambda\neq 1$ does not yield a conformal vector.
\end{observation}

\begin{remark}
Reading the original data: $\omega_{p}=\tfrac{1}{2}(\log p)(a^{(p)}_{-1}a^{(p)}_{-1}+b^{(p)}_{-1}b^{(p)}_{-1})|0\rangle$
is the Sugawara built from the \emph{rescaled} Heisenberg
$[a^{(p)}_{m},a^{(p)}_{n}]=m(\log p)\delta_{m+n}$, so $\omega_{p}$ is
canonical in the rescaled coordinate. Under this reading,
$c^{(p)}=2$ independent of $\log p$. The $\log p$ moves
$L_{0}$-spectrum but not central charge.
\end{remark}

\begin{theorem}[per-prime central charge]
\label{v4-arith:thm:per-prime-c}
\ClaimStatusProvedHere. $c^{(p)}=2$ for every prime $p$. The
$\log p$ rescaling is a metric on the Heisenberg dual lattice; it
enters conformal weights, not central charge.
\end{theorem}

\section{Infinite Tensor Product Is Not a VOA}
\label{v4-arith:pol:infinite-tensor-not-VOA}

\begin{observation}[infinite tensor fails Virasoro]
\label{v4-arith:obs:infinite-tensor-fails}
Stress tensor $\omega=\sum_{p}\omega_{p}$ is an infinite sum of
non-zero vectors in the Fock module; not a vector in the module.
On the restricted tensor product, the Virasoro commutator
$[L_{m},L_{-m}]=2m L_{0}+\tfrac{c}{12}m(m^{2}-1)$ has
$c=\sum_{p}c^{(p)}=\sum_{p}2=+\infty$. The central term is an
infinite scalar multiplying every state.
\end{observation}

\begin{corollary}
\label{v4-arith:cor:no-virasoro}
\ClaimStatusProvedHere. $\mathcal{V}^{\mathrm{prim}}$ is not a
VOA: the central term of the Virasoro bracket is divergent. Only
the Witt algebra (classical conformal, no central extension) acts
algebraically.
\end{corollary}

\section{$\zeta_{\mathbb{P}}(0)$ Does Not Exist}
\label{v4-arith:pol:prime-zeta-obstruction}

\begin{theorem}[natural boundary of prime zeta]
\label{v4-arith:thm:prime-zeta-natural-boundary}
\ClaimStatusProvedElsewhere (Landau--Walfisz 1920; Fr\"oberg 1968).
The prime zeta function $\zeta_{\mathbb{P}}(s)=\sum_{p}p^{-s}$ has
a natural boundary on $\mathrm{Re}(s)=0$: logarithmic singularities
at $s=\rho/n$ for every nontrivial zero $\rho$ of $\zeta$ and every
$n\geq 1$ are dense on $\mathrm{Re}(s)=0$.
$\zeta_{\mathbb{P}}$ cannot be analytically continued past the
imaginary axis.
\end{theorem}

\begin{corollary}[$\zeta_{\mathbb{P}}(0)$ undefined]
\label{v4-arith:cor:zeta-P-0-undefined}
\ClaimStatusProvedHere. The proposed regularisation
$c=2\zeta_{\mathbb{P}}(0)=-1$ is not well-defined.
\end{corollary}

\begin{remark}[category error in $c=-1$]
The alternative proposal $c=2\zeta(0)=-1$ silently replaces primes
by positive integers: $\zeta(0)=-1/2$ regularises
$\sum_{n\geq 1}1$, not $\sum_{p}1$. Primes are a density-zero
subset of $\mathbb{Z}_{>0}$; $\zeta(0)$ does not regularise a prime
sum.
\end{remark}

\section{Central Charge as Residue}
\label{v4-arith:pol:c-as-residue}

\begin{theorem}[regularised global central charge]
\label{v4-arith:thm:regularised-c}
\ClaimStatusProvedHere. The only finite regularisation of the
global central charge that survives contact with the functional
equation $\xi(s)=\xi(1-s)$ is
\begin{equation}
\boxed{\;c_{\mathrm{global}}\;=\;2\cdot\mathrm{Res}_{s=1}\zeta(s)\;=\;2.\;}
\end{equation}
Not a trace, not a sum. A residue.
\end{theorem}

\begin{proof}[Sketch]
Per-prime contribution to the partition is
$\mathrm{Tr}_{\mathcal{H}^{(p)}}p^{-sL_{0}}=\eta(p^{-s})^{-2}$;
Euler product $=\xi(s)$ after archimedean completion. Per-prime
$c^{(p)}=2$. The global zeta-regularised central charge extracts
the residue at the zeta pole where the trace diverges most sharply:
$2\cdot\mathrm{Res}_{s=1}\zeta=2\cdot 1=2$.
\end{proof}

\section{Witt versus Virasoro}
\label{v4-arith:pol:witt-virasoro}

\begin{theorem}[Witt acts, Virasoro does not]
\label{v4-arith:thm:witt-virasoro}
\ClaimStatusProvedHere. $\mathcal{V}^{\mathrm{prim}}$ has a Witt
action (classical conformal, no central extension) on every state
via $L_{m}=\sum_{p}L_{m}^{(p)}$ (finite sum on each restricted
vector). It has no Virasoro action algebraically; the quantum
anomaly (Schwinger central term) is infinite and cannot be
absorbed by any normal-ordering procedure that preserves the
Virasoro relation.
\end{theorem}

\begin{remark}[soft-algebra interpretation]
This matches the Hartnoll--Yang reading of
$\mathcal{V}^{\mathrm{prim}}$ as a partition function on the
gravitational superspace: asymptotic symmetries are Witt-level
(reparametrisation invariance); quantum anomalies live on the
genuine Virasoro-level. The arithmetic version has Witt
asymptotic symmetry but no honest Virasoro lift.
\end{remark}

\section{The Functional Equation as Arithmetic Modular Invariance}
\label{v4-arith:pol:functional-equation-modular}

\begin{theorem}[arithmetic modular invariance]
\label{v4-arith:thm:arithmetic-modular-invariance}
\ClaimStatusProvedHere. The partition function of
$\mathcal{V}^{\mathrm{prim}}$ is \emph{not} a genus-$1$ modular
form $Z(\tau)=\mathrm{tr}\,q^{L_{0}-c/24}$ but the completed
Riemann zeta $\xi(s)=\tfrac{1}{2}s(s-1)\pi^{-s/2}\Gamma(s/2)\zeta(s)$.
Its modular invariance is the functional equation $\xi(s)=\xi(1-s)$
-- the arithmetic analogue of $\tau\to-1/\tau$.
\end{theorem}

\section{The Arakelov Pairing Replaces the Weyl Anomaly}
\label{v4-arith:pol:arakelov-pairing}

\begin{proposition}[Deligne pairing = Weyl anomaly analogue]
\label{v4-arith:prop:deligne-pairing}
\ClaimStatusProvedElsewhere. Under the Deligne pairing on Arakelov
line bundles, $\langle\omega_{p},\omega_{p}\rangle_{\mathrm{Arakelov}}=\log p$.
Summed over primes this reproduces (up to regularisation) the
logarithmic derivative of $\zeta$.
\end{proposition}

\begin{remark}[interpretation]
On $\mathbb{C}$ the Weyl anomaly is $\langle T^{\mu}_{\mu}\rangle=-cR/24\pi$;
Polyakov--Liouville action uses this as the central observable. On
$\overline{C}=\overline{\mathrm{Spec}\,\mathbb{Z}}$ the analogous
observable is the Arakelov metric contribution to the height
pairing: $\log p$ per prime. The arithmetic anomaly is height-theoretic,
not Weyl-theoretic.
\end{remark}

\section{Summary of This Chapter}
\label{v4-arith:pol:summary}

\begin{enumerate}
\item Per-prime $c^{(p)}=2$; $\log p$ rescaling does not change $c$
(\Cref{v4-arith:thm:per-prime-c}).
\item Infinite tensor $\bigotimes_{p}\mathcal{H}^{(p)}$ is not a
VOA; central term diverges
(\Cref{v4-arith:cor:no-virasoro}).
\item $\zeta_{\mathbb{P}}(0)$ has natural boundary, undefined
(\Cref{v4-arith:cor:zeta-P-0-undefined}); $c=-1$ proposals involve
category errors.
\item Regularised global $c=2\cdot\mathrm{Res}_{s=1}\zeta=2$ --
a residue, not a sum
(\Cref{v4-arith:thm:regularised-c}).
\item Witt action is present; Virasoro action is absent
(\Cref{v4-arith:thm:witt-virasoro}).
\item Modular invariance $=$ functional equation
(\Cref{v4-arith:thm:arithmetic-modular-invariance}).
\item Weyl anomaly analogue is Arakelov/Deligne height pairing
(\Cref{v4-arith:prop:deligne-pairing}).
\end{enumerate}

Central charge, the physical observable, is finite and equals $2$
-- but it is a residue, not a trace. A sum over primes at $s=0$ is
a formal symbol measuring nothing at either boundary. The honest
outcome: the question ``what is $c$?'' was ill-posed on the face
of it; the repair reveals the Arakelov--$\zeta$ structure.
