\chapter{Arithmetic FLE Completion Ledger}
\label{v4-arith:fle-completion-ledger}

\emph{This chapter records the final repair of the arithmetic FLE
analogue. The corrected object is not a proved arithmetic version of
Gaitsgory's fundamental local equivalence. It is a Hochschild-trace
candidate equipped with six conditional windows, a Selmer-derived
spectral target, and a list of theorem-waiting equivalences whose
failure would stop the programme.}

\section{Corrected Spectral Target}
\label{v4-arith:fle-ledger:spectral-target}

\begin{definition}[Selmer-derived global spectral stack]
\label{v4-arith:fle-ledger:def:selmer-stack}
Fix:
\begin{itemize}
\item a finite set $S$ of places containing $\infty$, all primes
where $\bar\rho$ is ramified, and all primes where the local
condition is not unramified;
\item a residual representation
$\bar\rho\colon G_{\mathbb{Q},S}\to GL_{2}(k)$;
\item a determinant character $\delta$ lifting $\det\bar\rho$;
\item for each $v\in S$, a local condition
$\mathcal{X}^{\mathcal{L}_{v}}_{v}\hookrightarrow
\mathcal{X}^{W,\varphi\Gamma}_{GL_{2},v}$ inside the full
Weil/$(\varphi,\Gamma)$ local parameter stack.
\end{itemize}
The corrected global spectral stack is the derived Selmer fibre
product
\begin{equation}
\label{v4-arith:fle-ledger:eq:selmer-stack}
\mathcal{X}^{\mathrm{Sel}}_{\bar\rho,\delta,\mathcal{L}}
:=
\mathrm{Map}_{\bar\rho,\delta}(G_{\mathbb{Q},S},GL_{2})
\times^{\mathbf{R}}_{\prod_{v\in S}\mathrm{Map}(G_{\mathbb{Q}_{v}},GL_{2})}
\prod_{v\in S}\mathcal{X}^{\mathcal{L}_{v}}_{v}.
\end{equation}
It replaces every product/colimit presentation of a ``global
Emerton--Gee stack''. Local Emerton--Gee stacks supply local
$(\varphi,\Gamma)$ input; the global object is Selmer-derived.
\end{definition}

\begin{proposition}[Selmer tangent complex]
\label{v4-arith:fle-ledger:prop:selmer-tangent}
\ClaimStatusProvedHere. At a point
$\rho\in\mathcal{X}^{\mathrm{Sel}}_{\bar\rho,\delta,\mathcal{L}}$,
the tangent complex is the fixed-determinant Selmer complex
\begin{equation}
\label{v4-arith:fle-ledger:eq:selmer-tangent}
T_{\rho}\mathcal{X}^{\mathrm{Sel}}_{\bar\rho,\delta,\mathcal{L}}
\simeq
C^{\bullet}_{\mathcal{L}}(G_{\mathbb{Q},S},\mathrm{ad}^{0}\rho)[1],
\end{equation}
where $C^{\bullet}_{\mathcal{L}}$ is the mapping fibre of the
localisation map from global cochains to the quotients by the chosen
local tangent conditions.
\end{proposition}

\begin{proof}
The stack in
\eqref{v4-arith:fle-ledger:eq:selmer-stack} is a derived fibre
product. Applying the tangent functor gives the fibre of
\[
C^{\bullet}(G_{\mathbb{Q},S},\mathrm{ad}^{0}\rho)
\longrightarrow
\bigoplus_{v\in S}
C^{\bullet}(G_{\mathbb{Q}_{v}},\mathrm{ad}^{0}\rho)/
C^{\bullet}_{\mathcal{L}_{v}}(G_{\mathbb{Q}_{v}},\mathrm{ad}^{0}\rho),
\]
then shifts by $[1]$ by the usual deformation-stack convention.
\end{proof}

\section{Immediate Repairs Installed}
\label{v4-arith:fle-ledger:repairs}

\begin{theorem}[repair ledger]
\label{v4-arith:fle-ledger:thm:repairs}
\ClaimStatusProvedHere. The arithmetic branch is henceforth read
with the following corrections:
\begin{enumerate}
\item The synthesis has six conditional windows on
$\mathcal{V}^{\mathrm{prim}}$, not six proved equivalent
presentations.
\item The global spectral target is
$\mathcal{X}^{\mathrm{Sel}}_{\bar\rho,\delta,\mathcal{L}}$, not a
product or filtered colimit of local Emerton--Gee stacks.
\item Fargues--Scholze supplies local geometrisation, spectral
action, and Hecke functoriality; it does not prove F1.a.
\item F2.c, the restricted-tensor $E_{2}$ transport lemma, is
separate from the mixed-characteristic Iwahori $E_{2}$ local input
F2.b, which remains theorem-waiting.
\item The arithmetic BZN target is full Weil/$(\varphi,\Gamma)$.
$B\widehat{\mathbb{Z}}$ is only an inertia shadow.
\item Kim arithmetic Chern--Simons supplies classical CS torsors and
action lines. No boundary VOA or quantum boundary category is
constructed here.
\item $\mathcal{V}^{\mathrm{prim}}$ has no global VOA/OPE
structure. Chiral OPE language is confined to the archimedean
factor; finite places carry locally constant/Hecke data.
\item Yangian/RTT language is falsified as a Hopf-algebraic claim.
Only the commutative Bethe/Hecke-algebra heuristic survives.
\item The global arithmetic branch is conditional/theorem-waiting
except for explicitly local, formal, or representation-level
statements marked ProvedHere.
\end{enumerate}
\end{theorem}

\section{Theorem-Waiting Equivalences}
\label{v4-arith:fle-ledger:obligations}

\begin{conjecture}[F1.a, local categorical comparison]
\label{v4-arith:fle-ledger:conj:F1a}
\ClaimStatusConjectured. For every prime $p$ there should be a
local categorical comparison
\begin{equation}
D^{\mathrm{sph\text{-}fin}}\bigl([GL_{2}(\mathbb{Q}_{p})]\bigr)
\simeq
\mathrm{IndCoh}_{\mathrm{Nilp}}\bigl(\mathcal{X}^{\mathrm{EG},p}_{GL_{2}}\bigr),
\end{equation}
compatible with Hecke functors, central-character families, and the
Fargues--Scholze spectral action.
\end{conjecture}

\begin{remark}[current status of F1.a]
Known inputs do not prove the conjecture. Fargues--Scholze gives
local geometrisation and spectral action. Emerton--Gee gives local
$(\varphi,\Gamma)$ stacks. Dotto--Emerton--Gee gives a fully faithful
fixed-central-character functor for $GL_{2}(\mathbb{Q}_{p})$ in the
$p\geq 5$ range; it is not essential surjectivity and it does not
cover the full $p=2,3$ or central-character-family statement.
\end{remark}

\begin{conjecture}[adelic categorical descent]
\label{v4-arith:fle-ledger:conj:adelic-descent}
\ClaimStatusConjectured. The local categories in F1.a glue to the
global automorphic category
$D([GL_{2}])^{\mathrm{sph\text{-}fin}}$, with compatibility for
Whittaker models, Bernstein centres, the archimedean
$(\mathfrak{g},K)$ factor, and Eisenstein/continuous spectrum.
Representation-level restricted tensor products are classical; this
categorical descent is not proved here.
\end{conjecture}

\begin{conjecture}[mixed-characteristic Iwahori $E_{2}$]
\label{v4-arith:fle-ledger:conj:F2b}
\ClaimStatusConjectured. The mixed-characteristic local Iwahori
Hecke category $\mathrm{Hk}^{\mathrm{Iw}}_{GL_{2},p}$ carries an
$E_{2}$-monoidal structure compatible with diamond BD fusion and
with the Iwahori stratification. Current sources give spherical
fusion and $E_{1}$ convolution; the Iwahori $E_{2}$ enhancement is
the missing local input.
\end{conjecture}

\begin{conjecture}[full-target arithmetic BZN]
\label{v4-arith:fle-ledger:conj:full-bzn}
\ClaimStatusConjectured. For the Selmer-derived full
Weil/$(\varphi,\Gamma)$ target there should be an arithmetic
Ben-Zvi--Nadler comparison
\begin{equation}
\mathrm{HH}_{\bullet}\bigl(\mathrm{Hk}^{\mathrm{Iw}}_{GL_{2},\mathrm{glob}}\bigr)
\simeq
\mathcal{O}\bigl(\mathrm{LocSys}^{W,\varphi\Gamma}_{GL_{2}}\bigr)^{\mathrm{flat}},
\end{equation}
including formal-Noetherian Hochschild continuity and comparison of
the Arinkin--Gaitsgory nilpotent flat locus with the
Fargues--Scholze/Emerton--Gee flat locus.
\end{conjecture}

\begin{conjecture}[categorical arithmetic Whittaker theory]
\label{v4-arith:fle-ledger:conj:whittaker}
\ClaimStatusConjectured. The arithmetic Whittaker factorisation
should lift from representation-level Whittaker models to a
categorical equivalence compatible with the local comparisons,
adelic descent, and the Selmer-derived spectral stack. The current
evidence is representation-level Tate/Jacquet--Langlands
factorisation only.
\end{conjecture}

\begin{conjecture}[quantum ACS boundary category]
\label{v4-arith:fle-ledger:conj:acs-boundary}
\ClaimStatusConjectured. If arithmetic Chern--Simons is retained as
more than motivation, one must construct a quantum boundary category
$\partial_{q}\mathrm{ACS}$ and compare it with the Hochschild-trace
candidate. Kim's theory gives the classical arithmetic CS
torsor/action line; it does not construct this boundary category.
\end{conjecture}
