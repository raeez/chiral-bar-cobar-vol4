\chapter{Adelic Factorization Axioms}
\label{v4-arith:costello-gwilliam}

\emph{In the voice of Costello + Gwilliam (\emph{Factorization
Algebras in Quantum Field Theory}, CUP Vol.~I~2017, Vol.~II~2021).
We test the Beilinson--Drinfeld / Costello--Gwilliam factorization
axioms on $\overline{C}=\overline{\mathrm{Spec}\,\mathbb{Z}}$ and
determine the minimal adelic structure $\mathcal{V}^{\mathrm{prim}}$
satisfies.}

\section{The Ran Space of $\overline{\mathrm{Spec}\,\mathbb{Z}}$}
\label{v4-arith:cg:ran-space}

\begin{observation}[naive Ran is terminal]
\label{v4-arith:obs:ran-naive-terminal}
BD~\S3.4.1 defines $\mathrm{Ran}(X)$ for $X$ a smooth curve over a
field $k$ as the prestack whose $S$-points for a $k$-test scheme $S$
are non-empty finite subsets of $X(S)$. For $X=\mathrm{Spec}\,\mathbb{Z}$,
the functor of points is
\begin{equation*}
S\;\mapsto\;\mathrm{Spec}\,\mathbb{Z}(S)\;=\;\mathrm{Hom}_{\mathrm{Ring}}(\mathbb{Z},\Gamma(S,\mathcal{O}_{S}))\;=\;\{*\}
\end{equation*}
(singleton). The naive Ran space of $\mathrm{Spec}\,\mathbb{Z}$ is
therefore the terminal prestack; a single point. Primes are not
sections; they are closed points, i.e., morphisms
$\mathrm{Spec}\,\mathbb{F}_{p}\to\mathrm{Spec}\,\mathbb{Z}$ going the
wrong way for the functor-of-points definition.
\end{observation}

\begin{definition}[enriched Ran on Arakelov curve]
\label{v4-arith:def:ran-Arakelov}
On the Arakelov-compactified curve
$\overline{C}=\overline{\mathrm{Spec}\,\mathbb{Z}}$, redefine the Ran
space combinatorially:
\begin{equation}
\mathrm{Ran}(\overline{C})(R)\;=\;\{\text{non-empty finite multisets }\Sigma\subset|\overline{C}|\}
\end{equation}
for $R=\mathbb{Q}$-points (or more generally a $\mathbb{Q}$-algebra).
Topology: ind-set $\varinjlim_{n}\mathrm{Sym}^{n}(\mathbb{P}\sqcup\{\infty\})$.
This is a discrete combinatorial object; no scheme structure,
no tangent space.
\end{definition}

\begin{remark}
$\mathrm{Ran}(\overline{C})$ is discrete. BD factorization axioms
applied to a discrete Ran space become set-theoretic identities
carrying no infinitesimal content. The OPE singularity structure
at the diagonal is the content of BD factorization; on a discrete
Ran space, the diagonal has no infinitesimal neighbourhood and
the content vanishes.
\end{remark}

\section{The Factorization Isomorphism Is Trivial}
\label{v4-arith:cg:factorization-iso-trivial}

\begin{observation}
\label{v4-arith:obs:factorization-iso-trivial}
BD~\S3.4.4: a factorization algebra $A$ on $X$ assigns to each
$n\geq 1$ a $\mathcal{D}$-module $A^{(n)}$ on $X^{n}$ with
isomorphisms $j_{U}^{*}A^{(n)}\xrightarrow{\sim}(\boxtimes_{i}A^{(n_{i})})|_{U}$
over $U=X^{n}\setminus\Delta$. On
$\mathrm{Ran}(\overline{C})$, for distinct primes
$p\neq q$, the fibre
$\mathcal{V}^{\mathrm{prim}}|_{\{p,q\}}=\mathcal{H}^{(p)}\otimes\mathcal{H}^{(q)}$
by definition; a tautology.
\end{observation}

\begin{corollary}[no nontrivial factorization structure]
\label{v4-arith:cor:no-nontrivial-factorization}
\ClaimStatusProvedHere. The factorization iso over
$\overline{C}^{n}\setminus\Delta$ is the tautology ``direct
product of stalks is the product of stalks'', which carries no
information beyond the collection of stalks. The BD factorization
axiom is not satisfied non-trivially.
\end{corollary}

\section{The Diagonal and $\mathbb{Z}_{p}\,\widehat{\otimes}\,\mathbb{Z}_{p}$}
\label{v4-arith:cg:diagonal-Zp-tensor}

\begin{observation}
\label{v4-arith:obs:Zp-tensor-collapses}
The arithmetic diagonal at a finite place collapses:
\begin{equation}
\mathbb{Z}_{p}\,\widehat{\otimes}_{\mathbb{Z}}\,\mathbb{Z}_{p}\;\cong\;\mathbb{Z}_{p},
\end{equation}
because $\mathbb{Z}_{p}$ is $p$-adically complete and the diagonal
is an isomorphism after completion. The ``diagonal neighbourhood''
at a finite place is trivial: $\mathrm{Spec}(\mathbb{Z}_{p}\,\widehat{\otimes}\,\mathbb{Z}_{p})=\mathrm{Spec}\,\mathbb{Z}_{p}$.
\end{observation}

\begin{remark}
Fontaine-style crystalline structure (prismatic site of
Bhatt--Scholze arXiv:1905.08229) could provide a non-trivial
thickening of the $p$-adic diagonal, at the cost of Frobenius
crystalline data. Without that upgrade, no fusion map exists at
finite places.
\end{remark}

\begin{theorem}[archimedean vs finite-place asymmetry]
\label{v4-arith:thm:arch-vs-finite-asymmetry}
\ClaimStatusProvedHere. At the archimedean place, the diagonal
$\mathbb{C}\hookrightarrow\mathbb{C}^{2}$ is a smooth submanifold
with full Taylor expansion; BD factorization is ordinary
Beilinson--Drinfeld on a complex curve. At each finite place, the
diagonal has no thickening; BD factorization axiom is either
trivial or requires Fontaine prismatic data absent in the naive
construction.
\end{theorem}

\section{The Locally Constant ($E_{0}$) Factorization at Each Finite Place}
\label{v4-arith:cg:E0-finite-places}

\begin{proposition}[Francis arXiv:1206.5522~\S2.3]
\label{v4-arith:prop:locally-constant-discrete}
\ClaimStatusProvedElsewhere. A locally constant factorization
algebra on a topological manifold $M$ factors through the homotopy
type of $M$, equivalently is an $E_{n}$-algebra for $n=\dim M$.
On a totally disconnected space like $\mathrm{Spf}\,\mathbb{Z}_{p}$,
locally constant factorization degenerates to an $E_{0}$-algebra
(a pointed vector space).
\end{proposition}

\begin{definition}[minimal adelic factorization structure]
\label{v4-arith:def:minimal-adelic-factorization}
The minimal Costello--Gwilliam-compatible adelic structure on
$\mathcal{V}^{\mathrm{prim}}$ is:
\begin{itemize}
\item At $\infty$: genuine holomorphic factorization algebra on
$\mathbb{C}_{\infty}$ in the CG sense (Vol.~I~Def.~3.1.4), with
Heisenberg OPE $j(z)j(w)\sim k_{\infty}/(z-w)^{2}$, $k_{\infty}=1$.
\item At each finite $p$: locally constant prestack on
$\mathrm{Spf}\,\mathbb{Z}_{p}$ with constant stalk $\mathcal{H}^{(p)}$
and Adams endomorphism $\psi_{p}$; this is an $E_{0}$-algebra.
\item Gluing data: restricted tensor product over spherical vectors
$|0\rangle_{p}$ via Tate's thesis, giving an automorphic
representation of $GL_{2}(\mathbb{A}_{\mathbb{Q}})$.
\end{itemize}
\end{definition}

\section{The Holomorphic-Topological Structure}
\label{v4-arith:cg:HT-structure}

\begin{remark}[Costello HT factorization, arXiv:1303.2632; Costello 2020]
Costello's holomorphic-topological factorization algebras live on
$\mathbb{C}^{n_{H}}\times\mathbb{R}^{n_{T}}$ with $n_{H}$ holomorphic
and $n_{T}$ topological directions. The axioms blend holomorphic
OPE (on the $\mathbb{C}^{n_{H}}$ factor) and locally constant
($E_{n_{T}}$-algebra structure on the $\mathbb{R}^{n_{T}}$ factor).
\end{remark}

\begin{theorem}[adelic HT structure]
\label{v4-arith:thm:adelic-HT}
\ClaimStatusConjectured. On $\overline{C}$ the natural HT
structure is
\begin{equation*}
\underbrace{\mathbb{C}_{\infty}}_{\text{holomorphic}}
\;\times\;\underbrace{\mathrm{Spf}\,\widehat{\mathbb{Z}}}_{\text{adelic / }E_{0}\text{ at each place}},
\end{equation*}
with the profinite completion $\widehat{\mathbb{Z}}=\prod_{p}\mathbb{Z}_{p}$
totally disconnected. Locally constant factorization on
$\mathrm{Spf}\,\widehat{\mathbb{Z}}$ decomposes as a product of
$E_{0}$-stalks. The non-trivial chiral content concentrates at
$\infty$.
\end{theorem}

\begin{definition}[relative factorization with adelic base]
\label{v4-arith:def:relative-adelic-factorization}
$\mathcal{V}^{\mathrm{prim}}$ is a holomorphic factorization algebra
on $\mathbb{C}_{\infty}$ \emph{over the base}
$\mathrm{Spec}\,\widehat{\mathbb{Z}}$, with the base acting by
Adams operations $\{\psi_{p}\}_{p}$ and Hecke operators
$\{T_{p}\}_{p}$:
\begin{equation}
\mathcal{V}^{\mathrm{prim}}\;\in\;\mathrm{FactAlg}_{\mathbb{C}_{\infty}}\bigl(\mathrm{Mod}_{\widehat{\mathbb{Z}},\psi}\bigr),
\end{equation}
where $\mathrm{Mod}_{\widehat{\mathbb{Z}},\psi}$ is the category of
$\widehat{\mathbb{Z}}$-modules with compatible commuting
Adams/Hecke operations $\psi_{p}$ for each prime.
\end{definition}

\section{Five Axioms (F1)--(F5) for the Minimal Adelic Factorization}
\label{v4-arith:cg:five-axioms}

\begin{definition}[Costello--Gwilliam arithmetic factorization]
\label{v4-arith:def:CG-arithmetic-factorization}
$\mathcal{V}^{\mathrm{prim}}$ satisfies:

\paragraph{(F1) Archimedean factorization.} Holomorphic
factorization algebra on $\mathbb{C}_{\infty}$ in the sense of
CG~Vol.~I~Def.~3.1.4, with OPE
$j(z)j(w)\sim k_{\infty}/(z-w)^{2}$, $k_{\infty}=1$.

\paragraph{(F2) Coefficient category.} Values in
$\mathrm{Mod}_{\widehat{\mathbb{Z}},\psi}$:
$\widehat{\mathbb{Z}}$-modules with commuting Adams operations
$\psi_{p}\colon\mathcal{M}\to\mathcal{M}$ satisfying
$\psi_{p}\psi_{q}=\psi_{q}\psi_{p}$ and $\psi_{p}|_{\mathbb{Z}_{p}}=\mathrm{id}$.

\paragraph{(F3) Hecke-eigen decomposition.} Global sections
decompose as a restricted tensor product
\begin{equation*}
\mathcal{V}^{\mathrm{prim}}(\mathbb{C}_{\infty})\;=\;\bigotimes_{p}^{\prime}\mathcal{H}^{(p)}
\end{equation*}
with spherical vectors $|0\rangle_{p}$, compatibly with Adams /
Hecke operators.

\paragraph{(F4) Archimedean--finite compatibility.} The $p$-adic
valuation of the Heisenberg level $k_{\infty}$ is determined by
the Hecke eigenvalue $a_{p}$ via $v_{p}(k_{\infty})=f(a_{p})$ for
an explicit function $f$ fixed by Hartnoll--Yang
(arXiv:2502.02661, Thm~1.1 or its equivalent).

\paragraph{(F5) No $p$-adic OPE.} At each finite place $p$, the
local factor $\mathcal{H}^{(p)}$ satisfies \emph{no singular OPE};
the only non-trivial operation is the Adams endomorphism
$\psi_{p}$. The ``fusion''
$\mathcal{H}^{(p)}\otimes\mathcal{H}^{(p)}\to\mathcal{H}^{(p)}$ is
the commutative multiplication in the Fock module, not a BD
factorization-algebra fusion.
\end{definition}

\begin{theorem}[Costello--Gwilliam's verdict]
\label{v4-arith:thm:CG-verdict}
\ClaimStatusProvedHere. The minimal adelic factorization structure
compatible with the naive data on $\mathcal{V}^{\mathrm{prim}}$ is
(F1)--(F5). This is a factorization algebra on the archimedean
place with adelic-local-system coefficients, \emph{not} a BD
factorization algebra on $\overline{C}$ or $\mathrm{Spec}\,\mathbb{Z}$.
\end{theorem}

\section{What No BV-Quantization Provides the Finite-Place Structure}
\label{v4-arith:cg:BV-quantization-fails}

\begin{remark}[BV quantization requires action functional]
CG~Vol.~I~\S4 constructs a factorization algebra of observables as
a BV-quantization of a classical field theory $(\mathcal{E},S)$.
For the free boson on $C/\mathbb{C}$:
$\mathcal{E}=\Gamma(C,\Omega^{1}_{C})[-1]$, action
$S[\phi]=\int_{C}\phi\wedge\bar{\partial}\phi$, quantum
factorization algebra = Heisenberg VA
(CG~Vol.~I~Thm~4.6.1).
\end{remark}

\begin{observation}[no classical arithmetic field theory]
\label{v4-arith:obs:no-classical-arith-FT}
$\overline{\mathrm{Spec}\,\mathbb{Z}}$ has no sheaf of Kähler
differentials $\Omega^{1}_{\mathbb{Z}/\mathbb{Z}}$ in the ordinary
sense (the sheaf is zero). Arakelov geometry replaces it with a
metrised line bundle; but the ``classical free field at place $p$''
would require a section of $\Omega^{1}_{\overline{C}/\mathbb{F}_{1}}$
restricted to a neighbourhood of $p$; $\mathbb{F}_{1}$-geometry,
unconstructed.
\end{observation}

\begin{remark}
The Borger $\Lambda$-ring formalism supplies Adams operations
$\psi_{p}$ as natural endomorphisms, but $\psi_{p}$ are
endomorphisms, \emph{not a field theory}. No action functional,
no BV bracket, no Poisson structure. The Heisenberg $\mathcal{H}^{(p)}$
at finite $p$ is not a CG-quantization.
\end{remark}

\begin{theorem}[classical factorization at finite places]
\label{v4-arith:thm:classical-finite-places}
\ClaimStatusProvedHere. At a finite place $p$, $\mathcal{H}^{(p)}$
is a classical (commutative) factorization algebra on
$\mathrm{Spf}\,\mathbb{Z}_{p}$ equipped with a Frobenius-lift /
Adams endomorphism $\psi_{p}$. There is no $\hbar$ deformation
quantization at finite $p$ because there is no classical action
functional to quantize.
\end{theorem}

\section{Summary of This Chapter}
\label{v4-arith:cg:summary}

\begin{enumerate}
\item $\mathrm{Ran}(\overline{C})$ as a prestack in the BD sense
is terminal (\Cref{v4-arith:obs:ran-naive-terminal}); enriched
combinatorial Ran is discrete (\Cref{v4-arith:def:ran-Arakelov}).
\item The factorization iso over the complement of the diagonal is
a tautology; no non-trivial factorization content
(\Cref{v4-arith:cor:no-nontrivial-factorization}).
\item The $p$-adic diagonal collapses; no thickening, no fusion
at finite places
(\Cref{v4-arith:obs:Zp-tensor-collapses},
\Cref{v4-arith:thm:arch-vs-finite-asymmetry}).
\item At finite places: $E_{0}$-algebra
(\Cref{v4-arith:prop:locally-constant-discrete}); at $\infty$:
genuine BD factorization algebra.
\item The minimal structure is a factorization algebra on
$\mathbb{C}_{\infty}$ with $\widehat{\mathbb{Z}}$-module
coefficients and commuting Adams operations
(\Cref{v4-arith:def:relative-adelic-factorization},
\Cref{v4-arith:def:CG-arithmetic-factorization}).
\item No BV-quantization of an arithmetic classical field theory
supplies the finite-place structure
(\Cref{v4-arith:thm:classical-finite-places}).
\end{enumerate}

$\mathcal{V}^{\mathrm{prim}}$ is a factorization algebra on a curve
with an adelic local system decoration, not an $E_{1}$-chiral
algebra over $\mathrm{Spec}\,\mathbb{Z}$. The honest theorem of
this chapter is (F1)--(F5); stronger claims should be retracted.
