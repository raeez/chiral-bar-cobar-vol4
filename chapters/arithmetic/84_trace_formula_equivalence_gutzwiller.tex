\chapter{Trace Formula, Ordered Bar, and the Statistical Upside}
\label{v4-arith:tfe-gutzwiller}

\emph{This chapter fixes the status of the trace-formula discussion.
The sought trace formula is a distributional equality between a
zero/eigenvalue sum and a geometric sum over prime powers, with
archimedean and polar corrections. The ordered-incidence bar proves
the Euler-product skeleton; it does not prove the trace formula. A
3D arithmetic-gravity realization would add a possible Gutzwiller
route to zero statistics, but only after a semiclassical Hamiltonian
package and a Hilbert--Polya spectral-flow theorem are supplied.}

\section{The Trace Formula Target}
\label{v4-arith:tfe:weil}

\begin{definition}[Weil test-function class]
\label{v4-arith:tfe:defn:test-class}
\ClaimStatusDefinitional. Let \(\mathcal H\) be a standard Weil
test-function class: even Paley--Wiener/Schwartz functions whose
Fourier transforms are sufficiently regular for the explicit formula.
The exact analytic class depends on the chosen normalization of the
Weil--Guinand formula.
\end{definition}

\begin{theorem}[Weil--Guinand explicit formula]
\label{v4-arith:tfe:thm:weil}
\ClaimStatusProvedElsewhere. For \(h\in\mathcal H\), the completed
zeta function satisfies a distributional identity of the form
\begin{equation}
\label{v4-arith:tfe:eq:weil}
\sum_\rho h(\gamma_\rho)
=
W_\infty(h)+W_{\mathrm{pol}}(h)
-\sum_{p}\sum_{k\geq 1}
(\log p)p^{-k/2}
\bigl(\widehat h(k\log p)+\widehat h(-k\log p)\bigr),
\end{equation}
after the usual normalization \(s=\frac12+i\gamma\). Here
\(W_\infty\) is the gamma-factor contribution and
\(W_{\mathrm{pol}}\) is the contribution of the polar/trivial pieces.
\end{theorem}

\begin{equation}
\label{v4-arith:tfe:eq:W-arch}
W_{\mathrm{arch+pol}}(h):=W_\infty(h)+W_{\mathrm{pol}}(h).
\end{equation}

\begin{remark}[what the sum is over]
\label{v4-arith:tfe:rem:weil-sketch}
The geometric side is a sum over prime powers \((p,k)\). In the
Morishita dictionary this is a sum over iterates \(K_p^k\) of the
prime knot \(K_p\), with length \(k\log p\). In a Gutzwiller language
it is the periodic-orbit side. In the ordered-bar language it is the
logarithmic derivative of the Euler-product character. These are three
descriptions of the same prime-power index set; none supplies the
zero-side spectral operator by itself.
\end{remark}

\section{Connes' Trace Formula}
\label{v4-arith:tfe:connes}

\begin{theorem}[Connes--Meyer scaling trace]
\label{v4-arith:tfe:thm:connes-meyer}
\ClaimStatusProvedElsewhere. Connes' adele-class-space construction,
in Meyer's trace-formula formulation, realizes the Weil distribution
as a distributional scaling trace on the appropriate quotient/cokernel
space:
\begin{equation}
\label{v4-arith:tfe:eq:connes-trace}
\operatorname{Tr}_{\mathrm{Connes}}(h)
=
\sum_\rho h(\gamma_\rho)
\end{equation}
distributionally, with the same prime-power and archimedean terms as
\eqref{v4-arith:tfe:eq:weil}.
This is Connes--Meyer's distributional trace formula. It is not, by
itself, a Hilbert--P\'olya self-adjoint operator whose spectral measure
has determinant \(\xi_{\mathbb R}\).
\end{theorem}

\section{What the Ordered Bar Proves}
\label{v4-arith:tfe:3d-acs}

\begin{proposition}[ordered-bar Euler skeleton]
\label{v4-arith:tfe:lem:ord-bar}
\ClaimStatusProvedHereConditional \emph{(on the ordered-incidence
model of \Cref{v4-arith:3d-acs:thm:chi-hom-bar} and on the chosen
regularised character)}. The ordered-incidence bar has regularised
character
\begin{equation}
\label{v4-arith:tfe:eq:ord-bar-product}
\operatorname{ch}_{\mathrm{reg}}
H_\bullet(B^{\mathrm{ord}}_\bullet(\mathcal V^{\mathrm{prim}}))
=
\chi_\infty(s)\prod_p(1-p^{-s})^{-1},
\end{equation}
with the completed \(\xi(s)\) obtained only after the archimedean and
polar normalisations are inserted.
\end{proposition}

\begin{proof}
This is the Euler-character computation of the ordered-incidence model:
finite prime factors contribute the Bost--Connes Euler factors, while
the archimedean and polar pieces are part of the regularisation data.
The statement is a character calculation, not a Lefschetz trace formula.
\end{proof}

\begin{remark}[canonical quantisation boundary]
\label{v4-arith:tfe:rem:canq}
If the profinite ACS theory and boundary comparison are assumed, the
same regularised character can be read as a boundary partition function.
That reading is conditional on
\Cref{v4-arith:3d-acs:conj:qu-acs} and
\Cref{v4-arith:3d-acs:thm:bulk-boundary-Vprim}.
\end{remark}

\section{Trace-Formula Equivalence}
\label{v4-arith:tfe:equiv}

\begin{theorem}[conditional equivalence of trace functionals]
\label{v4-arith:tfe:thm:equiv}
\ClaimStatusConditional \emph{(on
\Cref{v4-arith:3d-acs:conj:qu-acs},
\Cref{v4-arith:3d-acs:cor:BDCG-ordered-comparison}, and the
regularised character equality
\Cref{v4-arith:3d-acs:cor:partition-bar})}. The ACS
argument-principle functional
\begin{equation}
\label{v4-arith:tfe:eq:z-pairing}
\mathcal Z_h:=
\frac{1}{2\pi i}\int_\Gamma
h(s)\frac{Z'(s)}{Z(s)}\,ds
\end{equation}
equals the Weil explicit-formula functional whenever
\(Z(s)=\xi(s)\) is assumed. Hence it agrees with the Connes--Meyer
scaling trace on \(\mathcal H\).
\end{theorem}

\begin{proof}
Under the stated assumptions \(Z=\xi\). Applying the argument
principle to \(\xi'/\xi\) gives the zero sum. Expanding
\(\xi'/\xi\) in the half-plane of convergence and continuing by the
standard explicit-formula contour shift gives the prime-power side
\eqref{v4-arith:tfe:eq:weil}. The equality is therefore inherited from
the classical explicit formula.
\end{proof}

\begin{corollary}[same information, different surface]
\label{v4-arith:tfe:cor:info-equiv}
\ClaimStatusConditional. Once \(Z=\xi\) is assumed, the ACS functional
and Connes' scaling trace encode the same distribution on Weil test
functions.
\end{corollary}

\section{No Free Zeros}
\label{v4-arith:tfe:no-free}

\begin{theorem}[no independent zero locator]
\label{v4-arith:tfe:thm:no-free-zeros}
\ClaimStatusProvedHere. The ordered-incidence bar does not determine
the individual zero ordinates \(\gamma_n\) independently of the
classical function \(\xi\).
\end{theorem}

\begin{proof}
The bar supplies the Euler-product character and, after conditional
regularisation, the scalar function \(\xi\). The locations of the zeros
are then the zero set of that scalar function. No additional
self-adjoint operator, spectral theorem, or positivity input is
constructed by the bar calculation. Therefore the bar reorganizes the
same data; it does not add an independent zero-location mechanism.
\end{proof}

\begin{remark}[Hilbert--Polya in view]
\label{v4-arith:tfe:rem:hilbert-polya}
A genuine Hilbert--Polya operator would change the status: its
self-adjointness would place the spectrum on the critical line. But
constructing such an operator is exactly the spectral-flow obstruction
isolated in \Cref{v4-arith:3d-acs:conj:spectral-flow}.
\end{remark}

\section{Arithmetic Gutzwiller}
\label{v4-arith:tfe:gutzwiller}

\begin{theorem}[classical Gutzwiller trace formula]
\label{v4-arith:tfe:thm:gutzwiller}
\ClaimStatusProvedElsewhere. For a quantum Hamiltonian with a suitable
chaotic semiclassical limit, the oscillatory density of states is
expressed asymptotically as a sum over isolated periodic classical
orbits:
\begin{equation}
\label{v4-arith:tfe:eq:gutzwiller}
d_{\mathrm{osc}}(E)\sim
\sum_\gamma A_\gamma
\exp(iET_\gamma-i\mu_\gamma\pi/2).
\end{equation}
\end{theorem}

\begin{theorem}[arithmetic Gutzwiller package]
\label{v4-arith:tfe:thm:arith-gutzwiller}
\ClaimStatusConditional \emph{(on a semiclassical ACS Hamiltonian,
an arithmetic phase space, and a prime-orbit correspondence)}. If the
ACS boundary Hamiltonian has a classical chaotic limit whose primitive
periodic orbits are indexed by primes \(p\) with primitive lengths
\(\log p\), and if the orbit amplitudes are the Weil amplitudes, then
the periodic-orbit expansion has the desired form
\begin{equation}
\label{v4-arith:tfe:eq:arith-gutz}
\sum_p\sum_{k\geq 1}
(\log p)p^{-k/2}e^{itk\log p}
\quad+\quad\text{complex conjugate and archimedean terms}.
\end{equation}
\end{theorem}

\begin{proof}
This is substitution of the assumed arithmetic orbit dictionary into
the classical Gutzwiller formula. The proof is formal because the
semiclassical ACS phase space, the chaotic flow, and the amplitude
calculation are all hypotheses.
\end{proof}

\section{Connes-Side Comparison}
\label{v4-arith:tfe:connes-obstruction}

\begin{theorem}[known Connes model is not a Gutzwiller system]
\label{v4-arith:tfe:thm:no-gutzwiller-connes}
\ClaimStatusProvedHere. The standard adele-class-space realization
used in the Connes--Meyer trace formula is a noncommutative scaling
trace construction, not a smooth symplectic Hamiltonian system of the
kind required by the classical Gutzwiller theorem.
\end{theorem}

\begin{proof}
The adele class space is modeled by a noncommutative quotient. The
scaling action gives an operator-algebraic trace, while Gutzwiller's
formula requires a smooth phase space, Hamiltonian flow, isolated
periodic orbits, and monodromy determinants. Those structures are not
part of the standard Connes--Meyer construction.
\end{proof}

\begin{remark}[partial geometrisations]
\label{v4-arith:tfe:rem:partial-geom}
This does not rule out other geometrizations of the adele class space.
It only records that the standard Connes--Meyer trace formula is not
itself a classical Gutzwiller mechanism.
\end{remark}

\section{Statistical Upside}
\label{v4-arith:tfe:bgs}

\begin{theorem}[Bohigas--Giannoni--Schmit principle]
\label{v4-arith:tfe:thm:bgs}
\ClaimStatusConjectured. Quantum systems whose classical limits are
chaotic are expected, after symmetry class is fixed, to exhibit random
matrix local spectral statistics.
\end{theorem}

\begin{equation}
\label{v4-arith:tfe:eq:gue}
R_2(u)=1-\left(\frac{\sin \pi u}{\pi u}\right)^2.
\end{equation}

\begin{theorem}[conditional statistical upside]
\label{v4-arith:tfe:thm:stat-upside}
\ClaimStatusConditional \emph{(on spectral flow, arithmetic
Gutzwiller, quantum chaos, and the relevant BGS universality theorem)}.
If the ACS Hamiltonian supplied by the spectral-flow package has a
chaotic semiclassical limit with prime periodic orbits, then the
local statistics of the zero ordinates are governed by the corresponding
random-matrix ensemble, in the expected zeta case by GUE statistics
\eqref{v4-arith:tfe:eq:gue}.
\end{theorem}

\begin{proof}
Spectral flow identifies the Hamiltonian spectrum with the zero
ordinates. Arithmetic Gutzwiller identifies the classical periodic
orbits with prime powers. The BGS principle then transfers chaotic
spectral statistics to the zero ordinates. Every substantive step is
one of the stated hypotheses.
\end{proof}

\begin{remark}[content beyond the explicit formula]
\label{v4-arith:tfe:rem:content-added}
The explicit formula fixes a distributional relation between zeros and
primes. It does not by itself prove Montgomery pair correlation or GUE
statistics. A genuine 3D Hamiltonian realization could add that
statistical mechanism, but only after the semiclassical and
spectral-flow hypotheses are solved.
\end{remark}

\section{Residual Conditionals}
\label{v4-arith:tfe:residual-conds}

The remaining obligations are:
\begin{enumerate}
\item profinite quantum ACS on \(\overline{\mathrm{Spec}\,\mathbb Z}\);
\item BD/CG comparison with the ordered-incidence model;
\item regularised character equality \(Z=\xi\);
\item spectral-flow/Hilbert--Polya operator;
\item full Weil--Guinand contour shift from the Euler skeleton;
\item arithmetic semiclassical phase space and Hamiltonian flow;
\item chaos and random-matrix universality for that flow.
\end{enumerate}

\section{Summary}
\label{v4-arith:tfe:summary}

\begin{enumerate}
\item The sought trace formula is a sum over prime powers
\((p,k)\), or equivalently iterated prime knots/closed orbits, plus
archimedean and polar terms.
\item The ordered bar proves the Euler skeleton, not the trace formula.
\item If \(Z=\xi\) is assumed, ACS and Connes evaluate the same Weil
distribution. This is trace-formula equivalence, not new zero
computation.
\item The possible new value of arithmetic gravity is statistical:
a Gutzwiller/BGS route to zero statistics.
\item That upside is conditional on the same missing spectral-flow
operator and on a new arithmetic quantum-chaos package.
\end{enumerate}
