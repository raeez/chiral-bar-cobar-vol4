\chapter{Arithmetic Chern--Simons Holography}
\label{v4-arith:gaiotto}

\emph{In the voice of Gaiotto. Class-$\mathcal{S}$, 4D
$\mathcal{N}=2$, Higgs-branch VOA (Beem--Lemos--Liendo--Peelaers--Rastelli--van~Rees
arXiv:1312.5344), Costello--Gaiotto arXiv:1804.06460,
Costello--Dimofte--Gaiotto arXiv:2005.00083. We ask whether
$\mathcal{V}^{\mathrm{prim}}$ can be obtained from a quantum
boundary category of an arithmetic 4D gauge theory or, failing
that, of an arithmetic 3D Chern--Simons theory. Kim's construction
will give only the classical arithmetic CS torsor/action line; the
finite-level quantum boundary category is constructed later in
Chapter~\ref{v4-arith:3d-acs-E1bar}, while the profinite
\(\mathcal V^{\mathrm{prim}}\) comparison remains an open input here.}

\section{Class $\mathcal{S}$ Fails Arithmetically}
\label{v4-arith:ga:class-S-fails}

\begin{theorem}[Gaiotto, arXiv:0904.2715]
\label{v4-arith:thm:gaiotto-class-S}
\ClaimStatusProvedElsewhere. Class-$\mathcal{S}$ builds
4D $\mathcal{N}=2$ theories $T[\Sigma,\mathfrak{g}]$ by
compactifying the 6D $(2,0)$ theory $\mathfrak{X}[\mathfrak{g}]$ on
a punctured Riemann surface $\Sigma$.
\end{theorem}

\begin{observation}[three obstructions]
\label{v4-arith:obs:class-S-obstructions}
Arithmetic class-$\mathcal{S}$ on $\overline{\mathrm{Spec}\,\mathbb{Z}}\times\mathbb{R}^{4}$
fails:
\begin{enumerate}
\item 6D $(2,0)$ requires a smooth oriented $6$-manifold;
$\overline{\mathrm{Spec}\,\mathbb{Z}}\times\mathbb{R}^{4}$ is
topologically $5$-dimensional (arithmetic $1$-dim plus $4$-dim
real). Connes' spectral dimension of $\overline{\mathrm{Spec}\,\mathbb{Z}}$
is $1$, not $2$ (Connes--Consani arXiv:0809.3190).
\item The self-dual $2$-form $B$ on an M5-brane has no Arakelov
analogue (no star on $2$-forms on a $1$-d arithmetic curve, where
$\Omega^{2}=0$).
\item Hitchin-moduli Higgs branches on
$\overline{\mathrm{Spec}\,\mathbb{Z}}$ are empty: the arithmetic
cotangent $K_{\overline{\mathrm{Spec}\,\mathbb{Z}}}$ has Arakelov
degree $\log(2\pi)>0$, violating Arakelov-Riemann-Roch.
\end{enumerate}
\end{observation}

\begin{corollary}
\label{v4-arith:cor:class-S-falsified}
\ClaimStatusProvedHere. Class-$\mathcal{S}$ as a generating
principle for arithmetic VAs is falsified.
\end{corollary}

\section{Coulomb and Higgs Branches Vanish Arithmetically}
\label{v4-arith:ga:CH-vanish}

\begin{theorem}[arithmetic branch vanishing]
\label{v4-arith:thm:arith-branch-vanishing}
\ClaimStatusProvedHere. On Arakelov $\overline{\mathrm{Spec}\,\mathbb{Z}}$,
$H^{0}(\overline{\mathrm{Spec}\,\mathbb{Z}},K^{k})=0$ for $k\geq 1$
by Minkowski's degree argument. The arithmetic Coulomb branch
$\mathcal{C}^{\mathrm{arith}}=\bigoplus_{k}H^{0}(\overline{C},K^{k})$
is zero. The arithmetic Higgs branch is empty. The theory is
topological.
\end{theorem}

\begin{remark}[signature of topological 4D $\to$ 3D reduction]
Vanishing Coulomb and Higgs branches with non-trivial Schur sector
is the signature of a topological 4D theory, which in the Schur
sector reduces to a 3D topological theory. The correct framework
shifts from 4D gauge theory to 3D arithmetic Chern--Simons.
\end{remark}

\section{Kim's Arithmetic Chern--Simons}
\label{v4-arith:ga:kim-ACS}

\begin{definition}[Kim 2015, arXiv:1510.05818]
\label{v4-arith:def:kim-ACS}
For a number field $F$ and finite Galois module $M$, arithmetic
Chern--Simons theory assigns to a Galois representation
$\rho\colon\pi_{1}^{\mathrm{\acute{e}t}}(\overline{\mathrm{Spec}\,\mathcal{O}_{F}})\to G$
the action
\begin{equation}
\mathrm{CS}_{F}(\rho)\;=\;\langle\rho^{*}c,[\overline{\mathrm{Spec}\,\mathcal{O}_{F}}]\rangle\;\in\;H^{3}(\overline{\mathrm{Spec}\,\mathcal{O}_{F}},\mu_{N})
\end{equation}
built from a class $c\in H^{3}(BG,\mu_{N})$. This is a genuine 3D
TQFT on arithmetic 3-manifolds (Mazur--Morishita arithmetic topology:
primes as knots, $\overline{\mathrm{Spec}\,\mathbb{Z}}$ as 3-manifold;
Morishita arXiv:0904.3399).
\end{definition}

\begin{remark}
Follow-up: Chung--Kim--Kim--Pappas--Park--Yoo arXiv:1706.03336.
Arithmetic CS is a rigorous 3D TQFT construction; it is the
missing arithmetic TQFT vertex that Witten's analysis
(\Cref{v4-arith:thm:witten-verdict}) identified as Deninger's
unconstructed dream.
\end{remark}

\section{$\mathcal{V}^{\mathrm{prim}}$ and the ACS Boundary Shadow}
\label{v4-arith:ga:V-prim-as-boundary}

\begin{theorem}[arithmetic CS boundary shadow]
\label{v4-arith:thm:arith-CS-VA-correspondence}
\ClaimStatusConjectured. Let
$\mathrm{ACS}[\overline{\mathrm{Spec}\,\mathbb{Z}},\mathbb{G}_{m},1]$
denote Kim arithmetic Chern--Simons at level~$1$ with gauge group
$\mathbb{G}_{m}$. Kim's theory supplies a classical arithmetic
CS torsor/action line. A quantum boundary category
$\partial_{q}\mathrm{ACS}$, if constructed, is expected to have
archimedean Heisenberg factorization shadow
\begin{equation}
\chi\bigl(\partial_{q}\mathrm{ACS}\bigr)(s)\;=\;\prod_{p}(1-p^{-s})^{-1},
\end{equation}
and to compare with the Hochschild-trace candidate
$\mathcal{V}^{\mathrm{prim}}$. This is not a constructed boundary
VOA at finite places.
\end{theorem}

\begin{proof}[Status]
Kim's construction is classical: it assigns CS actions/torsors to
Galois data. Witten's CS/WZW boundary theorem cannot be transported
verbatim to $\overline{\mathrm{Spec}\,\mathbb{Z}}$, since finite
places do not carry a complex boundary curve or a $p$-adic OPE.
Chapter~\ref{v4-arith:3d-acs-E1bar} constructs the finite-level
boundary category by transgression and integral transform. The
displayed Euler product remains a character-level expectation for the
renormalised profinite boundary comparison, not a proof of a boundary
vertex algebra.
\end{proof}

\begin{corollary}[character = $\zeta$]
\label{v4-arith:cor:character-is-zeta}
\ClaimStatusConjectured.
\begin{equation}
\chi(\partial_{q}\mathrm{ACS})(s)\;=\;\prod_{p}(1-p^{-s})^{-1}\;=\;\zeta(s),
\end{equation}
conditional on constructing the quantum boundary category and
identifying it with the Hochschild-trace candidate.
\end{corollary}

\section{Wilson Lines Are Labeled by Ramified Hecke Characters}
\label{v4-arith:ga:wilson-lines}

\begin{theorem}[Wilson-line labels as local Hecke data]
\label{v4-arith:thm:wilson-hecke}
\ClaimStatusConjectured. Wilson lines in $\mathrm{ACS}$ wrapped
around the ``knot'' corresponding to a prime $p$ (Mazur--Morishita
arithmetic topology) are $1$-dimensional defects in the arithmetic
$3$-fold, each \emph{labeled by} a local Hecke character $\chi_{p}$
of $\mathbb{Q}_{p}^{\times}$ (ramified or unramified). The label
is a closed point of the universal family of local characters,
and conjecturally a module over the quantum boundary category
specialized at $p$. Inserting a Wilson line with label $\chi_{p}$
should produce a boundary defect whose local datum at $p$ is exactly
the ramification datum of $\chi_{p}$.
Wilson-line labels are $0$-dimensional data; the Wilson line itself
is $1$-dimensional; the categorical boundary object carrying them
is not constructed here.
\end{theorem}

\begin{remark}[defect/puncture dictionary]
Class-$\mathcal{S}$ punctures $\leftrightarrow$ arithmetic
ramification lifts not to 6D codimension-$2$ defects but to Wilson
lines in arithmetic Chern--Simons. The defect-puncture
correspondence ``punctures $=$ defects'' becomes
``Wilson lines $=$ ramification'', preserving the structural
content while abandoning the 6D/4D framing.
\end{remark}

\section{BLLPRvR Surviving as Adelic Character}
\label{v4-arith:ga:BLLPRvR-adelic}

\begin{theorem}[arithmetic Schur index]
\label{v4-arith:thm:arithmetic-schur-index}
\ClaimStatusConjectured. The arithmetic Schur index is an adelic
product
\begin{equation}
\mathcal{I}^{\mathrm{arith}}(\{t_{p}\})\;=\;\mathrm{Tr}_{\mathcal{H}^{\mathrm{arith}}}\prod_{p}t_{p}^{\mathrm{Frob}_{p}\text{-weight}}.
\end{equation}
With $t_{p}=p^{-s}$ under the Hecke substitution,
$\mathcal{I}^{\mathrm{arith}}(s)=\prod_{p}(1-p^{-s})^{-1}=\zeta(s)$.
\end{theorem}

\begin{remark}
The BLLPRvR correspondence survives in holographic form: $\zeta(s)$
is a Schur index of the 3D arithmetic CS theory, not of a 4D gauge
theory. Class-$\mathcal{S}$ falsifies as a generating principle for
arithmetic VAs; holography with arithmetic CS succeeds.
\end{remark}

\section{Summary of This Chapter}
\label{v4-arith:ga:summary}

\begin{enumerate}
\item Class-$\mathcal{S}$ falsified: three 4D-gauge obstructions
(\Cref{v4-arith:cor:class-S-falsified}).
\item Coulomb/Higgs branches vanish arithmetically $\Rightarrow$
theory is topological $\Rightarrow$ reduces to 3D
(\Cref{v4-arith:thm:arith-branch-vanishing}).
\item Kim's arithmetic Chern--Simons (arXiv:1510.05818) is the
correct 3D TQFT
(\Cref{v4-arith:def:kim-ACS}).
\item The finite-level quantum boundary category is constructed in
Chapter~\ref{v4-arith:3d-acs-E1bar}; the renormalised profinite
boundary should compare with $\mathcal{V}^{\mathrm{prim}}$ and have
character $\zeta(s)$
(\Cref{v4-arith:thm:arith-CS-VA-correspondence}).
\item Wilson lines are $1$-dimensional defects labeled by ramified
Hecke characters (= expected modules of the quantum boundary
category specialized at $p$, $0$-dimensional data)
(\Cref{v4-arith:thm:wilson-hecke}).
\item BLLPRvR adelic character: $\zeta(s)$ is a Schur index of 3D
arithmetic CS (\Cref{v4-arith:thm:arithmetic-schur-index}).
\end{enumerate}

Kim arithmetic $\mathbb{G}_{m}$-Chern--Simons on
$\overline{\mathrm{Spec}\,\mathbb{Z}}$ at level~$1$ supplies the
classical 3D arithmetic TQFT input. Chapter~\ref{v4-arith:3d-acs-E1bar}
constructs the finite-level quantum boundary category; the
renormalised profinite comparison with $\mathcal{V}^{\mathrm{prim}}$
remains open.
