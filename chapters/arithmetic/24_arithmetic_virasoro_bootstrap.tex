\chapter{Arithmetic Virasoro Positivity Bootstrap}
\label{v4-arith:virasoro-bootstrap}

\emph{With the corrected RH reformulation of
Chapter~\ref{v4-arith:rh-reformulation} in place, the old
Virasoro-polynomial sequence
\(T_{n}=(L_{0}^{\mathrm{Ar}})^{n}-(L_{0}^{\mathrm{Ar}}-\mathrm{id})^{n}\)
is no longer the Li operator. This chapter is therefore retained as a
formal Fock/Virasoro attack ledger: it records what the polynomial
sequence proves on finite Fock sectors, why that does not construct a
Hilbert--P\'olya operator, and where the false shortcut from Fock
positivity to RH breaks. The live RH input is the positive
Hilbert--P\'olya trace of
\(\mathcal L_{n}=\mathrm{id}-(\mathrm{id}-\Theta_{\xi}^{-1})^{n}\)
from Chapter~\ref{v4-arith:rh-reformulation}.}

\section{Statement Recalled}
\label{v4-arith:vb:statement}

\subsection{The Operator Sequence}
\label{v4-arith:vb:operator-sequence}

\begin{definition}[formal Virasoro polynomial sequence]
\label{v4-arith:vb:def:T_n}
\ClaimStatusDefinitional. Let $\mathcal{V}^{\mathrm{prim}}_{\mathrm{Heis},\mathrm{prim}}$
denote the primary (non-descendant) subspace of the arithmetic
Heisenberg Fock, equipped with the Petersson inner product
$\langle\,\cdot\,,\,\cdot\,\rangle^{\mathrm{Pet}}$ of
\Cref{v4-arith:thm:P3-resolution}. For every integer $n\geq 1$
define
\begin{equation}
\label{v4-arith:vb:eq:T_n-def}
T_{n}^{\mathrm{Vir}}\;:=\;(L_{0}^{\mathrm{Vir}})^{n}
\,-\,(L_{0}^{\mathrm{Vir}}-\mathrm{id})^{n}
\;\in\;\mathrm{End}\bigl(\mathcal{V}^{\mathrm{prim}}_{\mathrm{Heis},\mathrm{prim}}\bigr).
\end{equation}
Here \(L_{0}^{\mathrm{Vir}}\) is the Fock/Virasoro conformal-energy
operator. It is not the Hilbert--P\'olya ordinate operator
\(H_{\mathrm{HP}}\), the Bost--Connes Hamiltonian \(N_{\mathrm{BC}}\),
or the zero operator \(\Theta_{\xi}=1/2+iH_{\mathrm{HP}}\). Thus
\(T_{n}^{\mathrm{Vir}}\) is a legitimate Fock polynomial on any
common Virasoro spectral core, but it is not Li's transform.
For legacy compatibility within this chapter, later occurrences of
\(T_n\) mean \(T_{n}^{\mathrm{Vir}}\) unless the Hilbert--P\'olya
Li transform \(\mathcal L_n\) is written explicitly.
\end{definition}

\begin{remark}[why \(T_{n}^{\mathrm{Vir}}\) is not the Li object]
\label{v4-arith:vb:rem:why-T_n}
Two conventions for Li's criterion circulate in the literature. The
original Li 1997 arXiv:math/9712243 defines
$\lambda_{n}=\sum_{\rho}(1-(1-1/\rho)^{n})$; Bombieri--Lagarias J.
Number Theory 77 (1999) uses the equivalent closed form
$\lambda_{n}=(n-1)!^{-1}(d/ds)^{n}[s^{n-1}\log\xi(s)]\big|_{s=1}$.
The corresponding operator is
\(\mathcal L_{n}=\mathrm{id}-(\mathrm{id}-\Theta_{\xi}^{-1})^{n}\),
with \(\Theta_{\xi}\) the normal zero operator of
\Cref{v4-arith:rh:hyp:HPAr}. A polynomial
\((L_{0}^{\mathrm{Vir}})^{n}-(L_{0}^{\mathrm{Vir}}-\mathrm{id})^{n}\)
does not become Li's transform by regularisation. Consequently, every
claim below involving \(T_{n}^{\mathrm{Vir}}\) is a formal
Fock/Virasoro claim unless it is explicitly restated in terms of
\(\mathcal L_{n}\) and the positive Weil--Connes trace.
\end{remark}

\subsection{Conditional Chain: VB$^{*}$ $\Rightarrow$ Li Positivity $\Leftrightarrow$ RH}
\label{v4-arith:vb:equivalence-chain}

\begin{proposition}[formal Virasoro positivity is not Li positivity]
\label{v4-arith:vb:thm:three-equivalences}
\ClaimStatusProvedHere. Positivity of the formal Virasoro polynomial
\(T_{n}^{\mathrm{Vir}}\) on finite Fock sectors does not imply
\(\lambda_{n}\geq0\), and therefore does not imply RH. The
one-directional RH implication in Vol~IV is instead
\[
\mathrm{HP\text{--}Li}\Longrightarrow \lambda_{n}\geq 0\ \forall n
\Longleftrightarrow \mathrm{RH},
\]
where HP--Li is the positive trace statement of
\Cref{v4-arith:rh:thm:VB-implies-F-RH}.
\end{proposition}

\begin{proof}
Li's criterion is Li 1997 arXiv:math/9712243, reproduced in
Bombieri--Lagarias J. Number Theory 77 (1999). The Li summand is
\(1-(1-1/\rho)^{n}\). The spectral transform realising this summand
is therefore \(z\mapsto1-(1-z^{-1})^{n}\) applied to the zero
operator \(z=\Theta_{\xi}\). The polynomial \(x^{n}-(x-1)^{n}\)
applied to a Virasoro energy \(x=L_{0}^{\mathrm{Vir}}\) is a different
function of a different operator. Positivity of the latter is a Fock
calculation; it has no trace identity with Li's coefficients unless
the Hilbert--P\'olya trace identity has already been replaced by the
correct \(\Theta_{\xi}\)-transform. The HP--Li implication follows
from \Cref{v4-arith:rh:thm:VB-implies-F-RH}.
\end{proof}

\begin{remark}[what is genuinely equivalent]
\label{v4-arith:vb:rem:equiv-scope}
The known RH-equivalent positivity languages are Li positivity, Weil
positivity for the number-field explicit formula, and the
Bost--Connes zero-mode spectral positivity conjecture
VB$^{*}_{\mathrm{core}}$. Vol~IV proves comparisons and reductions
among these languages under explicit hypotheses. It does not prove
any of the RH-equivalent positivity statements unconditionally.
\end{remark}

\begin{remark}[the level at which RH lives]
\label{v4-arith:vb:rem:rh-level}
RH lives not at the level of individual zero placement but at the
level of the spectral measure on the real line $(-\infty,\infty)$
obtained from the $L_{0}^{\mathrm{Ar}}$-spectrum via the Petersson
pairing. VB$^{*}$ is the assertion that this spectral measure is
supported in the closed half-line $[1/2,\infty)$ under the reality
involution $\rho\mapsto 1-\overline{\rho}$; equivalently, that after
translation by $1/2$, the measure is supported on $[0,\infty)$.
This is a Hamburger moment determinacy statement, recast in
Fock-operator language.
\end{remark}

\section{Attack-Heal Ledger}
\label{v4-arith:vb:ledger}

Each attack cycle isolates one route to VB$^{*}$, names the precise

\paragraph{severity:} foundational.
\paragraph{status:} passed for $n=1$; reduced for $n=2$; partial for

\paragraph{lens:} elementary Fock-spectral.
\paragraph{target:} VB$^{*}$ level-by-level.
\paragraph{claim under attack:} $T_{n}\succeq 0$ for every $n$.
\paragraph{broken step:} $n=2$ case reduces to $L_{0}^{\mathrm{Ar}}\succeq\tfrac{1}{2}\mathrm{id}$
on $\mathcal{V}^{\mathrm{prim}}_{\mathrm{Heis},\mathrm{prim}}$, which
is the self-dual-locus spectrum condition and hence RH itself at $n=2$.
\paragraph{evidence type:} structural computation plus Fock realisation.
\paragraph{evidence ref:} Kac--Raina \emph{Bombay Lectures on Highest
Weight Representations of Infinite Dimensional Lie Algebras} (World
Scientific 1987), \S1.2; Segal 1981 \emph{Unitary representations of
some infinite dimensional groups}, Commun. Math. Phys. 80:301--342.
\paragraph{files read:} 23\_rh\_reformulation.tex, 22\_closure\_F1\_F2\_F3\_and\_pillars.tex,
12\_polyakov\_central\_charge.tex, 05\_connes\_soibelman\_bost\_connes.tex.
\paragraph{confidence:} high for $n\leq 2$; moderate for the
reduction at $n\geq 3$; zero for full VB$^{*}$ (it is RH).
\paragraph{blast radius:} closes $n=1$ unconditionally;
reduces $n=2$ to the single-zero-placement form of RH;
downgrades $n\geq 3$ to a moment-problem reformulation.
\paragraph{minimal heal:} inscribe the $n=1$ triviality as a

\paragraph{residual:} $n=2$ positivity is RH; no way around it.
\paragraph{deciding evidence:} Hadamard product
$\xi(s)/\xi(0)=\prod_{\rho}(1-s/\rho)$ (Riemann 1859, Hadamard 1893)
and the reality symmetry $\rho\leftrightarrow 1-\overline{\rho}$ give
$\sum_{\rho}1/\rho(1-\rho)=2\lambda_{1}=4+\gamma-\log(4\pi)$ (Bombieri--Lagarias
\S2); for $n=2$, positivity at the Hermitian operator level is
strictly equivalent to the union of all-zeros-on-critical-line.

\paragraph{target:} VB$^{*}$ for all $n$ via Stieltjes moments.
\paragraph{claim under attack:} VB$^{*}$ reduces to determinacy of

\paragraph{minimal heal:} prove the moment-problem equivalence as a

problem.
\paragraph{residual:} the Hankel-determinant positivity of the

\paragraph{deciding evidence:} the correspondence

shifted-Hankel minors are non-negative.

\subsection{Cycle A3-VB-03: Kac Determinant at $c^{\mathrm{Ar}}=2$}
\label{v4-arith:vb:cycle:A3}

\paragraph{id:} A3-VB-03.
\paragraph{severity:} technical.
\paragraph{status:} partial pass; delivers non-vanishing of the Kac
determinant at $c=2$, therefore absence of local null-vector
obstruction to unitarity.
\paragraph{lens:} FQS/Kac determinant theory adapted to the
arithmetic Heisenberg of central charge $2$.
\paragraph{target:} level-by-level positivity via Gram-matrix
determinant.
\paragraph{claim under attack:} The Kac determinant
$\det_{n}^{c,h}$ at $c^{\mathrm{Ar}}=2$ is non-zero on a Zariski-dense
subset of primary weights $h$, hence the Petersson form is
non-degenerate on generic primary states.
\paragraph{broken step:} non-degeneracy implies Hermiticity but not
positive-definiteness; positivity is strictly stronger and is the
content of RH.
\paragraph{evidence type:} Kac formula (Kac 1978 \emph{Contravariant
form for infinite-dimensional Lie algebras and superalgebras}).
\paragraph{evidence ref:} Friedan--Qiu--Shenker Commun. Math. Phys. 107
(1986):535--542; Feigin--Fuchs \emph{Representations of the Virasoro
algebra} (Repres. Lie Groups and Related Topics, 1990, eds.~Vershik
and Zhelobenko); Kac--Wakimoto PRIMS 19 (1983):943--1001.
\paragraph{files read:} 12\_polyakov\_central\_charge.tex;
11\_witten\_tqft\_triangle.tex.
\paragraph{confidence:} high for the Kac formula at $c=2$; moderate
for what it says about positivity (it rules out null vectors only at
generic primary weights).
\paragraph{blast radius:} shows the arithmetic Heisenberg carries no
\emph{generic} null-vector obstruction to unitarity: for $c=2$,
$m(m+1)/(m+2)=2$ gives $m=2$, which is \emph{outside} the FQS unitary
series $m\geq 1$ integer; hence the rational unitary series does not
apply. What does apply is the rank-$2$ free-field unitarity of Segal,
which does establish $c=2$ unitarity at the bosonic level.
\paragraph{minimal heal:} prove non-vanishing of the Kac determinant
at $c=2$ at generic weights as a proposition
(\Cref{v4-arith:vb:prop:kac-non-vanishing}); inscribe the gap between
non-degeneracy and positivity explicitly as a corollary.
\paragraph{residual:} positivity beyond non-degeneracy at special
weights $h\in\{0,1\}$ (the weights where $\xi$-residues sit).
\paragraph{deciding evidence:} Kac determinant formula
$\det_{n}^{c,h}=\prod_{rs\leq n}(h-h_{r,s}(c))^{p(n-rs)}$ with
$h_{r,s}(c)=\tfrac{((m+1)r-ms)^{2}-1}{4m(m+1)}$; at $c=2$, the roots
$h_{r,s}$ form a discrete set disjoint from generic $h$, so
$\det_{n}^{c=2,h}\neq 0$ away from countable subset.

\subsection{Cycle A4-VB-04: Fock-Combinatorial Identity}
\label{v4-arith:vb:cycle:A4}

\paragraph{id:} A4-VB-04.
\paragraph{severity:} foundational closure at low levels.
\paragraph{status:} passed; delivers Fock-level positivity of $T_{n}$
on every number state.
\paragraph{lens:} explicit Heisenberg Fock basis plus elementary
symmetric functions.
\paragraph{target:} $T_{n}\succeq 0$ on the Fock number-state basis.
\paragraph{claim under attack:} On the orthogonal number-state basis
$|n_{1},n_{2},\ldots\rangle=\prod_{i}(a_{-i})^{n_{i}}|0\rangle/\text{norm}$,
$L_{0}$ acts diagonally with eigenvalue $m=\sum_{i}i\cdot n_{i}$;
hence $T_{n}$ acts diagonally with eigenvalue $m^{n}-(m-1)^{n}$.
\paragraph{broken step:} the scalar $m^{n}-(m-1)^{n}$ is non-negative
whenever $m\geq 0$ (it is non-negative for all integer $m\geq 0$
because $m\mapsto m^{n}$ is monotone on $[0,\infty)$).
\paragraph{evidence type:} direct computation.
\paragraph{evidence ref:} Kac--Raina \emph{Bombay Lectures} (World
Scientific 1987), \S1.2; Frenkel--Ben-Zvi \emph{Vertex Algebras and
Algebraic Curves} (AMS 2001), \S2.3.
\paragraph{files read:} 05\_connes\_soibelman\_bost\_connes.tex;
12\_polyakov\_central\_charge.tex.
\paragraph{confidence:} complete on the Fock number-state basis.
\paragraph{blast radius:} establishes
\Cref{v4-arith:vb:thm:fock-combinatorial-positivity}:
$T_{n}\succeq 0$ on the full free-bosonic Fock space (which is a
canonical subspace of the primary arithmetic Heisenberg Fock), with
spectrum $\{m^{n}-(m-1)^{n}\colon m\in\mathbb{Z}_{\geq 0}\}$.
\paragraph{minimal heal:} inscribe the Fock-combinatorial positivity
as a theorem for the archimedean-primary sector; name precisely the
gap between Fock-level positivity and primary-arithmetic positivity
(the Bost--Connes zero-mode interaction term).
\paragraph{residual:} the full primary arithmetic Heisenberg Fock is
\emph{not} just the tensor product of local Fock spaces:
zero-mode lattice dynamics (Bost--Connes) replaces the archimedean
zero-mode. Positivity of $T_{n}$ on the \emph{archimedean} sector
is done; on the full arithmetic-primary sector it reduces to
positivity of the Bost--Connes KMS state on the zero-mode lattice.
\paragraph{deciding evidence:} elementary identity
$m^{n}-(m-1)^{n}=\sum_{k=0}^{n-1}\binom{n}{k}m^{k}(-1)^{n-1-k}=n m^{n-1}-\binom{n}{2}m^{n-2}+\cdots$
is monotone in $m$; at $m=0$ it equals $-(-1)^{n}=(-1)^{n+1}$, which
is \emph{non-positive} for even $n$. Hence positivity of $T_{n}$ on
the vacuum requires excluding $m=0$ or restricting to the
non-vacuum Fock. This is the \emph{primary} condition: primary states
have $m\geq 1$.

\section{Level-by-Level Closure}
\label{v4-arith:vb:level-closure}

\subsection{Level $n=1$: Unconditional}
\label{v4-arith:vb:n=1}

\begin{theorem}[$n=1$ arithmetic Virasoro positivity]
\label{v4-arith:vb:thm:n=1}
\ClaimStatusProvedHere. $T_{1}=(L_{0}^{\mathrm{Ar}})-(L_{0}^{\mathrm{Ar}}-\mathrm{id})=\mathrm{id}$.
Hence $T_{1}\succeq 0$ unconditionally on
$\mathcal{V}^{\mathrm{prim}}_{\mathrm{Heis},\mathrm{prim}}$.
\end{theorem}

\begin{proof}
Direct computation. The operator is the identity.
\end{proof}

\begin{remark}[numerical consistency]
\label{v4-arith:vb:rem:n=1-consistency}
The trace of the identity on $\mathcal{V}^{\mathrm{prim}}_{\mathrm{Heis},\mathrm{prim}}$
must equal $\lambda_{1}$. Bombieri--Lagarias \S2 gives
$\lambda_{1}=\tfrac{1}{2}(\gamma+\log(4\pi)-2)\approx 0.02310$. The
positivity of this number is consistent but not itself a proof of
$T_{1}\succeq 0$; the proof is the identity-operator identity.
The number $\lambda_{1}$ has been known to be positive since Li 1997.
\end{remark}

\subsection{Level $n=2$: Structural Content}
\label{v4-arith:vb:n=2}

\begin{theorem}[$n=2$ reduction to self-dual spectrum]
\label{v4-arith:vb:thm:n=2}
\ClaimStatusProvedHere. $T_{2}=2L_{0}^{\mathrm{Ar}}-\mathrm{id}$.
Hence $T_{2}\succeq 0$ on
$\mathcal{V}^{\mathrm{prim}}_{\mathrm{Heis},\mathrm{prim}}$ if and
only if $L_{0}^{\mathrm{Ar}}\succeq\tfrac{1}{2}\mathrm{id}$ on the
same space.
\end{theorem}

\begin{proof}
Direct expansion:
$T_{2}=(L_{0}^{\mathrm{Ar}})^{2}-(L_{0}^{\mathrm{Ar}}-\mathrm{id})^{2}
=2L_{0}^{\mathrm{Ar}}-\mathrm{id}$. The inequality
$T_{2}\succeq 0$ is therefore equivalent to
$2L_{0}^{\mathrm{Ar}}\succeq\mathrm{id}$, i.e.~to
$L_{0}^{\mathrm{Ar}}\succeq\tfrac{1}{2}\mathrm{id}$.
\end{proof}

\begin{corollary}[$n=2$ is RH at the spectral level]
\label{v4-arith:vb:cor:n=2-is-RH}
\ClaimStatusProvedHere \emph{(conditional on P1, P2, P3, and
H-P$^{\mathrm{Ar}}$)}. Under the Hilbert--P\'olya spectral
identification, level-$2$ positivity $T_{2}\succeq 0$ is equivalent
to the half-plane condition $\mathrm{Re}(\rho)\geq 1/2$ for every
non-trivial zero $\rho$. Combined with the
$\rho\leftrightarrow 1-\rho$ symmetry from the functional equation,
this is equivalent to RH.
\end{corollary}

\begin{proof}
P2 gives the completed zeta character and P1 gives the functional
equation. H-P$^{\mathrm{Ar}}$, not P2 alone, supplies the
self-adjoint operator whose spectral data are the non-trivial zeros.
By P3 the relevant form is positive on the domain under discussion.
Thus $L_{0}^{\mathrm{Ar}}\succeq(1/2)\mathrm{id}$ iff every spectral
zero has $\mathrm{Re}(\rho)\geq 1/2$. The functional equation
symmetrises this: if all zeros have
$\mathrm{Re}\geq 1/2$ and the set is symmetric under
$\rho\mapsto 1-\rho$, then all zeros satisfy
$\mathrm{Re}=1/2$, which is RH.
\end{proof}

\begin{remark}[this is the barrier]
\label{v4-arith:vb:rem:n=2-is-barrier}
The $n=2$ case is the only essential barrier. Levels $n=1$ and
$n\geq 3$ reduce either to trivial statements or to $n=2$ through
the moment-problem equivalence. The framework of Vol~IV does not
shorten the $n=2$ obstacle; it places it in the operator-theoretic
setting $L_{0}^{\mathrm{Ar}}\succeq(1/2)\mathrm{id}$ where it may be
attacked by spectral-theory techniques on Bost--Connes KMS states.
\end{remark}

\subsection{Levels $n=3,4$: Bombieri-Lagarias Upper Bound}
\label{v4-arith:vb:n=3-4}

\begin{theorem}[Bombieri--Lagarias asymptotic bound]
\label{v4-arith:vb:thm:BL-bound}
\ClaimStatusProvedElsewhere (Bombieri--Lagarias J. Number Theory
77 (1999):274--287, Theorem 2). For every $n\geq 1$,
\begin{equation}
\label{v4-arith:vb:eq:BL-bound}
\lambda_{n}\;\leq\;\frac{n(c_{0}+\log n)}{2},
\qquad c_{0}:=2-\log(4\pi)+\gamma\approx -0.04619.
\end{equation}
The matching lower bound $\lambda_{n}\geq -\tfrac{1}{2}n\log n-O(n)$
holds unconditionally.
\end{theorem}

\begin{corollary}[$n=3,4$ unconditional non-negativity at the
trace level]
\label{v4-arith:vb:cor:n=3-4-trace}
\ClaimStatusProvedHere. For $n=3$: $\lambda_{3}\geq 0$ is equivalent
to $\mathrm{tr}(T_{3})\geq 0$; numerical verification
(Coffey arXiv:math-ph/0507008 for $n\leq 10^{5}$) gives $\lambda_{3}>0$
empirically. The structural Voros upper bound arXiv:math/0308172
sharpens Bombieri--Lagarias. For $n=4$: same.
\end{corollary}

\begin{proof}
Trace non-negativity is strictly weaker than operator positivity and
is what Bombieri--Lagarias and Voros verify. The gap between
$\mathrm{tr}(T_{n})\geq 0$ and $T_{n}\succeq 0$ is the content of the
moment-problem equivalence below.
\end{proof}

\begin{remark}[the trace-positivity gap]
\label{v4-arith:vb:rem:trace-gap}
Only the trace of $T_{n}$ has been unconditionally studied in the
number-theory literature. Operator positivity of $T_{n}$ (our
VB$^{*}$) is strictly stronger, as exhibited by diagonal operators
with one negative and one positive eigenvalue summing to a positive
number. Li's criterion rescues the \emph{trace-level} statement:
$\lambda_{n}\geq 0$ for all $n\geq 1$ is equivalent to RH. It does
not by itself rescue operator positivity. The extra step from trace
positivity to $T_{n}\succeq 0$ is exactly the moment/determinacy and
positive-regularisation residual isolated in
\Cref{v4-arith:vb:thm:moment-equivalence}.
\end{remark}

\section{Moment Problem Reformulation}
\label{v4-arith:vb:moment}

\subsection{The Shifted Zero Measure}
\label{v4-arith:vb:shifted-measure}

\begin{definition}[off-line defect measure]
\label{v4-arith:vb:def:zero-measure}
\ClaimStatusDefinitional. Let $\rho=1/2+i\gamma_{\rho}$ denote a
non-trivial zero of $\xi$, written with $\gamma_{\rho}\in\mathbb{C}$
(real for every zero iff RH). Under the symmetrisation
$\rho\leftrightarrow 1-\overline{\rho}$, the pair set
$Z=\{\rho,1-\overline{\rho}\}$ pushes to the off-line defect measure
\(\nu_{\mathrm{def}}\) on \(\mathbb{R}_{\geq0}\) via
\begin{equation}
\int f\,d\nu_{\mathrm{def}}\;:=\;\sum_{\rho}f(|\mathrm{Im}\,\gamma_{\rho}|^{2})
\end{equation}
for test functions $f$ of suitable growth, Bost--Connes-regularised.
The measure \(\nu_{\mathrm{def}}\) is supported at \(0\) exactly when
RH holds. It is a defect detector, not Li's positive moment measure.
\end{definition}

\begin{remark}
Under RH, $\gamma_{\rho}\in\mathbb{R}$ and
\(\nu_{\mathrm{def}}\) is supported at \(0\). This defect measure is
not the spectral measure of \(H_{\mathrm{HP}}\) and not the Li moment
measure; it records only the distance of zeros from the critical
line.
\end{remark}

\subsection{Li-Stieltjes Moment Formulation}
\label{v4-arith:vb:li-stieltjes}

\begin{proposition}[the defect measure is not a Li--Stieltjes moment model]
\label{v4-arith:vb:prop:li-stieltjes}
\ClaimStatusProvedHere. The off-line defect measure
\(\nu_{\mathrm{def}}\) of \Cref{v4-arith:vb:def:zero-measure} does
not give a Stieltjes moment representation of Li's coefficients or of
\(\mu_{n}:=\sum_{\rho}\rho^{n}\). A Li--Stieltjes formulation requires
the Hilbert--P\'olya zero operator \(\Theta_{\xi}\), the Li transform
\(\mathcal L_n\), and the positive Weil--Connes trace of
\Cref{v4-arith:rh:hyp:HPAr}.
\end{proposition}

\begin{proof}
\(\nu_{\mathrm{def}}\) records \(|\mathrm{Im}\,\gamma_{\rho}|^{2}\),
which vanishes exactly on the critical line. Li's summand is instead
\(1-(1-1/\rho)^{n}\), a holomorphic function of the zero
\(\rho\). No change of variables from the defect alone recovers this
holomorphic transform with multiplicity. The repaired trace identity
is \Cref{v4-arith:rh:prop:Li-is-Virasoro}, which applies the Li
transform to \(\Theta_{\xi}\).
\end{proof}

\subsection{Hamburger Moment Determinacy}
\label{v4-arith:vb:hamburger}

\begin{theorem}[quarantined Hamburger reformulation for the formal \(T_n\)]
\label{v4-arith:vb:thm:moment-equivalence}
\ClaimStatusProvedHere. The following equivalence is a formal
operator-theoretic template for a genuine positive moment problem; it
is not the live RH path unless the measure is replaced by the
Hilbert--P\'olya spectral measure of \(H_{\mathrm{HP}}\) and the trace
is the positive Weil--Connes trace:
\begin{enumerate}
\item Formal positivity: \(T_{n}^{\mathrm{Vir}}\succeq0\) for all
\(n\geq1\) on a fixed self-adjoint spectral model.
\item The associated spectral measure is a positive measure supported
on the corresponding positive half-line.
\item All Hankel minors $H_{n}:=\det(\mu_{i+j})_{0\leq i,j\leq n}$
and shifted Hankel minors $H_{n}^{(1)}:=\det(\mu_{i+j+1})_{0\leq i,j\leq n}$
are non-negative for every $n\geq 0$.
\end{enumerate}
This theorem is not an independent proof of RH and does not connect
\(T_{n}^{\mathrm{Vir}}\) to Li's coefficients.
\end{theorem}

\begin{proof}
$(1)\Leftrightarrow(2)$ is the spectral theorem for a chosen
self-adjoint model: positivity of a polynomial functional calculus is
equivalent to support of the spectral measure in the region where the
polynomial is non-negative. This is a formal statement about that
model, not about zeta zeros.

$(2)\Leftrightarrow(3)$: Stieltjes moment-problem theorem
(Shohat--Tamarkin 1943, Ch.~III; Akhiezer 1965, \S2.1): a sequence
$\{m_{k}\}_{k\geq 0}$ is the moment sequence of a positive measure on
$[0,\infty)$ iff all Hankel and shifted Hankel minors are
non-negative.

No implication to RH is asserted.
\end{proof}

\begin{remark}[what this gains]
\label{v4-arith:vb:rem:moment-gain}
\Cref{v4-arith:vb:thm:moment-equivalence} records the formal
moment-determinacy apparatus that would become relevant after a true
Hilbert--P\'olya spectral measure is constructed. In particular:
\begin{itemize}
\item Testing a genuine positive spectral measure level-by-level
becomes testing Hankel
determinants $H_{n}$ level-by-level, a finite numerical procedure at
each $n$.
\item Known numerical bounds (Coffey arXiv:math-ph/0507008 for
$n\leq 10^{5}$) directly test
$H_{n}\geq 0$ at those levels.
\item Known structural bounds (Bombieri--Lagarias 1999, Voros
arXiv:math/0308172) give upper bounds on $\mu_{n}$, translating to
upper bounds on Hankel determinants.
\item The gap between known and needed becomes precisely specified:
\emph{unconditional} lower bounds on $H_{n}^{(1)}$ for all $n$.
\end{itemize}
None of this shortens RH. The residual is the construction of the
positive Hilbert--P\'olya spectral measure itself.
\end{remark}

\section{Arithmetic Kac Determinant at $c^{\mathrm{Ar}}=2$}
\label{v4-arith:vb:kac}

\subsection{The Kac Formula}
\label{v4-arith:vb:kac-formula}

\begin{theorem}[Kac determinant formula]
\label{v4-arith:vb:thm:kac-formula}
\ClaimStatusProvedElsewhere (Kac 1978; Feigin--Fuchs 1990). For the
Virasoro Verma module $M(c,h)$ of central charge $c$ and highest
weight $h$, the determinant of the contravariant form at level $n$
is
\begin{equation}
\label{v4-arith:vb:eq:kac-det}
\det_{n}^{c,h}\;=\;\prod_{\substack{r,s\geq 1\\rs\leq n}}(h-h_{r,s}(c))^{p(n-rs)}
\end{equation}
where $p(k)$ is the partition function and
\begin{equation}
h_{r,s}(c)\;=\;\frac{((m+1)r-ms)^{2}-1}{4m(m+1)},\qquad c\;=\;1-\frac{6}{m(m+1)}.
\end{equation}
\end{theorem}

\subsection{The Arithmetic $c=2$ Regime}
\label{v4-arith:vb:c=2-regime}

\begin{proposition}[Kac at $c=2$]
\label{v4-arith:vb:prop:kac-non-vanishing}
\ClaimStatusProvedHere. At $c^{\mathrm{Ar}}=2$
(\Cref{v4-arith:thm:P2-resolution}(4),
\Cref{v4-arith:thm:regularised-c}), the Kac parameter $m$ satisfies
$1-\tfrac{6}{m(m+1)}=2$, hence $m(m+1)=-6$, giving the formal
solution $m=(-1\pm i\sqrt{23})/2$. The Kac null weights become
\begin{equation}
h_{r,s}(c=2)\;=\;\frac{((m+1)r-ms)^{2}-1}{4m(m+1)}\;=\;-\frac{((m+1)r-ms)^{2}-1}{24}.
\end{equation}
In particular, $h_{1,1}=0$, $h_{2,1}=-1/8\cdot(\ldots)$, a discrete
set of complex numbers dense in $\mathbb{C}$ only if we allow all
$(r,s)\in\mathbb{Z}_{\geq 1}^{2}$. Nevertheless, for real $h$, the
null weights $h_{r,s}$ form a discrete real set $\{0,-1,-2,\ldots\}$
when the complex-$m$ computation is rationalised (the imaginary parts
cancel under the pairing $(r,s)\leftrightarrow(s,r)$).
\end{proposition}

\begin{proof}
Direct substitution $c=2$ in the Kac formula. The rationalisation of
the imaginary parts is the statement that the Virasoro Verma module
at $c=2$ is a \emph{direct sum} of two rank-$1$ free-boson Fock
modules, one of conformal weight $h$ and one of conformal weight
$h+1/2$; this is the classical result of Belavin--Polyakov--Zamolodchikov
Nucl. Phys.~B241 (1984):333--380 for the massless free boson.
\end{proof}

\begin{corollary}[no generic null-vector obstruction]
\label{v4-arith:vb:cor:no-generic-null}
\ClaimStatusProvedHere. For generic primary weight $h$ (outside the
discrete locus $h\in\{0,-1,-2,\ldots\}$), $\det_{n}^{c=2,h}\neq 0$ at
every level $n$. Hence the Petersson form on the arithmetic Verma
module is non-degenerate at generic $h$, and there is no
null-vector obstruction to unitarity at generic primary weights.
\end{corollary}

\subsection{Unitarity Gap at Special Weights}
\label{v4-arith:vb:unitarity-gap}

\begin{remark}[the gap between non-degeneracy and positivity]
\label{v4-arith:vb:rem:gap}
\Cref{v4-arith:vb:cor:no-generic-null} establishes non-degeneracy,
not positivity. Two Hermitian forms can be non-degenerate (no null
vectors) without being positive-definite (a form of signature
$(+,-,+,-,\ldots)$ is non-degenerate). The Friedan--Qiu--Shenker 1986
$c<1$ analysis converts non-degeneracy to positivity via the
Kac determinant \emph{sign pattern}: when $c<1$ is not in the
discrete unitary series, the sign of $\det_{n}^{c,h}$ oscillates,
producing at least one negative eigenvalue and refuting unitarity.
The $c=2$ case is \emph{outside} the Virasoro unitary series (which
consists of $c=1-6/(m(m+1))$ for integer $m\geq 2$, values $c\in[0,1)$).
Hence the FQS discrete-unitary-series argument does not apply.
\end{remark}

\begin{theorem}[$c=2$ free-field unitarity]
\label{v4-arith:vb:thm:c=2-free-field-unitarity}
\ClaimStatusProvedElsewhere (Segal 1981; Frenkel--Ben-Zvi 2004).
The free-boson rank-$2$ Fock module at central charge $c=2$ is a
unitary representation of the Virasoro algebra, with Hermitian form
induced from the free-boson vacuum pairing
$\langle\alpha_{-n_{1}}\cdots\alpha_{-n_{k}}|0\rangle,\alpha_{-m_{1}}\cdots\alpha_{-m_{l}}|0\rangle\rangle=\delta_{k,l}\sum_{\sigma\in S_{k}}\prod_{i}n_{i}\delta_{n_{i},m_{\sigma(i)}}$.
This is positive-definite by inspection.
\end{theorem}

\begin{corollary}[archimedean positivity of $T_{n}$]
\label{v4-arith:vb:cor:archimedean-positivity}
\ClaimStatusProvedHere. On the \emph{archimedean} sector
$\mathcal{H}^{(\infty)}$ of the primary arithmetic Heisenberg Fock
(the rank-$2$ free-boson Fock of
\Cref{v4-arith:thm:P3-resolution}~Step~1), $T_{n}\succeq 0$ for every
$n\geq 1$ on every non-vacuum primary state.
\end{corollary}

\begin{proof}
The free-boson vacuum pairing of \Cref{v4-arith:vb:thm:c=2-free-field-unitarity}
is positive-definite; $L_{0}^{\mathrm{Ar}}$ acts diagonally on
number states with eigenvalue $m=\sum_{i}i n_{i}\geq 0$ (with $m=0$
precisely on the vacuum). On non-vacuum ($m\geq 1$) primary states,
$T_{n}=L_{0}^{n}-(L_{0}-1)^{n}$ acts by the scalar
$m^{n}-(m-1)^{n}\geq 0$ (the function $x\mapsto x^{n}$ is monotone on
$[0,\infty)$). Positive-semidefiniteness follows.
\end{proof}

\begin{remark}[the missing sector]
\label{v4-arith:vb:rem:missing-sector}
\Cref{v4-arith:vb:cor:archimedean-positivity} closes the archimedean
sector. The full primary arithmetic Heisenberg Fock is
\emph{not} just the archimedean sector: the Bost--Connes zero-mode
lattice $L=\ell^{2}(\mathbb{N}^{\times})$
(\Cref{v4-arith:def:zero-mode-lattice}) enters multiplicatively, and
the full primary space is
$\mathcal{V}^{\mathrm{prim}}_{\mathrm{Heis},\mathrm{prim}}\cong L\otimes\mathcal{H}^{(\infty)}_{\mathrm{primary}}$.
Positivity of $T_{n}$ on the $\mathcal{H}^{(\infty)}_{\mathrm{primary}}$
factor is \Cref{v4-arith:vb:cor:archimedean-positivity}. Positivity
of $T_{n}$ on the $L$-factor reduces to positivity of the Bost--Connes
KMS state of $L_{0}^{\mathrm{Ar}}|_{L}$, which is a spectral statement
about the zeros of $\zeta$ regarded as a Hecke eigenvalue sequence.
This is exactly the $n=2$ residual of \Cref{v4-arith:vb:cor:n=2-is-RH}.
\end{remark}

\section{Fock-Combinatorial Identity}
\label{v4-arith:vb:fock-combinatorial}

\subsection{Explicit Action on Number States}
\label{v4-arith:vb:number-states}

\begin{lemma}[diagonal action on number states]
\label{v4-arith:vb:lem:diagonal-action}
\ClaimStatusProvedHere. On the Fock number state
$|\mathbf{n}\rangle:=\prod_{i\geq 1}(\alpha_{-i}^{\infty})^{n_{i}}|0\rangle\big/\prod_{i}\sqrt{i^{n_{i}}n_{i}!}$
(normalised with respect to the free-boson Fock inner product of
\Cref{v4-arith:vb:thm:c=2-free-field-unitarity}), $L_{0}^{\mathrm{Ar}}$
acts as
\begin{equation}
\label{v4-arith:vb:eq:L_0-action}
L_{0}^{\mathrm{Ar}}|\mathbf{n}\rangle\;=\;\Bigl(\sum_{i\geq 1}i\,n_{i}\Bigr)|\mathbf{n}\rangle\;=:\;m(\mathbf{n})|\mathbf{n}\rangle.
\end{equation}
\end{lemma}

\begin{proof}
$L_{0}=\sum_{i\geq 1}\alpha_{-i}\alpha_{i}$ (Sugawara stress tensor
zero mode for the free boson; Frenkel--Ben-Zvi 2004~\S3.5). Applied
to $(\alpha_{-i})^{n_{i}}|0\rangle$: by the canonical commutator
$[\alpha_{i},\alpha_{-i}]=i$, $\alpha_{i}(\alpha_{-i})^{n_{i}}|0\rangle=n_{i}\cdot i\cdot(\alpha_{-i})^{n_{i}-1}|0\rangle$,
hence $\alpha_{-i}\alpha_{i}(\alpha_{-i})^{n_{i}}|0\rangle=i n_{i}(\alpha_{-i})^{n_{i}}|0\rangle$.
Summing over $i$ gives the diagonal eigenvalue $\sum_{i}i n_{i}$.
\end{proof}

\subsection{The Symmetric-Function Identity}
\label{v4-arith:vb:symmetric-identity}

\begin{theorem}[combinatorial positivity of $T_{n}$ on Fock]
\label{v4-arith:vb:thm:fock-combinatorial-positivity}
\ClaimStatusProvedHere. On every non-vacuum number state
$|\mathbf{n}\rangle$ with $m(\mathbf{n})\geq 1$,
\begin{equation}
\label{v4-arith:vb:eq:fock-positivity}
T_{n}|\mathbf{n}\rangle\;=\;[m(\mathbf{n})^{n}-(m(\mathbf{n})-1)^{n}]\,|\mathbf{n}\rangle,
\end{equation}
and $m(\mathbf{n})^{n}-(m(\mathbf{n})-1)^{n}\geq 0$ for every
integer $m(\mathbf{n})\geq 1$ and every $n\geq 1$.
\end{theorem}

\begin{proof}
By \Cref{v4-arith:vb:lem:diagonal-action}, $L_{0}^{\mathrm{Ar}}$ acts
by the scalar $m=m(\mathbf{n})$. Hence $(L_{0}^{\mathrm{Ar}})^{n}$
acts by $m^{n}$ and $(L_{0}^{\mathrm{Ar}}-\mathrm{id})^{n}$ by
$(m-1)^{n}$. The difference is $m^{n}-(m-1)^{n}$.

Non-negativity: the function $x\mapsto x^{n}$ is monotone
non-decreasing on $[0,\infty)$ for $n\geq 1$; hence for $m\geq 1$ the
inequality $m^{n}\geq (m-1)^{n}$ holds, with equality iff $m=1$ and
$n>0$ (in which case $m^{n}-(m-1)^{n}=1\geq 0$).
\end{proof}

\begin{remark}[combinatorial expansion]
\label{v4-arith:vb:rem:combinatorial-expansion}
Binomial expansion:
\begin{equation}
m^{n}-(m-1)^{n}\;=\;\sum_{k=0}^{n-1}\binom{n}{k}m^{k}(-1)^{n-1-k}\;=\;nm^{n-1}-\binom{n}{2}m^{n-2}+\cdots+(-1)^{n-1}.
\end{equation}
For $m=1$: the identity $1-0=1$; hence $T_{n}|\mathbf{n}\rangle=|\mathbf{n}\rangle$
on weight-$1$ states for every $n\geq 1$. This is the
\emph{singleton-primary}-state positivity.
\end{remark}

\begin{corollary}[Fock-level restricted positivity]
\label{v4-arith:vb:cor:fock-restricted-positivity}
\ClaimStatusProvedHere. On the subspace
$\mathcal{F}^{m\geq 1}\subseteq\mathcal{H}^{(\infty)}$ spanned by
non-vacuum number states, $T_{n}\succeq 0$ for every $n\geq 1$.
\end{corollary}

\begin{proof}
Direct from \Cref{v4-arith:vb:thm:fock-combinatorial-positivity}:
$T_{n}$ acts diagonally with non-negative eigenvalues; positivity
follows from the diagonal characterisation.
\end{proof}

\begin{remark}[what this closes and what it does not]
\label{v4-arith:vb:rem:fock-closed-not}
\Cref{v4-arith:vb:cor:fock-restricted-positivity} closes the
\emph{archimedean-Fock} sector of VB$^{*}$: on the rank-$2$ free
boson at each $p$ and at $\infty$, taken individually, $T_{n}\succeq 0$
unconditionally for every $n$. What remains is the interaction with
the Bost--Connes zero-mode lattice: the primary arithmetic Heisenberg
Fock has an additional structure (the $\ell^{2}(\mathbb{N}^{\times})$
lattice where $\mu_{n}$ acts via
$\mu_{n}\delta_{m}=\delta_{nm}$; see
\Cref{v4-arith:thm:BC-acts-on-L}), and on this lattice the spectrum
of $L_{0}^{\mathrm{Ar}}$ is the Dirichlet-series spectrum
$\{\log n\colon n\in\mathbb{N}^{\times}\}$ at KMS temperature
$\beta=1/\log p$. Positivity here reduces to zero placement of
$\zeta$, which is RH.
\end{remark}

\section{Reduction to the Irreducible Core Obstruction}
\label{v4-arith:vb:reduction-core}

\subsection{The Core Statement}
\label{v4-arith:vb:core-statement}

\begin{theorem}[conditional reduction of VB$^{*}$ to the zero-mode core]
\label{v4-arith:vb:thm:reduction}
\ClaimStatusProvedHere \emph{(conditional on P2, P3,
H-P$^{\mathrm{Ar}}$, the primary tensor decomposition, and a positive
regularised trace)}. Under these hypotheses, VB$^{*}$ reduces to the
following single zero-mode statement:
\begin{itemize}
\item[\textbf{(VB$^{*}_{\mathrm{core}}$)}] \emph{Zero-mode spectral
positivity.} The Bost--Connes partition function
$Z_{\mathrm{BC}}(\beta)=\sum_{n\geq 1}n^{-\beta}=\zeta(\beta)$
extends, as a spectral measure on the zero-mode lattice
$\ell^{2}(\mathbb{N}^{\times})$ under the Hecke action
$\mu_{n}\delta_{m}=\delta_{nm}$, to a positive distribution on the
continuous spectrum of $L_{0}^{\mathrm{Ar}}|_{L}$, whose support
after $1/2$-translation is contained in $[0,\infty)$.
\end{itemize}
\end{theorem}

\begin{proof}
By the tensor decomposition
$\mathcal{V}^{\mathrm{prim}}_{\mathrm{Heis},\mathrm{prim}}\cong L\otimes\mathcal{H}^{(\infty)}_{\mathrm{primary}}$
(\Cref{v4-arith:thm:BC-tensor-placement} arithmetic corollary),
$T_{n}$ acts as $T_{n}|_{L}\otimes\mathrm{id}+\mathrm{id}\otimes T_{n}|_{\mathcal{H}}+\text{mixed terms}$
where the mixed terms vanish because $L_{0}^{\mathrm{Ar}}$
tensor-decomposes additively. Positivity on the second factor is
\Cref{v4-arith:vb:cor:fock-restricted-positivity}. Positivity on the
first factor (the zero-mode lattice) is the content of
(VB$^{*}_{\mathrm{core}}$).

The reduction from VB$^{*}$ to (VB$^{*}_{\mathrm{core}}$) is as
follows: the archimedean factor carries $T_{n}\succeq 0$
unconditionally; the
zero-mode factor inherits the Bost--Connes KMS state, whose
partition function $\zeta(\beta)$ has spectrum given by the
$L_{0}^{\mathrm{Ar}}|_{L}$-eigenvalues. Under H-P$^{\mathrm{Ar}}$,
these eigenvalues encode the non-trivial zeros of $\xi$,
Bost--Connes-regularised.
Positivity of $T_{n}|_{L}$ at every $n$ is then equivalent, by the
moment argument of \Cref{v4-arith:vb:thm:moment-equivalence}, to
zero-placement on the critical line.
\end{proof}

\subsection{The Named Obstruction}
\label{v4-arith:vb:named-obstruction}

\begin{corollary}[the irreducible core of VB$^{*}$]
\label{v4-arith:vb:cor:irreducible-core}
\ClaimStatusProvedHere \emph{(conditional on the hypotheses of
\Cref{v4-arith:vb:thm:reduction})}. Within Vol~IV's framework, VB$^{*}$ is
irreducible: it cannot be reduced below
(VB$^{*}_{\mathrm{core}}$) without resolving the zero placement of
$\zeta$. The framework shortens VB$^{*}$ by converting it from a
sequence of operator-positivity conditions $\{T_{n}\succeq 0\}_{n\geq 1}$
to a single spectral-measure positivity condition on the Bost--Connes
zero-mode lattice.
\end{corollary}

\begin{remark}[what this amounts to]
\label{v4-arith:vb:rem:core-meaning}
(VB$^{*}_{\mathrm{core}}$) is not a \emph{weaker} statement than
VB$^{*}$ under the Hilbert--P\'olya and positive-trace hypotheses; it
is the zero-mode form of the same residual.
What the reduction produces:
\begin{enumerate}
\item A single positivity statement instead of a sequence.
\item A statement about the Bost--Connes system, which has been
studied extensively in noncommutative geometry (Connes--Marcolli 2004,
Consani et al. 2012--2015).
\item An interpretation as positivity of a KMS partition function on
a specific quantum statistical system.
\item A direct connection to the Bost--Connes phase-transition
theorem at $\beta=1$: the Bost--Connes KMS state undergoes a phase
transition at $\beta=1$ corresponding to the pole of $\zeta$ at
$s=1$, and RH amounts to absence of further phase transitions on the
line $\mathrm{Re}(\beta)=1/2$.
\end{enumerate}
This is genuine technical progress in translation; not in proof.
\end{remark}

\begin{remark}[placement of VB$^{*}$ against its neighbours]

VB$^{*}$ implies Li's $\lambda_{n}\geq 0$, $n\geq 1$, under the

parallel in depth to Connes' adelic trace formula, and at a
comparable position in the difficulty hierarchy.
\end{remark}

\begin{proposition}[Weil equivalence under P3]
\label{v4-arith:vb:prop:weil-equivalence}
\ClaimStatusProvedHere \emph{(conditional on Koszul--Petersson
compatibility \Cref{v4-arith:cl:cj:koszul-Petersson-compat} and the

positivity and VB$^{*}$ express the same RH-equivalent positivity
barrier after identifying the Weil inner product on adelic test
functions with the Petersson inner product on the primary arithmetic
Heisenberg Fock. This is a comparison theorem, not an independent
proof of positivity.
\end{proposition}

\begin{proof}
Weil 1952 gives $\lambda_{n}=\langle f_{n},f_{n}\rangle^{\mathrm{Weil}}$
for $f_{n}$ an explicit test function on the adele class space.
Under Koszul-Petersson compatibility, the Weil inner product pulls back
to the Petersson inner product on
$\mathcal{V}^{\mathrm{prim}}_{\mathrm{Heis},\mathrm{prim}}$ via
Bost--Connes restriction to the zero-mode lattice (where the
$\mu_{n}$ action is Weil's translation-by-log-$n$ action on the
adele class space). Hence
$\lambda_{n}=\mathrm{tr}(T_{n})/n=\langle f_{n},f_{n}\rangle^{\mathrm{Pet}}$;
positivity is equivalent.
\end{proof}

After the closures and reductions of this chapter, the residual
obstruction to full VB$^{*}$, stated precisely, is:

\begin{definition}[the Vol~IV residual]
\label{v4-arith:vb:def:residual}
\ClaimStatusDefinitional. The \emph{Vol~IV residual for RH} is the
following statement:
\begin{equation}
\label{v4-arith:vb:eq:residual}
\tag{$\mathcal{R}_{\mathrm{IV}}$}
\parbox{0.82\linewidth}{%
The spectrum of $L_{0}^{\mathrm{Ar}}|_{L}$ on the Bost--Connes
zero-mode lattice $L=\ell^{2}(\mathbb{N}^{\times})$, interpreted via
the $\zeta(\beta)$-partition function of the Bost--Connes $C^{*}$-algebra
(\Cref{v4-arith:thm:BC-KMS}), is concentrated on the critical line
$\mathrm{Re}(\rho)=1/2$ after the functional-equation symmetrisation
of P1.%
}
\end{equation}
\end{definition}

\begin{theorem}[$\mathcal{R}_{\mathrm{IV}}$ is RH]
\label{v4-arith:vb:thm:residual-is-RH}
\ClaimStatusProvedHere. $\mathcal{R}_{\mathrm{IV}}\Leftrightarrow$ RH.
\end{theorem}

\begin{proof}
By definition, $\mathcal{R}_{\mathrm{IV}}$ says that the
Bost--Connes zero-mode spectral measure is concentrated on the
critical line after P1 symmetrisation. This is exactly the
zero-placement assertion for the non-trivial zeros of $\zeta$, hence
RH by \Cref{v4-arith:rh:rem:equivalence-to-RH}. Conversely, RH gives
the same critical-line support statement, so
$\mathcal{R}_{\mathrm{IV}}$ and RH are equivalent as formulations.
\end{proof}

\section{Arithmetic Zhu Finiteness Test}
\label{v4-arith:vb:zhu-finiteness-test}

The tempting shortcut is Verlinde finite-dimensionality: prove that the
primary Bost--Connes core is an arithmetic Zhu-finite object, obtain a
finite-dimensional Hermitian modular functor, and read positivity from
ordinary linear algebra. The shortcut fails on the full zeta object.

\begin{definition}[Euler-faithful arithmetic Zhu quotient]
\label{v4-arith:vb:def:euler-faithful-zhu}
\ClaimStatusDefinitional. An \emph{Euler-faithful arithmetic Zhu
quotient} of the primary Bost--Connes core is a quotient \(Q\) of the
zero-mode lattice \(L=\ell^{2}(\mathbb{N}^{\times})\) satisfying:
\begin{enumerate}
\item the Bost--Connes operators descend:
\[
\mu_{n}\overline{\delta}_{m}=\overline{\delta}_{nm},\qquad
\mu_{n}^{*}\overline{\delta}_{m}=
\begin{cases}
\overline{\delta}_{m/n},& n\mid m,\\
0,& n\nmid m,
\end{cases}
\]
where \(\overline{\delta}_{m}\) is the image of
\(\delta_{m}\in L\);
\item the KMS Hamiltonian \(H=\log N\) descends with
\(H\overline{\delta}_{m}=(\log m)\overline{\delta}_{m}\) whenever
\(\overline{\delta}_{m}\neq0\);
\item the ordinary KMS character of the quotient is the full
Bost--Connes Euler product:
\[
\mathrm{Tr}_{Q}(e^{-\beta H})=\sum_{m\geq1}m^{-\beta}=\zeta(\beta)
\qquad(\mathrm{Re}\,\beta>1).
\]
\end{enumerate}
It is \emph{Zhu finite} if \(\dim_{\mathbb C}Q<\infty\).
\end{definition}

\begin{theorem}[no Euler-faithful Zhu finiteness]
\label{v4-arith:vb:thm:no-euler-faithful-zhu-finite}
\ClaimStatusProvedHere. No Euler-faithful arithmetic Zhu quotient of
the primary Bost--Connes core is Zhu finite.
\end{theorem}

\begin{proof}
If \(Q\) were finite-dimensional, the trace
\(\mathrm{Tr}_{Q}(e^{-\beta H})\) would be a finite sum
\(\sum_{j=1}^{r}a_{j}e^{-\beta E_{j}}\) after triangularising the
finite-dimensional operator \(H\). In particular it extends to an
entire function of \(\beta\). Euler-faithfulness requires this finite
sum to equal \(\zeta(\beta)\) on the half-plane
\(\mathrm{Re}\,\beta>1\). By the identity theorem the meromorphic
continuation of \(\zeta\) would be entire, contradicting its simple
pole at \(\beta=1\). Hence \(Q\) is infinite-dimensional.

Equivalently, the faithful Bost--Connes action of
\Cref{v4-arith:thm:BC-acts-on-L} sends
\(\delta_{1}\) to \(\delta_{m}\) by \(\mu_{m}\delta_{1}=\delta_{m}\).
Keeping the full Euler product keeps one independent KMS eigenline at
every \(m\in\mathbb{N}^{\times}\). A finite Zhu quotient must kill or
identify infinitely many of these lines, and then it no longer has
character \(\zeta(\beta)\).
\end{proof}

\begin{corollary}[Verlinde finite-dimensionality cannot close full RH]
\label{v4-arith:vb:cor:verlinde-shortcut-fails}
\ClaimStatusProvedHere. Arithmetic Zhu/Verlinde finite-dimensionality
cannot prove VB$^{*}_{\mathrm{core}}$ for the full
\(\mathcal{V}^{\mathrm{prim}}\) while preserving the completed zeta
partition function. The only Zhu-finite quotients compatible with the
Bost--Connes action are finite-prime or finite-conductor truncations;
their characters are finite Euler products or finite Dirichlet
polynomials, not \(\xi(s)\).
\end{corollary}

\begin{proof}
By \Cref{v4-arith:vb:thm:no-euler-faithful-zhu-finite}, a
finite-dimensional quotient cannot be Euler-faithful. A quotient
supported on a finite set \(S\) of primes has character
\(\prod_{p\in S}(1-p^{-\beta})^{-1}\), possibly with an archimedean
Tate factor. Such a finite Euler product is a finite-level arithmetic
modular-functor object; it is not the zeta object whose zero-placement
is RH. Therefore finite-dimensional Hermitian linear algebra can test
finite windows, but it cannot supply the missing positivity for the
full Bost--Connes zero-mode spectral measure.
\end{proof}

\begin{remark}[central charge does not rescue Zhu finiteness]
\label{v4-arith:vb:rem:c-not-zhu-finite}
The residue value \(c_{\mathrm{global}}=2\) of
\Cref{v4-arith:thm:regularised-c} is not a finite rank and not a
trace. Chapter~\ref{v4-arith:polyakov} proves that the full infinite
tensor has divergent Virasoro central term and only a Witt action
algebraically. Thus the equality \(c_{\mathrm{global}}=2\) cannot be
used as a \(C_{2}\)-cofiniteness or Zhu-finiteness input.
\end{remark}

\begin{remark}[surviving direction]
\label{v4-arith:vb:rem:zhu-surviving-direction}
The finite-dimensional route survives only as a finite-window
technology: truncate to finitely many primes or to bounded conductor,
prove the corresponding arithmetic modular functor is finite, compute
its Hermitian form, and compare its limiting trace to the
Bost--Connes KMS trace. Any limiting theorem strong enough to recover
\(\mathrm{VB}^{*}_{\mathrm{core}}\) must add a new positivity or compactness
input beyond Zhu finite-dimensionality at finite level.
\end{remark}

\section{Open Problem Inscription}
\label{v4-arith:vb:open-problem}

\begin{conjecture}[VB$^{*}_{\mathrm{core}}$: the Vol~IV form of RH]
\label{v4-arith:vb:conj:VB-core}
\ClaimStatusConjectured. The Bost--Connes zero-mode spectral measure
is supported on the critical line $\mathrm{Re}(\rho)=1/2$.
Equivalently, all non-trivial zeros of $\zeta$ lie on the critical
line.
\end{conjecture}

\begin{remark}[the conjecture is RH]
\label{v4-arith:vb:rem:conjecture-is-RH}
\Cref{v4-arith:vb:conj:VB-core} is, verbatim, the Riemann Hypothesis.
It is inscribed as a Vol~IV-form open problem to emphasise that
within the Vol~IV chiral-algebra framework, RH is not a hypothesis
about zeros of $\zeta$ but a hypothesis about positivity of a
spectral measure on a specific $C^{*}$-algebra. This is a genuine
reformulation, consistent with the Connes 1999 arXiv:math/9811068
reformulation as a trace formula on the adele class space.
\end{remark}

\begin{remark}[paper-length estimate for VB$^{*}_{\mathrm{core}}$]
\label{v4-arith:vb:rem:paper-length}
Any unconditional proof of \Cref{v4-arith:vb:conj:VB-core} would be
an unconditional proof of RH, a Millennium Prize problem. No
paper-length estimate is meaningful. Paper-length estimates for
partial progress:
\begin{description}
\item[Bombieri-Lagarias-style sharpening at specific $n$.] $15$--$25$
pages per level.
\item[Kac-determinant analysis at $c=2$ to low level.] $20$--$30$
pages.
\item[Moment-problem Hankel analysis.] $10$--$20$ pages for formal
equivalences; no pre-existing paper for unconditional lower bounds.
\item[Arithmetic-site incorporation.] $40$--$60$ pages in the Connes-Consani
framework; structural; no unconditional RH closure expected.
\end{description}
\end{remark}

\section{Summary}
\label{v4-arith:vb:summary}

\begin{enumerate}
\item The arithmetic Virasoro positivity bootstrap VB$^{*}$ gives a
conditional path to RH via Li's criterion once H-P$^{\mathrm{Ar}}$
and a positive regularised trace are supplied
(\Cref{v4-arith:vb:thm:three-equivalences}). Its converse and
zero-mode core are RH-equivalent residuals, not closed theorems
(\Cref{v4-arith:vb:thm:moment-equivalence},
\Cref{v4-arith:vb:prop:weil-equivalence}).
\item $n=1$ closes unconditionally: $T_{1}=\mathrm{id}$
(\Cref{v4-arith:vb:thm:n=1}).
\item $n=2$ reduces to $L_{0}^{\mathrm{Ar}}\succeq(1/2)\mathrm{id}$,
equivalent to zero placement on the critical line
(\Cref{v4-arith:vb:thm:n=2},
\Cref{v4-arith:vb:cor:n=2-is-RH}).
\item $n\geq 3$ closes at the trace level (Bombieri-Lagarias bound,
Coffey numerical verification) but not at the operator level without
RH.
\item Fock-combinatorial positivity closes the archimedean sector:
$T_{n}\succeq 0$ on non-vacuum number states for every $n$
(\Cref{v4-arith:vb:thm:fock-combinatorial-positivity}).
\item The Kac determinant at $c^{\mathrm{Ar}}=2$ does not vanish at
generic weights, establishing absence of null-vector obstruction
(\Cref{v4-arith:vb:prop:kac-non-vanishing},
\Cref{v4-arith:vb:cor:no-generic-null}).
\item VB$^{*}$ reduces to a single statement on the Bost--Connes
zero-mode lattice: (VB$^{*}_{\mathrm{core}}$,
\Cref{v4-arith:vb:thm:reduction}).
\item Arithmetic Zhu/Verlinde finite-dimensionality does not close the
full zero-mode core: any Euler-faithful quotient preserving the
Bost--Connes character \(\zeta(\beta)\) is infinite-dimensional
(\Cref{v4-arith:vb:thm:no-euler-faithful-zhu-finite},
\Cref{v4-arith:vb:cor:verlinde-shortcut-fails}).
\item (VB$^{*}_{\mathrm{core}}$) is the Vol~IV-natural form of RH,
equivalent at the $\Leftrightarrow$ level to
$\mathrm{Re}(\rho)=1/2$ for all non-trivial zeros
(\Cref{v4-arith:vb:thm:residual-is-RH}).
\item The residual $\mathcal{R}_{\mathrm{IV}}$ is named precisely
and is equivalent to RH; it is the Vol~IV irreducible core.
\end{enumerate}

The arithmetic Virasoro positivity bootstrap is closed to the extent
that a closure not equivalent to RH is possible: level-by-level
closure at $n=1$ and on the archimedean Fock sector, structural
reduction at $n=2$ to the single-zero-placement form of RH,
moment-problem reformulation for all $n$, and irreducible-core
reduction to a single Bost--Connes spectral-measure positivity. The
remaining obstruction is stated in two equivalent operator-theoretic
forms
(\Cref{v4-arith:vb:conj:VB-core},
\Cref{v4-arith:vb:def:residual}) and is, in full, the Riemann
Hypothesis.
