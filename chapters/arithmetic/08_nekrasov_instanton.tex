\chapter{The Arithmetic Instanton Partition Function}
\label{v4-arith:nekrasov}

\emph{In the voice of Nekrasov. Instanton counting, $\Omega$-background,
AGT, Maulik--Okounkov. We ask whether the Hartnoll--Yang $L(s)$ is
an arithmetic avatar of $Z_{\mathrm{inst}}$.}

\section{Nekrasov's $Z_{\mathrm{inst}}$ on $\mathbb{C}^{2}$}
\label{v4-arith:nk:Z-inst}

\begin{theorem}[Nekrasov, hep-th/0206161 Thm~2; Nekrasov--Okounkov hep-th/0306238 \S3]
\label{v4-arith:thm:Z-inst-classical}
\ClaimStatusProvedElsewhere. For $U(N)$ gauge theory on
$\mathbb{C}^{2}$ with equivariant torus action by
$(t_{1},t_{2})$ on $(z_{1},z_{2})\in\mathbb{C}^{2}$,
$\epsilon_{i}=\log t_{i}$:
\begin{equation*}
Z^{U(N)}_{\mathrm{inst}}(\vec a,\epsilon_{1},\epsilon_{2};\Lambda)
=\sum_{\vec\lambda}\Lambda^{N|\vec\lambda|}
\prod_{\alpha,\beta=1}^{N}\prod_{s\in\lambda^{(\alpha)}}
\frac{1}{E_{\alpha\beta}(s)(\epsilon_{1}+\epsilon_{2}-E_{\alpha\beta}(s))}
\end{equation*}
with $E_{\alpha\beta}(s)=a_{\alpha}-a_{\beta}-\epsilon_{1}L_{\lambda^{(\beta)}}(s)+\epsilon_{2}(A_{\lambda^{(\alpha)}}(s)+1)$.
\end{theorem}

\section{Dimension Count Fails}
\label{v4-arith:nk:dimension-fails}

\begin{observation}
\label{v4-arith:obs:dimension-fails}
$\mathbb{C}^{2}$ is real $4$-dimensional with a $T^{2}$ torus
action. The Arakelov-compactified arithmetic curve
$\overline{C}=\overline{\mathrm{Spec}\,\mathbb{Z}}$ has arithmetic
Krull dimension $1$; arithmetic Chow groups
$\widehat{\mathrm{CH}}^{*}(\overline{C})$ live only in degrees $0,1$.
There is no $\widehat{\mathrm{CH}}^{2}$ --- no instanton class.
\end{observation}

\begin{corollary}
\label{v4-arith:cor:no-direct-Z-inst}
\ClaimStatusProvedHere. A $U(N)$-gauge theory on the arithmetic
1-dimensional $\overline{C}$ is 2d Yang--Mills in character with
no instanton sector. The primon partition function $L(s)$ cannot
be a naive $Z^{U(2)}_{\mathrm{inst}}$ on $\overline{C}$.
\end{corollary}

\section{Pass to the Arithmetic Surface}
\label{v4-arith:nk:arithmetic-surface}

\begin{principle}[Deligne]
\label{v4-arith:princ:deligne-self-product}
Arakelov-theoretic constructions on $\mathrm{Spec}\,\mathbb{Z}$
become two-dimensional on the self-product. Durov's arXiv:0704.2030
and Connes--Consani arXiv:0809.1259 supply the $\mathbb{F}_{1}$
base over which $\overline{C}\times_{\mathbb{F}_{1}}\overline{C}$
is a surface.
\end{principle}

\begin{definition}[arithmetic surface setup]
\label{v4-arith:def:arithmetic-surface-setup}
On the self-product $\overline{C}\times_{\mathbb{F}_{1}}\overline{C}$:
\begin{itemize}
\item Torus action: $\mathbb{G}_{m}(\mathbb{F}_{1})^{2}=\{1\}^{2}$
literally, or $(\mathbb{Z}_{>0},\times)^{2}$ in Connes--Consani
semigroup form.
\item Fixed points: pairs $(p,q)$ of primes; or single primes $p$
on the diagonal.
\item Equivariant weights: $\epsilon_{p}=\log p$ per prime.
\end{itemize}
\end{definition}

\section{Rank Analysis: $U(1)^{\infty}$ per Prime}
\label{v4-arith:nk:rank-analysis}

\begin{observation}
\label{v4-arith:obs:rank-infinite}
The primon VA has Coulomb-branch moduli $\{a_{p}\}_{p}$ indexed by
primes, so the gauge rank is $|\mathbb{P}|=\aleph_{0}$. $L(s)$
corresponds to a $U(\infty)$ theory if we identify primes as
gauge-group indices.
\end{observation}

\begin{definition}[abelian arithmetic Nekrasov per prime]
\label{v4-arith:def:abelian-per-prime}
\begin{equation}
Z^{U(1)}_{\mathrm{inst},p}(a_{p},\epsilon_{1}^{(p)},\epsilon_{2}^{(p)};\Lambda_{p})
\;=\;\sum_{\lambda}\Lambda_{p}^{|\lambda|}
\prod_{s\in\lambda}\frac{1}{\epsilon_{1}^{(p)}L_{\lambda}(s)\,\epsilon_{2}^{(p)}(A_{\lambda}(s)+1)}.
\end{equation}
On the CY / Nekrasov--Okounkov slice $\epsilon_{1}+\epsilon_{2}=0$,
$\epsilon_{1}=-\epsilon_{2}=\hbar$:
\begin{equation}
Z^{U(1),\mathrm{CY}}_{\mathrm{inst}}\;=\;\sum_{\lambda}\frac{\Lambda^{|\lambda|}}{\prod_{s\in\lambda}h_{\lambda}(s)^{2}}
\;=\;\eta(\Lambda)^{-1}\Lambda^{1/24},
\end{equation}
the MacMahon-Nekrasov--Okounkov--Pandharipande plane-partition
generating function (MNOP, math/0312059~Thm~1).
\end{definition}

\begin{computation}
\label{v4-arith:comp:per-prime-Euler}
Setting $\Lambda_{p}=p^{-s}$:
\begin{equation}
Z^{U(1),\mathrm{CY}}_{\mathrm{inst},p}(s)\;=\;(1-p^{-s})^{-1}\cdot\text{(higher corrections)}.
\end{equation}
Product over primes:
\begin{equation}
\prod_{p}Z^{U(1),\mathrm{CY}}_{\mathrm{inst},p}(s)\;=\;L(s)\cdot R(s),
\end{equation}
with leading Euler factor matching the primon partition and
corrections $R(s)=\prod_{p}R_{p}(s)$.
\end{computation}

\section{The $\epsilon_{2}\to\infty$ Degeneration}
\label{v4-arith:nk:epsilon-2-degeneration}

\begin{definition}[degenerate per-prime limit]
\label{v4-arith:def:degenerate-per-prime}
Set per prime $p$:
\begin{equation*}
\epsilon_{1}^{(p)}=\log p,\quad\epsilon_{2}^{(p)}=+\infty.
\end{equation*}
In this limit $1/\epsilon_{2}^{(p)}\to 0$ and every Young diagram
with $|\lambda|\geq 2$ contributes $0$. The sum collapses to
\begin{equation}
Z^{U(1)}_{\mathrm{inst},p}\bigr|_{\mathrm{arith}}\;=\;(1-p^{-s})^{-1}.
\end{equation}
\end{definition}

\begin{theorem}[primon partition as degenerate Nekrasov]
\label{v4-arith:thm:primon-as-degenerate-nekrasov}
\ClaimStatusConjectured. The Hartnoll--Yang primon partition
function $L(s)$ is the product over primes of degenerate
$U(1)_{p}$ Nekrasov functions in the
$(\epsilon_{1}^{(p)},\epsilon_{2}^{(p)})=(\log p,\infty)$ limit:
\begin{equation}
\boxed{\;L(s,\rho)\;=\;\prod_{p}Z^{U(N)}_{\mathrm{inst},p}(s;\rho_{p})\bigr|_{b^{(p)}\to 0}^{\epsilon_{2}^{(p)}\to\infty}
\;=\;\prod_{p}\det(1-p^{-s}\rho_{p}(\mathrm{Frob}_{p}))^{-1}.\;}
\end{equation}
This is \emph{narrow}: $L(s)$ is the degenerate limit of Nekrasov,
not a full $Z_{\mathrm{inst}}$.
\end{theorem}

\section{Arakelov Coulomb Modulus}
\label{v4-arith:nk:Arakelov-Coulomb}

\begin{definition}[arithmetic Coulomb modulus]
\label{v4-arith:def:arithmetic-coulomb}
Identify $s$ with the arithmetic first Chern class:
\begin{equation}
s\;=\;\widehat{\deg}\,\mathcal{O}(s\cdot[\infty])\;\in\;\widehat{\mathrm{CH}}^{1}(\overline{C})\otimes\mathbb{R}\cong\mathbb{R}.
\end{equation}
The Nekrasov Coulomb modulus $\vec a$ is replaced by the single
arithmetic modulus $s$. The Hecke-character tuple $\{\theta_{p}\}$
is the arithmetic analogue of the non-trivial Coulomb splitting
$a_{\alpha}-a_{\beta}$ for $U(N)$.
\end{definition}

\begin{remark}[Gillet--Soulé Arakelov pairing]
On $\overline{C}\times\overline{C}$ the height pairing at
$\infty\times\infty$ is $s\cdot s'$ up to finite contributions
(Gillet--Soulé \emph{Arithmetic intersection theory},
Publ.~IHES~1992).
\end{remark}

\section{Nekrasov--Shatashvili Limit: $\log\zeta(s)=\sum_{p}\mathcal{W}^{\mathrm{NS}}_{p}(s)$}
\label{v4-arith:nk:NS-limit}

\begin{theorem}[arithmetic NS identity]
\label{v4-arith:thm:NS-identity}
\ClaimStatusProvedHere. The regulated per-prime NS superpotential
$\mathcal{W}^{\mathrm{NS}}_{p}(s)=\log(1-p^{-s})^{-1}=\sum_{k\geq 1}p^{-ks}/k$
sums over primes to
\begin{equation}
\boxed{\;\log\zeta(s)\;=\;\sum_{p}\mathcal{W}^{\mathrm{NS}}_{p}(s).\;}
\end{equation}
The Dedekind zeta logarithm is the sum over primes of the
arithmetic Nekrasov--Shatashvili superpotential, evaluated at
arithmetic Coulomb modulus $s$.
\end{theorem}

\section{AGT: Semiclassical Liouville Degenerates to Free Boson}
\label{v4-arith:nk:AGT-degenerate}

\begin{remark}[AGT obstruction]
AGT (Alday--Gaiotto--Tachikawa arXiv:0906.3219):
$Z^{U(2)}_{\mathrm{inst}}(\vec a,\epsilon_{1},\epsilon_{2};\Lambda)=
\mathcal{B}^{\mathrm{Liouville}}_{c,\Delta_{i}}(\vec q)$
with $c=1+6(b+b^{-1})^{2}$, $b^{2}=\epsilon_{1}/\epsilon_{2}$. In
the arithmetic limit $\epsilon_{2}\to\infty$, $b\to 0$: $c\to\infty$.
AGT as stated does not extend.
\end{remark}

\begin{theorem}[semiclassical-Liouville $\to$ free boson]
\label{v4-arith:thm:liouville-to-free-boson}
\ClaimStatusProvedElsewhere (Teschner arXiv:hep-th/0104158;
Harlow--Maltz--Witten arXiv:1108.4417). In the $b\to 0$ limit,
Virasoro degenerates to Heisenberg: the Liouville primary
$V_{\alpha}=e^{2\alpha\varphi}$ with $\alpha=\alpha_{0}+\alpha_{1}b$
becomes a free-boson vertex operator of charge $\alpha_{0}$;
Virasoro conformal blocks become free-boson correlators.
\end{theorem}

\begin{corollary}[arithmetic AGT at $b\to 0$]
\label{v4-arith:cor:arithmetic-AGT}
\ClaimStatusConjectured. Per prime,
\begin{equation}
Z^{U(1)}_{\mathrm{inst},p}\bigr|_{b^{(p)}\to 0}(\Lambda_{p})
\;=\;\langle V_{\alpha}|e^{\Lambda_{p}L_{-1}}|V_{\alpha}\rangle_{\mathrm{free\,boson}}
\;=\;(1-\Lambda_{p}/\text{gap})^{-1}.
\end{equation}
Taking $\Lambda_{p}=p^{-s}$ and gap~$1$: $(1-p^{-s})^{-1}$
= primon Heisenberg character per prime.
\end{corollary}

\begin{theorem}[Hartnoll--Yang primon VA = $b\to 0$ AGT]
\label{v4-arith:thm:HY-primon-as-bto0-AGT}
\ClaimStatusConjectured.
\begin{equation*}
L(s)\;=\;\prod_{p}Z^{U(1)}_{\mathrm{inst},p}\bigr|_{\mathrm{AGT,\,}b\to 0}(s)\;=\;\prod_{p}\langle\mathcal{H}^{(p)}_{k=1}\rangle_{\mathrm{partition}}.
\end{equation*}
The rank-$1$ Heisenberg per prime is the semiclassical-Liouville
degeneration of Virasoro at prime $p$, and the AGT map reduces to
a free-boson partition at each place.
\end{theorem}

\section{Maulik--Okounkov and the Bethe-Hecke Shadow}
\label{v4-arith:nk:maulik-okounkov}

\begin{theorem}[Maulik--Okounkov arXiv:1211.1287, Thm~1.5.1]
\label{v4-arith:thm:maulik-okounkov}
\ClaimStatusProvedElsewhere. The $R$-matrix on equivariant
cohomology of the instanton moduli space $\mathcal{M}_{N,k}$
generates the Yangian $Y(\widehat{\mathfrak{gl}}_{N})$.
$Z^{U(N)}_{\mathrm{inst}}$ is the trace of the holonomy of the MO
$R$-matrix.
\end{theorem}

\begin{definition}[primon Bethe-Hecke shadow]
\label{v4-arith:def:primon-yangian}
The \emph{primon Bethe-Hecke shadow} is the commutative
Bethe/Hecke algebra acting on the arithmetic instanton model. It is
not an arithmetic Yangian or RTT Hopf algebra. Fixed-point data may
still be indexed by $N$-tuples of Young diagrams per prime, but the
surviving arithmetic action is through Hecke transfer operators.
\end{definition}

\begin{theorem}[$L$-function as MO trace]
\label{v4-arith:thm:L-as-MO-trace}
\ClaimStatusHeuristic. $L(s,\rho)$ should be read as a
Bethe/Hecke transfer trace in the arithmetic instanton shadow. The
Maulik--Okounkov Yangian remains a complex-geometric analogy; no
arithmetic RTT $R$-matrix or primon Yangian is constructed.
\end{theorem}

\section{The Topological-Sector Reading (Compatible with Nekrasov)}
\label{v4-arith:nk:topological-sector}

\begin{remark}[Kurokawa/Deninger]
Alternative reading: $L(s)=\sum_{\text{arith.~topological classes}}p^{-s\cdot v_{p}(\text{class})}$,
where each term $n^{-s}$ corresponds to the arithmetic class with
$p$-adic valuation $v_{p}(n)$. This is the $\mathbb{F}_{1}$-zeta
perspective of Kurokawa (Proc.~Japan Acad.~2005) and Deninger
(arXiv:alg-geom/9404007): $\zeta(s)=\det(s-\Theta)_{\mathrm{reg}}^{-1}$
for a hypothetical cohomological operator $\Theta$.
\end{remark}

\begin{theorem}[motivic-topological compatibility]
\label{v4-arith:thm:motivic-topological-compatibility}
\ClaimStatusConjectured. The motivic instanton partition function
(Nekrasov--Okounkov \S4; Behrend motivic DT arXiv:math/0507523) on
the arithmetic surface localises in the $\epsilon_{2}\to\infty$
limit to the $\mathbb{F}_{1}$-cohomological partition sum. Young
diagrams collapse to their size $|\lambda|$, which labels the
instanton number $c_{2}^{\mathrm{Ar}}$, and the $c_{2}^{\mathrm{Ar}}$-sum
becomes the $v_{p}$-sum:
\begin{equation*}
Z^{U(N),\mathrm{arith}}_{\mathrm{inst},p}(s,\rho_{p})
\;=\;\det(1-p^{-s}\rho_{p}(\mathrm{Frob}_{p}))^{-1}.
\end{equation*}
\end{theorem}

\section{Summary of This Chapter}
\label{v4-arith:nk:summary}

\begin{enumerate}
\item Dimension mismatch: $\overline{C}$ is $1$-d, Nekrasov needs
$2$-d; pass to self-product $\overline{C}\times_{\mathbb{F}_{1}}\overline{C}$
(\Cref{v4-arith:princ:deligne-self-product}).
\item Gauge group: $U(1)^{\infty}$ abelian, rank $\aleph_{0}$
(\Cref{v4-arith:obs:rank-infinite}).
\item $\Omega$-background: $(\log p,\infty)$ per prime
(\Cref{v4-arith:def:degenerate-per-prime}).
\item Primon partition = degenerate arithmetic Nekrasov
(\Cref{v4-arith:thm:primon-as-degenerate-nekrasov}).
\item Coulomb modulus: $s=\widehat{\deg}\,\mathcal{O}([\infty])$
(\Cref{v4-arith:def:arithmetic-coulomb}).
\item $\log\zeta(s)=\sum_{p}\mathcal{W}^{\mathrm{NS}}_{p}(s)$
(\Cref{v4-arith:thm:NS-identity}).
\item AGT degenerates to free-boson per prime at $b\to 0$
(\Cref{v4-arith:cor:arithmetic-AGT}).
\item MO $R$-matrix degenerates only to a Bethe-Hecke transfer
heuristic on the arithmetic instanton shadow
(\Cref{v4-arith:thm:L-as-MO-trace}).
\end{enumerate}

$L(s)$ is the \emph{$(\epsilon_{2}\to\infty,\,b\to 0)$ semiclassical-Liouville
degenerate limit} of Nekrasov, not a full $Z_{\mathrm{inst}}$. The
full $\Omega$-deformed instanton structure is not visible in
$L(s)$ --- it is the undeformed-$\epsilon_{2}$-direction limit
with $\epsilon_{1}=\log p$ per prime.
