\chapter{The Arithmetic Yangian Collapses to Its Bethe Subalgebra}
\label{v4-arith:drinfeld-solo}

\emph{In the voice of Drinfeld. The Yangian $Y(\mathfrak{gl}_{n})$
(Drinfeld 1985, 1988), RTT presentation, $R$-matrix, YBE,
classical $r$-matrix. We ask whether an arithmetic Yangian
$Y^{\mathrm{arith}}(\mathfrak{gl}_{n})$ exists and find that its
Hopf-algebraic part collapses to a commutative Bethe subalgebra,
with the arithmetic structure carried by the automorphic
modularity condition.}

\section{The Four Obstructions to a Naive Arithmetic Yangian}
\label{v4-arith:dr:four-obstructions}

\begin{theorem}[four structural obstructions]
\label{v4-arith:thm:four-obstructions}
\ClaimStatusProvedHere. Yang's R-matrix $R(u)=1+P/u$ has no
arithmetic analogue:
\begin{enumerate}
\item \textbf{Discrete-continuous mismatch.} The spectral parameter
$u$ is continuous and $\mathrm{Spec}\,\mathbb{Z}$ is discrete;
primes cannot be spectral parameters.
\item \textbf{No cross-prime subtraction.} $u-v$ has no analogue
across distinct primes ($\mathbb{Q}_{p}\not\hookrightarrow\mathbb{Q}_{q}$
for $p\neq q$).
\item \textbf{No meromorphic 1-form on $\mathrm{Spec}\,\mathbb{Z}$.}
Costello 4D Chern--Simons (arXiv:1709.09993) requires a
meromorphic $1$-form $\omega$ on the spectral-parameter direction;
$\mathrm{Spec}\,\mathbb{Z}$ has none.
\item \textbf{No $\zeta$-valued shuffle kernel.} Arithmetic
shuffle-kernel constructions with $\zeta$-valued kernels fail the
cocycle condition (Feigin--Odesskii).
\end{enumerate}
\end{theorem}

\begin{corollary}[no arithmetic Yangian as Hopf algebra]
\label{v4-arith:cor:no-arith-yangian}
\ClaimStatusProvedHere. $Y^{\mathrm{arith}}(\mathfrak{gl}_{n})$
does not exist as a noncommutative Hopf algebra deforming
$U(\mathfrak{gl}_{n}[z])$ with an arithmetic $R$-matrix and RTT
relation.
\end{corollary}

\section{The Bethe-Algebra Collapse}
\label{v4-arith:dr:bethe-collapse}

\begin{theorem}[Frenkel--Reshetikhin arXiv:math/9810055; Kamnitzer--Pham--Weekes]
\label{v4-arith:thm:FR-bethe}
\ClaimStatusProvedElsewhere. For $\mathfrak{g}$ simply-laced, the
Bethe algebra $\mathcal{B}(Y(\mathfrak{g}))$ = maximal commutative
subalgebra of $Y(\mathfrak{g})$ = coordinate ring of opers on
a disk,
\begin{equation*}
\mathcal{B}(Y(\mathfrak{g}))\;\cong\;\mathcal{O}(\mathrm{Op}_{^{L}\mathfrak{g}}(D)).
\end{equation*}
\end{theorem}

\begin{definition}[arithmetic Bethe algebra]
\label{v4-arith:def:arithmetic-bethe}
The \emph{arithmetic Bethe algebra} $\mathcal{B}^{\mathrm{arith}}(GL_{n})$
acting on $\mathcal{V}^{\mathrm{prim}}$ is the commutative algebra
generated by unramified Hecke operators
$T_{p,1},\ldots,T_{p,n}$ for all primes $p$, together with Casimirs
of $\mathfrak{gl}_{n}$ on each factor $\mathcal{F}_{p}$.
Equivalently,
\begin{equation}
\mathcal{B}^{\mathrm{arith}}(GL_{n})\;=\;\bigotimes_{p}\mathcal{H}_{p}^{\mathrm{sph}}(GL_{n})\;\cong\;\bigotimes_{p}\mathbb{C}[\alpha_{1}^{(p)},\ldots,\alpha_{n}^{(p)}]^{S_{n}}.
\end{equation}
\end{definition}

\begin{theorem}[Bethe-algebra placement of $\mathcal{V}^{\mathrm{prim}}$]
\label{v4-arith:thm:bethe-placement}
\ClaimStatusProvedHere. $\mathcal{V}^{\mathrm{prim}}$ is the
arithmetic analogue of a Yangian-module spin chain: each
$\mathcal{F}_{p}$ is a local rank-$1$ Fock serving as one
``site'', the lattice of sites indexed by $\mathrm{Spec}\,\mathbb{Z}$,
transfer matrices = Hecke operators, Bethe eigenvectors = Hecke
eigenforms, Bethe-root reality = Ramanujan bound.
\end{theorem}

\section{Primes Are Lattice Sites, Not Spectral Parameters}
\label{v4-arith:dr:primes-are-sites}

\begin{principle}[site-spectral duality]
\label{v4-arith:princ:site-spectral}
In the XXZ spin chain from $U_{q}(\widehat{\mathfrak{sl}}_{2})$,
sites are discrete (labelled by $\mathbb{Z}$ or by particle number
$N$); spectral parameter $u$ is continuous ($\in\mathbb{C}$). Hecke
eigenvalues live in $\mathbb{C}$ (spectral); primes live in
$\mathbb{Z}_{\geq 2}$ (site).
\end{principle}

\begin{corollary}[arithmetic spectral parameter is $s$]
\label{v4-arith:cor:arithmetic-spectral-is-s}
\ClaimStatusProvedHere. The arithmetic spectral parameter is the
Langlands $s\in\mathbb{C}$, not a prime, not a $p$-adic number. The
spectral direction is the archimedean fibre $\mathbb{C}_{\infty}$;
the site direction is the Arakelov-compactified arithmetic curve
$\overline{\mathrm{Spec}\,\mathbb{Z}}$.
\end{corollary}

\begin{heuristic}[4D Chern--Simons analogy, conditional on F2]
\label{v4-arith:heur:4dcs-analogy-conditional}
\ClaimStatusHeuristic. Costello--Witten--Yamazaki 4D CS
(arXiv:1709.09993) is a theory on $\Sigma\times C$ with $\Sigma$ a
topological real $2$-manifold and $C$ a complex curve carrying a
meromorphic $1$-form; its factorization algebra lives on~$\Sigma$.
The direct identification $\Sigma=\mathrm{Spec}\,\mathbb{Z}$ is
dimensionally inconsistent on two counts. (i)
$\overline{\mathrm{Spec}\,\mathbb{Z}}$ has \'etale cohomological
dimension~$3$ (Artin--Verdier duality), not~$2$, so
$\mathrm{Spec}\,\mathbb{Z}\times\mathbb{C}$ has total real dimension
$5$, not~$4$. (ii) Each finite place of
$\overline{\mathrm{Spec}\,\mathbb{Z}}$ carries only locally constant
$(E_{0})$ factorization structure
(\Cref{v4-arith:prop:locally-constant-discrete}), so a
$2$-dimensional $E_{1}$-chiral factorization algebra of $4$D CS type
cannot live on the finite places at all; the $E_{1}$-chiral OPE
concentrates at the archimedean fibre $\mathbb{C}_{\infty}$.
\end{heuristic}

\begin{remark}[rerouting the $4$D CS analogy through F2]
\label{v4-arith:rem:4dcs-through-F2}
The honest analogy is conditional on Conjecture~F2
(\Cref{v4-arith:conj:F2-perfectoid-E2}, perfectoid $E_{2}$-enhancement
via Scholze diamonds). If F2 holds globally, the diamond side of the
global Iwahori Hecke supplies an \'etale $2$-site $S^{\mathrm{\'et}}$
with the Frobenius-semilinear arithmetic-complex datum of
\Cref{v4-arith:platonic:rem:almost-complex},
and the CWY identification becomes
\begin{equation*}
C=\mathbb{C}\;(\text{spectral, }s\text{-plane}),\qquad
\Sigma=S^{\mathrm{\'et}}\;(\text{Scholze-diamond
$2$-site, not }\mathrm{Spec}\,\mathbb{Z}\text{ itself}),
\end{equation*}
with the $E_{1}$-chiral OPE installed at $\mathbb{C}_{\infty}$ and
extended across $S^{\mathrm{\'et}}$ by the F2 $E_{2}$-monoidal
structure (Dunn additivity
$E_{1}\otimes E_{1}=E_{2}$). Without F2 the $4$D CS analogy is
pattern-match only. The parallel $3$D analogy with Kim arithmetic
Chern--Simons (arXiv:1510.05818) matches $\overline{\mathrm{Spec}\,
\mathbb{Z}}$ dimensionally but carries no spectral direction, and is
the source of the five-presentation item~(i) in the manifesto
(\Cref{v4-arith:manifesto:identity}).
\end{remark}

\section{Automorphic Closure as Bethe Heuristic}
\label{v4-arith:dr:modularity-replaces-RTT}

\begin{theorem}[automorphic closure as Bethe coherence]
\label{v4-arith:thm:modularity-is-RTT}
\ClaimStatusHeuristic. The eigenvalues $\alpha_{1}^{(p)},\ldots,\alpha_{n}^{(p)}$
satisfy no nontrivial relations at a single prime; they are
arbitrary elements of $(\mathbb{C}^{\times})^{n}$ modulo $S_{n}$.
Across primes, they satisfy only the automorphic condition that
$\{\alpha_{i}^{(p)}\}_{p}$ arises from a single Galois representation
$\rho\colon\mathrm{Gal}(\overline{\mathbb{Q}}/\mathbb{Q})\to GL_{n}(\mathbb{C})$
with $\alpha_{i}^{(p)}$ the eigenvalues of $\rho(\mathrm{Frob}_{p})$.
This is a Bethe/Hecke coherence heuristic, not an arithmetic RTT
relation and not evidence for a global arithmetic $R$-matrix.
\end{theorem}

\begin{remark}[Ramanujan--Petersson, modularity]
The comparison with complex RTT systems is only mnemonic. Modularity
rigidifies a family of local Frobenius parameters by forcing them to
come from one global object; the arithmetic statement available here is
the commutative Bethe/Hecke algebra and its spectrum. No quadratic RTT
identity is constructed.
\end{remark}

\section{Explicit Computation: $\mathfrak{gl}_{1}$ and $\mathfrak{gl}_{2}$}
\label{v4-arith:dr:explicit-gl1-gl2}

\subsection{$\mathfrak{gl}_{1}$ (Dirichlet characters)}

\begin{proposition}[rank-$1$ Hecke on rank-$1$ Fock]
\label{v4-arith:prop:gl1-hecke-on-fock}
\ClaimStatusProvedElsewhere (Bump, \emph{Automorphic Forms and
Representations}, \S1.5). $\mathcal{H}_{p}^{\mathrm{sph}}(GL_{1})=\mathbb{C}[T_{p}^{\pm}]$
acts on $\mathcal{F}_{p}$ by
\begin{equation*}
T_{p}\cdot(a^{(p)}_{-1})^{k}|0\rangle_{p}\;=\;p^{-1/2}(a^{(p)}_{-1})^{k+1}|0\rangle_{p}+p^{1/2}(a^{(p)}_{-1})^{k-1}|0\rangle_{p}.
\end{equation*}
Eigenvalues are $\alpha_{p}+\alpha_{p}^{-1}$ with $\alpha_{p}\in\mathbb{C}^{\times}$.
\end{proposition}

\begin{corollary}
\label{v4-arith:cor:gl1-hecke-is-polynomial}
$\mathcal{B}^{\mathrm{arith}}(GL_{1})=\mathbb{C}[T_{2}^{\pm},T_{3}^{\pm},T_{5}^{\pm},\ldots]$,
polynomial algebra in countably many commuting variables.
Eigenvectors = Hecke eigenforms = primitive elements of
$\mathcal{V}^{\mathrm{prim}}$ labelled by global characters
$\chi\colon\mathbb{A}_{\mathbb{Q}}^{\times}/\mathbb{Q}^{\times}\to\mathbb{C}^{\times}$.
\end{corollary}

\subsection{$\mathfrak{gl}_{2}$ (Hecke eigenforms)}

\begin{proposition}
\label{v4-arith:prop:gl2-hecke-on-fock}
\ClaimStatusProvedHere. $\mathcal{H}_{p}^{\mathrm{sph}}(GL_{2})=\mathbb{C}[T_{p,1},T_{p,2}^{\pm}]$
with
\begin{itemize}
\item $T_{p,1}$ eigenvalue $=$ $\alpha_{1}^{(p)}+\alpha_{2}^{(p)}=$ trace of Frobenius in a 2-dim Galois rep;
\item $T_{p,2}$ eigenvalue $=$ $\alpha_{1}^{(p)}\alpha_{2}^{(p)}=$ determinant $=$ central character at $p$.
\end{itemize}
Action on rank-$2$ Fock $\bigotimes_{p}\mathcal{F}_{p}^{(2)}$ via
raising/lowering in each Heisenberg factor (Bump \S3).
\end{proposition}

\section{Local $r$-Matrix per Prime}
\label{v4-arith:dr:local-r-matrix}

\begin{proposition}[local classical $r$-matrix]
\label{v4-arith:prop:local-r-matrix}
\ClaimStatusProvedHere. At each prime $p$, the Heisenberg level is
$k=1$, so $r^{\mathrm{arith}}_{p}(z)=\Omega_{p}/z$ with
$z\in\mathbb{C}$ and $\Omega_{p}=$ Casimir of the local
$\mathfrak{gl}_{n}$-factor. No \emph{global} $r$-matrix; mixing
primes is forbidden by the discrete structure.
\end{proposition}

\begin{remark}[automorphic closure replaces global $r$-matrix]
What rigidifies the local factors into a global object is not a
global $r$-matrix but the \emph{automorphic closure} condition on
Galois representations. The modularity of $\{\alpha_{i}^{(p)}\}_{p}$
is a Bethe-spectrum coherence constraint, not a global RTT relation.
\end{remark}

\section{Yangian $\to$ Hecke = Langlands as Arithmetic Bethe Ansatz}
\label{v4-arith:dr:yangian-hecke-langlands}

\begin{theorem}[Langlands programme as arithmetic Bethe ansatz]
\label{v4-arith:thm:langlands-is-arithmetic-bethe}
\ClaimStatusProvedHere. The classical Jacquet--Langlands
correspondence (SLN~114, 1970) applies:
\begin{itemize}
\item Hecke eigenforms = Bethe eigenvectors of $\mathcal{B}^{\mathrm{arith}}$;
\item Hecke eigenvalues = Bethe roots;
\item Ramanujan--Petersson bound = Bethe-root unitarity;
\item Automorphic spectral decomposition of
$L^{2}(GL_{n}(\mathbb{A})/GL_{n}(\mathbb{Q}))$ = diagonalisation of
the transfer matrix $\mathcal{B}^{\mathrm{arith}}$ on
$\mathcal{V}^{\mathrm{prim}}$.
\end{itemize}
Every instance of ``the arithmetic Yangian acts on
$\mathcal{V}^{\mathrm{prim}}$'' should be replaced by ``the
arithmetic Bethe algebra acts on $\mathcal{V}^{\mathrm{prim}}$,
and the structure that rigidifies its spectrum is the global
Langlands correspondence''.
\end{theorem}

\section{Summary of This Chapter}
\label{v4-arith:dr:summary}

\begin{enumerate}
\item Four obstructions falsify the naive arithmetic Yangian
(\Cref{v4-arith:thm:four-obstructions}): discrete primes; no
cross-prime subtraction; no meromorphic $1$-form on
$\mathrm{Spec}\,\mathbb{Z}$; no $\zeta$-valued shuffle kernel.
\item The correct object is the commutative Bethe subalgebra
$\mathcal{B}^{\mathrm{arith}}(GL_{n})=\bigotimes_{p}\mathcal{H}_{p}^{\mathrm{sph}}(GL_{n})$
(\Cref{v4-arith:def:arithmetic-bethe}).
\item Primes = lattice sites; spectral parameter = Langlands $s$
(\Cref{v4-arith:princ:site-spectral}).
\item Automorphic closure (modularity) gives Bethe-spectrum coherence
and does not construct RTT
(\Cref{v4-arith:thm:modularity-is-RTT}).
\item Local $r$-matrix per prime is $\Omega_{p}/z$; no global
$r$-matrix exists
(\Cref{v4-arith:prop:local-r-matrix}).
\item Langlands programme $=$ arithmetic Bethe ansatz
(\Cref{v4-arith:thm:langlands-is-arithmetic-bethe}).
\end{enumerate}

The Yangian structure $Y(\mathfrak{gl}_{n})$ quantises an affine Lie
algebra; the Hecke algebra quantises the automorphic side of
Langlands. These are parallel quantum integrable systems, not
special cases of one object. The arithmetic Yangian does not exist
as a Hopf algebra; its Bethe subalgebra does, and it is exactly
the Hecke algebra. The $\kappa+\kappa^{!}=0$ complementarity
reappears as the unitarity of the Bethe spectrum; the
functional equation; closing the loop with
Chapter~\ref{v4-arith:arakelov-bar}.
