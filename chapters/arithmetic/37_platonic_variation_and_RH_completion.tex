\chapter{The Platonic Variation and RH Completion}
\label{v4-arith:platonic-rh-completion}

\emph{The preceding chapters isolate the arithmetic branch's correct
geometric variation. The base is not a scheme-theoretic surface and
not a literal almost-complex manifold at finite primes. It is an
operadic adelic surface: holomorphic at $\infty$, Frobenius-oriented
at finite primes, and glued by the pointed $E_{2}$ restricted tensor
product of Iwahori Hecke categories. This chapter records the
upgraded object and the exact theorem by which it drives the
Riemann Hypothesis completion inside Vol~IV.}

\section{The Right Variation}
\label{v4-arith:platonic:right-variation}

\begin{definition}[Platonic $2$-arithmetic surface]
\label{v4-arith:platonic:def:S2}
\ClaimStatusDefinitional. A \emph{Platonic $2$-arithmetic surface}
for the primary arithmetic branch is the stratified adelic-diamond
factorisation datum
\begin{equation}
\label{v4-arith:platonic:eq:S2}
\mathfrak{S}^{\mathrm{Ar}}_{2}
\;:=\;
\bigl(|\overline{C}|,\{S_{v},\mathfrak{J}^{\mathrm{Ar}}_{v},\mathcal{C}_{v},\eta_{v},u_{v}\}_{v},
\bigotimes_{v\in|\overline{C}|}^{\prime}(\mathcal{C}_{v},u_{v})\bigr),
\end{equation}
where:
\begin{itemize}
\item $|\overline{C}|=\mathbb{P}\sqcup\{\infty\}$ is the set of
places of $\overline{\mathrm{Spec}\,\mathbb{Z}}$;
\item $S_{\infty}=\mathbb{C}_{\infty}$,
$\mathfrak{J}^{\mathrm{Ar}}_{\infty}=i$, and
$\mathcal{C}_{\infty}$ is the archimedean holomorphic
factorisation category with Harish--Chandra
$K_{\infty}$-spherical unit $u_{\infty}$;
\item for $p<\infty$, $S_{p}$ is the Fargues--Scholze
perfectoid-diamond Hecke chart, $\mathfrak{J}^{\mathrm{Ar}}_{p}$
is the Frobenius-semilinear untilt-orientation with Adams
operation $\psi_{p}$ and prismatic/Nygaard comparison datum, and
$\mathcal{C}_{p}=\mathrm{Hk}^{\mathrm{Iw}}_{GL_{2},p}$ is the full
non-semisimple Iwahori Hecke category with spherical unit
$u_{p}=|0\rangle_{p}$;
\item $\eta_{v}\colon\mathrm{Sp}\to\mathcal{C}_{v}$ is the pointed
$E_{2}$-algebra map with image $u_{v}$; at almost every finite place
$u_{v}$ is the monoidal unit, and at the remaining places it is a
compact central idempotent algebra object;
\item the tensor product in~\eqref{v4-arith:platonic:eq:S2} is the
Tate restricted tensor product of pointed $E_{2}$-monoidal stable
presentable $\infty$-categories in the sense of
\Cref{v4-arith:tate:setup:data}.
\end{itemize}
The word ``surface'' refers to the operadic dimension: the global
category is $E_{2}$-monoidal. It does not assert a smooth
scheme-theoretic surface over $\mathbb{Z}$.
\end{definition}

\begin{remark}[almost complex versus arithmetic complex]
\label{v4-arith:platonic:rem:almost-complex}
The symbol $\mathfrak{J}^{\mathrm{Ar}}$ is literal almost-complex
geometry only on $S_{\infty}=\mathbb{C}_{\infty}$. At a finite prime
it is the Frobenius-oriented substitute: a semilinear orientation of
the diamond chart together with Adams and prismatic comparison data.
The finite place supplies no tangent-bundle endomorphism satisfying
$J^{2}=-1$; its role is to make the $E_{2}$ Iwahori Hecke datum
functorial.
This convention governs every occurrence of
$\mathfrak{J}^{\mathrm{Ar}}$ in the arithmetic branch.
\end{remark}

\begin{remark}[source anchors and dependency separation]
\label{v4-arith:platonic:rem:sources}
The finite charts use Scholze diamonds and Fargues--Scholze local
Langlands geometry as catalogued in
\Cref{v4-arith:app:A:langlands} and
\Cref{v4-arith:app:A:prismatic}. The
archimedean factorisation $D$-module fibre is the
Beilinson--Drinfeld/Francis--Gaitsgory/Costello--Gwilliam fibre of
\Cref{v4-arith:app:A:beilinson-etc} and
\Cref{v4-arith:app:A:costello}. The
arithmetic $\mathcal{D}^{\dagger}$ and overconvergent
$F$-isocrystal language is only a local coefficient shadow, with
primary sources recorded in
\Cref{v4-arith:app:A:arith-Ddagger}. F3 and the wild-prime BZN
comparison affect the identification of
$\mathcal{V}^{\mathrm{prim}}$; they do not alter the definition of
$\mathfrak{S}^{\mathrm{Ar}}_{2}$.
\end{remark}

\begin{theorem}[upgrade of the aCFT base]
\label{v4-arith:platonic:thm:base-upgrade}
\ClaimStatusProvedHere. Replacing the base datum (B) of
\Cref{v4-arith:def:aCFT} by
$\mathfrak{S}^{\mathrm{Ar}}_{2}$ preserves the primary aCFT
inscription and upgrades its finite-place geometry: the archimedean
fibre remains genuinely chiral, each finite fibre remains
locally-constant with Adams/Frobenius structure, and the global
object remains $E_{2}$-monoidal by the restricted tensor product
construction.
\end{theorem}

\begin{proof}
At $\infty$ the datum is unchanged: it is the holomorphic
factorisation algebra on $\mathbb{C}_{\infty}$ of
\Cref{v4-arith:def:CG-arithmetic-factorization}. At finite $p$,
\Cref{v4-arith:thm:arch-vs-finite-asymmetry} and
\Cref{v4-arith:prop:locally-constant-discrete} identify the local
factorisation structure as $E_{0}$ with Adams operation; the
Frobenius-semilinear orientation records exactly this finite-place
arithmetic data. The global $E_{2}$ structure is the pointed
restricted tensor product of \Cref{v4-arith:tate:cor:F2-closed}.
Thus all axioms of \Cref{v4-arith:def:aCFT} survive, with the finite
geometry expressed by its correct arithmetic substitute.
\end{proof}

\section{Factorisable Arithmetic Modules}
\label{v4-arith:platonic:fact-modules}

\begin{definition}[factorisable arithmetic crystal]
\label{v4-arith:platonic:def:fact-crystal}
\ClaimStatusDefinitional. A \emph{factorisable arithmetic
crystal} on $\mathfrak{S}^{\mathrm{Ar}}_{2}$ is a functor
\begin{equation}
\label{v4-arith:platonic:eq:fact-crystal}
\mathscr{D}^{\mathrm{Ar}}\colon
\mathrm{Fin}(|\overline{C}|)^{\mathrm{op}}
\longrightarrow
\mathrm{Alg}_{\mathcal{O}}(\mathrm{Pr}^{\mathrm{L}}_{\mathrm{st}})
\end{equation}
such that
\[
\mathscr{D}^{\mathrm{Ar}}(S)\simeq\bigotimes_{v\in S}\mathcal{C}_{v},
\qquad
\mathscr{D}^{\mathrm{Ar}}(S_{1}\sqcup S_{2})
\simeq
\mathscr{D}^{\mathrm{Ar}}(S_{1})\otimes
\mathscr{D}^{\mathrm{Ar}}(S_{2})
\]
for disjoint finite subsets $S_{1},S_{2}\subset|\overline{C}|$,
with transition maps given by insertion of the spherical units
$u_{v}$. The archimedean value is a genuine holomorphic
factorisation $D$-module in the Beilinson--Drinfeld sense. The
finite-place value is an arithmetic coefficient category: an
Iwahori Hecke category on a diamond chart, or on the spectral side
an ind-coherent category with nilpotent singular support; in
differential language it may be represented by overconvergent
$F$-isocrystal or arithmetic $\mathcal{D}^{\dagger}$ data on a
$p$-adic formal model.
Here ``represented'' means local coefficient shadow on a chosen
smooth $p$-adic formal chart. The definition is on the
Hecke/diamond side; the global identification with
$\mathrm{IndCoh}_{\mathrm{Nilp}}(\mathcal{X}^{\mathrm{Sel}}_{\bar\rho,\det,\mathcal{L}})$
remains the F1/LACD/BC-diamond-glue comparison, not a definitional
equivalence.
\end{definition}

\begin{proposition}[no loss of the classical $D$-module intuition]
\label{v4-arith:platonic:prop:Dmodule-upgrade}
\ClaimStatusProvedHere. The factorisable arithmetic crystal of
\Cref{v4-arith:platonic:def:fact-crystal} contains the classical
factorisation $D$-module picture as its archimedean fibre and
contains the finite-prime arithmetic replacement as Frobenius
crystalline/diamond coefficient data. Hence it is the correct
upgrade of ``factorisable $D$-module'' for
$\mathfrak{S}^{\mathrm{Ar}}_{2}$.
\end{proposition}

\begin{proof}
Beilinson--Drinfeld factorisation $D$-modules require a smooth
curve over a field. This hypothesis is satisfied by the
archimedean curve $\mathbb{C}_{\infty}$ and fails at
$\mathrm{Spd}\,\mathbb{Q}_{p}$ as a classical $D$-module base. At
finite $p$ the available primary objects are the perfectoid
Iwahori Hecke category and its spectral ind-coherent avatar; these
are exactly the values $\mathcal{C}_{p}$ in
\Cref{v4-arith:platonic:def:S2}. The functorial factorisation law
is the discrete Ran law of
\Cref{v4-arith:tate:thm:BD-automatic-discrete}. At $p=2,3$ no
semisimple truncation is taken; $\mathcal{C}_{p}$ includes the wild
Ext contribution of
\Cref{v4-arith:bzn-rcompletion:thm:F3-BZN-Ar-corrected}.
\end{proof}

\section{The Arithmetic-Chiral Bar Complex on the Variation}
\label{v4-arith:platonic:bar}

\begin{definition}[Platonic arithmetic-chiral bar complex]
\label{v4-arith:platonic:def:bar}
\ClaimStatusDefinitional. For a primary object
$\mathcal{V}^{\mathrm{prim}}$ on
$\mathfrak{S}^{\mathrm{Ar}}_{2}$ let
$\mathrm{Fin}_{\mathrm{rec}}(|\overline C|)$ be the filtered poset
of finite sets $S=S_{f}\sqcup\{\infty\}$ containing the ramified
places and closed under the Parshin--Gersten boundary maps of the
symbols appearing in the stage. Write
$C^{\bullet}_{\mathrm{PG},S}(\overline C)$ for the resulting finite
Parshin--Gersten complex, with omitted-place boundary terms killed by
the inserted spherical units. Define
\begin{equation}
\label{v4-arith:platonic:eq:finite-bar}
B^{\mathrm{Ar}}_{S}(\mathcal{V}^{\mathrm{prim}})
\;:=\;
B^{\mathrm{ch}}(\mathcal{V}_{\infty}^{\mathrm{prim}})
\widehat{\otimes}
C^{\bullet}_{\mathrm{PG},S}(\overline C)
\widehat{\otimes}
\widehat{\bigotimes}_{p\in S_{f}}
B^{\Lambda_{p}}_{E_{1}}(\mathcal{V}_{p}^{\mathrm{prim}}),
\end{equation}
with transition maps inserting the closed spherical vector at new
finite places. The global arithmetic-chiral bar is
\[
B^{\mathrm{Ar}}(\mathcal{V}^{\mathrm{prim}})
\;:=\;
\operatorname*{colim}_{S\in\mathrm{Fin}_{\mathrm{rec}}(|\overline C|)}
B^{\mathrm{Ar}}_{S}(\mathcal{V}^{\mathrm{prim}}).
\]
Choose an order $\infty<p_{1}<\cdots<p_{r}$ on $S$. Its finite-stage
differential is the signed total differential
\begin{equation}
\label{v4-arith:platonic:eq:bar-differential}
D_{S}\;:=\;d_{\infty}+(-1)^{|\cdot|_{\infty}}d_{\mathrm{PG},S}
\;+\;\sum_{j=1}^{r}(-1)^{|\cdot|_{\infty}+|\cdot|_{\mathrm{PG}}+
\sum_{i<j}|\cdot|_{p_{i}}}\,d_{p_{j}},
\end{equation}
where
\begin{equation}
\label{v4-arith:platonic:eq:bar-local-differentials}
d_{\infty}=d_{\mathrm{int}}+d_{\mathrm{res}}+d_{\mathrm{dR}},
\qquad
d_{p}=d_{\Lambda,\mathrm{int}}+\iota_{p}(\tau_{S})
+d_{\mathrm{bar},p}^{\mathrm{loc}}.
\end{equation}
Here $\tau_{S}$ is the Kato--Parshin tame-symbol twisting cochain.
The finite residue term is the Gersten part of $\tau_{S}$; it is not
a copied complex-analytic $d\log(p-q)$ operator. The signed
totalization is independent of the chosen order by the symmetric
braiding.
\end{definition}

\begin{lemma}[closed spherical units]
\label{v4-arith:platonic:lem:closed-units}
\ClaimStatusProvedHere. For every finite place $p$, the spherical
unit $u_{p}$ is closed for the local arithmetic bar differential:
$d_{p}(u_{p})=0$.
\end{lemma}

\begin{proof}
The internal differential kills the monoidal unit. The
Parshin--Gersten term kills it because the tame symbol of the unit is
zero. The local bar differential kills it by the augmentation-unit
identity in the $E_{1}$ bar construction. At $p=2,3$ this is computed
inside the full non-semisimple Iwahori category; the wild Ext summand
changes the trace category, not the closedness of the unit.
\end{proof}

\begin{proposition}[finite-stage strictification]
\label{v4-arith:platonic:prop:finite-stage-strictification}
\ClaimStatusProvedHere. For every
$S\in\mathrm{Fin}_{\mathrm{rec}}(|\overline C|)$, the twisting cochain
$\tau_{S}$ satisfies the Maurer--Cartan equation
\[
d_{\mathrm{PG},S}\tau_{S}+\tau_{S}d_{\mathrm{PG},S}
\;+\;\frac{1}{2}[\tau_{S},\tau_{S}]=0
\]
in the completed convolution Lie algebra of the finite-stage
Parshin--Gersten complex. Hence the signed total differential
\eqref{v4-arith:platonic:eq:bar-differential} satisfies
$D_{S}^{2}=0$ on the chain level.
\end{proposition}

\begin{proof}
The Gersten differential squares to zero by the Gersten resolution.
The displayed Maurer--Cartan equation is exactly the
Kato--Parshin reciprocity law for the tame-symbol boundary, restricted
to the reciprocity-closed stage $S$. Boundaries at omitted places are
zero because new places enter through the closed spherical units of
\Cref{v4-arith:platonic:lem:closed-units}. The Koszul signs in
\eqref{v4-arith:platonic:eq:bar-differential} are the standard tensor
totalization signs; after inserting them, the cross terms of
$D_{S}^{2}$ are precisely the three terms of the Maurer--Cartan
equation. Thus $D_{S}^{2}=0$ strictly, not merely after passing to
cohomology.
\end{proof}

\begin{proposition}[comparison with the Parshin--Gersten bar]
\label{v4-arith:platonic:prop:bar-comparison}
\ClaimStatusProvedHere. The arithmetic bar complex of
\Cref{v4-arith:def:arith-bar-concrete} is the finite-support
presentation of the colimit
$B^{\mathrm{Ar}}(\mathcal{V}^{\mathrm{prim}})$ after replacing the
single Iwasawa algebra $\Lambda$ by the finite-stage adelic system
$\{\Lambda_{p}\}_{p\in S_{f}}$ and totalizing with
\eqref{v4-arith:platonic:eq:bar-differential}.
\end{proposition}

\begin{proof}
Every element of the Parshin--Gersten complex has finite support in
the set of places. It therefore lies in some
$C^{\bullet}_{\mathrm{PG},S}(\overline C)$ with
$S\in\mathrm{Fin}_{\mathrm{rec}}(|\overline C|)$. The transition maps
insert closed spherical units, so the finite-support union is the
filtered colimit. The $\Lambda$-action of
\Cref{v4-arith:def:arith-bar-concrete} decomposes on a finite stage as
the completed tensor product of the local $\Lambda_{p}$-actions. This
identifies the older global presentation with the restricted tensor
presentation.
\end{proof}

\begin{theorem}[bar differential closure on the primary scope]
\label{v4-arith:platonic:thm:bar-closure}
\ClaimStatusProvedHere. On the abelian primary Heisenberg scope,
the differential~\eqref{v4-arith:platonic:eq:bar-differential}
satisfies $D_{S}^{2}=0$ at every reciprocity-closed finite stage and
is compatible with the transition maps. Therefore
$B^{\mathrm{Ar}}(\mathcal{V}^{\mathrm{prim}})$ is a well-defined
arithmetic-chiral bar complex on
$\mathfrak{S}^{\mathrm{Ar}}_{2}$.
\end{theorem}

\begin{proof}
The archimedean summand squares to zero by the Vol~I chiral bar
construction and Vol~II's Costello--Gwilliam comparison. At finite
places, \Cref{v4-arith:platonic:prop:finite-stage-strictification}
gives the chain-level square-zero statement. Local-global
factorisation compatibility
\Cref{v4-arith:pos-costello:cor:C1-closed}, pro-nilpotent convergence
\Cref{v4-arith:pos-costello:thm:C3-closed}, the archimedean
classical-limit check \Cref{v4-arith:pos-costello:thm:C4-closed}, and
functoriality \Cref{v4-arith:pos-costello:thm:C5-closed} supply the
ambient completed coalgebra and functoriality. Transition maps are
chain maps by \Cref{v4-arith:platonic:lem:closed-units}. The
comparison with the older Parshin--Gersten model is
\Cref{v4-arith:platonic:prop:bar-comparison}.
\end{proof}

\section{RH Completion}
\label{v4-arith:platonic:RH-completion}

\begin{definition}[RH completion package]
\label{v4-arith:platonic:def:RH-package}
\ClaimStatusDefinitional. The \emph{Vol~IV RH completion package}
is the conjunction of:
\begin{itemize}
\item the Platonic surface $\mathfrak{S}^{\mathrm{Ar}}_{2}$ of
\Cref{v4-arith:platonic:def:S2};
\item the primary arithmetic-chiral bar complex of
\Cref{v4-arith:platonic:def:bar};
\item the arithmetic Heisenberg Koszul biconditional, conditional on
the arithmetic Positselski completion input of
\Cref{v4-arith:pos-costello:cor:arith-Heis-Koszul};
\item the arithmetic Petersson positive-form package, conditional on
the scoped Koszul--Petersson compatibility input recorded in
\Cref{v4-arith:kp:cor:P3-unconditional};
\item H-P$^{\mathrm{Ar}}$ in the equivalent spectral-realisation
form of \Cref{v4-arith:hpvbconv:thm:HPAr-equivalence};
\item the positive Stieltjes--Hamburger spectral package:
VB$^{*}$ on the primary Fock, equivalently the strengthened
HB+/BD+ forms of
\Cref{v4-arith:hpvbconv:thm:VBstar-converse-spectral-regularity}.
\end{itemize}
\end{definition}

\begin{theorem}[Platonic RH completion]
\label{v4-arith:platonic:thm:RH-completion}
\ClaimStatusProvedHere. If the Vol~IV RH completion package of
\Cref{v4-arith:platonic:def:RH-package} is realised, then the
Riemann Hypothesis holds.
\end{theorem}

\begin{proof}
The surface and bar parts give the correct primary arithmetic
chiral-algebra object. The conditional Koszul input gives P1 in the
abelian Heisenberg scope
(\Cref{v4-arith:pos-costello:cor:arith-Heis-Koszul}); P2 is Tate's
functional-equation/Zhu-modularity input from
\Cref{v4-arith:thm:P2-resolution}; the conditional Petersson package
supplies the positive Hermitian form needed for P3 in the scoped
arithmetic branch
\Cref{v4-arith:kp:cor:P3-unconditional}. H-P$^{\mathrm{Ar}}$
realises the self-adjoint spectral operator with the required
Bost--Connes trace. The positive Stieltjes--Hamburger package gives
VB$^{*}$ on the primary Fock; determinacy alone is not the positivity
input. Then
\Cref{v4-arith:rh:thm:VB-implies-F-RH} gives F-RH, and
\Cref{v4-arith:rh:rem:equivalence-to-RH} identifies F-RH with RH.
\end{proof}

\begin{corollary}[single remaining spectral-positivity theorem]
\label{v4-arith:platonic:cor:single-positivity}
\ClaimStatusProvedHere. After the upgrades of this chapter and after
the conditional P1/P3 inputs have been supplied, the full RH route may
be stated as one spectral-positivity theorem:
the Bost--Connes spectral realisation of
\Cref{v4-arith:hpvbconv:thm:HPAr-equivalence} has a positive
Stieltjes--Hamburger zero measure on the primary Fock and is
determinate; equivalently the strengthened HB+ or BD+ condition
holds. This theorem is sufficient, together with P1, P2, and P3 in
the stated conditional scopes, to prove RH through
\Cref{v4-arith:platonic:thm:RH-completion}.
\end{corollary}

\begin{proof}
H-P$^{\mathrm{Ar}}$ supplies the operator and trace. The stated
positive Stieltjes--Hamburger package supplies the missing
operator-positivity by
\Cref{v4-arith:hpvbconv:thm:VBstar-converse-spectral-regularity}.
The remaining implication is
\Cref{v4-arith:platonic:thm:RH-completion}.
\end{proof}

\section{Summary}
\label{v4-arith:platonic:summary}

\begin{enumerate}
\item The correct surface is $\mathfrak{S}^{\mathrm{Ar}}_{2}$:
archimedean complex geometry plus finite Frobenius-diamond
orientation, glued as a pointed $E_{2}$ restricted tensor product.
\item The correct factorisable module is a factorisable arithmetic
crystal: classical factorisation $D$-module at $\infty$,
diamond/ind-coherent/$F$-isocrystal coefficient data at finite
places.
\item The correct arithmetic-chiral bar complex is the filtered
colimit of finite-stage bars
\eqref{v4-arith:platonic:eq:finite-bar} with
Gersten/Parshin finite residue terms.
\item The RH completion is the conditional theorem
\Cref{v4-arith:platonic:thm:RH-completion}: P1/P3 in their scoped
conditional forms, H-P$^{\mathrm{Ar}}$, and positive
determinacy/HB+/BD+ drive the primary arithmetic aCFT to RH.
\end{enumerate}
