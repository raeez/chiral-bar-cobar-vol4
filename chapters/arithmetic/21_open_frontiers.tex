\section{F1: Global Arithmetic Arinkin--Gaitsgory}
\label{v4-arith:of:F1}

\begin{conjecture}[global arithmetic AG]
\label{v4-arith:conj:F1-global-AG}
For $G=GL_{2}$, there is an equivalence of dg-categories
\begin{equation}
\mathrm{D}([GL_{2}])^{\mathrm{sph\text{-}fin}}\;\simeq\;\mathrm{IndCoh}_{\mathrm{Nilp}}\bigl(\mathcal{X}^{\mathrm{EG,glob}}_{GL_{2}}\bigr),
\end{equation}
where:
\begin{itemize}
\item $[GL_{2}]:=GL_{2}(\mathbb{Q})\backslash GL_{2}(\mathbb{A}_{\mathbb{Q}})/Z(\mathbb{A})K_{\infty}$
is the global automorphic double coset;
\item $\mathcal{X}^{\mathrm{EG,glob}}_{GL_{2}}=\varinjlim_{S}\mathrm{Spf}(R_{\bar\rho,S})^{\mathrm{rig}}$
is the global Emerton--Gee stack of Galois deformations
(Emerton--Gee arXiv:1908.07185);
\item $\mathrm{IndCoh}_{\mathrm{Nilp}}$ denotes ind-coherent sheaves
with nilpotent singular support (Arinkin--Gaitsgory arXiv:1201.6343).
\end{itemize}
Under the equivalence, $\mathcal{V}^{\mathrm{prim}}$ corresponds to
the structure sheaf $\mathcal{O}$ of the Emerton--Gee stack.
\end{conjecture}

\begin{remark}[status]
\textbf{Local theorem-grade inputs, not the displayed equivalence.}
Fargues--Scholze arXiv:2102.13459 construct the local geometrisation,
spectral action, and Langlands-parameter package via the
Fargues--Fontaine curve and diamonds. Emerton--Gee arXiv:1908.07185
construct the local $(\varphi,\Gamma)$-module stacks. These sources
do not prove the categorical equivalence displayed in F1.
Dotto--Emerton--Gee arXiv:2603.26887 gives a fully faithful
$GL_{2}(\mathbb{Q}_{p})$ functor for $p\geq 5$ with fixed central
character; it is not an equivalence. Caraiani--Scholze arXiv:1710.10543
prove a unitary-Shimura-variety result in a different global setting.

\textbf{Conjecture globally} for $GL_{2}/\mathbb{Q}$ in full
generality. The local categorical comparison and the global gluing
across primes require perfectoid compatibility that is not yet fully
worked out.

\textbf{What a proof would accomplish.} Realize $\mathcal{V}^{\mathrm{prim}}$
as a natural object on the spectral side of arithmetic Langlands,
connecting its Hochschild-trace presentation (Chapter~\ref{v4-arith:hochschild-trace})
to the Galois-deformation side.
\end{remark}

\section{F2: $E_{2}$-Enhancement via Scholze Diamonds}
\label{v4-arith:of:F2}

\begin{conjecture}[perfectoid $E_{2}$-enhancement]
\label{v4-arith:conj:F2-perfectoid-E2}
The local Iwahori Hecke category $\mathrm{Hk}^{\mathrm{Iw}}_{GL_{2},p}$
carries a canonical $E_{2}$-monoidal structure coming from the
Beilinson--Drinfeld Grassmannian over $\mathrm{Spec}\,\mathbb{Z}_{p}$
in Scholze's perfectoid (diamond) setting. Globally, the product
$\bigotimes'_{p}\mathrm{Hk}^{\mathrm{Iw}}_{GL_{2},p}$ inherits a
well-defined $E_{2}$-monoidal structure.
\end{conjecture}

\begin{remark}[role of this conjecture]
Dunn additivity $E_{1}\otimes E_{1}=E_{2}$ applied to the
$E_{2}$-monoidal structure gives the $E_{1}$-algebra structure on
the Hochschild trace $\mathrm{HH}_{\bullet}(\mathrm{Hk}^{\mathrm{Iw}}_{GL_{2},\mathrm{glob}})$.
Without perfectoid $E_{2}$-enhancement, the $E_{1}$ Hochschild-trace
structure on $\mathcal{V}^{\mathrm{prim}}$ is not available at the
categorical-trace level.
\end{remark}

\begin{remark}[status]
\textbf{Theorem locally.} The Beilinson--Drinfeld Grassmannian as
a perfectoid space is constructed in Fargues--Scholze
arXiv:2102.13459; the $E_{2}$-monoidal structure is a known
consequence. Zhu's Witt-vector Grassmannian (arXiv:1603.05593;
Bhatt--Scholze arXiv:1507.06490) gives the equal-characteristic
analog.

\textbf{Conjecture globally}: the Scholze-diamond gluing across
primes, producing a restricted-tensor-product $E_{2}$-category, is
open.
\end{remark}

\section{F3: Arithmetic Ben-Zvi--Nadler}
\label{v4-arith:of:F3}

\begin{conjecture}[arithmetic Ben-Zvi--Nadler]
\label{v4-arith:conj:F3-arith-BZN}
There is an equivalence of $E_{1}$-algebras
\begin{equation}
\mathrm{HH}_{\bullet}\bigl(\mathrm{Hk}^{\mathrm{Iw}}_{GL_{2},\mathrm{glob}}\bigr)\;\xrightarrow{\sim}\;\mathcal{V}^{\mathrm{prim}}
\end{equation}
realising the CS-Hecke-trace-boundary triangle arithmetically:
$\mathcal{V}^{\mathrm{prim}}=\partial\,\mathrm{ACS}=\mathrm{HH}_{\bullet}(\mathrm{Hk}^{\mathrm{Iw}}_{\mathrm{glob}})$
as a boundary Hochschild-trace/factorisation class.
\end{conjecture}

\begin{remark}[status]
\textbf{Theorem over $\mathbb{C}$.} Ben-Zvi--Nadler
arXiv:0904.1247 establishes
$\mathrm{HH}_{\bullet}(\mathrm{D}(G/G))=\mathcal{O}(\mathrm{LocSys}_{G}(S^{1}))^{\mathrm{flat}}$
for complex reductive $G$.

\textbf{Conjecture over $\mathbb{Z}$.} The arithmetic adaptation
requires:
\begin{enumerate}
\item Conjecture F1 (global arithmetic AG);
\item Conjecture F2 (perfectoid $E_{2}$-enhancement);
\item Explicit computation of the Hochschild trace on arithmetic
$\mathrm{Hk}^{\mathrm{Iw}}_{\mathrm{glob}}$ matching $\mathcal{V}^{\mathrm{prim}}$'s
$L$-function character.
\end{enumerate}

\textbf{What F3 accomplishes.} Unifies Gaiotto's CS presentation
(\Cref{v4-arith:thm:arith-CS-VA-correspondence}) with
Bezrukavnikov--Bernstein's Hochschild presentation
(\Cref{v4-arith:thm:crowning-placement}) with Beilinson-solo's
bar-cobar presentation (\Cref{v4-arith:thm:bar-verifications}) via
a single arithmetic Ben-Zvi--Nadler equivalence.
\end{remark}

\subsection{F4: $\mathsf{B}$-Row Arithmetic Realisation}
\label{v4-arith:of:F4-B-row}

\begin{conjecture}[arithmetic $\mathsf{B}$-row ceiling]
\label{v4-arith:conj:F4-B-row}
There exists an arithmetic avatar of the Vol~III Mukai-enhanced
K3 Heisenberg such that $\kappa^{\mathrm{Ar}}+(\kappa^{!})^{\mathrm{Ar}}=8$
arithmetically. The witness is a Bruinier Heegner Chern-class
reciprocity on a $p$-adic Shimura variety, at each prime $p$ of
good reduction.
\end{conjecture}

\begin{remark}
Ingredients available: Kudla--Rapoport arithmetic intersection
numbers on unitary Shimura varieties; Bruinier's singular-theta
lifts; Borcherds products on $\mathrm{Sp}_{4}(\mathbb{Z})$. Open:
integral assembly into an arithmetic $\mathsf{B}$-row theorem.
\end{remark}

\subsection{F5: Arithmetic Chain Homotopy $h^{\mathrm{Ar}}_{A}$}
\label{v4-arith:of:F5-chain-homotopy}

\begin{conjecture}[universal arithmetic chain homotopy]
\label{v4-arith:conj:F5-chain-homotopy}
There exists a universal chain homotopy $h^{\mathrm{Ar}}_{A}$ on
the arithmetic bar complex $B^{\mathrm{ch,Ar}}(\mathcal{V}^{\mathrm{prim}})$
satisfying
\begin{equation}
h^{\mathrm{Ar}}_{A}\;=\;\frac{h^{\mathrm{LV}}_{A}}{\kappa+\kappa^{!}}
\end{equation}
(where $h^{\mathrm{LV}}_{A}$ is the Losev--Vol~III universal
chain homotopy), with obstruction class $\mathrm{obs}^{(1)}\in H^{2}(\mathfrak{grt}^{\mathrm{ell}},\mathfrak{sp}\otimes\mathfrak{sp})$
vanishing iff $r_{\max}=2$.
\end{conjecture}

\subsection{F6: Chenevier Determinant at $p=691$}
\label{v4-arith:of:F6-chenevier-691}

\begin{conjecture}[Chenevier determinant identity]
\label{v4-arith:conj:F6-chenevier-691}
At $p=691$ (Ramanujan $\Delta_{12}$ Eisenstein congruence),
\begin{equation}
\det_{\mathrm{Chenevier}}\bigr|_{p=691}\;=\;\chi_{\mathrm{cyc}}^{11}\cdot\varepsilon_{B_{12}},
\end{equation}
where $\varepsilon_{B_{12}}$ is the mod-$691$ Kummer character
attached to the numerator of $B_{12}$.
\end{conjecture}

\begin{remark}[connection]
This is the arithmetic avatar of the Chenevier determinant
appearing in the Vol~III universal trace identity. The prime
$p=691$ is the seed of the arithmetic $\mathsf{B}$-row structure
(Chapter~\ref{v4-arith:hochschild-trace},
\Cref{v4-arith:thm:crowning-placement}).
\end{remark}

\section{Dependency Graph}
\label{v4-arith:of:dependency-graph}

The three closing conjectures F1, F2, F3 are hierarchically
ordered:
\begin{center}
F1 $\to$ F2 $\to$ F3 $\to$ (all six-window presentations synthesise to a single theorem)
\end{center}
F1 supplies the spectral-side equivalence; F2 supplies the
$E_{2}$-enhancement making Hochschild trace $E_{1}$-algebraic; F3
identifies the trace with $\mathcal{V}^{\mathrm{prim}}$. With all
three, the arithmetic-branch synthesis
(\Cref{v4-arith:thm:six-window-placement}) becomes a theorem.

\section{Roadmap to Inscription}
\label{v4-arith:of:roadmap}

The arithmetic branch of Vol~IV, as of this inscription, is a
curated synthesis. To convert to theorem-status inscription
throughout:

\begin{enumerate}
\item \textbf{First paper (40--60 pages).} Prove F3 (arithmetic
Ben-Zvi--Nadler) over $\mathbb{Q}$ under the assumption that F1
holds. This gives a conditional proof of the six-window
synthesis.
\item \textbf{Second paper (80--120 pages).} Prove F1 (global
arithmetic AG) for $GL_{2}/\mathbb{Q}$, importing Fargues--Scholze
local theorems and filling the global gluing gap.
\item \textbf{Third paper (60--80 pages).} Prove F2 (perfectoid
$E_{2}$-enhancement) globally.
\item \textbf{Fourth paper, if ready.} Prove the
load-bearing theorem
\Cref{v4-arith:thm:G-row-Riemann-functional-equation} directly,
independent of F1--F3, using the arithmetic Koszul-resolution
computation of Hochschild cochains.
\end{enumerate}

Papers (1)--(3) close the cross-Langlands loop. Paper (4) is
independent and its completion would inscribe the arithmetic
$\mathsf{G}$-row as a Vol~I Theorem-C corollary.
