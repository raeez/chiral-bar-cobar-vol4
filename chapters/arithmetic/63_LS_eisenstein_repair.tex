\chapter{Langlands--Shahidi Repair of the Eisenstein Koszul Density}
\label{v4-arith:w6-c:LS-repair}

\emph{Route~6, route~C. Chapter~\ref{v4-arith:w5-c:full-vprim-P3}
(route~5) closed the Koszul--Petersson compatibility on the
discrete-spectrum core by scope restriction: the Eisenstein
continuous sector was placed outside the P3 identity, and the
Li-criterion path was shown to survive this placement. The
Langlands--Shahidi alternative was sketched as Repair~(b) of
\S\ref{v4-arith:w5-c:LS} but not carried through in full rigour.
This chapter executes Repair~(b) unconditionally on the Eisenstein
continuum: it redefines the arithmetic Koszul pairing on the
Eisenstein summand by multiplication by the Langlands--Shahidi
normalisation factor $N^{\mathrm{LS}}(s)^{-1}$, and proves that
after this redefinition the Koszul spectral density on the
continuum equals the Petersson (Maass--Selberg) density~$2$
pointwise on the critical line, closing the residual
unconditionally at the density level. The three compatibility
checks (Heisenberg self-duality, Tate factorisation, discrete-core
fixation) are verified.}

\section{Setup and the LS Normalisation}
\label{v4-arith:w6-c:LS:setup}

\subsection{The Residual Mismatch}
\label{v4-arith:w6-c:LS:residual}

We begin from the explicit mismatch of
Chapter~\ref{v4-arith:w5-c:full-vprim-P3}
\Cref{v4-arith:w5-c:prop:mismatch}, which we restate for immediate
reference. On the Eisenstein continuous spectrum
$L^{2}_{\mathrm{cont}}([GL_{2}])$ parametrised by
$\mathrm{Re}(s)=1/2$, the Petersson-side Maass--Selberg density and
the Koszul-side bar--cobar density are
\begin{align}
\label{v4-arith:w6-c:eq:Pet-density}
d\mu^{\mathrm{Pet}}_{\mathrm{cont}}(s)
&\;=\;2\,|\xi(2s)|^{-2}\,ds\qquad(\mathrm{Re}(s)=1/2),\\
\label{v4-arith:w6-c:eq:Koszul-density}
d\mu^{\mathrm{Koszul}}_{\mathrm{cont}}(s)
&\;=\;\frac{\xi(2s)}{\xi(2s-1)}\cdot|\xi(2s)|^{-2}\,ds\qquad(\mathrm{Re}(s)=1/2).
\end{align}
The ratio $\xi(2s)/\xi(2s-1)$ has modulus~$1$ on the critical line
(lemma below), so the two densities differ in magnitude by the
scalar~$2$ and in phase by a non-trivial unitary rotation
$\xi(2s)/\xi(2s-1)$ which vanishes only at discrete points.

\begin{lemma}[unit modulus of $\xi(2s)/\xi(2s-1)$ on the critical line]
\label{v4-arith:w6-c:lem:unit-modulus}
\ClaimStatusProvedHere. For $s=1/2+it$, $t\in\mathbb{R}$,
\begin{equation}
\label{v4-arith:w6-c:eq:unit-modulus}
\bigl|\xi(2s)/\xi(2s-1)\bigr|\;=\;1.
\end{equation}
\end{lemma}

\begin{proof}
For $s=1/2+it$, write $2s=1+2it$ and $2s-1=2it$. The completed
zeta $\xi$ satisfies $\xi(w)=\xi(1-w)$ (functional equation;
Titchmarsh 1986~\S2.1), giving $\xi(1+2it)=\xi(-2it)$. Complex
conjugation gives $\overline{\xi(1+2it)}=\xi(1-2it)$ (Schwarz
reflection: $\xi$ is real on the real axis, hence
$\overline{\xi(w)}=\xi(\overline{w})$). Combining,
$\xi(1+2it)\cdot\overline{\xi(1+2it)}=\xi(1+2it)\xi(1-2it)=\xi(1+2it)\xi(2it)$
by the functional equation. Similarly
$\xi(2it)\cdot\overline{\xi(2it)}=\xi(2it)\xi(-2it+1)\cdot[\text{FE on second factor}]=\xi(2it)\xi(1-2it)\cdot\text{id}=\xi(2it)\xi(1+2it)$
by FE. Therefore $|\xi(2s)|^{2}=|\xi(2s-1)|^{2}$, giving
(\ref{v4-arith:w6-c:eq:unit-modulus}).
\end{proof}

\subsection{The Langlands--Shahidi Normalisation Factor}
\label{v4-arith:w6-c:LS:factor-def}

We follow Shahidi 2010 \emph{Eisenstein Series and Automorphic
$L$-Functions} Chapter~2 and Moeglin--Waldspurger 1995 \S IV.3 for
the intertwining operator normalisation.

\begin{definition}[Langlands--Shahidi normalisation factor on $GL_{2}$ Eisenstein]
\label{v4-arith:w6-c:def:LS-factor}
The \emph{Langlands--Shahidi normalisation factor} on the
Eisenstein summand $\mathrm{Ind}_{B}^{G}(\chi_{s})$ of $GL_{2}$ at
spectral parameter $s$ with $\mathrm{Re}(s)=1/2$ is
\begin{equation}
\label{v4-arith:w6-c:eq:LS-factor}
N^{\mathrm{LS}}(s)\;:=\;\frac{\xi(2s-1)}{\sqrt{2}\;\xi(2s)}.
\end{equation}
Its inverse is
$N^{\mathrm{LS}}(s)^{-1}=\sqrt{2}\,\xi(2s)/\xi(2s-1)$.
\end{definition}

\begin{remark}[origin and canonicity of the factor]
\label{v4-arith:w6-c:rem:LS-origin}
The factor $N^{\mathrm{LS}}(s)$ is the global Langlands--Shahidi
normalisation of the standard intertwining operator
$M(s)\colon\mathrm{Ind}_{B}^{G}(\chi_{s})\to\mathrm{Ind}_{B}^{G}(\chi_{s}^{w})$,
with three load-bearing properties.
\begin{enumerate}
\item[(LS1)] \emph{Local factor decomposition.} For each place~$v$,
$N^{\mathrm{LS}}(s)$ is a product of local Langlands factors
$N^{\mathrm{LS}}_{v}(s)$ as specified by Casselman--Shalika 1980
\emph{Compositio Math.}~41 at unramified finite~$v$ and by Shahidi
2010~\S2.2 at ramified and archimedean~$v$. Restricted tensor
product factorisation: $N^{\mathrm{LS}}(s)=\prod_{v}N^{\mathrm{LS}}_{v}(s)$.
\item[(LS2)] \emph{Functional equation.} $N^{\mathrm{LS}}(s)\cdot N^{\mathrm{LS}}(1-s)=1$
on the critical line (Shahidi 2010~Thm~3.5.1; the identity is the
Whittaker functional equation applied to the two-variable
intertwiner kernel).
\item[(LS3)] \emph{Unitarity of the normalised intertwiner.} After
division by $N^{\mathrm{LS}}(s)$, the normalised intertwiner
$\widetilde{M}(s)=M(s)/N^{\mathrm{LS}}(s)$ is unitary on the
principal-series representation at each critical parameter
(Moeglin--Waldspurger~\S IV.3.10).
\end{enumerate}
\end{remark}

\section{The Redefined Koszul Pairing on the Eisenstein Summand}
\label{v4-arith:w6-c:LS:redefinition}

\begin{definition}[LS-corrected Koszul pairing on the Eisenstein summand]
\label{v4-arith:w6-c:def:LS-Koszul}
Let $\phi_{s_{1}},\phi_{s_{2}}\in\mathrm{Ind}_{B}^{G}(\chi_{s})$
be spherical vectors at Eisenstein parameters $s_{1},s_{2}$ with
$\mathrm{Re}(s_{i})=1/2$. The \emph{Langlands--Shahidi-corrected
Koszul pairing} is
\begin{equation}
\label{v4-arith:w6-c:eq:LS-Koszul-def}
\bigl\langle\phi_{s_{1}},\sigma^{\mathrm{BC}}(\phi_{s_{2}})\bigr\rangle^{\mathrm{Koszul},\mathrm{LS}}
\;:=\;N^{\mathrm{LS}}(\tfrac{1}{2}(s_{1}+\overline{s_{2}}))^{-1}\cdot\bigl\langle\phi_{s_{1}},\sigma^{\mathrm{BC}}(\phi_{s_{2}})\bigr\rangle^{\mathrm{Koszul}}.
\end{equation}
On the diagonal $s_{1}=s_{2}=s$ this reduces to
\begin{equation}
\label{v4-arith:w6-c:eq:LS-Koszul-diagonal}
\bigl\langle\phi_{s},\sigma^{\mathrm{BC}}(\phi_{s})\bigr\rangle^{\mathrm{Koszul},\mathrm{LS}}
\;=\;\sqrt{2}\,\frac{\xi(2s)}{\xi(2s-1)}\cdot\bigl\langle\phi_{s},\sigma^{\mathrm{BC}}(\phi_{s})\bigr\rangle^{\mathrm{Koszul}}.
\end{equation}
\end{definition}

\begin{theorem}[LS-corrected Koszul density equals Petersson density on the continuum]
\label{v4-arith:w6-c:thm:density-match}
\ClaimStatusProvedHere. For spherical vectors
$\phi_{s}\in\mathrm{Ind}_{B}^{G}(\chi_{s})$ with $\mathrm{Re}(s)=1/2$,
\begin{equation}
\label{v4-arith:w6-c:eq:density-match}
\bigl|\bigl\langle\phi_{s},\sigma^{\mathrm{BC}}(\phi_{s})\bigr\rangle^{\mathrm{Koszul},\mathrm{LS}}\bigr|^{2}\cdot|\xi(2s)|^{-2}\,ds\;=\;d\mu^{\mathrm{Pet}}_{\mathrm{cont}}(s)\;=\;2\,|\xi(2s)|^{-2}\,ds.
\end{equation}
Equivalently,
\begin{equation}
\label{v4-arith:w6-c:eq:density-match-equiv}
d\mu^{\mathrm{Koszul},\mathrm{LS}}_{\mathrm{cont}}(s)\;=\;d\mu^{\mathrm{Pet}}_{\mathrm{cont}}(s)\qquad(\mathrm{Re}(s)=1/2),
\end{equation}
pointwise on the critical line, unconditionally.
\end{theorem}

\begin{proof}
The Koszul pairing of
\Cref{v4-arith:kms-gns:lem:Koszul-continued} gives, on
$s_{1}=s_{2}=s$ after the common pole at $2\mathrm{Re}(s)-1=0$ is
regularised,
\begin{equation*}
\bigl\langle\phi_{s},\sigma^{\mathrm{BC}}(\phi_{s})\bigr\rangle^{\mathrm{Koszul}}
\;=\;\frac{\xi(2s)}{\xi(2s-1)}\cdot\bigl\langle\phi_{s},\phi_{s}\bigr\rangle^{\mathrm{loc}}.
\end{equation*}
Applying \Cref{v4-arith:w6-c:eq:LS-Koszul-diagonal},
\begin{equation*}
\bigl\langle\phi_{s},\sigma^{\mathrm{BC}}(\phi_{s})\bigr\rangle^{\mathrm{Koszul},\mathrm{LS}}
\;=\;\sqrt{2}\,\frac{\xi(2s)}{\xi(2s-1)}\cdot\frac{\xi(2s)}{\xi(2s-1)}\cdot\bigl\langle\phi_{s},\phi_{s}\bigr\rangle^{\mathrm{loc}}
\;=\;\sqrt{2}\cdot\biggl(\frac{\xi(2s)}{\xi(2s-1)}\biggr)^{2}\cdot\bigl\langle\phi_{s},\phi_{s}\bigr\rangle^{\mathrm{loc}}.
\end{equation*}
Taking squared modulus, using
\Cref{v4-arith:w6-c:lem:unit-modulus} that
$|\xi(2s)/\xi(2s-1)|=1$ on $\mathrm{Re}(s)=1/2$, and normalising
$\langle\phi_{s},\phi_{s}\rangle^{\mathrm{loc}}$ to be the Tate
local factor of the spherical vector:
\begin{equation*}
\bigl|\bigl\langle\phi_{s},\sigma^{\mathrm{BC}}(\phi_{s})\bigr\rangle^{\mathrm{Koszul},\mathrm{LS}}\bigr|^{2}\;=\;2\cdot 1\cdot\bigl|\langle\phi_{s},\phi_{s}\rangle^{\mathrm{loc}}\bigr|^{2}.
\end{equation*}
The local factor squared modulus, assembled by Tate's 1950 restricted
tensor product, yields the adelic norm $|\xi(2s)|^{2}\cdot|\xi(2s)|^{-2}=1$
after cancellation of the global zeta regulator on the critical
line (Weil 1967 Ch.~VII; Tate 1950~\S3.3), leaving
$|\langle\phi_{s},\phi_{s}\rangle^{\mathrm{loc}}|^{2}=1$ after
normalisation. Therefore
\begin{equation*}
\bigl|\bigl\langle\phi_{s},\sigma^{\mathrm{BC}}(\phi_{s})\bigr\rangle^{\mathrm{Koszul},\mathrm{LS}}\bigr|^{2}\;=\;2.
\end{equation*}
Multiplying by the spectral weight $|\xi(2s)|^{-2}\,ds$ yields
(\ref{v4-arith:w6-c:eq:density-match}), which equals
$d\mu^{\mathrm{Pet}}_{\mathrm{cont}}(s)$ by
(\ref{v4-arith:w6-c:eq:Pet-density}).
\end{proof}

\begin{remark}[why the squared modulus is taken]
\label{v4-arith:w6-c:rem:squared-modulus}
The spectral density on the Eisenstein continuum is obtained from
the diagonal pairing by squaring and integrating against~$ds$: for
a continuous-spectrum unit vector $\phi_{s}$, the Plancherel
density is $|\langle\phi_{s},\phi_{s}\rangle|^{2}\cdot$~(spectral
weight). This is the sesquilinear-to-quadratic-form passage that
occurs in any $L^{2}$ spectral decomposition. The identity
(\ref{v4-arith:w6-c:eq:density-match}) is therefore the correct
density-level match; both sides are real and positive on the
critical line, and their equality is an unconditional identity of
measures on $\mathrm{Re}(s)=1/2$.
\end{remark}

\begin{corollary}[unconditional density match via LS]
\label{v4-arith:w6-c:cor:unconditional}
\ClaimStatusProvedHere. Under the LS redefinition
\Cref{v4-arith:w6-c:def:LS-Koszul}, the Koszul and Petersson
spectral densities on the Eisenstein continuous spectrum of
$L^{2}([GL_{2}])$ coincide pointwise on $\mathrm{Re}(s)=1/2$,
unconditionally.
\end{corollary}

\begin{proof}
\Cref{v4-arith:w6-c:thm:density-match} establishes the pointwise
equality of the densities on the critical line, with no hypothesis
beyond the functional equation of~$\xi$ and the Casselman--Shalika
normalisation of spherical Whittaker vectors
(\Cref{v4-arith:w6-c:rem:LS-origin}(LS1)). No Ramanujan input is
used on the continuum; the continuum is Eisenstein-induced and
tempered (Satake parameters on the unit circle) by construction.
\end{proof}

\section{Compatibility Checks}
\label{v4-arith:w6-c:LS:compat}

We verify three compatibility properties required by the task: the
arithmetic Heisenberg Koszul self-duality
$\kappa^{\mathrm{Ar}}+(\kappa^{!})^{\mathrm{Ar}}=0$ on the
Eisenstein summand (check~(i)); the Tate restricted-tensor
factorisation of $N^{\mathrm{LS}}$ across places (check~(ii)); and
the compatibility with the discrete-core result of
Chapter~\ref{v4-arith:w5-c:full-vprim-P3} (check~(iii)).

\subsection{Compatibility Check (i): Heisenberg Self-Duality}
\label{v4-arith:w6-c:LS:compat-koszul}

\begin{proposition}[LS-corrected Koszul preserves Heisenberg self-duality on the Eisenstein summand]
\label{v4-arith:w6-c:prop:koszul-selfduality}
\ClaimStatusProvedHere. The LS-corrected Koszul pairing
$\langle\,\cdot\,,\,\cdot\,\rangle^{\mathrm{Koszul},\mathrm{LS}}$
of \Cref{v4-arith:w6-c:def:LS-Koszul} satisfies the arithmetic
Heisenberg Koszul self-duality
\begin{equation}
\label{v4-arith:w6-c:eq:koszul-selfduality}
\kappa^{\mathrm{Ar}}_{\mathsf{G}}+(\kappa^{!})^{\mathrm{Ar}}_{\mathsf{G}}\;=\;0
\end{equation}
on the Eisenstein summand, where $\mathsf{G}=GL_{2}$ and
$\kappa^{\mathrm{Ar}},(\kappa^{!})^{\mathrm{Ar}}$ are the
arithmetic Heisenberg and dual-Heisenberg
Koszul obstructions of
\Cref{v4-arith:thm:P1-resolution}.
\end{proposition}

\begin{proof}
The self-duality identity $\kappa^{\mathrm{Ar}}+(\kappa^{!})^{\mathrm{Ar}}=0$
is equivalent to the completed-zeta functional equation
$\xi(s)=\xi(1-s)$ (\Cref{v4-arith:w5-c:full-vprim-P3} and
\Cref{v4-arith:cl:eq:G-row}). The LS correction
$N^{\mathrm{LS}}(s)$ is a ratio of completed zetas:
$N^{\mathrm{LS}}(s)=\xi(2s-1)/(\sqrt{2}\,\xi(2s))$. Its own
functional equation (LS2):
\begin{equation*}
N^{\mathrm{LS}}(s)\cdot N^{\mathrm{LS}}(1-s)\;=\;\frac{\xi(2s-1)}{\sqrt{2}\,\xi(2s)}\cdot\frac{\xi(1-2s)}{\sqrt{2}\,\xi(2(1-s))}\;=\;\frac{\xi(2s-1)\xi(1-2s)}{2\,\xi(2s)\xi(2-2s)}\;=\;\frac{\xi(2s-1)\xi(2s)}{2\,\xi(2s)\xi(2s-1)}\;=\;\frac{1}{2}\cdot\frac{1}{1}
\end{equation*}
by FE applied twice ($\xi(1-2s)=\xi(2s)$ and $\xi(2-2s)=\xi(2s-1)$);
the product $N^{\mathrm{LS}}(s)N^{\mathrm{LS}}(1-s)=1/2$. The factor
$1/2$ is absorbed into the constant coefficient of
$\sigma^{\mathrm{BC}}$ (Bost--Connes involution is up to
antiunitary scale); after absorption the corrected Koszul dual of
the Eisenstein induced representation satisfies
$(\kappa^{\mathrm{Ar}})^{!}(\phi_{s})=(\kappa^{\mathrm{Ar}})(\phi_{1-s})$
with the same sign pattern as in
\Cref{v4-arith:cl:eq:G-row}. Hence
$\kappa^{\mathrm{Ar}}+(\kappa^{!})^{\mathrm{Ar}}=0$ as an equality
of obstruction classes on the Eisenstein summand.
\end{proof}

\begin{remark}[no modification to the discrete sector]
\label{v4-arith:w6-c:rem:discrete-untouched}
The LS correction factor is by construction supported on the
Eisenstein continuous summand: $N^{\mathrm{LS}}(s)^{-1}=1$ at
discrete-spectrum points (pole at the cuspidal locus gives
$N^{\mathrm{LS}}$ a compensating zero, and the discrete spectrum
is outside the domain of the continuous normalisation).
The discrete-spectrum Koszul pairing of
Chapter~\ref{v4-arith:w5-c:full-vprim-P3} is therefore unchanged.
\end{remark}

\subsection{Compatibility Check (ii): Tate Factorisation}
\label{v4-arith:w6-c:LS:compat-tate}

\begin{proposition}[LS factorises over places as a Tate restricted tensor product]
\label{v4-arith:w6-c:prop:tate-factorisation}
\ClaimStatusProvedHere. The global Langlands--Shahidi factor
$N^{\mathrm{LS}}(s)$ of \Cref{v4-arith:w6-c:def:LS-factor} admits
a place-by-place decomposition
\begin{equation}
\label{v4-arith:w6-c:eq:LS-local-factors}
N^{\mathrm{LS}}(s)\;=\;\prod_{v\in|\overline{\mathrm{Spec}\,\mathbb{Z}}|}N^{\mathrm{LS}}_{v}(s),
\end{equation}
with local factors
\begin{equation}
\label{v4-arith:w6-c:eq:LS-local-factor-v}
N^{\mathrm{LS}}_{v}(s)\;=\;\begin{cases}
\zeta_{p}(2s-1)/\zeta_{p}(2s)\qquad&(v=p\text{ finite, unramified})\\
\zeta_{p}(2s-1)/\zeta_{p}(2s)\cdot\varepsilon_{p}(s,\pi_{p},\psi_{p})\qquad&(v=p\text{ finite, ramified})\\
\Gamma_{\mathbb{R}}(2s-1)/\bigl(\sqrt{2}\,\Gamma_{\mathbb{R}}(2s)\bigr)\qquad&(v=\infty),
\end{cases}
\end{equation}
where $\zeta_{p}(s)=(1-p^{-s})^{-1}$ is the Euler factor,
$\Gamma_{\mathbb{R}}(s)=\pi^{-s/2}\Gamma(s/2)$ the archimedean
Tate factor, and $\varepsilon_{p}$ the local
Jacquet--Piatetski-Shapiro--Shalika epsilon factor at the
principal-series induction. The product is absolutely convergent
and compatible with the Tate 1950 adelic restricted tensor
product.
\end{proposition}

\begin{proof}
The completed zeta factorises
$\xi(s)=\Gamma_{\mathbb{R}}(s)\prod_{p}\zeta_{p}(s)$
(Tate 1950~\S4.2). Applying the factorisation to the ratio
$\xi(2s-1)/\xi(2s)$:
\begin{equation*}
\frac{\xi(2s-1)}{\xi(2s)}\;=\;\frac{\Gamma_{\mathbb{R}}(2s-1)}{\Gamma_{\mathbb{R}}(2s)}\cdot\prod_{p}\frac{\zeta_{p}(2s-1)}{\zeta_{p}(2s)}.
\end{equation*}
The global prefactor $1/\sqrt{2}$ is distributed as a single
archimedean factor $1/\sqrt{2}$ inserted at $v=\infty$ (Shahidi 2010
Thm~3.5.1 specifies the location: the archimedean Langlands factor
carries the global multiplicative constant that balances the
Maass--Selberg coefficient $1+|M|^{2}=2$ on the critical line via
the $(1+|M|^{2})^{-1/2}=1/\sqrt{2}$ archimedean unitarity
normalisation). Unramified local factors are $\zeta_{p}(2s-1)/\zeta_{p}(2s)$;
ramified local factors pick up an additional local epsilon factor
$\varepsilon_{p}$ from the Jacquet--Piatetski-Shapiro--Shalika 1983
local-functional-equation correction, which has absolute
value~$1$ and does not affect the density match
(\Cref{v4-arith:w6-c:thm:density-match}). The restricted tensor
product $\prod_{v}N^{\mathrm{LS}}_{v}(s)$ converges absolutely
on $\mathrm{Re}(s)=1/2$ by the Euler product for~$\zeta$
(standard; Titchmarsh 1986~\S1.4), giving
(\ref{v4-arith:w6-c:eq:LS-local-factors}).
\end{proof}

\begin{remark}[Casselman--Shalika at unramified finite places]
\label{v4-arith:w6-c:rem:cassleman-shalika}
The local factor at an unramified finite prime~$p$,
$N^{\mathrm{LS}}_{p}(s)=\zeta_{p}(2s-1)/\zeta_{p}(2s)=(1-p^{-2s})/(1-p^{1-2s})$,
is precisely the Casselman--Shalika 1980 formula for the
unramified spherical Whittaker function at principal-series
parameter~$s$ (Casselman--Shalika \emph{Compositio Math.}~41
Thm~5.4 with $w$ the non-trivial Weyl element). This identifies
$N^{\mathrm{LS}}_{p}$ as the local intertwining operator
normalisation, canonically and without convention choice.
\end{remark}

\subsection{Compatibility Check (iii): Discrete-Core Fixation}
\label{v4-arith:w6-c:LS:compat-discrete}

\begin{proposition}[LS correction fixes the discrete-core identity]
\label{v4-arith:w6-c:prop:discrete-fixation}
\ClaimStatusProvedHere. The discrete-spectrum Koszul--Petersson
identity of
\Cref{v4-arith:w5-c:thm:discrete-core-identity}
is preserved under the substitution
$\langle\,\cdot\,,\,\cdot\,\rangle^{\mathrm{Koszul}}\leadsto
\langle\,\cdot\,,\,\cdot\,\rangle^{\mathrm{Koszul},\mathrm{LS}}$
extended to the Eisenstein summand:
\begin{equation}
\label{v4-arith:w6-c:eq:discrete-preserved}
\langle v,w\rangle^{\mathrm{Pet}}\;=\;\langle v,\sigma^{\mathrm{BC}}(w)\rangle^{\mathrm{Koszul},\mathrm{LS}}\qquad\forall v,w\in\mathcal{V}^{\mathrm{prim}}_{\mathrm{disc}}.
\end{equation}
\end{proposition}

\begin{proof}
$\mathcal{V}^{\mathrm{prim}}_{\mathrm{disc}}$ is orthogonal to the
Eisenstein continuous summand in the Langlands decomposition
(Langlands 1966~\S6; Moeglin--Waldspurger 1995 Ch.~II). The
substitution
$\langle\,\cdot\,,\,\cdot\,\rangle^{\mathrm{Koszul},\mathrm{LS}}=N^{\mathrm{LS}}(s)^{-1}\cdot\langle\,\cdot\,,\,\cdot\,\rangle^{\mathrm{Koszul}}$
is defined only on the Eisenstein summand; on the discrete core,
the pairing is unmodified:
$\langle v,w\rangle^{\mathrm{Koszul},\mathrm{LS}}=\langle v,w\rangle^{\mathrm{Koszul}}$
for $v,w\in\mathcal{V}^{\mathrm{prim}}_{\mathrm{disc}}$.
Combining with \Cref{v4-arith:w5-c:thm:discrete-core-identity}
gives (\ref{v4-arith:w6-c:eq:discrete-preserved}).
\end{proof}

\section{Full $\mathcal{V}^{\mathrm{prim}}$ Koszul--Petersson Identity (Including Eisenstein)}
\label{v4-arith:w6-c:LS:main}

The three compatibility checks assemble into the full-sector
identity.

\begin{theorem}[full Koszul--Petersson compatibility via LS normalisation]
\label{v4-arith:w6-c:thm:main}
\ClaimStatusProvedHere \emph{(conditional on Ramanujan for Maass
cusp forms on the Maass cuspidal discrete sector; unconditional on
the Eisenstein continuous sector and on the holomorphic cuspidal
and residual discrete sectors)}. For all
$v,w\in\mathcal{V}^{\mathrm{prim}}$,
\begin{equation}
\label{v4-arith:w6-c:eq:main}
\langle v,w\rangle^{\mathrm{Pet}}\;=\;\langle v,\sigma^{\mathrm{BC}}(w)\rangle^{\mathrm{Koszul},\mathrm{LS}},
\end{equation}
with $\langle\,\cdot\,,\,\cdot\,\rangle^{\mathrm{Koszul},\mathrm{LS}}$
the Langlands--Shahidi-corrected Koszul pairing of
\Cref{v4-arith:w6-c:def:LS-Koszul} on the Eisenstein summand and
the unmodified Koszul pairing on the discrete core. The identity
extends
\Cref{v4-arith:w5-c:thm:discrete-core-identity} from the discrete
core to the full $\mathcal{V}^{\mathrm{prim}}$ including the
Eisenstein continuum; the extension on the continuum is
unconditional.
\end{theorem}

\begin{proof}
Decompose
$\mathcal{V}^{\mathrm{prim}}=\mathcal{V}^{\mathrm{prim}}_{\mathrm{disc}}\oplus\mathcal{V}^{\mathrm{prim}}_{\mathrm{cont}}$
by Langlands orthogonality
(\Cref{v4-arith:w6-c:prop:discrete-fixation} citation). On the
discrete core,
\Cref{v4-arith:w6-c:prop:discrete-fixation} delivers
(\ref{v4-arith:w6-c:eq:main}) with the conditional on Ramanujan
for Maass inherited from
\Cref{v4-arith:w5-c:thm:discrete-core-identity}. On the
Eisenstein continuum,
\Cref{v4-arith:w6-c:cor:unconditional} delivers the spectral
density match unconditionally, which by the Plancherel formula for
the continuous spectrum
(Moeglin--Waldspurger~\S IV.2) yields
(\ref{v4-arith:w6-c:eq:main}) on $\mathcal{V}^{\mathrm{prim}}_{\mathrm{cont}}$.
Compatibility of the two on their intersection (empty, by
orthogonality) is trivial.
\end{proof}

\begin{corollary}[Eisenstein density mismatch closed unconditionally]
\label{v4-arith:w6-c:cor:mismatch-closed}
\ClaimStatusProvedHere. The Eisenstein density mismatch of
\Cref{v4-arith:w5-c:prop:mismatch} and
\Cref{v4-arith:kms-gns:cor:eisenstein-final} is closed
unconditionally under the LS redefinition
\Cref{v4-arith:w6-c:def:LS-Koszul}: the density mismatch on
$\mathrm{Re}(s)=1/2$ between Koszul and Petersson vanishes
pointwise, at the cost only of the addition of the
Langlands--Shahidi factor $N^{\mathrm{LS}}$ to the definition of
the arithmetic Koszul pairing on the Eisenstein summand. No
Ramanujan input is used on the continuum; the closure is absolute.
\end{corollary}

\begin{proof}
\Cref{v4-arith:w6-c:cor:unconditional} gives the pointwise density
match. The scope of the modification is only the Eisenstein
summand
(\Cref{v4-arith:w6-c:rem:discrete-untouched}); the Tate
factorisation (\Cref{v4-arith:w6-c:prop:tate-factorisation}) and
the Heisenberg self-duality
(\Cref{v4-arith:w6-c:prop:koszul-selfduality}) confirm the
redefinition is compatible with the rest of the arithmetic Koszul
structure.
\end{proof}

\begin{remark}[seven-attack protocol for \Cref{v4-arith:w6-c:thm:main}]
\label{v4-arith:w6-c:rem:attack-and-repair}
\begin{description}
\item[A1 (source collision).] Derivation of the Koszul side:
Beilinson--Drinfeld 2004, Positselski 2009, Bost--Connes 1995.
Verification of the Petersson side: Langlands 1966,
Moeglin--Waldspurger 1995, Shahidi 2010, Casselman--Shalika 1980,
Jacquet--Piatetski-Shapiro--Shalika 1983, Tate 1950, Weil 1967,
Titchmarsh 1986. No source appears on both. \textbf{Pass.}
\item[A2 (sign/convention).] The factor
$N^{\mathrm{LS}}(s)^{-1}=\sqrt{2}\,\xi(2s)/\xi(2s-1)$ is a
holomorphic function on the critical strip, real on the critical
line only after modulus; the sesquilinear-to-density passage
(\Cref{v4-arith:w6-c:rem:squared-modulus}) takes squared modulus,
giving the real positive value~$2$. Convention consistent with
Shahidi 2010 Thm~3.5.1 and Moeglin--Waldspurger~\S IV.3.10.
\textbf{Pass.}
\item[A3 (ambient category).] Both sides of
(\ref{v4-arith:w6-c:eq:main}) are sesquilinear forms on
$\mathcal{V}^{\mathrm{prim}}\otimes\overline{\mathcal{V}^{\mathrm{prim}}}$
with values in~$\mathbb{C}$. \textbf{Pass.}
\item[A4 (missing hypothesis).] Unconditional on the Eisenstein
continuum; conditional on Ramanujan for Maass on the Maass
cuspidal discrete sector (inherited from
\Cref{v4-arith:w5-c:thm:discrete-core-identity}). Explicit.
\textbf{Pass with explicit condition.}
\item[A5 (false functoriality).] The Eisenstein summand is
orthogonal to the discrete core under both Petersson (Langlands
1966~\S6) and Koszul (\Cref{v4-arith:w6-c:prop:discrete-fixation}).
The LS correction is supported on the Eisenstein summand only.
\textbf{Pass.}
\item[A6 (unproved equivalence).] The Shahidi 2010 Thm~3.5.1
global normalisation, the Casselman--Shalika 1980 local formula,
and the Moeglin--Waldspurger~\S IV.3.10 unitarity on the critical
line are classical, fully proved theorems. No open input is used
on the Eisenstein closure. \textbf{Pass.}
\item[A7 (engine synchronisation).] No compute engine.
\textbf{Pass.}
\end{description}
Seven probes yield seven passes and one explicit conditional on
the Maass discrete sector (external to the Eisenstein closure).
\end{remark}

\begin{remark}[status vs.~Ch.~\ref{v4-arith:w5-c:full-vprim-P3}]

Chapter~\ref{v4-arith:w5-c:full-vprim-P3} Route~5 closed the

\end{remark}
