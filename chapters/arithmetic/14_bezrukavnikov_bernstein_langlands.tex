\section{The Naive $\mathrm{Bun}_{GL_{2}}(\mathrm{Spec}\,\mathbb{Z})$ Is a Point}
\label{v4-arith:bb:bun-is-point}

\begin{theorem}[Minkowski]
\label{v4-arith:thm:minkowski}
\ClaimStatusProvedElsewhere. $\pi_{1}^{\mathrm{\acute{e}t}}(\mathrm{Spec}\,\mathbb{Z})=1$:
every \'etale cover of $\mathrm{Spec}\,\mathbb{Z}$ is trivial. Class
number of $\mathbb{Q}$ is $1$.
\end{theorem}

\begin{corollary}
\label{v4-arith:cor:Bun-GL2-is-point}
\ClaimStatusProvedHere. $\mathrm{Bun}_{GL_{2}}(\mathrm{Spec}\,\mathbb{Z})=\{\mathbb{Z}^{2}\}$,
a single isomorphism class of rank-$2$ projective $\mathbb{Z}$-modules.
$D$-modules on this groupoid are single vector spaces, cannot carry
Hecke-eigen content.
\end{corollary}

\section{The Spherical Hecke Category Collapses the OPE}
\label{v4-arith:bb:spherical-collapses-OPE}

\begin{theorem}[Mirkovi\'c--Vilonen]
\label{v4-arith:thm:mirkovic-vilonen}
\ClaimStatusProvedElsewhere. Geometric Satake gives
$\mathrm{Perv}_{GL_{2}(\mathcal{O})}(\mathrm{Gr}_{GL_{2}})\xrightarrow{\sim}\mathrm{Rep}(\widehat{GL_{2}})=\mathrm{Rep}(GL_{2})$.
\end{theorem}

\begin{observation}[type mismatch]
\label{v4-arith:obs:type-mismatch-perv-VA}
$\mathcal{H}^{(p)}$ is an $E_{1}$-algebra in VAs (hence
$E_{2}$ in complexes); $\mathrm{Perv}(\mathrm{Gr})$ is a monoidal
category under convolution. Type error at the naive level. The
spherical Hecke algebra
$\mathcal{H}_{p}=C_{c}(GL_{2}(\mathbb{Z}_{p})\backslash GL_{2}(\mathbb{Q}_{p})/GL_{2}(\mathbb{Z}_{p}))$
is commutative, so it carries no OPE and cannot be a vertex
algebra.
\end{observation}

\section{The Iwahori Hecke Category Carries Convolution Data}
\label{v4-arith:bb:iwahori-carries-OPE}

\begin{theorem}[Bezrukavnikov arXiv:1209.0403]
\label{v4-arith:thm:bezrukavnikov}
\ClaimStatusProvedElsewhere. Equivalences
\begin{equation*}
\mathrm{D}^{b}\mathrm{Coh}^{GL_{2}\times\mathbb{G}_{m}}(\widetilde{\mathcal{N}})\xrightarrow{\sim}\mathrm{D}^{b}\mathrm{Coh}(\mathrm{St}_{GL_{2}})\xrightarrow{\sim}\mathrm{D}^{b}_{\mathrm{mix}}(\mathrm{Perv}_{\mathrm{Iw}}(\mathrm{Fl}_{GL_{2}})).
\end{equation*}
The Iwahori-fixed direction carries the noncommutative convolution
datum needed for the Hochschild trace: the affine flag
variety admits an Iwahori stratification indexed by the affine
Weyl group $W^{\mathrm{aff}}_{GL_{2}}=\mathbb{Z}\rtimes W_{f}$.
\end{theorem}

\begin{definition}[global Iwahori Hecke category]
\label{v4-arith:def:global-iwahori-hecke}
\begin{equation}
\mathrm{Hk}^{\mathrm{Iw}}_{GL_{2},\mathrm{glob}}\;:=\;\bigotimes_{p}^{\prime}\mathrm{Hk}^{\mathrm{Iw}}_{GL_{2},p},\quad\mathrm{Hk}^{\mathrm{Iw}}_{GL_{2},p}=\mathrm{D}^{b}_{\mathrm{mix}}(\mathrm{Perv}_{\mathrm{Iw}}(\mathrm{Fl}_{GL_{2},\mathbb{Q}_{p}})).
\end{equation}
\end{definition}

\section{The Emerton--Gee Spectral Side}
\label{v4-arith:bb:emerton-gee}

\begin{theorem}[Emerton--Gee arXiv:1908.07185]
\label{v4-arith:thm:emerton-gee}
\ClaimStatusProvedElsewhere. There is a Noetherian formal algebraic
stack $\mathcal{X}^{\mathrm{EG}}_{GL_{2}}$ over $\mathrm{Spec}\,\mathbb{Z}_{p}$
whose points parametrise $(\varphi,\Gamma)$-modules of rank~$2$.
The global spectral object used below is not a product or colimit of
these local stacks; it is the Selmer-derived global deformation stack
with fixed residual representation, determinant, and local
conditions.
\end{theorem}

\begin{theorem}[arithmetic Arinkin--Gaitsgory]
\label{v4-arith:thm:arithmetic-AG}
\ClaimStatusConjectured (theorem-waiting locally, conjectural
globally).
\begin{equation}
\mathrm{D}^{\mathrm{sph\text{-}fin}}([GL_{2}])\;\simeq\;\mathrm{IndCoh}_{\mathrm{Nilp}}(\mathcal{X}^{\mathrm{Sel}}_{\bar\rho,\det,\mathcal{L}}).
\end{equation}
Fargues--Scholze arXiv:2102.13459 supplies local geometrization,
spectral action, and Hecke functoriality, while Emerton--Gee
supplies the local $(\varphi,\Gamma)$ stack. The displayed
automorphic-to-$\mathrm{IndCoh}_{\mathrm{Nilp}}$ equivalence is not
proved there; it is theorem-waiting locally and conjectural under
global adelic descent.
\end{theorem}

\section{The Hochschild Trace Placement}
\label{v4-arith:bb:hochschild-placement}

\begin{theorem}[crowning placement]
\label{v4-arith:thm:crowning-placement}
\ClaimStatusConjectured.
\begin{equation}
\boxed{\;\mathcal{V}^{\mathrm{prim}}\;=\;\mathrm{HH}_{\bullet}\bigl(\mathrm{Hk}^{\mathrm{Iw}}_{GL_{2},\mathrm{glob}}\bigr)\;}
\end{equation}
$E_{1}$-algebra from Dunn additivity ($E_{1}\otimes E_{1}=E_{2}$)

Ben-Zvi--Nadler arXiv:0904.1247 identifies
$\mathrm{HH}_{\bullet}(\mathrm{D}(G/G))=\mathcal{O}(\mathrm{LocSys}_{G}(S^{1}))^{\mathrm{flat}}$.
\end{theorem}

\begin{remark}[adelic inertia circle]
The inertia circle at $p$ is $\widehat{\mathbb{Z}}\subset W_{\mathbb{Q}_{p}}$;
the free loop space is $\mathrm{LocSys}_{GL_{2}}$ on this inertia
circle; its functions are the trace. This is the arithmetic
analogue of Ben-Zvi--Nadler's ``character theory of a complex group''.
\end{remark}

\section{Connection to Gaiotto's 3D Chern--Simons Presentation}
\label{v4-arith:bb:connection-gaiotto}

\begin{proposition}[three-way equivalence]
\label{v4-arith:prop:three-way-equivalence}
\ClaimStatusConjectured. The three presentations of
$\mathcal{V}^{\mathrm{prim}}$ are equivalent:
\begin{enumerate}
\item (Gaiotto) Kim arithmetic CS gives a classical boundary-shadow window for $\mathcal{V}^{\mathrm{prim}}$;
\item (Bezrukavnikov--Bernstein) $\mathcal{V}^{\mathrm{prim}}=\mathrm{HH}_{\bullet}(\mathrm{Hk}^{\mathrm{Iw}}_{GL_{2},\mathrm{glob}})$;
\item (Costello--Gwilliam) $\mathcal{V}^{\mathrm{prim}}\in\mathrm{FactAlg}_{\mathbb{C}_{\infty}}(\mathrm{Mod}_{\widehat{\mathbb{Z}},\psi})$
with (F1)--(F5).
\end{enumerate}
These are conditional comparison windows, not proved equivalent
presentations. Ben-Zvi--Nadler's CS-trace-Hecke triangle is the
complex template for gluing them; the arithmetic gluing is F3.
\end{proposition}

\section{Hecke Trace = Eichler--Selberg}
\label{v4-arith:bb:hecke-trace}

\begin{theorem}[healed Hecke-trace identity]
\label{v4-arith:thm:healed-hecke-trace}
\ClaimStatusProvedElsewhere. For each $N\geq 1$, $k\geq 2$:
\begin{equation}
\mathrm{tr}\bigl(T_{p}\mid S_{k}(\Gamma_{1}(N))\bigr)\;=\;-\sum_{t^{2}<4p}P_{k}(t,p)H_{w}(t^{2}-4p)+\text{boundary terms},
\end{equation}
with $P_{k}(t,p)=(\alpha^{k-1}-\beta^{k-1})/(\alpha-\beta)$,
$\alpha\beta=p$, $\alpha+\beta=t$, $H_{w}$ the Hurwitz class
number. This is the Eichler--Selberg trace formula.
\end{theorem}

\begin{theorem}[trace-class tabulation]
\label{v4-arith:thm:trace-class-tabulation}
\ClaimStatusConjectured. $\mathcal{V}^{\mathrm{prim}}$ represents
the universal family of Hecke characters on the global automorphic
category $\mathrm{D}([GL_{2}])^{\mathrm{sph\text{-}fin}}$: it is
the $2$-arithmetic-dimensional object whose closed-point
specializations are individual Hecke characters. Restricted to the
subspace $S_{k}(\Gamma_{1}(N))$ for fixed $(k,N)$, the fibre of the
universal family recovers the Eichler--Selberg trace formula
coefficient by coefficient; individual Hecke eigenforms are
$0$-dimensional points of the family.
\end{theorem}

\section{Summary of This Chapter}
\label{v4-arith:bb:summary}

\begin{enumerate}
\item $\mathrm{Bun}_{GL_{2}}(\mathrm{Spec}\,\mathbb{Z})$ is a
point (\Cref{v4-arith:cor:Bun-GL2-is-point}); naive
$D$-module-on-$\mathrm{Bun}$ reading fails.
\item Spherical Hecke collapses the OPE; the correct local model
is the Iwahori-equivariant affine Hecke category
(\Cref{v4-arith:thm:bezrukavnikov}).
\item The spectral side is the Selmer-derived global stack, built
from local Emerton--Gee/$(\varphi,\Gamma)$ stacks with fixed
residual representation, determinant, and local conditions.
\item Arithmetic Arinkin--Gaitsgory: local FS/EG inputs exist, but
the automorphic-to-$\mathrm{IndCoh}_{\mathrm{Nilp}}$ equivalence is
theorem-waiting locally and conjectural globally
(\Cref{v4-arith:thm:arithmetic-AG}).
\item Crowning placement:
$\mathcal{V}^{\mathrm{prim}}=\mathrm{HH}_{\bullet}(\mathrm{Hk}^{\mathrm{Iw}}_{GL_{2},\mathrm{glob}})$
(\Cref{v4-arith:thm:crowning-placement}).
\item Three-way equivalence with Gaiotto's CS and Costello--Gwilliam's
factorization (\Cref{v4-arith:prop:three-way-equivalence}).
\item Hecke-trace tabulation recovers Eichler--Selberg
(\Cref{v4-arith:thm:trace-class-tabulation}).
\end{enumerate}

$\mathcal{V}^{\mathrm{prim}}$ is an arithmetic Hochschild
character. It encodes on the automorphic side the universal trace
of the identity on the global Iwahori Hecke category of $GL_{2}$;
on the spectral side, through the conjectural arithmetic Langlands
equivalence, the structure sheaf $\mathcal{O}$ of the global
Galois-deformation stack. The rank-$2$ Heisenberg presentation
per prime is the Whittaker model of this trace, with the two
generators diagonalising the Satake parameters.
