\section{Spec(Z) as Etale 3-Space}
\label{v4-arith:3d-acs:spec-z}

Mazur 1973 (\emph{Notes on étale cohomology of number fields},
Ann.~Sci.~ENS~6) observed that $\mathrm{Spec}\,\mathcal{O}_K$ has étale
cohomological dimension $3$ for $K$ a number field, with Artin--Verdier
duality supplying a fundamental class
$[\mathrm{Spec}\,\mathcal{O}_K]\in H^{3}_{\mathrm{\acute{e}t}}(\mathrm{Spec}\,\mathcal{O}_K,\widehat{\mathbb{Z}})$
analogous to a 3-manifold's orientation class. Morishita 2012
\emph{Knots and Primes} (arXiv:0904.3399) systematizes the dictionary:
primes $\mathfrak{p}\subset\mathcal{O}_K$ behave as knots in a closed
$3$-manifold; $\mathrm{Spec}\,\mathcal{O}_{K,\mathfrak{p}}$ is the
tubular neighborhood; the inertia subgroup $I_{\mathfrak{p}}\subset\mathrm{Gal}(\overline{K}/K)$
is the meridian; linking numbers become Legendre symbols; the
Alexander--Fox module becomes the Iwasawa module. For $K=\mathbb{Q}$,
Arakelov compactification $\overline{\mathrm{Spec}\,\mathbb{Z}}$ closes
the 3-space by adding the archimedean boundary component at $\infty$.

\begin{remark}[cohomological 3-dimensionality]
\label{v4-arith:3d-acs:rem:cohom-3dim}
$\mathrm{Spec}\,\mathbb{Z}$ is not a smooth real 3-manifold. Its
3-dimensionality is cohomological: the étale site supports non-trivial
$H^1,H^2,H^3$ computable via global class-field theory and
Artin--Verdier. The Chern-Simons action functional below takes values
in étale cohomology classes rather than in differential forms; its
quantization is in Galois-representation-theoretic terms rather than
path-integral terms in the differential-geometric sense.
\end{remark}

\section{Kim Classical Arithmetic Chern-Simons}
\label{v4-arith:3d-acs:kim}

Kim 2015 \emph{Arithmetic Chern-Simons Theory I} (arXiv:1510.05818), and
the non-abelian extension Chern-Dan-Kim-Park 2018 (arXiv:1609.03012),
define the classical action
\begin{equation}
\label{v4-arith:3d-acs:eq:kim-action}
\mathrm{CS}^{\mathrm{Ar}}_{M,c}(\rho):=\langle\rho^{*}c,[M]\rangle\in\mathbb{Q}/\mathbb{Z},
\end{equation}
for $M=\mathrm{Spec}\,\mathcal{O}_K$, $\rho:\pi_{1}^{\mathrm{\acute{e}t}}(M)\to G$
a continuous representation into a finite group $G$, $c\in H^{3}(BG,\mathbb{Q}/\mathbb{Z})$
a level class, and $[M]$ the Artin--Verdier fundamental class. Summation
over $\rho$ gives the classical partition function
$Z^{\mathrm{cl,Ar}}_{\mathrm{CS},c}(M)=\sum_{\rho}\frac{e^{2\pi i\mathrm{CS}^{\mathrm{Ar}}_{M,c}(\rho)}}{|\mathrm{Aut}(\rho)|}$.
Chapter~\ref{v4-arith:gaiotto} already uses this classical torsor as a
boundary action-line input for $\mathcal{V}^{\mathrm{prim}}$; the present
chapter upgrades first to finite state sums and then states the
profinite quantum-TQFT problem.

\section{The Arithmetic Kac-Moody Lift}
\label{v4-arith:3d-acs:km-lift}

Chapter~\ref{v4-arith:virasoro-bootstrap} equips $\mathcal{V}^{\mathrm{prim}}_{\mathrm{Heis}}$

of a conformal vector places $\mathcal{V}^{\mathrm{prim}}$ inside the

\begin{theorem}[arithmetic Kac-Moody enhancement]
\label{v4-arith:3d-acs:thm:km-lift}
\ClaimStatusConditional \emph{(on the $\mathsf{G}$-row resolution

Chapter~\ref{v4-arith:all-genus-koszul})}. The arithmetic Heisenberg

\label{v4-arith:3d-acs:eq:km-enh}
\widehat{\mathcal{V}}^{\mathrm{Ar}}_{k^{\mathrm{Ar}}}
:=\mathcal{V}^{\mathrm{prim}}_{\mathrm{Heis}}\otimes\mathrm{Ind}_{k^{\mathrm{Ar}}}\bigl(L\mathfrak{gl}_{2}^{\mathrm{Ar}}\bigr),
\end{equation}
at level $k^{\mathrm{Ar}}=\kappa^{\mathrm{Ar}}_{\mathsf{G}}$, where
$L\mathfrak{gl}_{2}^{\mathrm{Ar}}:=\bigotimes'_{p}L\mathfrak{gl}_{2}^{(p)}$

\label{v4-arith:3d-acs:rem:critical-level}

\end{remark}

\section{3D Quantum Arithmetic Chern-Simons}
\label{v4-arith:3d-acs:qu-acs}

\begin{definition}[finite arithmetic field groupoid]
\label{v4-arith:3d-acs:def:finite-field-groupoid}
\ClaimStatusDefinitional. Let \(M\) be an arithmetic 3-space in the
Mazur--Artin--Verdier sense, for example
\(\overline{\mathrm{Spec}\,\mathcal O_{K}}\) with its compact-support
fundamental class. Let \(G\) be a finite group and let
\(\alpha\in H^{3}(BG,\mathbb C^{\times})\) be a level. The finite
arithmetic field groupoid is
\[
\mathcal F_{G}(M):=\operatorname{Map}_{\mathrm{cont}}
\bigl(\Pi^{\mathrm{\acute et}}_{1}(M),BG\bigr),
\]
the groupoid of continuous \(G\)-local systems on the etale homotopy
type of \(M\). For a field \(\rho\in\mathcal F_{G}(M)\), the action is
\[
S_{\alpha}^{\mathrm{Ar}}(\rho):=
\left\langle \rho^{*}\alpha,[M]\right\rangle\in\mathbb C^{\times}.
\]
\end{definition}

\begin{theorem}[finite-level arithmetic state sum and gluing]
\label{v4-arith:3d-acs:thm:finite-level-ACS}
\ClaimStatusProvedHere. For a closed arithmetic \(3\)-space \(M\) for
which \(\mathcal F_G(M)\) is a finite groupoid, the assignment
\[
Z^{\mathrm{Ar}}_{G,\alpha}(M)
:=\int_{\mathcal F_{G}(M)}S_{\alpha}^{\mathrm{Ar}}
=\sum_{[\rho]\in\pi_{0}\mathcal F_{G}(M)}
\frac{\langle\rho^{*}\alpha,[M]\rangle}{|\operatorname{Aut}(\rho)|}
\]
is Kim's arithmetic Chern--Simons partition sum for the class
\(\alpha\). If an arithmetic gluing \(M=M_{1}\cup_{\Sigma}M_{2}\)
has finite field groupoids and satisfies étale descent
\[
\mathcal F_G(M)\simeq
\mathcal F_G(M_1)\times_{\mathcal F_G(\Sigma)}\mathcal F_G(M_2),
\]
then \(Z^{\mathrm{Ar}}_{G,\alpha}\) satisfies the corresponding
finite-groupoid gluing formula.
\end{theorem}

\begin{proof}
The construction uses only finite homotopy types. To a closed
arithmetic 3-space \(M\), assign the finite groupoid
\(\mathcal F_{G}(M)\) and integrate the locally constant function
\(S_{\alpha}^{\mathrm{Ar}}\) by groupoid cardinality. If
\[
M=M_{1}\cup_{\Sigma}M_{2}
\]
is a gluing along an arithmetic boundary, then etale descent for
local systems gives a homotopy pullback
\[
\mathcal F_{G}(M)\simeq
\mathcal F_{G}(M_{1})\times_{\mathcal F_{G}(\Sigma)}
\mathcal F_{G}(M_{2}).
\]
The fundamental class is additive under gluing, so the action
multiplies:
\[
\langle\rho^{*}\alpha,[M]\rangle
=
\langle\rho_{1}^{*}\alpha,[M_{1}]\rangle
\langle\rho_{2}^{*}\alpha,[M_{2}]\rangle .
\]
Fubini for finite groupoid cardinality then gives the displayed gluing
formula. Kim's action is exactly the same evaluation of a pulled-back
degree-3 class on the Artin--Verdier fundamental class, written in
\(\mathbb Q/\mathbb Z\) and exponentiated here into
\(\mathbb C^{\times}\).
\end{proof}

\begin{definition}[finite-level boundary category]
\label{v4-arith:3d-acs:def:finite-boundary-category}
\ClaimStatusDefinitional. For an arithmetic boundary \(\Sigma\), let
\(\tau_{\Sigma}(\alpha)\) be the transgression of
\(\alpha\in H^{3}(BG,\mathbb C^{\times})\) to a \(2\)-cocycle on the
field groupoid \(\mathcal F_{G}(\Sigma)\). The finite-level boundary
category is
\[
\mathcal B_{G,\alpha}(\Sigma):=
\operatorname{Sh}_{\tau_{\Sigma}(\alpha)}
\bigl(\mathcal F_{G}(\Sigma)\bigr),
\]
the category of finite-dimensional \(\tau_{\Sigma}(\alpha)\)-twisted
local systems on \(\mathcal F_{G}(\Sigma)\).
\end{definition}

\begin{theorem}[finite-level bulk-boundary span]
\label{v4-arith:3d-acs:thm:finite-bulk-boundary}
\ClaimStatusConditional \emph{(on a finite arithmetic cobordism
category whose morphisms satisfy the gluing hypothesis of
\Cref{v4-arith:3d-acs:thm:finite-level-ACS})}. The finite-level theory
\(Z^{\mathrm{Ar}}_{G,\alpha}\) assigns
\(\mathcal B_{G,\alpha}(\Sigma)\) to the arithmetic boundary
\(\Sigma\). If \(M\) is a cobordism from \(\Sigma_{0}\) to
\(\Sigma_{1}\), then \(M\) acts by the integral transform
\[
\operatorname{Sh}_{\tau_{\Sigma_{0}}(\alpha)}
\bigl(\mathcal F_{G}(\Sigma_{0})\bigr)
\longrightarrow
\operatorname{Sh}_{\tau_{\Sigma_{1}}(\alpha)}
\bigl(\mathcal F_{G}(\Sigma_{1})\bigr)
\]
whose kernel is the action local system on \(\mathcal F_{G}(M)\).
For a closed \(M\), the categorical trace of this transform is
\(Z^{\mathrm{Ar}}_{G,\alpha}(M)\).
\end{theorem}

\begin{proof}
Restriction of local systems gives a span
\[
\mathcal F_{G}(\Sigma_{0})\leftarrow \mathcal F_{G}(M)
\rightarrow \mathcal F_{G}(\Sigma_{1}).
\]
The transgressed cocycle records the boundary anomaly of the
degree-3 class \(\alpha\); tensoring by the action local system and
then pushing forward along the right leg gives the displayed integral
transform. Composition of cobordisms is composition of spans. The
homotopy-pullback formula from
\Cref{v4-arith:3d-acs:thm:finite-level-ACS} makes the two composite
kernels canonically isomorphic. When the boundary is empty, the
trace reduces to groupoid-cardinality integration over
\(\mathcal F_{G}(M)\), which is precisely
\(Z^{\mathrm{Ar}}_{G,\alpha}(M)\).
\end{proof}

\begin{proposition}[finite boundary Hochschild trace]
\label{v4-arith:3d-acs:prop:finite-boundary-HH}
\ClaimStatusProvedHere. Assume \(\mathcal F_G(\Sigma)\) is a finite
groupoid and work over \(\mathbb C\). Then the finite boundary category
\[
\mathcal B_{G,\alpha}(\Sigma)=
\operatorname{Sh}_{\tau_\Sigma(\alpha)}(\mathcal F_G(\Sigma))
\]
has Hochschild trace equal to the twisted inertia trace of
\(\mathcal F_G(\Sigma)\):
\[
\operatorname{HH}_{0}\bigl(\mathcal B_{G,\alpha}(\Sigma)\bigr)
\cong
\Gamma\bigl(\pi_0 I\mathcal F_G(\Sigma),
L_{\tau_\Sigma(\alpha)}\bigr),
\qquad
\operatorname{HH}_{n>0}\bigl(\mathcal B_{G,\alpha}(\Sigma)\bigr)=0.
\]
Here \(I\mathcal F_G(\Sigma)\) is the inertia groupoid, and
\(L_{\tau_\Sigma(\alpha)}\) is the transgressed local system on
inertia.
\end{proposition}

\begin{proof}
A finite twisted sheaf category on a finite groupoid is Morita
equivalent to modules over the corresponding finite-dimensional
twisted groupoid algebra. Over \(\mathbb C\) this algebra is separable:
it is a finite product of twisted finite group algebras, hence
semisimple. Hochschild homology of a separable algebra with coefficients
in itself vanishes in positive degrees. Degree zero is the algebra
modulo commutators, which is the space of twisted class functions on
conjugacy classes, equivalently sections of the transgressed local
system on the inertia groupoid.
\end{proof}

\begin{corollary}[Wilson lines at finite level]
\label{v4-arith:3d-acs:cor:finite-wilson}
\ClaimStatusProvedHere. A prime \(p\) viewed as a knot in the
Morishita dictionary supports Wilson defects labelled by
representations of the local stabiliser group of a point of
\(\mathcal F_{G}(\mathrm{Spec}\,\mathbb Q_{p})\). For a
one-dimensional label \(\chi\), the Wilson expectation is the
corresponding Frobenius trace \(\chi(\mathrm{Frob}_{p})\).
\end{corollary}

\begin{proof}
Removing the prime \(p\) and fixing monodromy around the meridian is
the arithmetic analog of prescribing a conjugacy class around a knot.
The defect category is the module category of the stabiliser of that
local monodromy. A one-dimensional module evaluates a loop by its
character, hence the meridian/Frobenius loop contributes
\(\chi(\mathrm{Frob}_{p})\).
\end{proof}

\begin{remark}[status]
\label{v4-arith:3d-acs:rem:qu-acs-status}
The closed finite-level state sum is theorem-grade:
\Cref{v4-arith:3d-acs:thm:finite-level-ACS} constructs it from étale
local systems, the Artin--Verdier fundamental class, and finite
groupoid cardinality. Functorial finite-level TQFT language still
requires a specified arithmetic cobordism category. The larger
residual is the renormalised profinite limit
\(G_{N}\to GL_{2}(\widehat{\mathbb Z})\), the comparison of its
archimedean boundary category with \(\mathcal V^{\mathrm{prim}}\), and
the identification of its renormalised partition function with
\(\xi(k^{\mathrm{Ar}})\).

There is a concrete obstruction to treating the closed finite tower as
the zeta partition function. On the closed base
\(\overline{\mathrm{Spec}\,\mathbb Z}\), finite etale local systems are
trivial by Hermite--Minkowski, so the bare finite state sum for
\(G_N=GL_2(\mathbb Z/N\mathbb Z)\) is the groupoid-cardinality scalar
\(|G_N|^{-1}\), or \(1\) after finite-gauge normalisation. This scalar
does not contain prime weights \(p^{-s}\), the archimedean
\(\Gamma_{\mathbb R}\)-factor, or the polar correction. A zeta-valued
profinite ACS theory therefore requires extra data: compatible
continuous level classes, a counterterm or measure system on the
congruence tower, a profinite boundary category, a Morita comparison
with the local Iwahori trace factors, and a regularised character map.
\end{remark}

\begin{theorem}[bulk-boundary comparison with \(\mathcal V^{\mathrm{prim}}\)]
\label{v4-arith:3d-acs:thm:bulk-boundary-Vprim}
\ClaimStatusConditional \emph{(on the renormalised profinite limit in
\Cref{v4-arith:3d-acs:conj:qu-acs} and on the local Satake/Iwahori
trace identifications of Chapters~\ref{v4-arith:tate-tensor} and
\ref{v4-arith:hochschild-trace})}. Let
\[
\mathcal B_{\infty}^{\mathrm{Ar}}
:=\varprojlim_{N}^{\mathrm{ren}}
\mathcal B_{G_{N},\alpha_{N}}(\{\infty\})
\]
be the renormalised archimedean boundary category of the profinite
ACS tower. If its finite-place compact generators identify with the
local Iwahori Hecke generators and its archimedean vacuum sector
identifies with the rank-\(2\) Heisenberg Kac-Moody vacuum, then
\[
\operatorname{HH}_{\bullet}(\mathcal B_{\infty}^{\mathrm{Ar}})
\simeq \mathcal V^{\mathrm{prim}}.
\]
Equivalently, the boundary factorization algebra of
\(\mathrm{ACS}^{\mathrm{qu}}_{k^{\mathrm{Ar}}}\) is
\(\mathcal V^{\mathrm{prim}}\).
\end{theorem}

\begin{proof}
At finite level,
\Cref{v4-arith:3d-acs:thm:finite-bulk-boundary} identifies the
boundary state category with twisted sheaves on the boundary field
groupoid. Passing through the congruence tower gives a renormalised
inverse system of such categories. The stated compact-generation
hypothesis is exactly the assertion that the local factors of this
system are the Iwahori Hecke categories used in the definition of
\(\mathcal V^{\mathrm{prim}}\). Hochschild trace is local for the
restricted tensor product under the compact-generation and spherical
unit hypotheses of Chapter~\ref{v4-arith:tate-tensor}. Thus the
finite places contribute the local trace factors
\(\mathcal H^{(p)}\), while the archimedean vacuum contributes
\(\mathcal H^{(\infty)}\). Their restricted tensor product is the
Hochschild trace class \(\mathcal V^{\mathrm{prim}}\) by
Chapter~\ref{v4-arith:hochschild-trace}. This proves the displayed
boundary identification, conditional precisely on the profinite
renormalisation and the local trace comparison.
\end{proof}

\section{The Ordered Incidence Bar}
\label{v4-arith:3d-acs:E1-bar}

The primes of \(\mathbb Z\) carry the auxiliary total order
\(2<3<5<7<\cdots<\infty\), with the archimedean place placed last.
This order is not geometric structure on
\(\overline{\mathrm{Spec}\,\mathbb Z}\). It defines an incidence model
whose comparison with genuine arithmetic factorization homology is
kept separate below.

\begin{definition}[ordered incidence category]
\label{v4-arith:3d-acs:def:ordered-incidence-category}
\ClaimStatusDefinitional. Put \(V=\mathbb P\cup\{\infty\}\), with
the order \(2<3<5<\cdots<\infty\). Let
\(\mathsf{Inc}_{<}(V)\) be the category whose objects are finite
ordered block decompositions of finite ordered subsets of \(V\).
A singleton block is written \(v\), and the merge of two adjacent
blocks is written by concatenation. Morphisms are generated by
order-preserving adjacent contractions
\[
\{v_{1}<\cdots<v_{i}<v_{i+1}<\cdots<v_{n}\}
\longrightarrow
\{v_{1}<\cdots<(v_{i}v_{i+1})<\cdots<v_{n}\},
\]
where \(v_{i}v_{i+1}\) denotes the formal merged block carrying the
product operation of the coefficient system, not a new arithmetic
place. The simplicial nerve of
\(\mathsf{Inc}_{<}(V)\) is the one-dimensional ordered collision
model for the arithmetic Ran space.
\end{definition}

\begin{definition}[ordered incidence homology]
\label{v4-arith:3d-acs:def:ordered-chiral-homology}
\ClaimStatusDefinitional. Let
\(\mathcal H=\{\mathcal H^{(v)}\}_{v\in V}\) be the per-place
coefficient system of \(\mathcal V^{\mathrm{prim}}\), with adjacent
products
\[
m_{v,w}\colon \mathcal H^{(v)}\otimes \mathcal H^{(w)}
\longrightarrow \mathcal H^{(vw)}
\qquad (v<w).
\]
Here \(\mathcal H^{(vw)}\) means the coefficient assigned to the
merged block \(vw\).
Let \(F_{\mathcal H}\colon\mathsf{Inc}_{<}(V)\to\mathrm{Ch}_{\mathbb C}\)
send an ordered block decomposition \(B_{1}|\cdots|B_{r}\) to
\[
F_{\mathcal H}(B_{1}|\cdots|B_{r})
:=\mathcal H^{(B_{1})}\otimes\cdots\otimes\mathcal H^{(B_{r})},
\]
with contractions acting by the corresponding adjacent products. The
ordered incidence homology is the derived colimit
\[
\int_{\overline{\mathrm{Spec}\,\mathbb Z}}^{\mathrm{ord}}
\mathcal V^{\mathrm{prim}}
:=
\operatorname*{L\,colim}_{\mathsf{Inc}_{<}(V)}F_{\mathcal H}.
\]
\end{definition}

\begin{definition}[ordered incidence bar]
\label{v4-arith:3d-acs:defn:E1-bar}
\ClaimStatusDefinitional. Let $\mathcal{V}^{\mathrm{prim}}$ be the
arithmetic Heisenberg with per-place factors $\mathcal{H}^{(p)}$ (finite
primes, $E_0$ in the sense of Drinfeld--Kapranov,
Ch.~\ref{v4-arith:drinfeld-kapranov}) and $\mathcal{H}^{(\infty)}$
(archimedean, $E_1$-chiral). The \emph{ordered incidence bar} is the
normalized chain complex of the simplicial replacement of
\(F_{\mathcal H}\):
\begin{equation}
\label{v4-arith:3d-acs:eq:E1-bar}
B^{\mathrm{ord}}_{\bullet}(\mathcal{V}^{\mathrm{prim}})
:=
N_{\bullet}\operatorname{srep}_{\mathsf{Inc}_{<}(V)}(F_{\mathcal H}).
\end{equation}
In degree \(n\), this is the direct sum over \(n\)-simplices
\(C_{0}\to\cdots\to C_{n}\) of \(\mathsf{Inc}_{<}(V)\) of
\(F_{\mathcal H}(C_{0})\), modulo degeneracies. Faces are induced by
composition in \(\mathsf{Inc}_{<}(V)\), hence by adjacent contractions.
Thus all targets are coefficients of formal merged blocks, not new
arithmetic places.
\end{definition}

\begin{theorem}[ordered incidence homology via normalized bar]
\label{v4-arith:3d-acs:thm:chi-hom-bar}
\ClaimStatusProvedHere. The ordered incidence homology of
\(\mathcal{V}^{\mathrm{prim}}\) is computed by the ordered incidence
bar:
\begin{equation}
\label{v4-arith:3d-acs:eq:chi-hom-bar}
\int_{\overline{\mathrm{Spec}\,\mathbb{Z}}}^{\mathrm{ord}}\mathcal{V}^{\mathrm{prim}}
\;\simeq\;
H_{\bullet}\bigl(B^{\mathrm{ord}}_{\bullet}(\mathcal{V}^{\mathrm{prim}})\bigr).
\end{equation}
\end{theorem}

\begin{proof}
This is the standard computation of a derived colimit by the
normalized chain complex of the simplicial replacement of the
coefficient functor \(F_{\mathcal H}\). The incidence category was
defined so that its generating morphisms are adjacent contractions;
therefore the normalized differential is the alternating sum of the
corresponding contraction maps.
\end{proof}

\begin{corollary}[comparison with BD/CG chiral homology]
\label{v4-arith:3d-acs:cor:BDCG-ordered-comparison}
\ClaimStatusConditional \emph{(on constructing an arithmetic
Beilinson--Drinfeld/Costello--Gwilliam Ran model for
\(\overline{\mathrm{Spec}\,\mathbb Z}\), finite-place collision
thickenings, constructible descent, ordered-to-unordered descent, and
completion exactness identifying the arithmetic chiral homology of
Chapters~\ref{v4-arith:arakelov-bar} and
\ref{v4-arith:costello-gwilliam} with the ordered incidence model)}.
One has
\[
\int_{\overline{\mathrm{Spec}\,\mathbb Z}}\mathcal V^{\mathrm{prim}}
\simeq

\section{Deninger Skeleton from Arithmetic Chiral Homology}

\begin{definition}[pre-Deninger arithmetic coefficient package]
\label{v4-arith:3d-acs:def:deninger-enhanced-algebra}
\ClaimStatusDefinitional. The \emph{pre-Deninger arithmetic

with \(N_{p}\delta_{p^{r}}=r(\log p)\delta_{p^{r}}\) and
\(N_{\mathrm{BC}}=\sum_{p}N_{p}\) on the restricted tensor product;
\item the archimedean Tate-Gaussian sector
\(\mathcal A_{\infty}^{\mathrm{Tate}}\), whose Mellin character is
\(\Gamma_{\mathbb R}(s)\);
\item a polar Tate complex
\(\mathsf P^{\bullet}=\mathbb C_{0}[0]\oplus\mathbb C_{1}[2]\),

\item the Bost--Connes Euler flow
\(\phi_{\mathrm{BC}}^{t}:=\exp(-tN_{\mathrm{BC}})\) on the finite

\(\mathcal A_{\infty}^{\mathrm{Tate}}\).
\end{enumerate}
The last flow records closed-orbit lengths \(\log p\). It is not the
Deninger Frobenius flow on the conjectural middle cohomology.

\label{v4-arith:3d-acs:def:chiral-deninger-complex}

\begin{proposition}[unconditional Bost--Connes Euler skeleton]
\label{v4-arith:3d-acs:prop:deninger-prime-side}
\ClaimStatusProvedHere. The finite Bost--Connes zero-mode sector has
ordinary KMS character
\[
\mathrm{Tr}\bigl(e^{-sN_{\mathrm{BC}}}\mid\mathcal A_{\mathrm{fin}}\bigr)
=\prod_{p}\sum_{r\geq0}p^{-rs}
=\prod_{p}(1-p^{-s})^{-1}
=\zeta(s)
\qquad(\mathrm{Re}\,s>1),
\]
and logarithmic differentiation gives the prime-orbit sum
\[
-\frac{\zeta'}{\zeta}(s)
=\sum_{p}\sum_{r\geq1}(\log p)\,p^{-rs}.
\]
This is the raw non-archimedean Euler/log-prime skeleton in the
half-plane of absolute convergence.
\end{proposition}

\begin{proof}
On \(\mathcal H_{p}^{\mathrm{BC}}\),
\(\mathrm{Tr}(e^{-sN_{p}})=\sum_{r\geq0}e^{-sr\log p}\). The
restricted tensor product multiplies these traces over primes, giving
the Euler product for \(\zeta(s)\) in the half-plane of convergence.
Logarithmic differentiation turns each local factor into the Dirichlet
series over iterates of the closed orbit at \(p\), with length
\(\log p\).
\end{proof}

\begin{remark}[what is not proved by the Euler skeleton]
\label{v4-arith:3d-acs:rem:euler-skeleton-not-trace}
\Cref{v4-arith:3d-acs:prop:deninger-prime-side} is not yet the
Weil--Guinand distribution. Passing from
\(-\zeta'/\zeta(s)\) in \(\mathrm{Re}\,s>1\) to the finite-place term
on critical-line test functions requires a separate contour shift,
Fourier/Mellin convention, the \(n^{-1/2}\) weight, the sign and factor
\(2\), plus the archimedean and polar terms. Those analytic
normalisations are part of the trace-formula residual, not part of the
ordered-bar computation.
\end{remark}

\qquad \rho=\frac12+i\gamma_{\rho},

\begin{proposition}[conditional formal Deninger package]
\label{v4-arith:3d-acs:thm:conditional-deninger-cohomology}
