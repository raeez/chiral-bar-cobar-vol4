\chapter{$\mathrm{R}_{\mathrm{analytic}}$ Audit: Is the Sharp Moment Asymptotic Classical-Analytic?}
\label{v4-arith:R-analytic-audit}

\emph{Chapter~\ref{v4-arith:hpar-det} (henceforth ``Ch~41'' or
``$\delta$'s chapter'') isolated three residuals for the minimal RH
completion path: weight existence (W), Weil explicit formula at
Gaussian test functions (H), and sharp moment asymptotic
$\mathrm{R}_{\mathrm{analytic}}$ asserting
$\mu_{2n}\asymp C^{n}n!^{2}$ unconditionally. In
\Cref{v4-arith:hpar-det:rem:residual-vs-RH} $\delta$ characterises
$\mathrm{R}_{\mathrm{analytic}}$ as ``strictly weaker than RH'' and
``a significant analytic-number-theory advance but not, by itself,
RH''. Wave-4 audit agent~C inspects this characterisation. The
verdict of this chapter is that $\delta$'s claim is \emph{partially
correct but substantially misleading}: the IMPLICATION
``$\mathrm{R}_{\mathrm{analytic}}\Rightarrow$ RH'' is indeed not
known; in that narrow technical sense
$\mathrm{R}_{\mathrm{analytic}}$ is not RH-equivalent. But
$\delta$'s further claim that ``Carleman divergence holds
unconditionally via Bombieri--Lagarias modulo closing the
polylog-vs-factorial gap'' contains an algebraic error which
invalidates the unconditional Carleman-divergence conclusion, and
$\delta$'s framing of $\mathrm{R}_{\mathrm{analytic}}$ as
``classical-analytic'' conflates the statement form with the
statement difficulty. This chapter produces the honest reduction:
$\mathrm{R}_{\mathrm{analytic}}$ has classical-analytic FORM but
RH-adjacent DIFFICULTY, and the partial Bombieri--Lagarias
Carleman route, correctly computed, is inconclusive rather than
divergent. $\delta$'s wave-3 ledger is therefore tightened, not
overturned: $\mathrm{R}_{\mathrm{analytic}}$ remains the single
analytic residual, but it is not ``merely'' classical.}

\subsection{The Upper Bound on $\mu_{2n}$ from BL and RvM}
\label{v4-arith:r-audit:upper-bound}

$\delta$'s \Cref{v4-arith:hpar-det:prop:BL-carleman-partial} proves
(assuming the symmetrised shifted-zero moments can be bounded from
Riemann--von Mangoldt zero-counting and the BL Li-coefficient
machinery)
\begin{equation}
\label{v4-arith:r-audit:eq:upper-bound}
\mu_{2n}\;\leq\;\frac{(2n)!}{(2\pi)^{4n}}\cdot(\log n)^{4n}\cdot(1+o(1)).
\end{equation}
We do not dispute this upper bound in this audit; the saddle-point
derivation in $\delta$'s proof is standard.

\subsection{Corrected Stirling Evaluation}
\label{v4-arith:r-audit:corrected-stirling}

\begin{lemma}[Stirling for $(2n)!^{1/(2n)}$]
\label{v4-arith:r-audit:lem:stirling}
\ClaimStatusProvedHere. By Stirling's formula
$n!=\sqrt{2\pi n}\,(n/e)^{n}(1+O(1/n))$,
\begin{equation}
(2n)!^{1/(2n)}\;=\;\frac{2n}{e}\cdot(4\pi n)^{1/(4n)}\cdot(1+O(1/n)).
\end{equation}
Since $(4\pi n)^{1/(4n)}\to 1$ as $n\to\infty$, one has
$(2n)!^{1/(2n)}\sim 2n/e$, i.e., the $(2n)!^{1/(2n)}$ power
grows \emph{linearly} in $n$, not as $\sqrt{n}$.
\end{lemma}

\begin{proof}
$(2n)!=\sqrt{4\pi n}\,(2n/e)^{2n}(1+O(1/n))$ gives
\begin{align*}
(2n)!^{1/(2n)}
&=\bigl[\sqrt{4\pi n}\,(2n/e)^{2n}\bigr]^{1/(2n)}\cdot(1+O(1/n))^{1/(2n)}\\
&=(4\pi n)^{1/(4n)}\cdot(2n/e)\cdot\bigl(1+O(1/n^{2})\bigr).
\end{align*}
Since $(4\pi n)^{1/(4n)}=\exp(\tfrac{1}{4n}\log(4\pi n))
=1+\tfrac{\log(4\pi n)}{4n}+O(n^{-2}(\log n)^{2})\to 1$, the claim
follows.
\end{proof}

\begin{proposition}[corrected Carleman-sum lower bound from BL upper bound]
\label{v4-arith:r-audit:prop:BL-carleman-corrected}
\ClaimStatusProvedHere. From
\Cref{v4-arith:r-audit:eq:upper-bound},
\begin{equation}
\label{v4-arith:r-audit:eq:mu-power}
\mu_{2n}^{1/(2n)}\;\leq\;
\frac{1}{(2\pi)^{2}}\cdot(2n)!^{1/(2n)}\cdot(\log n)^{2}\cdot(1+o(1))
\;\sim\;\frac{2n}{e(2\pi)^{2}}\cdot(\log n)^{2}.
\end{equation}
Hence
\begin{equation}
\label{v4-arith:r-audit:eq:mu-inverse-lower}
\mu_{2n}^{-1/(2n)}\;\geq\;
\frac{e(2\pi)^{2}}{2n(\log n)^{2}}\cdot(1-o(1))
\;=\;\frac{c_{\mathrm{BL}}}{n(\log n)^{2}}
\end{equation}
for a constant $c_{\mathrm{BL}}>0$. Summing,
\begin{equation}
\label{v4-arith:r-audit:eq:carleman-bound}
\sum_{n\geq 2}\mu_{2n}^{-1/(2n)}\;\geq\;
c_{\mathrm{BL}}\cdot\sum_{n\geq 2}\frac{1}{n(\log n)^{2}}.
\end{equation}
\end{proposition}

\begin{proof}
Direct substitution of \Cref{v4-arith:r-audit:lem:stirling} into
the $1/(2n)$-th power of the upper bound
\Cref{v4-arith:r-audit:eq:upper-bound}. The factor $(\log n)^{4n}$
inside the bound becomes $(\log n)^{+2}$ after the
$1/(2n)$-th-root, and when inverted for $\mu_{2n}^{-1/(2n)}$ becomes
$(\log n)^{-2}$. The factor $(2\pi)^{-4n}$ becomes $(2\pi)^{-2}$
and when inverted becomes $(2\pi)^{+2}$. The factor
$(2n)!^{1/(2n)}\sim 2n/e$ in the numerator of $\mu_{2n}^{1/(2n)}$
becomes $e/(2n)$ in $\mu_{2n}^{-1/(2n)}$.
\end{proof}

\begin{corollary}[the BL Carleman route is \emph{inconclusive}, not divergent]
\label{v4-arith:r-audit:cor:inconclusive}
\ClaimStatusProvedHere. The lower bound on the Carleman sum from
\Cref{v4-arith:r-audit:prop:BL-carleman-corrected} is
\begin{equation}
\label{v4-arith:r-audit:eq:bertrand}
\sum_{n\geq 2}\frac{1}{n(\log n)^{2}}\;<\;+\infty,
\end{equation}
a classical Bertrand series which \emph{converges}. Hence the
lower bound on $\sum_{n}\mu_{2n}^{-1/(2n)}$ produced by the BL
upper bound on $\mu_{2n}$ is finite. This does \emph{not} prove
the Carleman sum diverges; it is strictly inconclusive.
\end{corollary}

\begin{proof}
The Bertrand series $\sum_{n\geq 2}1/(n(\log n)^{\alpha})$
converges iff $\alpha>1$, by the integral comparison
$\int_{2}^{\infty}dx/(x(\log x)^{\alpha})
=(\alpha-1)^{-1}(\log 2)^{-(\alpha-1)}$ for $\alpha>1$. For
$\alpha=2$ this integral is finite, confirming convergence.

A lower bound on $\sum \mu_{2n}^{-1/(2n)}$ which is itself finite
does not establish divergence of $\sum \mu_{2n}^{-1/(2n)}$. The
Carleman sum may still diverge or converge; the BL upper bound on
$\mu_{2n}$ with the $(\log n)^{4n}$ log-factor is simply too
weak to decide.
\end{proof}

\begin{remark}[where $\delta$'s error originates]
\label{v4-arith:r-audit:rem:delta-error}
$\delta$'s \Cref{v4-arith:hpar-det:eq:carleman-partial} asserts
$\sum \mu_{2n}^{-1/(2n)}\geq c'\sum 2\pi(\log n)/\sqrt{2n}$. This
expression has $\sqrt{n}$ in the denominator instead of $n$, and
$(\log n)^{+1}$ in the numerator instead of $(\log n)^{-2}$. The
resulting sum $\sum (\log n)/\sqrt{n}$ indeed diverges, but the
transcription from $\mu_{2n}^{-1/(2n)}$ via Stirling should
produce $1/(n(\log n)^{2})$ per \Cref{v4-arith:r-audit:lem:stirling}
plus power-rearrangement of $(\log n)^{4n}$, giving a CONVERGENT
Bertrand series. The source of the error is compatible with
$\delta$ computing $(2n)!^{1/(2n)}\sim\sqrt{2n}$ instead of
$\sim 2n/e$, a slip between Stirling's formula and the incorrect
root-of-factorial approximation $n!^{1/n}\sim\sqrt{n}$ (which is
wrong; the correct asymptotic is $n!^{1/n}\sim n/e$).
\end{remark}

\begin{remark}[the RH-heuristic Carleman divergence is salvageable]
\label{v4-arith:r-audit:rem:heuristic-saves}
If one substitutes the RH-heuristic two-sided asymptotic
$\mu_{2n}\asymp C^{n}(n!)^{2}$ (no $(\log n)^{4n}$ correction) of
\Cref{v4-arith:hpvbconv:prop:carleman-insufficient}, then
$\mu_{2n}^{1/(2n)}\asymp C^{1/2}\cdot n/e$ giving
$\mu_{2n}^{-1/(2n)}\asymp e/(C^{1/2}n)$ and the Carleman sum
\emph{does} diverge at the harmonic rate $\sum 1/n$. But this
heuristic is \emph{not} established unconditionally; the
BL-plus-Conrey machinery has the $(\log n)^{4n}$ gap
(\Cref{v4-arith:hpar-det:cor:unconditional-gap}) which puts the
true sum into the log-convergent regime of
\Cref{v4-arith:r-audit:cor:inconclusive}. Hence Carleman's
criterion from BL-plus-Conrey alone is inconclusive; the
classical-heuristic form is what requires unconditional
confirmation.
\end{remark}

\section{Gap Analysis}
\label{v4-arith:r-audit:gap-analysis}

\subsection{Can BL's Polylog Bound be Sharpened to $n!^{2}$ Classically?}
\label{v4-arith:r-audit:can-BL-sharpen}

The audit prompt asks: can the BL polylog upper bound be improved
to an unconditional $\mu_{2n}\asymp n!^{2}$ bound using only
classical analysis, or does this require RH-content? We resolve this
as follows.

\begin{proposition}[BL does not bound $\mu_{2n}$; the quantities are
structurally different]
\label{v4-arith:r-audit:prop:BL-vs-mu}
\ClaimStatusProvedHere. Bombieri--Lagarias Theorem~2
(\Cref{v4-arith:vb:thm:BL-bound}) bounds
$\lambda_{n}=\sum_{\rho}[1-(1-1/\rho)^{n}]$; this is a heavily
oscillatory unit-modulus-phase sum in $\rho$. Under RH, $|1-1/\rho|=1$
for every non-trivial zero, so the BL bound is a statement about
phase-cancellation, not magnitude growth. The moment sum
$\mu_{2n}=\sum_{\rho}(1/4+\gamma_{\rho}^{2})^{n}$ is a monotone
positive sum (under RH) whose growth is governed by the density of
zeros high in the critical line, via Riemann--von Mangoldt. These two
quantities are related only through H-P$^{\mathrm{Ar}}$-conditional
spectral-trace identities, not through elementary bounds.
\end{proposition}

\begin{proof}
Direct: BL bounds $\lambda_{n}$ linearly in $n\log n$; the
unconditional lower bound is $\lambda_{n}\geq-(n/2)\log n-O(n)$
(\Cref{v4-arith:vb:thm:BL-bound}). These are polylog bounds. A
polylog bound on $\lambda_{n}$ says nothing about $\mu_{2n}$ because
$\lambda_{n}$ has cancellations which $\mu_{2n}$ does not. The
identity $\mathrm{tr}(T_{n})=n\lambda_{n}$ holds only at the
trace level under H-P$^{\mathrm{Ar}}$ (cf.~Ch~41 Cycle~5
\Cref{v4-arith:hpar-det:cycle-5}).
\end{proof}

\begin{proposition}[sharpening the unconditional $\mu_{2n}$ upper bound from
$(\log n)^{4n}$ to $(\log n)^{o(n)}$ requires zero-density improvements]
\label{v4-arith:r-audit:prop:closing-gap-is-density-RH}
\ClaimStatusProvedHere. The $(\log n)^{4n}$ factor in
\Cref{v4-arith:r-audit:eq:upper-bound} comes from the
Conrey-style fluctuation envelope in the RvM zero-counting:
$N(T)\sim (T/2\pi)\log(T/2\pi)$ with second-order error
$O(\log T)$, so that the integral representation of $\mu_{2n}$
picks up a $(\log T)^{\cdot}$ factor at the saddle cutoff
$T\sim\sqrt{n}\log n$. Removing this factor to obtain
$\mu_{2n}\leq C^{n}n!^{2}$ \emph{with no logarithmic inflation}
requires improving the zero-counting asymptotic from the
unconditional $N(T)=(T/2\pi)\log(T/2\pi)-T/2\pi+O(\log T)$ to a
form where the error term does not amplify by a factor of $\log T$
at each moment-integral saddle. This is a zero-density-hypothesis
improvement, a classical analytic-number-theory programme that is
currently open and is not known to follow from RH alone.
\end{proposition}

\begin{proof}
The RvM estimate $N(T)=(T/2\pi)\log(T/2\pi)-T/2\pi+O(\log T)$
(Edwards~1974 Thm.~6.7) has error $O(\log T)$. In the saddle-point
evaluation $\mu_{2n}\approx \int_{0}^{\infty}t^{2n}(dN/dt)dt$, the
saddle is at $t\sim\sqrt{n}\log n$. The error term $O(\log T)$
evaluated at the saddle contributes a factor of
$O(\log(\sqrt{n}\log n))\sim O(\log n)$ at the saddle, and the
$n$-fold application (from the $n$-th moment) amplifies this to
$O((\log n)^{4n})$ in the final bound (the factor of 4 comes from
the $\mu_{2n}$ being the $2n$-th moment, with two such
amplifications in the saddle-point evaluation and two from the
$t^{2}$ factor).

To close this to $(\log n)^{o(n)}$ would require an error term for
$N(T)$ of size $O(1)$ or $O(T^{\epsilon})$ for every $\epsilon>0$.
The current best unconditional error term is $O(T/\log T)$ at best,
which is strictly weaker than the $O(\log T)$ required. Improving
$N(T)$'s error to $O(1)$ is known as the ``density-1 hypothesis''
for the zeros of $\zeta$, a form of the density hypothesis
open since Bombieri~1965.
\end{proof}

\begin{remark}[classical-analytic form, RH-adjacent difficulty]
\label{v4-arith:r-audit:rem:form-vs-difficulty}
\Cref{v4-arith:r-audit:prop:closing-gap-is-density-RH} establishes
the key distinction: the \emph{form} of
$\mathrm{R}_{\mathrm{analytic}}$ is classical-analytic (a moment
asymptotic), but its \emph{difficulty} is RH-adjacent. The gap
between BL-Conrey's polylog-inflated bound and the target
$C^{n}n!^{2}$ bound is precisely the density-of-zeros hypothesis for
$\zeta$, which is a major open problem on the same level as RH.
Density-of-zeros is strictly weaker than RH (density-1-hypothesis
does not imply all zeros are on the line, only that exceptional
zeros are sparse), so $\mathrm{R}_{\mathrm{analytic}}\not\Rightarrow$RH
is consistent, but $\mathrm{R}_{\mathrm{analytic}}$ is not known
unconditionally either.
\end{remark}

\subsection{Axis Distinction: Form vs Implication vs Difficulty}
\label{v4-arith:r-audit:axis-distinction}

The confusion in both $\delta$'s summary and the audit-prompt
hypothesis stems from blurring three distinct axes. We enumerate
them separately.

\begin{description}
\item[\textbf{Form axis:}] The \emph{statement} of
$\mathrm{R}_{\mathrm{analytic}}$ is a moment asymptotic
$C_{-}^{n}n!^{2}\leq\mu_{2n}\leq C_{+}^{n}n!^{2}$. This is
classical-analytic: the object is a moment sequence, the technique
is asymptotic analysis with saddle-point.
\item[\textbf{Implication axis:}] Does
$\mathrm{R}_{\mathrm{analytic}}\Rightarrow$ RH? \emph{No}. Off-line
zeros with subleading $2n$-th moment contribution are consistent
with $\mathrm{R}_{\mathrm{analytic}}$. Does
RH$\Rightarrow\mathrm{R}_{\mathrm{analytic}}$? \emph{Partially yes
under regularisation}: under RH, Riemann--von Mangoldt plus
saddle-point gives $\mu_{2n}\asymp (2\pi)^{-4n}(2n)!$ up to
logarithmic corrections, which is $C^{n}n!^{2}$ up to Stirling;
so RH (plus the density-hypothesis refinements) does give
$\mathrm{R}_{\mathrm{analytic}}$.
\item[\textbf{Difficulty axis:}] How hard is it to prove
$\mathrm{R}_{\mathrm{analytic}}$ unconditionally? \emph{Very hard};
the remaining gap (\Cref{v4-arith:r-audit:prop:closing-gap-is-density-RH})
is a density-of-zeros hypothesis, a major open problem of
analytic-number-theory at the level of the ``density-1 hypothesis''
and related; strictly weaker than RH but adjacent.
\end{description}

\begin{center}
\begin{tabular}{|l|l|l|l|}
\hline
 & \textbf{Form} & \textbf{Implication to RH} & \textbf{Difficulty} \\
\hline
R$_{\mathrm{analytic}}$ & Classical-analytic & Does not imply RH & RH-adjacent \\
 & (moment asymptotic) & (weaker than RH in & (density-of-zeros \\
 & & one direction) & unsolved) \\
\hline
\end{tabular}
\end{center}

$\delta$'s error in \Cref{v4-arith:r-audit:claim-paraphrase} is to
infer from the form being classical-analytic that the difficulty is
classical. This conflates the first and third axes. The second axis
(implication) supports $\delta$: $\mathrm{R}_{\mathrm{analytic}}$
is indeed strictly weaker than RH. But the third axis does not:
the difficulty is RH-adjacent.

\section{Chain Audit: $\det \Leftrightarrow \mathrm{VB}^{*} \Leftrightarrow \mathrm{RH}$}
\label{v4-arith:r-audit:chain-audit}

The audit-prompt hypothesis postulates the chain
\[
\text{determinacy}\;\Leftrightarrow\;\mathrm{VB}^{*}\text{ main direction}\;\Leftrightarrow\;\mathrm{RH}.
\]
Each step fails.

\subsection{Step 1: Determinacy $\Leftrightarrow$ VB$^{*}$ (Which Direction?)}
\label{v4-arith:r-audit:step-1}

\begin{proposition}[determinacy is equivalent to the VB$^{*}$ converse, not the main direction]
\label{v4-arith:r-audit:prop:det-converse-not-main}
\ClaimStatusProvedHere. Ch~36
\Cref{v4-arith:hpvbconv:thm:VBstar-converse-spectral-regularity}
asserts
\[
(\mathrm{VB^{*}c})\;\Leftrightarrow\;(\mathrm{Det})\;\Leftrightarrow\;
(\mathrm{HB+})\;\Leftrightarrow\;(\mathrm{BD+})
\]
under P2+P3+H-P$^{\mathrm{Ar}}$, where (VB$^{*}$c) is the statement
``F-RH $\Rightarrow$ VB$^{*}$''. The \emph{main direction}
(VB$^{*}\Rightarrow$ F-RH) is
\Cref{v4-arith:vb:thm:three-equivalences} and is a separate
conditional implication; it is NOT equivalent to determinacy.
\end{proposition}

\begin{proof}
Direct citation. Ch~36
\Cref{v4-arith:hpvbconv:rem:converse-does-not-help} further notes:
``The forward direction VB$^{*}\Rightarrow$F-RH is the separate
content of \Cref{v4-arith:rh:thm:VB-implies-F-RH}, which is a
conditional theorem under P2, P3, H-P$^{\mathrm{Ar}}$, and a positive
Bost--Connes regularised trace. The four-way equivalence of the
converse does not close the forward implication''. Hence
determinacy is tied to the converse, not the main direction.
\end{proof}

\subsection{Step 2: Determinacy vs RH}
\label{v4-arith:r-audit:step-2}

\begin{proposition}[determinacy is strictly finer than RH]
\label{v4-arith:r-audit:prop:det-strictly-finer}
\ClaimStatusProvedHere. Ch~36
\Cref{v4-arith:hpvbconv:cor:three-strictly-finer} asserts that each
of VB$^{*}$-converse, determinacy, and B\'aez-Duarte sharp decay is
strictly finer than RH. In particular, determinacy is
\emph{strictly stronger} than RH, not equivalent. The audit-prompt
hypothesis ``$\text{Det}\Leftrightarrow\mathrm{RH}$'' is wrong
in one direction: $\text{Det}\Rightarrow\mathrm{RH}$ holds only
when combined with H-P$^{\mathrm{Ar}}$ (Ch~36
\Cref{v4-arith:hpvbconv:thm:HPAr-implies-reduction}), and
RH$\Rightarrow\text{Det}$ is the strictly-finer-than-RH content that
is not proved from RH alone.
\end{proposition}

\begin{proof}
Ch~36 \Cref{v4-arith:hpvbconv:rem:strict-content} explicitly:
``No structural property of the zero distribution of $\xi$
independent of RH is known to force determinacy. Hence the
VB$^{*}$ converse requires an additional arithmetic input beyond
RH.'' Hence $\mathrm{RH}\not\Rightarrow\mathrm{Det}$. The reverse
implication $\mathrm{Det}\Rightarrow\mathrm{RH}$ is also not
automatic; it requires H-P$^{\mathrm{Ar}}$ plus the VB$^{*}$ main
direction (Ch~24 \Cref{v4-arith:vb:thm:three-equivalences}). So
the two are logically independent without auxiliary hypotheses.
\end{proof}

\subsection{Corrected Chain Diagram}
\label{v4-arith:r-audit:corrected-chain}

\begin{remark}[corrected position of $\mathrm{R}_{\mathrm{analytic}}$ in the chain]
\label{v4-arith:r-audit:diag:chain}
Under P2, P3, H-P$^{\mathrm{Ar}}$:
\begin{center}
$\mathrm{R}_{\mathrm{analytic}}\;\Longrightarrow\;\text{(Carleman criterion)}\;\Longrightarrow\;\text{Det}\;\Longleftrightarrow\;(\mathrm{VB}^{*}\text{ converse})$
\end{center}
Independently:
\begin{center}
$\mathrm{VB}^{*}\text{ main direction}\;\Longrightarrow\;\text{RH}$ (conditional, Ch~24 Prop.)
\end{center}
Combined, with both directions:
\begin{center}
$\text{H-P}^{\mathrm{Ar}}+\mathrm{R}_{\mathrm{analytic}}+\mathrm{VB}^{*}\text{ main direction}\;\Longrightarrow\;\mathrm{VB}^{*}\;\Longleftrightarrow\;\text{RH}+\text{Det}$
\end{center}
(Ch~36 \Cref{v4-arith:hpvbconv:thm:HPAr-implies-reduction}).
\end{remark}

\begin{remark}[$\mathrm{R}_{\mathrm{analytic}}$ closes the converse, not the forward]
\label{v4-arith:r-audit:rem:closes-converse}
In the corrected diagram, $\mathrm{R}_{\mathrm{analytic}}$ contributes
to the VB$^{*}$ \emph{converse} direction (RH$\Rightarrow$VB$^{*}$),
by forcing Hamburger determinacy which together with H-P$^{\mathrm{Ar}}$
and RH closes VB$^{*}$. $\mathrm{R}_{\mathrm{analytic}}$ does not
by itself close the VB$^{*}$ \emph{main} direction nor RH. The
main-direction closure requires VB$^{*}$ proper (as hypothesis) or
some alternative route to the Li positivity, which is not supplied
by $\mathrm{R}_{\mathrm{analytic}}$.
\end{remark}

\section{Verdict}
\label{v4-arith:r-audit:verdict}

\begin{theorem}[audit verdict on $\delta$'s claim about $\mathrm{R}_{\mathrm{analytic}}$]
\label{v4-arith:r-audit:thm:verdict}
\ClaimStatusProvedHere. Of the four claims extracted from $\delta$'s
Ch~41 in \Cref{v4-arith:r-audit:claim-1}--\Cref{v4-arith:r-audit:claim-paraphrase}:
\begin{enumerate}
\item[\textbf{(1)}] \textbf{Form}: $\mathrm{R}_{\mathrm{analytic}}$ has
classical-analytic form (moment asymptotic). \textbf{CORRECT}.
\item[\textbf{(2)}] \textbf{Strict weakness}: $\mathrm{R}_{\mathrm{analytic}}
\not\Rightarrow\mathrm{RH}$. \textbf{CORRECT} (subject to the
caveat that the converse $\mathrm{RH}\Rightarrow\mathrm{R}_{\mathrm{analytic}}$
also requires density-of-zeros refinements beyond RH alone).
\item[\textbf{(3)}] \textbf{Unconditional partial Carleman
divergence}: the BL+Conrey route gives a Carleman sum lower bound
of the form $\sum(\log n)/\sqrt{n}$ that diverges.
\textbf{INCORRECT} at the arithmetic level:
\Cref{v4-arith:r-audit:prop:BL-carleman-corrected} shows the
correct lower bound is $\sum 1/(n(\log n)^{2})$, a Bertrand series
that \emph{converges}, hence the BL route is inconclusive, not
divergent.
\item[\textbf{(4)}] \textbf{Summary paraphrase ``classical-analytic,
not RH-equivalent''}: \textbf{PARTIALLY CORRECT}. Not RH-equivalent
in the implication-direction sense, yes; but ``classical-analytic''
conflates form with difficulty, and the actual difficulty
(closing $(\log n)^{4n}\to 1$ via density-of-zeros improvements) is
RH-adjacent, not classical-elementary.
\end{enumerate}
\end{theorem}

\begin{proof}
(1) is \Cref{v4-arith:r-audit:cycle-4} item~1; the form is
manifestly a moment asymptotic.

(2) is \Cref{v4-arith:hpar-det:rem:residual-vs-RH} combined with
the off-line-zeros robustness argument.

(3) is \Cref{v4-arith:r-audit:prop:BL-carleman-corrected} combined
with \Cref{v4-arith:r-audit:cor:inconclusive}. The Stirling
correction $(2n)!^{1/(2n)}\sim 2n/e$ (not $\sqrt{2n}$) changes the
Carleman sum from divergent to convergent.

(4) is \Cref{v4-arith:r-audit:rem:form-vs-difficulty} combined with
\Cref{v4-arith:r-audit:prop:closing-gap-is-density-RH}: closing the
logarithmic gap requires density-of-zeros improvements which, while
strictly weaker than RH, constitute a major open programme of
analytic number theory, not a classical closure.
\end{proof}

\begin{corollary}[the audit-prompt hypothesis is also partially wrong]
\label{v4-arith:r-audit:cor:hypothesis-also-wrong}
\ClaimStatusProvedHere. The audit prompt's conjecture
``$\mathrm{R}_{\mathrm{analytic}}$ IS RH-equivalent because
determinacy $\Leftrightarrow$ VB$^{*}$ main direction
$\Leftrightarrow$ RH'' is incorrect in two places:
\begin{enumerate}
\item Determinacy is equivalent to VB$^{*}$ \emph{converse}, not
main direction (\Cref{v4-arith:r-audit:prop:det-converse-not-main}).
\item Determinacy is strictly finer than RH, not equivalent
(\Cref{v4-arith:r-audit:prop:det-strictly-finer}).
\end{enumerate}
Hence the chain determinacy $\Leftrightarrow$ RH has no direct
proof, and $\mathrm{R}_{\mathrm{analytic}}$ at the implication
level is genuinely weaker than RH.
\end{corollary}

\begin{proof}
Direct citations of Ch~24
\Cref{v4-arith:vb:thm:three-equivalences} and Ch~36
\Cref{v4-arith:hpvbconv:thm:VBstar-converse-spectral-regularity}.
Both texts are unambiguous about the direction of the equivalence
and about the strict finer-ness of determinacy above RH.
\end{proof}

\section{Honest Reduction}
\label{v4-arith:r-audit:honest-reduction}

We now produce the honest reduction of $\mathrm{R}_{\mathrm{analytic}}$
that $\delta$'s Ch~41 aimed to produce.

\begin{theorem}[honest reduction of $\mathrm{R}_{\mathrm{analytic}}$]
\label{v4-arith:r-audit:thm:honest-reduction}
\ClaimStatusProvedHere. The sharp moment asymptotic
$\mathrm{R}_{\mathrm{analytic}}:\mu_{2n}\asymp C^{n}n!^{2}$
unconditionally satisfies:
\begin{enumerate}
\item \emph{Form}: classical-analytic (a moment asymptotic).
\item \emph{Position in chain}: under P2+P3+H-P$^{\mathrm{Ar}}$,
$\mathrm{R}_{\mathrm{analytic}}\Rightarrow$Carleman criterion
$\Rightarrow$Det$\Leftrightarrow$VB$^{*}$ converse; does not imply
RH directly.
\item \emph{Implication to RH}:
$\mathrm{R}_{\mathrm{analytic}}\not\Rightarrow$RH (strictly weaker
in the implication direction).
\item \emph{Difficulty}: closing the unconditional
$(\log n)^{4n}$ gap requires density-of-zeros hypotheses for

difficulty but technically strictly weaker than RH.
\item \emph{Unconditional partial Carleman route via BL+Conrey}:
\emph{inconclusive}, not divergent; the correct Bertrand-series
lower bound $\sum 1/(n(\log n)^{2})$ converges.
\end{enumerate}
Hence $\mathrm{R}_{\mathrm{analytic}}$ is classical in FORM,
strictly weaker than RH in IMPLICATION, but RH-adjacent in
DIFFICULTY. $\delta$'s phrasing
``classical-analytic, not RH-equivalent'' captures 2--3 but elides
4; the phrasing is partially correct and partially misleading.
\end{theorem}

\begin{proof}
(1) is \Cref{v4-arith:r-audit:axis-distinction}.
(2) is \Cref{v4-arith:r-audit:diag:chain}.
(3) is \Cref{v4-arith:hpar-det:rem:residual-vs-RH}.
(4) is \Cref{v4-arith:r-audit:prop:closing-gap-is-density-RH}.
(5) is \Cref{v4-arith:r-audit:cor:inconclusive}.
\end{proof}

\section{Summary}
\label{v4-arith:r-audit:summary}

\begin{enumerate}
\item \textbf{$\delta$'s claim that $\mathrm{R}_{\mathrm{analytic}}$ is
``classical-analytic, not RH-equivalent'' is partially correct
and partially misleading}
(\Cref{v4-arith:r-audit:thm:verdict}). Classical-analytic FORM is
correct; not-RH-equivalent in IMPLICATION is correct; but
classical-analytic DIFFICULTY is incorrect. The correct difficulty
is RH-adjacent (density-of-zeros programme).
\item \textbf{$\delta$'s unconditional Carleman-divergence argument
via BL+Conrey has an arithmetic error}
(\Cref{v4-arith:r-audit:cor:inconclusive}). The correct lower bound
on the Carleman sum from BL+Conrey bounds is the Bertrand series
$\sum 1/(n(\log n)^{2})$, which converges. The BL route is
inconclusive, not divergent.
\item \textbf{The audit-prompt hypothesis ``determinacy
$\Leftrightarrow$ VB$^{*}$ main direction $\Leftrightarrow$ RH''
is also incorrect}
(\Cref{v4-arith:r-audit:cor:hypothesis-also-wrong}). Determinacy is
equivalent to the VB$^{*}$ \emph{converse} direction (Ch~36) and
is strictly finer than RH, not equivalent.
\item \textbf{The honest reduction of $\mathrm{R}_{\mathrm{analytic}}$}
(\Cref{v4-arith:r-audit:thm:honest-reduction}) separates three
axes: form (classical), implication direction (does not imply RH),
and difficulty (RH-adjacent via density-of-zeros).
$\mathrm{R}_{\mathrm{analytic}}$ remains the single analytic
residual of the minimal path of Vol~IV, but its closure is not
mere bookkeeping: it is a genuine open problem of analytic number
theory at the zero-density level.
\item \textbf{$\delta$'s three-residual decomposition survives}: the
three residuals (W, H, $\mathrm{R}_{\mathrm{analytic}}$) of $\delta$'s
\Cref{v4-arith:hpar-det:cor:wave3-structure} remain the tightest
operational reduction of the Ch~37 RH completion path. Only the
arithmetic of $\delta$'s Carleman claim and $\delta$'s difficulty
framing are corrected.
\item \textbf{Centennial obstacle clarified}: closing
$\mathrm{R}_{\mathrm{analytic}}$ requires either (a) a density-of-zeros
improvement pushing the Conrey lower bound from 40\%-of-zeros-on-line
to density-1-hypothesis with sub-logarithmic error, or (b) an
alternative positivity or measure-regularisation route that
produces $\mu_{2n}\asymp C^{n}n!^{2}$ directly without
density-of-zeros input. Either route is a substantive
analytic-number-theory programme, not a classical exercise. The
Vol~IV minimal path to RH therefore rests on three outputs of
significant analytic work (W, H, $\mathrm{R}_{\mathrm{analytic}}$),
not on a mixture of classical and hard statements.
\end{enumerate}
