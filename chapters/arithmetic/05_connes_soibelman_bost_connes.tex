\section{The Bost--Connes $C^{*}$-Algebra Explicit}
\label{v4-arith:cs:bost-connes-explicit}

\begin{definition}[Bost--Connes 1995, \S1]
\label{v4-arith:def:bost-connes-algebra}
The Bost--Connes Q-lattice $C^{*}$-algebra $C_{\mathbb{Q}}^{*}$ has
generators
\begin{align}
\{\mu_{n}\}_{n\in\mathbb{N}^{\times}},\quad
\{\mu_{n}^{*}\}_{n\in\mathbb{N}^{\times}},\quad
\{e(\gamma)\}_{\gamma\in\mathbb{Q}/\mathbb{Z}},
\end{align}
with relations
\begin{align}
\mu_{n}\mu_{m}\;&=\;\mu_{nm}, \\
\mu_{n}^{*}\mu_{n}\;&=\;1, \\
\mu_{n}\mu_{n}^{*}\;&=\;e_{n}\;=\;\frac{1}{n}\sum_{\gamma:n\gamma=0}e(\gamma), \\
e(\gamma_{1})e(\gamma_{2})\;&=\;e(\gamma_{1}+\gamma_{2}), \\
\mu_{n}e(\gamma)\mu_{n}^{*}\;&=\;\rho_{n}(e(\gamma))\;=\;\frac{1}{n}\sum_{\delta:n\delta=\gamma}e(\delta).
\end{align}
The $e(\gamma)$ generate the group $C^{*}$-algebra
$C^{*}(\mathbb{Q}/\mathbb{Z})$; the $\mu_{n}$ furnish an endomorphism
semigroup action; $\mu_{n}^{*}\mu_{n}=1$, $\mu_{n}\mu_{n}^{*}=e_{n}$
makes $\mu_{n}$ a Toeplitz isometry. The crossed product
$C^{*}(\mathbb{Q}/\mathbb{Z})\rtimes\mathbb{N}^{\times}$ yields
$C_{\mathbb{Q}}^{*}$.
\end{definition}

\begin{theorem}[KMS trace, Bost--Connes~Prop.~3.3 and Thm~5.4]
\label{v4-arith:thm:BC-KMS}
\ClaimStatusProvedElsewhere. The KMS${}_{\beta}$ state $\varphi_{\beta}$
on $C_{\mathbb{Q}}^{*}$ for $\beta>1$ is unique and extremal; its
partition function is
\begin{equation}
Z_{\mathrm{BC}}(\beta)\;=\;\mathrm{tr}(e^{-\beta H})\;=\;\sum_{n\geq 1}n^{-\beta}\;=\;\zeta(\beta),
\end{equation}
where $H=\log N$ generates $\sigma_{t}(\mu_{n})=n^{it}\mu_{n}$. The
$\zeta$-partition function is explicitly motivated by Julia's
primon gas (\emph{Statistical theory of numbers}, Springer
Proc.~Phys.~47, 1990).
\end{theorem}

\section{The Structural Type Mismatch}
\label{v4-arith:cs:type-mismatch}

\begin{observation}
\label{v4-arith:obs:CCR-vs-Cuntz-Toeplitz}
The Heisenberg CCR $[a_{m}^{(p)},a_{n}^{(q)}]=m\delta_{m+n,0}\delta_{pq}$
is a relation on unbounded self-adjoint operators on a Fock space
with countable spectrum; $\mu_{n}\mu_{n}^{*}=e_{n}$ is a
Toeplitz--Cuntz relation on bounded partial isometries with
spectrum $\{1/n\}\cup\{0\}$. These are distinct operator-algebraic
structures.
\end{observation}

\begin{corollary}[no $C^{*}$-algebra homomorphism]
\label{v4-arith:cor:no-CC-to-Fock-homomorphism}
\ClaimStatusProvedHere. No algebra homomorphism
$C_{\mathbb{Q}}^{*}\to\mathrm{End}(\mathcal{V}^{\mathrm{prim}})$ sending
$\mu_{n}\mapsto\text{anything bounded on Fock with the right spectrum}$
exists, because the Fock representation has no rank-$1/n$
projections.
\end{corollary}

\section{Action on the Zero-Mode Lattice}
\label{v4-arith:cs:zero-mode-action}

\begin{definition}[zero-mode lattice]
\label{v4-arith:def:zero-mode-lattice}
Let $L\subset\mathcal{V}^{\mathrm{prim}}$ be the lattice
\begin{equation}
L\;=\;\Bigl\{\prod_{p}|n_{p}\rangle_{p}\colon n=\prod_{p}p^{n_{p}}\in\mathbb{N}^{\times}\Bigr\},
\end{equation}
where $|n_{p}\rangle_{p}=(a^{(p)}_{-1})^{n_{p}}|0\rangle_{p}/\sqrt{n_{p}!}$
is the $n_{p}$-th Fock number state. Then $L\cong\ell^{2}(\mathbb{N}^{\times})$
as a Hilbert space.
\end{definition}

\begin{theorem}[Bost--Connes acts on $L$]
\label{v4-arith:thm:BC-acts-on-L}
\ClaimStatusProvedHere. $C_{\mathbb{Q}}^{*}$ acts faithfully on
the zero-mode lattice $L\subset\mathcal{V}^{\mathrm{prim}}$ via:
\begin{itemize}
\item $\mu_{n}\delta_{m}=\delta_{nm}$,
\item $\mu_{n}^{*}\delta_{m}=\delta_{m/n}$ if $n\mid m$, else zero,
\item $e(\gamma)\delta_{m}=\zeta_{m}(\gamma)\delta_{m}$ with
$\zeta_{m}$ a primitive $m$-th root of unity.
\end{itemize}
This is the Bost--Connes representation from 1995~\S1.2, Prop.~1,
transported to the zero-mode lattice of $\mathcal{V}^{\mathrm{prim}}$.
\end{theorem}

\begin{observation}[no extension off-lattice]
\label{v4-arith:obs:no-off-lattice-extension}
The Bost--Connes action does not extend to $\mathrm{End}(\mathcal{V}^{\mathrm{prim}})$
because off-lattice states (coherent states, superposition of
number states, descendants with higher Heisenberg modes) have no
Bost--Connes action.
\end{observation}

\section{The Tensor Decomposition}
\label{v4-arith:cs:tensor-decomposition}

\begin{theorem}[tensor-product placement of $\mathcal{V}^{\mathrm{prim}}$]
\label{v4-arith:thm:BC-tensor-placement}
\ClaimStatusProvedHere. For each finite $N$, let
$\mathcal{V}^{\mathrm{prim}}_{N}=\bigotimes_{p\leq N}\mathcal{H}^{(p)}$
be the rank-$\pi(N)$ truncation. Then as graded vector spaces,
\begin{equation}
\mathcal{V}^{\mathrm{prim}}_{N}\;\cong\;
\underbrace{C_{\mathbb{Q},N}^{*}\bigr|_{\mathrm{lattice}}}_{\text{arithmetic}}
\;\otimes\;
\underbrace{\mathcal{H}^{\otimes\pi(N)}_{\mathrm{descendant}}}_{\text{universal}}
\end{equation}
where the left factor is the Bost--Connes action on the rank-$\pi(N)$
zero-mode lattice, and the right factor is the universal
Heisenberg-descendant tower of rank $\pi(N)$ carrying all
higher-mode data $a^{(p)}_{-n}$ for $n\geq 2$.
\end{theorem}

\begin{corollary}
\label{v4-arith:cor:arithmetic-content-on-lattice}
\ClaimStatusProvedHere. All arithmetic content of
$\mathcal{V}^{\mathrm{prim}}$ lives in the Bost--Connes factor
(left-hand side). The Heisenberg descendant tower depends only on
the rank $\pi(N)$, not on the arithmetic nature of the primes; it
is a universal rank-$\pi(N)$ Heisenberg dressing.
\end{corollary}

\section{Kesten--McKay = Plancherel of Unramified $GL_{2}(\mathbb{Q}_{p})$}
\label{v4-arith:cs:kesten-mckay-plancherel}

\begin{theorem}[Kesten 1959; McKay 1981; Lubotzky 1994~Thm~7.4.5]
\label{v4-arith:thm:kesten-mckay-plancherel}
\ClaimStatusProvedElsewhere. The Kesten--McKay distribution on
$[-2\sqrt{p},2\sqrt{p}]$ for $q=p$ prime,
\begin{equation}
d\nu_{p}(x)\;=\;\frac{(p+1)\sqrt{4p-x^{2}}}{2\pi((p+1)^{2}-x^{2})}\,dx,
\end{equation}
is the spectral measure of the adjacency operator on the
$(p+1)$-regular tree, equivalently the $p$-adic Bruhat--Tits tree
of $\mathrm{PGL}_{2}(\mathbb{Q}_{p})$, equivalently the Plancherel
measure of the spherical-function algebra on the tree.
\end{theorem}

\begin{corollary}
\label{v4-arith:cor:kesten-mckay-is-local-hecke}
\ClaimStatusProvedElsewhere. The Kesten--McKay averaging of HY's
primon partition is the Plancherel averaging of the local
Bost--Connes Hecke algebra at $p$. Equivalently: averaging against
$\nu_{p}$ is the Plancherel decomposition of the unramified
spherical representation of $GL_{2}(\mathbb{Q}_{p})$.
\end{corollary}

\begin{remark}[not bar-cobar averaging]
\label{v4-arith:rem:not-bar-cobar-averaging}
The ``arithmetic averaging map'' $\mathrm{av}^{\mathrm{arith}}$ is not
the operadic averaging $\mathfrak{g}^{E_{1}}\to\mathfrak{g}^{\mathrm{mod}}$
of Vol~I. It is the Plancherel decomposition of an automorphic
representation. The two have the same formal shape (projection
onto a symmetric sector weighted by a measure) but distinct
structural content. The operadic averaging lives on chiral
cohomology; the Plancherel averaging lives on spherical
representations.
\end{remark}

\section{Hochschild Comparison: Connes--Marcolli vs Claimed Theorem H$^{\mathrm{arith}}$}
\label{v4-arith:cs:HH-comparison}

\begin{theorem}[Connes--Marcolli math/0404128 \S1.8]
\label{v4-arith:thm:CM-HH-BC}
\ClaimStatusProvedElsewhere.
\begin{align}
HH^{0}(C_{\mathbb{Q}}^{*})\;&=\;\mathbb{C}\quad(\text{centre: scalars}), \\
HH^{1}(C_{\mathbb{Q}}^{*})\;&=\;\mathbb{C}\quad(\text{outer derivations, generated by }\delta=\frac{d}{dt}\sigma_{t}), \\
HH^{2}(C_{\mathbb{Q}}^{*})\;&=\;0\quad(\text{type-III}_{1}\text{ factor}), \\
HH^{k}(C_{\mathbb{Q}}^{*})\;&=\;0\quad\text{for }k\geq 2.
\end{align}
\end{theorem}

\begin{observation}
\label{v4-arith:obs:CM-stricter-than-H-arith}
Connes--Marcolli's $HH^{\bullet}$ has finite-dimensional
$HH^{0}$ and $HH^{1}$ and vanishing $HH^{\geq 2}$. This is
\emph{strictly sharper} than the claimed Theorem H$^{\mathrm{arith}}$
concentration in $\{0,1,2\}$.
\end{observation}

\begin{theorem}[healed Theorem H$^{\mathrm{arith}}$]
\label{v4-arith:thm:H-arith-healed}
\ClaimStatusConjectured. $HH^{\bullet}(\mathcal{V}^{\mathrm{prim}})$ has:
\begin{itemize}
\item $HH^{0}=\mathbb{Z}$ (Bost--Connes centre),
\item $HH^{1}=\bigoplus_{p}\log p\cdot\mathbb{Z}$ (KMS
generators, \Cref{v4-arith:thm:CM-HH-BC}),
\item $HH^{2}=\Lambda$-torsor related to Witt cotangent
(conjectural).
\end{itemize}
Concentration in $\{0,1,2\}$ holds over $\mathbb{Q}$; integral
concentration is Conjecture.
\end{theorem}

\section{Does $\mathcal{V}^{\mathrm{prim}}$ Add Anything to Bost--Connes?}
\label{v4-arith:cs:new-content}

\begin{question}
Does the VA structure on $\mathcal{V}^{\mathrm{prim}}$ produce new
invariants beyond what Bost--Connes already encodes?
\end{question}

\begin{answer}[universal, non-arithmetic]
\label{v4-arith:ans:universal-non-arithmetic}
\ClaimStatusProvedHere. The VA enhancement on
$\mathcal{V}^{\mathrm{prim}}_{N}$ (rank $\pi(N)$) reproduces the
Heisenberg archetype $\mathsf{G}$ of the Vol~I landscape census,
with $\kappa=\pi(N)$, $r_{\max}=2$, chiral Hochschild concentrated
in $\{0,1,2\}$. These invariants depend only on the rank $\pi(N)$,
not on the primes themselves. The arithmetic indexing by primes is
invisible to the VA invariants.
\end{answer}

\begin{corollary}[placement in five-archetype classification]
\label{v4-arith:cor:placement-G-archetype}
\ClaimStatusProvedHere. $\mathcal{V}^{\mathrm{prim}}$ sits in
archetype $\mathsf{G}$ (Heisenberg / Gaussian), depth $r_{\max}=2$,
$\kappa_{N}=\pi(N)$ (finite) or formal series (infinite).
\emph{This is not a new archetype}; it is a particular infinite-rank
limit of the Heisenberg class.
\end{corollary}

\section{The Correct Identification}
\label{v4-arith:cs:correct-identification}

\begin{theorem}[healed placement of $\mathcal{V}^{\mathrm{prim}}$]
\label{v4-arith:thm:CS-healed-placement}
\ClaimStatusProvedHere. $\mathcal{V}^{\mathrm{prim}}$ is best
understood as the restricted tensor product
\begin{equation}
\mathcal{V}^{\mathrm{prim}}_{N}\;=\;\bigl(\text{Bost--Connes module at rank }\pi(N)\bigr)
\otimes\bigl(\text{universal Heisenberg rank-}\pi(N)\text{ VA}\bigr).
\end{equation}
All arithmetic content lives in the Bost--Connes factor (Hecke
algebra, KMS states, $\zeta$/$L$-function partition function,
Plancherel--Kesten--McKay averaging, prime indexing). The VA
factor contributes the universal Heisenberg descendant tower,
depending only on rank, devoid of arithmetic.
\end{theorem}

\begin{remark}
$\mathcal{V}^{\mathrm{prim}}$ is a \emph{module over} Bost--Connes,
not a chiral-algebraic extension. The VA structure and the
arithmetic symmetries commute, and the VA factor contributes
nothing arithmetic beyond indexing cardinality.
\end{remark}

\begin{remark}[role of Vol~III $\mathsf{B}$-row]
A genuine VA-arithmetic enhancement --- a chiral algebra whose
stress tensor encodes arithmetic data (Hecke eigenvalues, not just
prime indexing), with chiral Hochschild cohomology recovering
$HH^{\bullet}(C_{\mathbb{Q}}^{*})$ as a subquotient --- would sit
in the $\mathsf{B}$-row of Theorem~C (Borcherds/BKM archetype,
$K^{\kappa}=8$) by the universal positive-geometry template
(Vol~III, AP929--936). $\mathcal{V}^{\mathrm{prim}}$ does not
achieve this; it stays in class $\mathsf{G}$. Upgrading requires
Borcherds-product encoding of Hecke data at the level of the
stress tensor, which is the Vol~III $\Phi_{N}=c_{N}(0)/2$ programme.
\end{remark}

\section{Summary of This Chapter}
\label{v4-arith:cs:summary}

\begin{enumerate}
\item Bost--Connes $C^{*}_{\mathbb{Q}}$ has $\zeta(\beta)$ as KMS
partition function (\Cref{v4-arith:thm:BC-KMS}).
\item CCR vs Toeplitz--Cuntz are structurally distinct operator
algebras; no $C^{*}$-homomorphism exists
(\Cref{v4-arith:cor:no-CC-to-Fock-homomorphism}).
\item Bost--Connes acts faithfully on the zero-mode lattice
$L\subset\mathcal{V}^{\mathrm{prim}}$, not on the whole VA
(\Cref{v4-arith:thm:BC-acts-on-L}).
\item The decomposition
$\mathcal{V}^{\mathrm{prim}}=\text{Bost--Connes}\otimes\text{universal Heisenberg}$
(\Cref{v4-arith:thm:BC-tensor-placement}): all arithmetic on the
left, all VA on the right.
\item Kesten--McKay = Plancherel of unramified
$GL_{2}(\mathbb{Q}_{p})$ = Bost--Connes local Hecke spectrum =
Soibelman CoHA on Jordan quiver
(\Cref{v4-arith:cor:kesten-mckay-is-local-hecke},
\Cref{v4-arith:prop:HY-BC-Soibelman-dictionary}).
\item Connes--Marcolli $HH^{\bullet}$ is strictly sharper than
the claimed Theorem H$^{\mathrm{arith}}$
(\Cref{v4-arith:thm:CM-HH-BC}).
\item $\mathcal{V}^{\mathrm{prim}}$ sits in class $\mathsf{G}$
with $\kappa=\pi(N)$, $r_{\max}=2$; upgrading to $\mathsf{B}$-row
requires the Vol~III Borcherds encoding
(\Cref{v4-arith:cor:placement-G-archetype}).
\end{enumerate}

The chiral-algebraic enhancement over Bost--Connes is universal
and non-arithmetic; the arithmetic content is already in the
$C^{*}$-algebra. The upgrade to a genuine VA-arithmetic object is
the Vol~III programme.
