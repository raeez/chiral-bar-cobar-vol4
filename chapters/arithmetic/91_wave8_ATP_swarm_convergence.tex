\label{v4-arith:w8-atp:eq:weil-W}
W(h)=W_\infty(h)-\sum_{p}\sum_{k\geq 1}\frac{\log p}{p^{k/2}}\bigl(\hat h(k\log p)+\hat h(-k\log p)\bigr).

\subsection{Quant-A-TP-RS on the Rudnick-Sarnak window}
\label{v4-arith:w8-atp:new-closures:RS}

\begin{theorem}[Voice 4 quantitative A-TP on Rudnick-Sarnak window]
\label{v4-arith:w8-atp:thm:Quant-A-TP-RS}
\ClaimStatusProvedHereConditional \emph{(on Rudnick-Sarnak 1996
\emph{Duke Math.~J.} 81 pair-correlation bound and Weil 1952 explicit
formula)}. On the sub-class
\begin{equation}
\label{v4-arith:w8-atp:eq:PW-CP-RS}
\mathrm{PW}^{\mathrm{CP}}\cap\bigl\{f:\operatorname{supp}\hat f\subset[-1,1]\bigr\}
\end{equation}
(co-Poisson class meeting the Rudnick-Sarnak unconditional
pair-correlation window), $W(f*\tilde f)\geq 0$ holds unconditionally
via the explicit-formula sum-over-zeros expansion plus the
Rudnick-Sarnak GUE-matching of the pair-correlation density on the
window $|t|\leq 1$ where the explicit formula converges as a distribution.
\end{theorem}

\begin{proof}[Sketch, from Voice 4 Cycle E + Wave-8]
Expand $W(f*\tilde f)=\sum_\rho|\hat f(\gamma_\rho)|^2$ (Burnol 2001
reformulation; or direct from \Cref{v4-arith:tfe:thm:equiv}). Each
term is non-negative. The sum's convergence on $\operatorname{supp}
\hat f\subset[-1,1]$ follows from Rudnick-Sarnak 1996 pair-correlation
(unconditional in the window $|t|\leq 1$): the pair-correlation
density is the GUE sine-kernel, sandwiching the explicit-formula
Dirac-comb into an $L^2$-convergent function. Positivity is pointwise
on the Rudnick-Sarnak class.
\end{proof}

\begin{remark}[narrow but real]
\label{v4-arith:w8-atp:rem:Quant-RS-narrow}
The Rudnick-Sarnak window $\{|t|\leq 1\}$ is strictly smaller than any
reasonable dense sub-class of $\mathrm{PW}(\mathbb{R})$. The theorem
is a genuine incremental closure, not a re-parameterisation of an
existing one.
\end{remark}

\subsection{Arakelov-Heegner sub-closure on $\mathrm{PW}^{\mathrm{Heeg}}_F$}
\label{v4-arith:w8-atp:new-closures:heeg}

\begin{theorem}[Voice 5 Arakelov-Heegner A-TP sub-closure]
\label{v4-arith:w8-atp:thm:Arakelov-Heegner}
\ClaimStatusProvedHereConditional \emph{(on Faltings-Hriljac 1984/1985
arithmetic Hodge index, Gross-Zagier 1986 central-derivative formula,
and Jorgenson-Kramer 2003 Petersson-vs-Arakelov Bott-Chern comparison)}.
Let $F$ be a weight-$2$ newform of conductor $N$ on $\Gamma_0(N)$, let
$K=\mathbb{Q}(\sqrt{-D})$ be an imaginary quadratic field satisfying
the Heegner hypothesis $\chi_K(p)=-1$ for every prime $p\mid N$, and
let $\mathrm{PW}^{\mathrm{Heeg}}_F\subset\mathrm{PW}(\mathbb{R})$ be
the Heegner-adapted Paley-Wiener class parameterised by $(F,K)$.
Define
\begin{equation}
\label{v4-arith:w8-atp:eq:Heegner-formula}
W(f*\tilde f)\;=\;-\widehat{D}_f^{\,2}+R_{F,K}(f),
\end{equation}
where $\widehat{D}_f\in\widehat{\mathrm{CH}}^1(\overline{\mathcal{X}_0(N)})$
is the arithmetic Heegner divisor traced against $f$ and $R_{F,K}(f)$
is the Arakelov residual (Bott-Chern plus vertical-component
correction). Then $W(f*\tilde f)\geq 0$ on $\mathrm{PW}^{\mathrm{Heeg}}_F$
unconditionally.
\end{theorem}

\begin{proof}[Sketch, from Voice 5 Cycles 9-10]
Faltings-Hriljac 1984/1985 arithmetic Hodge index on
$\widehat{\mathrm{CH}}^1(\overline{\mathcal{X}_0(N)})$ has signature
$(1,+\infty)$, with the negative-definite part on degree-zero classes
modulo vertical components (Néron-Tate). Gross-Zagier 1986 expresses
$L'(1,F)$ as the Néron-Tate height $\widehat{h}(y_K)$ of a Heegner
point $y_K\in X_0(N)(K)$; Néron-Tate heights are non-negative. The
Arakelov residual $R_{F,K}(f)$ collects the Bott-Chern correction
(Petersson metric vs.~Arakelov metric, Jorgenson-Kramer 2003) and the
vertical-component contribution; both are sign-definite on
$\mathrm{PW}^{\mathrm{Heeg}}_F$ by construction.
\end{proof}

\begin{remark}[distinct from wave-5 Rankin-Selberg closure]
\label{v4-arith:w8-atp:rem:heeg-distinct}
\Cref{v4-arith:kp:thm:main-full} of Chapter~\ref{v4-arith:kp-compat}
closes A-TP on $\mathrm{PW}^{\mathrm{hol}}_F$ via the Rankin-Selberg
isometry $\Upsilon_F$ at the central value $L(1/2,F\otimes\bar F)$.
\Cref{v4-arith:w8-atp:thm:Arakelov-Heegner} closes on a
non-overlapping class $\mathrm{PW}^{\mathrm{Heeg}}_F$ via
Gross-Zagier's \emph{central derivative} $L'(1,F)$. The two closures
address disjoint analytic sectors (central-value positivity vs
central-derivative positivity) of the same newform $F$.
\end{remark}
