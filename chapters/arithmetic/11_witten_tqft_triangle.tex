\chapter{The Automorphic Vertex of the Langlands--TQFT Triangle}
\label{v4-arith:witten}

\emph{In the voice of Witten. Kapustin--Witten (hep-th/0604151)
realised geometric Langlands as S-duality in $\mathcal{N}=4$
SYM on $\Sigma\times C$. We test whether the Hartnoll--Yang
BKL--primon correspondence supports a genuine arithmetic TQFT
bulk.}

\section{Kapustin--Witten Does Not Lift to Arithmetic}
\label{v4-arith:wit:KW-does-not-lift}

\begin{theorem}[Kapustin--Witten setup]
\label{v4-arith:thm:KW-setup}
\ClaimStatusProvedElsewhere. Geometric Langlands for $C/\mathbb{C}$
arises as S-duality of 4D $\mathcal{N}=4$ SYM on $\Sigma\times C$
with $\Sigma$ topologically twisted and $C$ holomorphically
twisted. Categories $D(\mathrm{Bun}_{G})$ and
$\mathrm{QCoh}(\mathrm{Loc}_{^{L}G})$ appear as boundary
conditions; S-duality $\tau\mapsto-1/\tau$ exchanges them.
\end{theorem}

\begin{observation}[arithmetic obstruction]
\label{v4-arith:obs:KW-arithmetic-obstruction}
$\overline{C}=\overline{\mathrm{Spec}\,\mathbb{Z}}$ is not a smooth
complex curve, not a Riemann surface, not an algebraic curve over
any field. The holomorphic twist of $\mathcal{N}=4$ requires a
complex structure on the compactified direction;
$\overline{\mathrm{Spec}\,\mathbb{Z}}$ has none. The residue field
at a closed point $\mathfrak{p}=(p)$ is $\mathbb{F}_{p}$, not
$\mathbb{C}$.
\end{observation}

\begin{corollary}[no arithmetic 4D SYM]
\label{v4-arith:cor:no-arith-4D-SYM}
\ClaimStatusProvedHere. ``$\mathcal{N}=4$ SYM on $\Sigma\times\overline{\mathrm{Spec}\,\mathbb{Z}}$''
is not a definition of any theory in 2026's mathematics.
Kapustin--Witten setup fails at the first step: no holomorphic
twist, no topological $\Sigma$-coupling, no $\tau\mapsto-1/\tau$.
\end{corollary}

\section{BLLPRvR Does Not Accommodate $\mathcal{V}^{\mathrm{prim}}$}
\label{v4-arith:wit:BLLPRvR-fails}

\begin{theorem}[BLLPRvR, arXiv:1312.5344]
\label{v4-arith:thm:BLLPRvR}
\ClaimStatusProvedElsewhere. The BLLPRvR correspondence assigns a
VA $\chi(\mathcal{T})$ to any 4D $\mathcal{N}=2$ SCFT $\mathcal{T}$
with $c_{2d}=-12c_{4d}$ and $k_{2d}=-\tfrac{1}{2}k_{4d}$.
\end{theorem}

\begin{observation}
\label{v4-arith:obs:BLLPRvR-rank-mismatch}
The primon VA has $c=\aleph_{0}$ (countable infinity) and flavour
rank $=|\mathbb{P}|=\aleph_{0}$. No 4D $\mathcal{N}=2$ SCFT has
flavour symmetry of rank $\aleph_{0}$: rank of flavour symmetry is
bounded by gauge rank plus hypermultiplet content, both finite.
\end{observation}

\begin{corollary}
\label{v4-arith:cor:BLLPRvR-fails-arith}
\ClaimStatusProvedHere. $\mathcal{V}^{\mathrm{prim}}$ is not the
BLLPRvR VA of any 4D $\mathcal{N}=2$ SCFT. Class-$\mathcal{S}$ on
an infinite-genus / infinite-puncture arithmetic curve would
require accepting $\overline{\mathrm{Spec}\,\mathbb{Z}}$ as a
Riemann surface (same obstruction as
\Cref{v4-arith:obs:KW-arithmetic-obstruction}).
\end{corollary}

\section{3D Chern--Simons with Integer Level Does Not Help}
\label{v4-arith:wit:3D-CS-fails}

\begin{observation}[non-integer level]
\label{v4-arith:obs:non-integer-level}
Witten's CS/WZW correspondence gives $CS_{k}(G)$ on $\Sigma\times\mathbb{R}$
with boundary VOA $V_{k}(\mathfrak{g})$ at integer level $k$. A
$\mathcal{V}^{\mathrm{prim}}$-boundary CS requires
$\prod_{p}U(1)_{p}$ at levels $k_{p}=1/\log p$ --- non-integer,
ill-defined on non-trivial 3-manifolds.
\end{observation}

\begin{corollary}
\label{v4-arith:cor:no-CS-VOA-arith}
\ClaimStatusProvedHere. No finite-level abelian CS delivers
$\mathcal{V}^{\mathrm{prim}}$ as boundary.
\end{corollary}

\section{BKL--Primon Is Semiclassical Coarse-Graining}
\label{v4-arith:wit:BKL-coarse-graining}

\begin{observation}[Selberg $\zeta$ vs primon $\zeta$]
\label{v4-arith:obs:selberg-vs-primon}
The Selberg trace formula on $\mathrm{SL}(2,\mathbb{Z})\backslash\mathbb{H}^{2}$
gives the spectral density of the Laplacian as a sum over Maass
cusp forms plus continuous Eisenstein. The Eisenstein scattering
phase involves $\zeta(2it)/\zeta(2it+1)$ on the critical line; the
primon partition involves $\zeta(s)$ with pole at $s=1$. Different
occurrences of $\zeta$: scattering matrix entry vs partition
function.
\end{observation}

\begin{theorem}[HY is coarse-graining, not exact]
\label{v4-arith:thm:HY-coarse-graining}
\ClaimStatusProvedElsewhere (Hartnoll--Yang arXiv:2502.02661,
\S3). The primon-BKL correspondence is a thermodynamic
coarse-graining, not an exact spectral match. The Selberg
continuous spectrum above $s=1/4$ is replaced by a coarse-grained
primon count at high temperature.
\end{theorem}

\section{The Automorphic Vertex}
\label{v4-arith:wit:automorphic-vertex}

\begin{theorem}[placement: automorphic vertex]
\label{v4-arith:thm:automorphic-vertex}
\ClaimStatusProvedHere. $\mathcal{V}^{\mathrm{prim}}$ sits on the
automorphic vertex of the Langlands--TQFT triangle:
\begin{equation*}
\underbrace{\bigl\{\text{auto reps of }GL_{1}(\mathbb{A}_{\mathbb{Q}})\bigr\}}_{\mathcal{V}^{\mathrm{prim}}\text{ here}}\;\longleftrightarrow\;\underbrace{\bigl\{\text{1D Galois reps}\bigr\}}_{\text{class field theory}}\;\longleftrightarrow\;\underbrace{\bigl\{\text{arithmetic TQFT}\bigr\}}_{\text{Deninger dream}}.
\end{equation*}
Its graded character is an $L$-function; its Heisenberg
decomposition is the Euler product. Class field theory
(Artin reciprocity) realises the first two vertices; the third
is Deninger's conjectural programme.
\end{theorem}

\section{The Asymptotic-Symmetry Reading: End-of-Time Sector}
\label{v4-arith:wit:asymptotic-symmetry}

\begin{proposition}[Damour--Henneaux--Nicolai hep-th/0207267]
\label{v4-arith:prop:DHN-billiard}
\ClaimStatusProvedElsewhere. BKL dynamics for 4D pure gravity on
$T^{3}$ reduces to null billiards on the Weyl chamber of $AE_{3}$,
a rank-$3$ hyperbolic Kac--Moody algebra; the fundamental domain
is a modification of $\mathrm{SL}_{2}(\mathbb{Z})\backslash\mathbb{H}^{2}$.
\end{proposition}

\begin{observation}[asymptotic-symmetry placement]
\label{v4-arith:obs:asymptotic-symmetry-placement}
For $\mathcal{V}^{\mathrm{prim}}$ to be a soft-mode algebra at the
BKL end-of-time boundary, it must decompose into Weyl-group
representations of $W(AE_{3})$. The primon Heisenberg generators
are indexed by primes (countably many); the $AE_{3}$ root lattice
has rank $3$. Literal identification fails.
\end{observation}

\begin{remark}
The correct reading: $\mathcal{V}^{\mathrm{prim}}$ is a bosonic
Fock quantisation of the Eisenstein continuous sector of the
gravitational superspace, not the full asymptotic-symmetry algebra.
The partition-function match of HY is a thermodynamic
coarse-graining of the Eisenstein density of states.
\end{remark}

\section{Vol III CY-to-Chiral Does Not Accept Arithmetic Input}
\label{v4-arith:wit:vol3-obstruction}

\begin{observation}
\label{v4-arith:obs:vol3-obstruction}
Vol III's $\Phi_{d}=\mathrm{Sp}^{\mathrm{ch}}_{\Sigma_{d-1},C}\circ\Phi^{\mathrm{FA}}_{d}$
requires CY$_{d}$ input on a $\mathbb{C}$-analytic variety.
$\overline{\mathrm{Spec}\,\mathbb{Z}}$ is an arithmetic $1$-scheme
over $\mathbb{Z}$, not a Calabi--Yau. Vol III's machinery does not
accept arithmetic input.
\end{observation}

\begin{remark}
Some interaction: if a CY$_{3}$ manifold carries
$\mathrm{SL}_{2}(\mathbb{Z})$ action on its elliptic-fibration
moduli (e.g.\ K3 $\times$ $E$), the modular-form side of $\Phi_{3}$
carries arithmetic data. The Borcherds product $\Phi_{N}$
($N\in\{1,2,3,4,6\}$) lives on $\mathrm{Sp}_{4}(\mathbb{Z})$-moduli.
These are automorphic objects; they sit on the same (automorphic)
vertex as $\mathcal{V}^{\mathrm{prim}}$.
\end{remark}

\section{Verdict: Automorphic Only}
\label{v4-arith:wit:verdict-automorphic}

\begin{theorem}[Witten's verdict]
\label{v4-arith:thm:witten-verdict}
\ClaimStatusProvedHere. No TQFT bulk for $\mathcal{V}^{\mathrm{prim}}$
exists in 2026's mathematical physics at the level of rigorous
construction. Every candidate framework (Kapustin--Witten,
BLLPRvR, Costello--Gaiotto HT-twist, Vol~III $\Phi_{d}$) fails at
a specific structural obstruction. $\mathcal{V}^{\mathrm{prim}}$
is automorphic-vertex, not TQFT-vertex, in the Langlands--TQFT
triangle.
\end{theorem}

\begin{remark}[what would change the verdict]
A rigorous construction of Deninger's $\mathbb{F}_{1}$-geometric
foundation, making $\overline{\mathrm{Spec}\,\mathbb{Z}}$ a
``curve over $\mathbb{F}_{1}$'' with a genuine Frobenius-type
action of $\mathbb{R}_{>0}$, would supply the TQFT-bulk vertex.
The arithmetic site of Connes--Consani arXiv:1405.4527 points in
this direction.
\end{remark}

\section{Summary of This Chapter}
\label{v4-arith:wit:summary}

\begin{enumerate}
\item Kapustin--Witten does not lift: no complex structure on
$\overline{\mathrm{Spec}\,\mathbb{Z}}$
(\Cref{v4-arith:cor:no-arith-4D-SYM}).
\item BLLPRvR fails: infinite-rank mismatch
(\Cref{v4-arith:cor:BLLPRvR-fails-arith}).
\item 3D CS fails: non-integer levels
(\Cref{v4-arith:cor:no-CS-VOA-arith}).
\item HY BKL--primon is thermodynamic coarse-graining
(\Cref{v4-arith:thm:HY-coarse-graining}), not exact pairing.
\item Vol III $\Phi_{d}$ does not accept arithmetic input
(\Cref{v4-arith:obs:vol3-obstruction}).
\item Placement: automorphic vertex of Langlands--TQFT triangle
(\Cref{v4-arith:thm:automorphic-vertex}); TQFT vertex is
Deninger's unconstructed programme.
\end{enumerate}

The TQFT-bulk reading of $\mathcal{V}^{\mathrm{prim}}$ is
dispositional; the automorphic reading is rigorous. The
Hartnoll--Yang correspondence motivates constructing an arithmetic
TQFT but does not provide one.
