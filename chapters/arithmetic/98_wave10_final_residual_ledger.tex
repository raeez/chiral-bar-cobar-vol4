\chapter{Wave-10 Final Residual Ledger: Platonic Closure of the Arithmetic Branch}
\label{v4-arith:w10-final:chapter}

\emph{This chapter records the post-wave-10 arithmetic ledger. It
does not assert RH. It isolates exactly what remains after the
Koszul, chiral-homology, and 3D arithmetic Chern--Simons constructions
have been pushed to their natural boundary: one RH-equivalent
statement, four upstream conjectures outside the internal Vol~IV
construction, one named sub-RH Ramanujan hypothesis, and two small
density-zero residuals. The original Vol~IV HZ-IV decorator campaign
is recorded separately; it belongs to the realization programme proper,
not to the arithmetic-branch mission.}

\section{W10-A Bonus Theorem}
\label{v4-arith:w10-final:w10-a}

\begin{theorem}[W10-A bonus theorem]
\label{v4-arith:w10-final:thm:cs-descent-atp-rh}
\ClaimStatusProvedHereConditional \emph{(on the Clausen--Scholze
condensed-Hilbert descent of
\Cref{v4-arith:w10-a:conj:condensed-descent} and A-TP
\Cref{v4-arith:w5-a:conj:archimedean-tate-positivity})}. The
conjunction
\[
\text{Clausen--Scholze condensed-Hilbert descent}\quad+\quad
\text{A-TP}
\]
implies the Riemann Hypothesis. The descent alone does not decide RH.
After the descent is assumed, A-TP is the only RH-equivalent
obstruction remaining in the W10-A bulk package.
\end{theorem}

\begin{proof}
The implication is
\Cref{v4-arith:w10-a:thm:descent-plus-ATP-equals-RH}. The descent
constructs the condensed Hilbert bulk and identifies the
Wilson-surface action with the Arakelov scaling operator. It does
not assert positivity of the scaling quadratic form. That positivity
is precisely A-TP, equivalent to RH by
\Cref{v4-arith:w9-r2:thm:purity-ATP-RH}. Hence descent plus A-TP
implies RH, while descent without A-TP supplies the bulk and not the
zero-line assertion.
\end{proof}

\begin{remark}[logical shape]
\label{v4-arith:w10-final:rem:w10-a-shape}
The theorem separates construction from positivity. The condensed
descent is the construction statement for the quantum ACS bulk. A-TP
is the positivity statement. Only the second carries RH-equivalent
content.
\end{remark}

equivalent to RH by
\Cref{v4-arith:w9-r2:thm:purity-ATP-RH}.

\begin{proof}

and Chapter~\ref{v4-arith:w8-atp:chapter} identify the same residual
through the direct, Nyman--Beurling, de Branges, Gaussian,
Mellin--Barnes, Hadamard, Gram, Guinand--Weil, adelic-scattering,
Arthur, geometric-Langlands, Iwasawa, quantum-chaos, and Arakelov
coordinates. Chapter~\ref{v4-arith:w9-r2:archimedean-wilson-surface}
identifies it with RH. No later wave removes this residual.

\section{Tier 2 Supplement: The Sub-RH Ramanujan Hypothesis}
\label{v4-arith:w10-final:sub-rh}

\begin{hypothesis}[GRC for $\mathrm{GL}_{4}/\mathbb{Q}$ in the Selberg class]
\label{v4-arith:w10-final:hyp:GRC-GL4}
\ClaimStatusConditional. Every cuspidal automorphic representation of
$\mathrm{GL}_{4}/\mathbb{Q}$ in the Selberg-class envelope relevant to
the Rankin--Selberg square of a Maass form is tempered at every finite
place.
\end{hypothesis}

\begin{proposition}[GRC-$\mathrm{GL}_{4}$ closes the Maass sector]
\label{v4-arith:w10-final:prop:GRC-GL4}
\ClaimStatusProvedHere. Conditional on
\Cref{v4-arith:w10-final:hyp:GRC-GL4}, full Ramanujan holds for the
Maass forms entering the full $\mathcal V^{\mathrm{prim}}$ P3
identity. This closes the Maass sector of
\Cref{v4-arith:w10-c:thm:main}. The hypothesis is not classical RH;
Vol~IV records it as a named sub-RH conditional independent of the
RH-equivalent A-TP obstruction.
\end{proposition}

\begin{proof}
Chapter~\ref{v4-arith:w10-c:chapter} proves that the Arthur--Ramakrishnan
sub-RH hypothesis implies full Ramanujan for Maass forms
\Cref{v4-arith:w10-c:thm:AR-implies-Ramanujan}, and reduces that
hypothesis to GRC for $\mathrm{GL}_{4}/\mathbb{Q}$
\Cref{v4-arith:w10-c:prop:sub-RH-strict}. The same chapter records
the exact conditional P3 identity in
\Cref{v4-arith:w10-c:thm:main}. Since A-TP concerns the zero
placement of $\zeta$ through Weil positivity and GRC-$\mathrm{GL}_{4}$
concerns Satake temperedness for a higher-rank automorphic family,
the two residuals occupy different Selberg-class axes.
\end{proof}

\section{Structural Assessment}
\label{v4-arith:w10-final:assessment}

\begin{theorem}[post-wave-10 arithmetic near-closure]
\label{v4-arith:w10-final:thm:near-closure}
\ClaimStatusProvedHere. The arithmetic branch is at near-closure in
the following precise sense. The pre-wave-5 residual corpus has been
reduced to:
\begin{enumerate}
\item one RH-equivalent statement: A-TP;
\item four upstream conjectures outside the branch's internal
constructions: Wild-EG enhancement, $\Pi$-flat pro-\'etale flatness,
arithmetic AG singular support, and Clausen--Scholze
condensed-Hilbert descent;
\item one named sub-RH hypothesis: GRC for
$\mathrm{GL}_{4}/\mathbb{Q}$;
\item two small residuals of combined scope $3$--$6$~pp and density
zero.
\end{enumerate}
The original Vol~IV HZ-IV decorator campaign remains open and
separate.
\end{theorem}

\begin{proof}
Item (1) is \Cref{v4-arith:w10-final:prop:tier1}. Item (2) is
\Cref{v4-arith:w10-final:thm:four-upstream}. Item (3) is
\Cref{v4-arith:w10-final:prop:GRC-GL4}. Item (4) is
\Cref{v4-arith:w10-final:prop:small-residuals}. The separation from
the original realization programme is
\Cref{v4-arith:w10-final:prop:hziv-separate}. No residual outside
these classes is left by Chapters~\ref{v4-arith:w5-synth:chapter}--\ref{v4-arith:w10-d:chapter}.
\end{proof}

\begin{remark}[what this does not say]
\label{v4-arith:w10-final:rem:not-rh}
The theorem does not prove RH. It says the arithmetic branch has
isolated the RH content into A-TP and has pushed the other large
obstructions into named upstream conjectures. Any proof of RH through
this branch still needs A-TP, or an equivalent theorem.
\end{remark}
