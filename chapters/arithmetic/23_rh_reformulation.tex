\chapter{The Riemann Hypothesis Reformulation}
\label{v4-arith:rh-reformulation}

\emph{With P1, P2, P3 resolved to the extent available
(Chapter~\ref{v4-arith:closure}), the Riemann Hypothesis admits a
precise reformulation as an arithmetic chiral-algebra statement:
$\mathcal{V}^{\mathrm{prim}}$ is a unitary self-dual arithmetic
chiral algebra whose character has its non-trivial zeros on the
self-dual locus. This chapter inscribes the reformulation, its
equivalent Li-positivity form, and its placement against existing
RH programs.}

\section{The Conjecture F-RH}
\label{v4-arith:rh:F-RH-statement}

\begin{conjecture}[arithmetic unitary bootstrap: F-RH]
\label{v4-arith:conj:F-RH}
$\mathcal{V}^{\mathrm{prim}}$ admits a unique positive-definite
Hermitian structure compatible with arithmetic Koszul self-duality
$\kappa^{\mathrm{Ar}}+(\kappa^{!})^{\mathrm{Ar}}=0$ and with
arithmetic Zhu modularity. Under this structure, the non-trivial
zeros of the entire character $\xi_{\mathbb R}(s)$ of
$\mathcal{V}^{\mathrm{prim}}$ lie on the self-dual locus
$\mathrm{Re}(s)=1/2$.
\end{conjecture}

\begin{remark}[equivalence to RH]
\label{v4-arith:rh:rem:equivalence-to-RH}
The Tate character supplied by P2 is the meromorphic completed zeta
function
\[
\Lambda_{\zeta}(s):=\pi^{-s/2}\Gamma(s/2)\zeta(s),
\]
with simple poles at \(s=0,1\). The entire Li--Hadamard object is
\[
\xi_{\mathbb R}(s):=\frac12s(s-1)\Lambda_{\zeta}(s).
\]
Throughout this chapter \(\xi\) means \(\xi_{\mathbb R}\) unless
\(\Lambda_{\zeta}\) is written explicitly. The zeros of
\(\xi_{\mathbb R}\) are exactly the non-trivial zeros of \(\zeta\);
the gamma poles cancel the negative even trivial zeros of
\(\zeta\), and the factor \(s(s-1)\) removes the Tate poles.
Hence F-RH \(\Leftrightarrow\) RH.
\end{remark}

\section{Three-Pillar Implication Structure}
\label{v4-arith:rh:three-pillar}

\begin{theorem}[F-RH from three pillars plus bootstrap]
\label{v4-arith:rh:thm:three-pillar-implication}
\ClaimStatusProvedHere. Suppose:
\begin{itemize}
\item[(P1)] $\kappa^{\mathrm{Ar}}_{\mathsf{G}}+(\kappa^{!})^{\mathrm{Ar}}_{\mathsf{G}}=0$
for $\mathcal{V}^{\mathrm{prim}}_{\mathrm{Heis}}$
(\Cref{v4-arith:thm:P1-resolution}).
\item[(P2)] $\xi(s)$ satisfies arithmetic Zhu modularity with
$c^{\mathrm{Ar}}=2$ (\Cref{v4-arith:thm:P2-resolution}).
\item[(P3)] $\mathcal{V}^{\mathrm{prim}}$ carries a positive-definite
Petersson inner product compatible with arithmetic Koszul self-duality
(\Cref{v4-arith:thm:P3-resolution}).
\item[(VB)] \emph{Arithmetic Virasoro bootstrap.} A unitary
self-dual arithmetic chiral algebra with character invariant
under the $\mathbb{Z}/2$-involution $s\mapsto 1-s$ and with
arithmetic Zhu modularity has all non-trivial character zeros on
the self-dual locus $\mathrm{Re}(s)=1/2$.
\end{itemize}
Then F-RH holds. Equivalently, RH holds.
\end{theorem}

\begin{proof}
P1 gives the functional equation $\xi(s)=\xi(1-s)$, expressing
invariance of the character under the arithmetic $S$-modular
involution. P2 provides the Zhu flat connection forcing $\xi$ to
be an eigenfunction of the arithmetic Virasoro. P3 gives the
positive-definite Hermitian form making $\mathcal{V}^{\mathrm{prim}}$
unitary. VB then applies to deliver zero-placement on
$\mathrm{Re}(s)=1/2$, which is F-RH and hence RH.
\end{proof}

\begin{remark}[VB is the single open ingredient]
\label{v4-arith:rh:rem:VB-single-open}
With P1, P2, P3 resolved to Theorem or Theorem-conditional,
F-RH reduces to VB, the arithmetic Virasoro bootstrap. This is a
single precisely-stated open obligation. It is the arithmetic
analogue of the $c<1$ Friedan--Qiu--Shenker bootstrap for unitary
Virasoro representations, adapted to $c=2$ and the arithmetic
modular setting.
\end{remark}

\section{Li-Positivity as Arithmetic Virasoro Unitarity}
\label{v4-arith:rh:li-positivity}

\subsection{Li's Criterion}

\begin{theorem}[Li 1997, arXiv:math/9712243]
\label{v4-arith:rh:thm:Li-criterion}
\ClaimStatusProvedElsewhere. RH is equivalent to $\lambda_{n}\geq 0$
for all $n\geq 1$, where
\begin{equation}
\lambda_{n}\;:=\;\sum_{\rho}\biggl(1-\Bigl(1-\frac{1}{\rho}\Bigr)^{n}\biggr)\;=\;\frac{1}{(n-1)!}\frac{d^{n}}{ds^{n}}\bigl[s^{n-1}\log\xi(s)\bigr]\bigg|_{s=1}
\end{equation}
(sum over non-trivial zeros of $\zeta$, counted with multiplicity).
\end{theorem}

Li's $\lambda_{n}$ admits a direct Hilbert--P\'olya transform
interpretation. The transform lives on the zero operator
\(\Theta_{\xi}\), not on the Virasoro/Fock energy operator.

\subsection{Hilbert--P\'olya Transform Computation}

\begin{hypothesis}[arithmetic Hilbert--P\'olya, H-P$^{\mathrm{Ar}}$]
\label{v4-arith:rh:hyp:HPAr}
There exists a Hilbert completion \(\mathcal H_{\xi}\) of the primary
arithmetic Heisenberg core, a dense invariant core
\(\mathcal D_{\xi}\subset\mathcal H_{\xi}\), and a self-adjoint
unbounded operator
\[
H_{\mathrm{HP}}:\mathcal D_{\xi}\subset\mathcal H_{\xi}\longrightarrow
\mathcal H_{\xi}
\]
in the Petersson/GNS inner product. Put
\[
\Theta_{\xi}:=\frac12\mathrm{id}+iH_{\mathrm{HP}} .
\]
The regularised determinant and trace distribution of
\(\Theta_{\xi}\) satisfy
\begin{equation}
\label{v4-arith:rh:eq:HP-determinant}
\det_{\infty}(s\mathrm{id}-\Theta_{\xi})=\xi_{\mathbb R}(s),
\end{equation}
and, for every admissible Weil--Connes Paley--Wiener test function
\(h\),
\begin{equation}
\label{v4-arith:rh:eq:HP-trace}
\mathrm{Tr}_{\mathrm{reg}}\bigl(h(H_{\mathrm{HP}})\bigr)
=\sum_{\rho}h(\gamma_{\rho}),
\qquad \rho=\frac12+i\gamma_{\rho}.
\end{equation}
The symbolic kernel \(\sum_{\rho}e^{-t\rho}\) is only the
distributional wave-kernel shorthand obtained from
\eqref{v4-arith:rh:eq:HP-trace} by admissible test-function
regularisation; it is not an ordinary heat trace of a non-negative
self-adjoint operator.
\end{hypothesis}

\begin{remark}[status of H-P$^{\mathrm{Ar}}$]
\label{v4-arith:rh:rem:HPAr-status}
H-P$^{\mathrm{Ar}}$ is the Hilbert--P\'olya conjecture stated
inside the arithmetic chiral-algebra framework. It is not
derivable from P1, P2, P3 alone: P2 supplies only a \emph{formal}
Virasoro action with commutation relations at the level of
generating functions, and P3 supplies the Petersson domain and
inner product on the scoped sph-finite cuspidal core. The
functional-analytic upgrade to a genuine self-adjoint ordinate
operator \(H_{\mathrm{HP}}\), together with the determinant identity
\eqref{v4-arith:rh:eq:HP-determinant} and the distributional trace
identity \eqref{v4-arith:rh:eq:HP-trace}, is an independent input and
is treated here as an explicit hypothesis.
\end{remark}

\begin{remark}[reconciliation with Ch.~\ref{v4-arith:polyakov}: primary sector]
\label{v4-arith:rh:rem:Virasoro-reconciliation}
Chapter~\ref{v4-arith:polyakov} (Polyakov) proves that the
\emph{full} $\mathcal{V}^{\mathrm{prim}}$ carries only a Witt
algebra, not an honest Virasoro: the infinite-tensor central term
$c_{\mathrm{full}}=\sum_{p}c^{(p)}$ diverges, and no
normal-ordering procedure preserves the Virasoro bracket on the
full tensor. $c_{\mathrm{global}}=2$
(\Cref{v4-arith:thm:c-as-residue}) exists only as the residue
$2\cdot\mathrm{Res}_{s=1}\zeta(s)$, not as a trace.

The operator \(H_{\mathrm{HP}}\) of H-P$^{\mathrm{Ar}}$ is therefore
\emph{not} a Virasoro zero-mode on the full
\(\mathcal{V}^{\mathrm{prim}}\). It is a separate spectral
realisation operator. The restriction of the per-prime Virasoro
action \(L_{m}=\sum_{p}L_{m}^{(p)}\) to a primary finite-support core
is still useful for Fock computations, but the claim that the central
extension regularises to \(c_{\mathrm{reg}}=2\) on a completed primary
sector is itself a substantive conjecture:
\end{remark}

\begin{conjecture}[primary-Virasoro reconciliation]
\label{v4-arith:rh:conj:primary-Virasoro}
On the primary subspace
$\mathcal{V}^{\mathrm{prim}}_{\mathrm{Heis},\mathrm{prim}}$, the
Witt action of Ch.~\ref{v4-arith:polyakov} lifts to an honest
Virasoro action with centrally-extended bracket
\begin{equation}
[L_{m}^{\mathrm{Ar}},L_{n}^{\mathrm{Ar}}]\;=\;(m-n)L_{m+n}^{\mathrm{Ar}}+\frac{c_{\mathrm{reg}}}{12}(m^{3}-m)\delta_{m+n,0},\qquad c_{\mathrm{reg}}=2,
\end{equation}
with $c_{\mathrm{reg}}$ matching the residue
$c_{\mathrm{global}}=2\cdot\mathrm{Res}_{s=1}\zeta(s)$ of
\Cref{v4-arith:thm:c-as-residue}. Equivalently, the per-prime
central-term sum $\sum_{p}c^{(p)}$, divergent on the full
$\mathcal{V}^{\mathrm{prim}}$, is per-state finite on
$\mathcal{V}^{\mathrm{prim}}_{\mathrm{Heis},\mathrm{prim}}$ and
regularises to $2$ in the zeta-regularised sense.
\end{conjecture}

\begin{remark}[operator dictionary]
\label{v4-arith:rh:rem:spectral-convention}
Four operators must remain distinct. \(L_{0}^{\mathrm{Vir}}\) is the
Fock/Virasoro conformal-energy operator. \(N_{\mathrm{BC}}\) is the
Bost--Connes Hamiltonian, with eigenvalues \(\log m\) on the
zero-mode lattice and partition function \(\zeta(\beta)\).
\(H_{\mathrm{HP}}\) is the hypothetical self-adjoint ordinate
operator. \(\Theta_{\xi}=1/2+iH_{\mathrm{HP}}\) is the normal
zero operator whose determinant is \(\xi_{\mathbb R}\). Conflating
these four objects turns Euler-product positivity into a false
Hilbert--P\'olya argument.
\end{remark}

\begin{remark}[Tate-pole and trivial-zero normalisation]
\label{v4-arith:rh:rem:trivial-zero}
\(\Lambda_{\zeta}(s)=\pi^{-s/2}\Gamma(s/2)\zeta(s)\) is meromorphic,
not entire: the negative even trivial zeros of \(\zeta\) are cancelled
by gamma poles, and \(\Lambda_{\zeta}\) has poles at \(0\) and \(1\).
The entire function
\(\xi_{\mathbb R}(s)=\frac12s(s-1)\Lambda_{\zeta}(s)\) removes these
Tate poles and has as zeros exactly the non-trivial zeros. Thus
H-P$^{\mathrm{Ar}}$ is a statement about the primary determinant
\(\xi_{\mathbb R}\), while the meromorphic Tate character
\(\Lambda_{\zeta}\) belongs to the cohomological/superdeterminant
surface.
\end{remark}

\begin{proposition}[Li coefficient = Hilbert--P\'olya transform trace]
\label{v4-arith:rh:prop:Li-is-Virasoro}
\ClaimStatusProvedHere \emph{(conditional on
\Cref{v4-arith:rh:hyp:HPAr})}.
Assume H-P$^{\mathrm{Ar}}$. For \(n\geq1\), define the normal
Li-transform operator
\begin{equation}
\label{v4-arith:rh:eq:Li-transform-operator}
\mathcal L_{n}:=\mathrm{id}-\bigl(\mathrm{id}-\Theta_{\xi}^{-1}\bigr)^{n}.
\end{equation}
Then
\begin{equation}
\label{v4-arith:rh:eq:Li-transform-trace}
\lambda_{n}=\mathrm{Tr}_{\mathrm{reg}}(\mathcal L_{n}).
\end{equation}
This is a trace identity for the Hilbert--P\'olya zero operator
\(\Theta_{\xi}\), not a polynomial identity in
\(L_{0}^{\mathrm{Vir}}\) or \(N_{\mathrm{BC}}\).
\end{proposition}

\begin{proof}
The determinant identity
\(\det_{\infty}(s\mathrm{id}-\Theta_{\xi})=\xi_{\mathbb R}(s)\)
identifies the spectral multiset of \(\Theta_{\xi}\), in the
regularised determinant sense, with the non-trivial zero multiset.
Applying the holomorphic functional calculus to
\(f_{n}(z)=1-(1-z^{-1})^{n}\) gives
\[
\mathrm{Tr}_{\mathrm{reg}}(f_{n}(\Theta_{\xi}))
=\sum_{\rho}\left(1-\left(1-\frac1{\rho}\right)^{n}\right),
\]
which is Li's coefficient by \Cref{v4-arith:rh:thm:Li-criterion}.
\end{proof}

\begin{corollary}[Li positivity from positive HP trace]
\label{v4-arith:rh:cor:VB-implies-Li}
\ClaimStatusProvedHere \emph{(conditional on P2, P3, H-P$^{\mathrm{Ar}}$)}.
Assume the regularised Hilbert--P\'olya trace is positive on the
Weil positive cone and that the Li transforms
\(\mathcal L_{n}\) of \eqref{v4-arith:rh:eq:Li-transform-operator}
belong to the closure of that positive cone after the standard
Li/Weil approximation. Then \(\lambda_{n}\geq0\) for every
\(n\geq1\). The converse is \emph{not} an operator-positivity theorem.
\end{corollary}

\begin{proof}
By the positivity assumption,
\(\mathrm{Tr}_{\mathrm{reg}}(\mathcal L_{n})\geq0\). The trace
identity \eqref{v4-arith:rh:eq:Li-transform-trace} identifies this
number with \(\lambda_{n}\). The reverse implication would require a
determinate positive trace representation of the full zero measure
from the scalar Li sequence; that is the classical Li/Weil positivity
problem, not a consequence of P1--P3.
\end{proof}

\begin{theorem}[HP--Li positivity implies F-RH]
\label{v4-arith:rh:thm:VB-implies-F-RH}
\ClaimStatusProvedHere \emph{(conditional on P2, P3, H-P$^{\mathrm{Ar}}$)}.
Assume P2, P3, and H-P$^{\mathrm{Ar}}$. If HP--Li positivity holds:
\begin{itemize}
\item[\textbf{(HP--Li)}] \emph{Positive Hilbert--P\'olya trace.}
\(\mathrm{Tr}_{\mathrm{reg}}(\mathcal L_{n})\geq0\) for every
\(n\geq1\), where \(\mathcal L_{n}\) is
\eqref{v4-arith:rh:eq:Li-transform-operator},
\end{itemize}
then F-RH holds. Equivalently, RH holds.
\end{theorem}

\begin{proof}
HP--Li gives \(\lambda_{n}\geq0\) for all \(n\geq1\) by
\Cref{v4-arith:rh:prop:Li-is-Virasoro}. Li positivity is equivalent
to RH (Li 1997,
\Cref{v4-arith:rh:thm:Li-criterion}); RH $\Leftrightarrow$ F-RH
(\Cref{v4-arith:rh:rem:equivalence-to-RH}).
\end{proof}

\begin{remark}[implication is one-directional]
\label{v4-arith:rh:rem:one-directional}
The prior draft of this chapter asserted F-RH \(\Leftrightarrow\)
VB$^{*}$ through the polynomial
\((L_{0}^{\mathrm{Ar}})^{n}-(L_{0}^{\mathrm{Ar}}-\mathrm{id})^{n}\).
That polynomial is not Li's transform and \(L_{0}^{\mathrm{Ar}}\)
was being used for three incompatible operators. The correct
one-directional statement is HP--Li positivity
\(\Rightarrow\) F-RH, after H-P$^{\mathrm{Ar}}$ supplies
\(\Theta_{\xi}\). The converse---whether F-RH supplies a positive
Weil trace representation of the whole zero distribution---is open
and is exactly the Hilbert--P\'olya/Connes--Deninger obstruction.
\end{remark}

\begin{remark}[what is gained by the reformulation]
\label{v4-arith:rh:rem:what-is-gained}
The reformulation does not prove RH. What it produces:
\begin{enumerate}
\item A precise spectral target (H-P$^{\mathrm{Ar}}$ plus HP--Li
positivity) equivalent to RH once the positive trace is supplied.
\item A structural reason why RH should be true: unitarity of
$\mathcal{V}^{\mathrm{prim}}$ (Petersson positivity) should propagate
through the Weil/Connes trace functional to give positivity of the
Li transforms.
\item A bridge between RH-programs (Connes, Deninger, Meyer) and
classical VOA bootstrap methods (Friedan--Qiu--Shenker,
Cardy, Zhu).
\item A computational target: admissible approximants to the
Li/Weil test functions can be evaluated against the Bost--Connes and
Hecke trace packages. Positivity is a distributional trace question,
not finite Fock positivity of \(T_{n}\).
\end{enumerate}
\end{remark}

\section{Residual Obligations}
\label{v4-arith:rh:residual}

After the structural reduction of Chapter~\ref{v4-arith:closure} and
the reformulation of this chapter, RH reduces to the following
five precisely-stated obligations, any one of which closes the
circuit at different levels:

\begin{enumerate}
\item[\textbf{(HP--Li)}] \emph{Positive Hilbert--P\'olya trace.}
\(\mathrm{Tr}_{\mathrm{reg}}(\mathcal L_{n})\geq0\) for every
\(n\geq1\), where \(\mathcal L_{n}\) is the Li transform of
\(\Theta_{\xi}\)
(\Cref{v4-arith:rh:thm:VB-implies-F-RH}).
\emph{Sufficient for RH given P2, P3, H-P$^{\mathrm{Ar}}$.}
\item[\textbf{(H-P$^{\mathrm{Ar}}$)}] \emph{Arithmetic Hilbert--P\'olya.}
There is a self-adjoint ordinate operator \(H_{\mathrm{HP}}\) such
that \(\Theta_{\xi}=1/2+iH_{\mathrm{HP}}\) has determinant
\(\xi_{\mathbb R}\) and Weil--Connes trace distribution equal to the
non-trivial-zero distribution
(\Cref{v4-arith:rh:hyp:HPAr}).
\emph{Required for the HP--Li \(\Rightarrow\) F-RH chain; not
derivable from P1, P2, P3.}
\item[(arith.~Positselski)] Arithmetic Positselski equivalence
(\Cref{v4-arith:thm:arithmetic-positselski}).
\emph{Required for full P1 beyond the forward implication.}
\item[(Koszul--Pet compat)] Identification of arithmetic Koszul
pairing with adelic Petersson pairing
(\Cref{v4-arith:cl:cj:koszul-Petersson-compat}).
\emph{Required for P3.}
\item[($\mathrm{L}_{\mathrm{ACD}}$)] Adelic categorical descent
for $\mathrm{D}^{\mathrm{sph\text{-}fin}}$
(\Cref{v4-arith:cl:cor:F1-single-lemma}).
\emph{Required for F1, hence for six-window placement.}
\item[(F3.BZN$^{\mathrm{Ar}}$)] Arithmetic Ben-Zvi--Nadler
identification (\Cref{v4-arith:thm:F3-reduction}).
\emph{Required for F3, hence for the categorical trace presentation
of $\mathcal{V}^{\mathrm{prim}}$.}
\end{enumerate}

Items (arith.~Positselski), (Koszul--Pet compat), H-P$^{\mathrm{Ar}}$,
and HP--Li positivity lie on the P1--P2--P3 spine of F-RH and together
form the minimal sufficient set for RH through Vol~IV's
chiral-algebra framework. Items ($\mathrm{L}_{\mathrm{ACD}}$) and
(F3.BZN$^{\mathrm{Ar}}$) are independent of RH proper and lie on
the six-window-presentation spine; they would inscribe
$\mathcal{V}^{\mathrm{prim}}$ as a theorem-grade object but do
not themselves close RH.

\section{The Minimal Path to RH}
\label{v4-arith:rh:minimal-path}

\begin{theorem}[minimal sufficient path]
\label{v4-arith:rh:thm:minimal-path}
\ClaimStatusProvedHere. The minimal sufficient path to RH within
Vol~IV's framework is:
\begin{center}
\begin{tikzcd}[row sep=1em, column sep=small]
\text{P1} \ar[r, "\text{arith.~Positselski}"] & \text{P1 functional equation}\\
\text{P2} \ar[r, "\text{Tate 1950}"] & \text{P2 unconditional}\\
\text{P3} \ar[r, "\text{Koszul--Pet compat}"] & \text{P3 scoped unitarity}\\
\text{(P1, P2, P3, H-P}^{\mathrm{Ar}}\text{)} \ar[r, "\text{HP--Li}"] & \text{F-RH} \ar[r, "\Leftrightarrow"] & \text{RH}
\end{tikzcd}
\end{center}
Four inputs suffice: arithmetic Positselski, Koszul--Pet
compatibility, arithmetic Hilbert--P\'olya (H-P$^{\mathrm{Ar}}$),
and positive Hilbert--P\'olya Li trace (HP--Li).
\end{theorem}

\begin{proof}
Structural closure of the diagram: P2 closes unconditionally as the
completed-zeta character/explicit-formula input
(\Cref{v4-arith:thm:P2-resolution}); P1 is the functional-equation
spine conditional on arithmetic Positselski
(\Cref{v4-arith:thm:P1-resolution}); P3 is the positive-form input
conditional on the scoped Koszul--Petersson compatibility
(\Cref{v4-arith:thm:P3-resolution}). Given these inputs plus
H-P$^{\mathrm{Ar}}$,
\Cref{v4-arith:rh:thm:VB-implies-F-RH} gives
HP--Li \(\Rightarrow\) F-RH, and F-RH $\Leftrightarrow$ RH via Li's
criterion (\Cref{v4-arith:rh:rem:equivalence-to-RH},
\Cref{v4-arith:rh:thm:Li-criterion}).
\end{proof}

\begin{remark}[paper-length estimate for the minimal path]
\label{v4-arith:rh:rem:minimal-path-paper}
\begin{description}
\item[arith.~Positselski] 30--50 pages; Iwasawa-theoretic Positselski.
\item[Koszul--Pet compat] 10--20 pages; compatibility verification.
\item[H-P$^{\mathrm{Ar}}$] Unknown. This is the Hilbert--P\'olya
operator-construction problem in the arithmetic Fock setting; the
domain theorem alone does not supply the zeta-zero spectrum.
\item[HP--Li] Unknown. This is the 100-year-old open problem in
positive-trace form. If tractable through Li coefficients, the
$n=2$ case ($\lambda_{2}\geq 0$) has been
verified numerically to $n\sim 10^{5}$ zeros of $\zeta$ (Coffey
2005; Voros 2003 for the first few). A general proof remains open.
\end{description}
The arithmetic framework does not shorten the Hilbert--P\'olya
obstacle, but it places the missing positivity in a precise
chain-level trace setting where VOA and factorisation techniques may
be tested without confusing them with the zero-spectrum theorem.
\end{remark}

\section{Relation to the Arithmetic Branch Load-Bearing Theorem}
\label{v4-arith:rh:load-bearing-relation}

The arithmetic branch's load-bearing theorem
(\Cref{v4-arith:thm:G-row-Riemann-functional-equation},
\Cref{v4-arith:thm:P1-resolution}) is
\begin{equation}
\kappa^{\mathrm{Ar}}_{\mathsf{G}}+(\kappa^{!})^{\mathrm{Ar}}_{\mathsf{G}}=0\;\Longleftrightarrow\;\xi(s)=\xi(1-s).
\end{equation}
This delivers the functional equation (geometric self-duality of
\(\xi_{\mathbb R}\)), not RH (spectral zero-placement). RH is
strictly beyond the load-bearing theorem, requiring the additional
positive Hilbert--P\'olya trace input.

F-RH is the Vol~IV capstone question: it asks whether the Koszul
self-duality that delivered the functional equation, combined with
the positivity and modularity that Vol~IV inscribes, suffices to
deliver the spectral zero-placement that RH asserts. The answer
within Vol~IV's framework is: yes, provided H-P$^{\mathrm{Ar}}$ and
HP--Li positivity are verified. Those two statements are the
Vol~IV-natural Hilbert--P\'olya form of RH.

\section{Summary of This Chapter}

\begin{enumerate}

\item F-RH is implied by VB (arithmetic Virasoro bootstrap;

\Cref{v4-arith:rh:thm:VB-implies-F-RH}) via Li's criterion, given
H-P$^{\mathrm{Ar}}$. The converse direction F-RH $\Rightarrow$
HP--Li is open and is the positive-trace realisation problem.
\item Li's coefficients $\lambda_{n}$ are Hilbert--P\'olya
transform traces of \(\Theta_{\xi}\)
(\Cref{v4-arith:rh:prop:Li-is-Virasoro}).
\item F-RH subsumes Li positivity and Weil positivity; it refines
Hilbert--P\'olya and aligns with Connes more than Deninger.
\item The minimal closure path to RH is four inputs:
arith.~Positselski, Koszul--Pet compat, H-P$^{\mathrm{Ar}}$, and
HP--Li positivity
(\Cref{v4-arith:rh:thm:minimal-path}). This is the
\emph{four-input conditional spine} audited in
Ch.~\ref{v4-arith:w3-audit-B} and made precise as

carries its own open content (five Positselski sub-items,

zeta-zero spectral matching, RH-equivalent positive trace),
and the conjunction is the honest minimal path.
\item RH is strictly beyond the branch's load-bearing theorem

positive-trace statement of RH. The operator-dictionary repair of

positive spectral-realisation problem
(\Cref{v4-arith:hpvbconv:cor:three-strictly-finer}) and does not
help the forward direction.
\end{enumerate}
